1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.LoadingContainer = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Colibri.UI.Templates['Colibri.UI.LoadingContainer']);
 15         this.AddClass('app-loading-container-component');
 16         
 17         this._loadingcontainer = this.Children('icon-container');
 18         this._progress = this.Children('progress');
 19     }
 20 
 21     /**
 22      * Loading icon
 23      * @type {String}
 24      */
 25     set icon(value) {
 26         this._loadingcontainer.value = value;
 27     }
 28     /**
 29      * Loading icon
 30      * @type {String}
 31      */
 32     get icon() {
 33         return this._loadingcontainer.value;
 34     }
 35 
 36     /**
 37      * Show icon
 38      * @type {boolean}
 39      */
 40     set shownIcon(value) {
 41         this._loadingcontainer.shown = value === 'true' || value === true;
 42     }
 43     /**
 44      * Show icon
 45      * @type {boolean}
 46      */
 47     get shownIcon() {
 48         return this._loadingcontainer.shown;
 49     }
 50 
 51     /**
 52      * Progress value
 53      * @type {number}
 54      */
 55     set progress(value) {
 56         this._progress.progress = value;
 57     }
 58     /**
 59      * Progress value
 60      * @type {number}
 61      */
 62     get progress() {
 63         return this._progress.progress;
 64     }
 65 
 66     /**
 67      * Progress bar color
 68      * @type {string}
 69      */
 70     set progressColor(value) {
 71         this._progress.color = value;
 72     }
 73     /**
 74      * Progress bar color
 75      * @type {string}
 76      */
 77     get progressColor() {
 78         return this._progress.color;
 79     }
 80 
 81     /**
 82      * Opacity
 83      * @type {number}
 84      */
 85     set opacity(value) {
 86         this._element.css('opacity', value)
 87     }
 88 
 89     /**
 90      * Opacity
 91      * @type {number}
 92      */
 93     get opacity() {
 94         return this._element.css('opacity');
 95     }
 96 
 97     /**
 98      * Start progress
 99      * @param {number} timer timer value
100      * @param {number} speed speed
101      */
102     StartProgress(timer, speed) {
103         this.BringToFront();
104         this.Children('progress').Start(timer, speed);
105     }
106 
107     /**
108      * Pause timer
109      */
110     PauseProgress() {
111         this.BringToFront();
112         this.Children('progress').Pause();
113     }
114 
115     /**
116      * Stop timer
117      */
118     StopProgress(timer) {
119         this.BringToFront();
120         this.Children('progress').Stop();
121     }
122 
123     /**
124      * Show component
125      */
126     Show() {
127         super.Show();
128         this.BringToFront();        
129     }
130 
131 }
