1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.FlexBox = class extends Colibri.UI.Component {
  7 
  8     static Horizontal = 'row';
  9     static Vertical = 'column'
 10 
 11     constructor(name, container, element) {
 12         super(name, container, element || Element.create('div'));
 13         this.AddClass('app-component-flexbox');
 14 
 15         // this.wrap = false;
 16     }
 17 
 18     get wrap() {
 19         let flexWrap = this._element.css('flex-wrap');
 20         switch (flexWrap) {
 21             case 'wrap':
 22                 return true;
 23             case 'nowrap':
 24                 return false;
 25         }
 26     }
 27 
 28     set wrap(value) {
 29         if (value) {
 30             this._element.css('flex-wrap', 'wrap');
 31         } else {
 32             this._element.css('flex-wrap', 'nowrap');
 33         }
 34     }
 35 
 36     get direction() {
 37         return this._element.css('flex-direction');
 38     }
 39 
 40     set direction(value) {
 41         if ([Colibri.UI.FlexBox.Horizontal, Colibri.UI.FlexBox.Vertical].indexOf(value) !== -1) {
 42             this._element.css('flex-direction', value);
 43         }
 44     }
 45 }
