1 /**
  2  * @class
  3  * @extends Colibri.UI.Forms.Field
  4  * @memberof Colibri.UI.Forms
  5  */
  6 Colibri.UI.Forms.Color = class extends Colibri.UI.Forms.Field {
  7 
  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-color-field');
 14 
 15         const contentContainer = this.contentContainer;
 16 
 17         this._flex = new Colibri.UI.FlexBox(this.name + '_flex', contentContainer);
 18         this._color = new Colibri.UI.Pane(this._name + '_color', this._flex);
 19         this._input = new Colibri.UI.Input(this.name + '_input', this._flex);
 20         this._button = new Colibri.UI.Button(this.name + '_button', this._flex);
 21         this._flex.shown = this._color.shown = this._input.shown = this._button.shown = true;
 22         this._input.loading = this._input.hasIcon = this._input.hasClearIcon = false;
 23         
 24         const icon = new Colibri.UI.Icon(this.name + '_buttonicon', this._button);
 25         icon.shown = true;
 26         icon.value = Colibri.UI.SelectArrowIcon;
 27 
 28         if(this._fieldData?.params?.readonly === undefined) {
 29             this.readonly = false;    
 30         }
 31         else {
 32             this.readonly = this._fieldData?.params?.readonly;
 33         }
 34         if(this._fieldData?.params?.enabled === undefined) {
 35             this.enabled = true;
 36         }
 37         else {
 38             this.enabled = this._fieldData.params.enabled;
 39         }
 40 
 41         this.placeholder = this._fieldData?.placeholder;
 42 
 43         this.value = this._fieldData?.default ?? '';
 44 
 45         this._button.AddHandler('Clicked', (event, args) => this.__buttonClicked(event, args));
 46 
 47         this._input.AddHandler('Changed', (event, args) => {
 48             this._color.styles = {backgroundColor: this._input.value};
 49             this.Dispatch('Changed', Object.assign(args, {component: this}));
 50         });
 51         this._input.AddHandler('KeyUp', (event, args) => this.Dispatch('KeyUp', args));
 52         this._input.AddHandler('KeyDown', (event, args) => this.Dispatch('KeyDown', args));
 53         this._input.AddHandler('Clicked', (event, args) => this.Dispatch('Clicked', args));
 54 
 55     }
 56 
 57     /** @private */
 58     _showPopup() {
 59         
 60         this._colorPopup = new Colibri.UI.Color(this.name + '_color', document.body);
 61         this._colorPopup.parent = this.contentContainer;
 62         this._colorPopup.hasShadow = true;
 63         this._colorPopup.shown = true;
 64         this._colorPopup.BringToFront();
 65         this._colorPopup.AddHandler('Changed', (event, args) => {
 66             this._input.value = this._colorPopup.value.hex;
 67             this._color.styles = {backgroundColor: this._input.value};
 68             this.Dispatch('Changed', args);
 69         });
 70         this._colorPopup.AddHandler('ShadowClicked', (event, args) => {
 71             console.log('disposing');
 72             this._colorPopup.Dispose();
 73         });
 74         this._colorPopup.value = this._input.value;
 75         this._colorPopup.top = this._input.top + this._input.height;
 76         this._colorPopup.left = this._input.left;
 77         this._colorPopup.width = Math.max(this._input.width, 379);
 78         
 79     }
 80 
 81     /**
 82      * @private
 83      * @param {Colibri.Events.Event} event event object
 84      * @param {*} args event arguments
 85      */ 
 86     __buttonClicked(event, args) {
 87         this._showPopup();
 88     }
 89 
 90     /** @protected */
 91     _registerEvents() {
 92         super._registerEvents();
 93         // если нужно добавить что то
 94     }
 95 
 96     /**
 97      * Focus on component
 98      */
 99     Focus() {
100         this._input.Focus();
101     }
102     
103     /**
104      * Readonly
105      * @type {boolean}
106      */
107     get readonly() {
108         return this._input.readonly;
109     }
110 
111     /**
112      * Readonly
113      * @type {boolean}
114      */
115     set readonly(value) {
116         value = this._convertProperty('Boolean', value);
117         this._input.readonly = value;
118         this._button.enabled = !value;
119     }
120 
121     /**
122      * Placeholder
123      * @type {string}
124      */
125     get placeholder() {
126         return this._input.placeholder;
127     }
128 
129     /**
130      * Placeholder
131      * @type {string}
132      */
133     set placeholder(value) {
134         value = this._convertProperty('String', value);
135         this._input.placeholder = value;
136     }
137 
138     /**
139      * Value
140      * @type {string}
141      */
142     get value() {
143         let value = this._input.value;
144         if(this._fieldData?.params?.emptyAsNull && !value) {
145             value = null;
146         }
147         if(this._fieldData?.params?.eval) {
148             value = eval(this._fieldData?.params?.eval);
149         }
150         return value;
151     }
152 
153     /**
154      * Value
155      * @type {string}
156      */
157     set value(value) {
158         this._input.value = value ?? '';
159         this._color.styles = {backgroundColor: this._input.value};
160     }
161 
162     /**
163      * Enable/Disable
164      * @type {boolean}
165      */
166     get enabled() {
167         return this._input.enabled;
168     }
169 
170     /**
171      * Enable/Disable
172      * @type {boolean}
173      */
174     set enabled(value) {
175         value = this._convertProperty('Boolean', value);
176         this._input.enabled = value;
177         this._button.enabled = value;
178     }
179 
180     /**
181      * Tab index
182      * @type {number}
183      */
184     get tabIndex() {
185         return this._input && this._input.tabIndex;
186     }
187     /**
188      * Tab index
189      * @type {number}
190      */
191     set tabIndex(value) {
192         this._input && (this._input.tabIndex = value === true ? Colibri.UI.tabIndex++ : value);
193     }
194 
195 }
