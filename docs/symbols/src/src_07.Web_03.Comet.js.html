1 /**
  2  * Handles the connection to the Comet server and message communication.
  3  * @class 
  4  * @extends Colibri.Events.Dispatcher
  5  * @memberof Colibri.Web
  6  */
  7 Colibri.Web.Comet = class extends Colibri.Events.Dispatcher {
  8 
  9     _url = '';
 10     _storage = window.localStorage;
 11     _store = null;
 12     _storeMessages = '';
 13     _storeUnread = '';
 14 
 15     _settings = null;
 16     _ws = null;
 17     _connected = false;
 18     _user = null;
 19     _clientId = null;
 20 
 21     /**
 22      * Creates a new Comet object.
 23      * @param {object} settings - Settings for the Comet connection.
 24      */
 25     constructor(settings) {
 26         super();
 27         
 28         this._clientId = this._generateDeviceId();
 29         this._settings = settings;
 30         this.RegisterEvent('MessageReceived', false, 'Когда получили новое сообщение');
 31         this.RegisterEvent('MessagesMarkedAsRead', false, 'Когда сообщения помечены как прочтенные');
 32         this.RegisterEvent('MessageRemoved', false, 'Когда сообщение удалено');
 33         this.RegisterEvent('EventReceived', false, 'Когда произошло событие');
 34         this.RegisterEvent('ConnectionError', false, 'Не смогли подключиться');
 35         this.RegisterEvent('Connected', false, 'Успешно подключились');
 36         
 37     }
 38 
 39     /**
 40      * Destructor to close WebSocket connection when the object is destroyed.
 41      */
 42     destructor() {
 43         super.destructor();
 44         if(this._ws) {
 45             this._ws.close();
 46         }
 47     }
 48 
 49     /**
 50      * Generates a unique device ID for the Comet connection.
 51      * @returns {string} - The generated device ID.
 52      */
 53     _generateDeviceId() {
 54         let deviceId = App.Browser.Get('device-id');
 55         if(!deviceId) {
 56             deviceId = String.MD5(Date.Mc() + '');
 57             App.Browser.Set('device-id', deviceId);
 58         }
 59         return deviceId;
 60     }
 61 
 62     /**
 63      * Initializes the WebSocket connection.
 64      */
 65     _initConnection() {
 66         this._ws && this._ws.close();
 67         this._ws = new WebSocket('wss://' + this._settings.host + ':' + this._settings.port + '/client/' + this._clientId);
 68         this._ws.onopen = () => this.__onCometOpened();
 69         this._ws.onmessage = (message) => this.__onCometMessage(message);
 70         this._ws.onerror = error => this.__onCometError(error);
 71     }
 72     
 73     /**
 74      * Initializes the Comet object with user data and storage settings.
 75      * @param {object} userData - Data of the user.
 76      * @param {Colibri.Storages.Store} store - Storage object.
 77      * @param {string} storeMessages - Key where messages will be stored.
 78      */
 79     Init(userData, store, storeMessages) {
 80 
 81         this._user = userData.guid;
 82         this._userName = userData.name;
 83         this._store = store;
 84         this._storeMessages = storeMessages;
 85         
 86         this._initConnection();
 87         this._saveToStore();
 88 
 89         Colibri.Common.StartTimer('comet-timer', 5000, () => {
 90             if(this._ws && this._ws.readyState !== 1) {
 91                 console.log('connection closed, may be server down');
 92                 this._initConnection();
 93             }
 94         });
 95 
 96     }
 97 
 98     /**
 99      * Disconnects from the Comet server.
100      */
101     Disconnect() {
102         if(this._ws) {
103             this._ws.close();
104             this._ws = null;
105         }
106     }
107 
108     /**
109      * Handles the WebSocket connection open event.
110      */
111     __onCometOpened() {
112         this._connected = true;
113         this.Command(this._user, 'register', {name: this._userName});
114     }
115 
116     /**
117      * Handles incoming messages from the Comet server.
118      * @param {object} message - The received message object.
119      */
120     __onCometMessage(message) {
121         message = JSON.parse(message.data);
122         if(message.action == 'connection-success') {
123             console.log('Connection to Comet Server ok');
124         }
125         else if(message.action == 'register-success') {
126             console.log('User registered successfuly');
127         }
128         else if(message.action == 'register-error') {
129             console.log('User registration error');
130         }
131         else if(message.action == 'message') {
132             message.id = message.id ?? Number.Rnd4();
133             message.date = new Date();
134             message.read = false;
135             var messages = this._getStoredMessages();
136             messages.push(message);
137             this._setStoredMessages(messages);
138             this._saveToStore();
139             this.Dispatch('MessageReceived', {message: message});
140         }
141         else {
142             this.Dispatch('EventReceived', {event: message});
143         }
144     }
145 
146     /**
147      * Handles WebSocket connection errors.
148      * @param {object} error - The error object.
149      */
150     __onCometError(error) {
151         
152         console.log('#{ui-comet-connection-error}');
153         // App.Notices.Add(new Colibri.UI.Notice('#{ui-comet-connection-error}'));
154         // Colibri.Common.StopTimer('comet-timer');
155 
156     } 
157 
158 
159     /**
160      * Retrieves stored messages from local storage.
161      * @returns {array} - Array of stored messages.
162      */
163     _getStoredMessages() {
164         let messages = this._storage.getItem('comet.messages');
165         if (!messages)
166             messages = [];
167         else
168             messages = JSON.parse(messages);
169         return messages;
170     }
171 
172     /**
173      * Saves messages to local storage.
174      * @param {array} messages - Array of messages to be stored.
175      */
176     _setStoredMessages(messages) {
177         this._storage.setItem('comet.messages', JSON.stringify(messages));
178     }
179 
180     /**
181      * Saves messages to the application store.
182      */
183     _saveToStore() {
184         let messages = this._getStoredMessages();
185         let unreadCount = 0;
186         messages.forEach(message => unreadCount += (message.read !== true ? 1 : 0));
187         this._store.Set(this._storeMessages, {messages: messages, unread: unreadCount});
188     }
189 
190     /**
191      * Clears stored messages.
192      */
193     ClearMessages() {
194         this._setStoredMessages([]);
195         this._saveToStore();
196     }
197 
198     /**
199      * Marks all messages as read.
200      */
201     MarkAsRead() {
202         let messages = this._getStoredMessages();
203         messages.forEach(message => message.read = true);
204         this._setStoredMessages(messages);
205         this._saveToStore();
206         this.Dispatch('MessagesMarkedAsRead', {});
207     }
208 
209     /**
210      * Removes a message from storage.
211      * @param {object} message - The message to be removed.
212      */
213     RemoveMessage(message) {
214         let messages = this._getStoredMessages();
215         messages = messages.filter(m => m.id != message.id);
216         this._setStoredMessages(messages);
217         this._saveToStore();
218         this.Dispatch('MessageRemoved', {});
219     }
220 
221     /**
222      * Sends a command to the Comet server.
223      * @param {string} userGuid - The GUID of the user.
224      * @param {string} action - The action to be performed.
225      * @param {object} message - The message data.
226      */
227     Command(userGuid, action, message = null) {
228         try {
229             if(this._ws.readyState === 1) {
230                 this._ws.send(JSON.stringify({action: action, user: userGuid, message: message, domain: document.domain}));
231             }
232             else {
233                 console.log('server goes away');
234             }
235         }
236         catch(e) {
237             console.log(e);
238         }
239     }
240 
241     /**
242      * Sends a message to a specific user.
243      * @param {string} userGuid - The GUID of the recipient user.
244      * @param {string} action - The action to be performed.
245      * @param {string} message - The message content.
246      * @returns {string|null} - The ID of the sent message.
247      */
248     SendTo(userGuid, action, message = null) {
249         try {
250             const id = Date.Mc();
251             if(this._ws.readyState === 1) {
252                 this._ws.send(JSON.stringify({action: action, recipient: userGuid, message: {text: message, id: id}, domain: document.domain}));
253                 return id;
254             }
255             else {
256                 console.log('server goes away');
257             }
258         }
259         catch(e) {
260             console.log(e);
261         }
262         return null;
263     }
264 
265     /**
266      * Gets the client ID for the Comet connection.
267      * @returns {string} - The client ID.
268      */
269     get clientId() {
270         return this._clientId;
271     }
