 28     /** 
 29      * Private max z-index calculator 
 30      * @static
 31      */
 32     static _getZIndex(elements = null) {
 33         return (elements ?? [...document.querySelectorAll('body *')]).reduce((accumulator, current_value) => {
 34             current_value = +getComputedStyle(current_value).zIndex;
 35             if (current_value === current_value) {
 36                 return Math.max(accumulator, current_value) 
 37             }
 38             return accumulator;
 39         }, 0);
 40     }
 41 
 42     /** 
 43      * Registers mutation observer for calculating current maximum of z-index
 44      * @static 
 45      */
 46     static registerMutationObserver() {
 47         // fixing start max z-index 
 48         Colibri.UI.maxZIndex = Colibri.UI._getZIndex();
 49         Colibri.Common.StartTimer('z-index-timer', 10000, () => {
 50             Colibri.UI.maxZIndex = Colibri.UI._getZIndex();
 51         });
 52         new MutationObserver((mutationList, observer) => {
 53             let elements = [];
 54             for(const mut of mutationList) {
 55                 elements.push(mut.target);
 56             }
 57             Colibri.UI.maxZIndex = Math.max(Colibri.UI.maxZIndex, Colibri.UI._getZIndex(elements));
 58         }).observe(document.body, {
 59             attributes: true, 
 60             attributeFilter: ['style', 'class'], 
 61             childList: true,
 62             subtree: true
 63         });
 64     }

 75     /**
 76      * Search for the component by path in document or given parent
 77      * @param {string} componentPath path for searching
 78      * @param {Colibri.UI.Component} parent search component within given parent component
 79      * @returns {Colibri.UI.Component}
 80      * @static
 81      */
 82     static Find(componentPath, parent = null) {
 83         const path = componentPath.split('/');
 84         const query = '[data-object-name="' + path.join('"] [data-object-name="') + '"]';
 85         const component = (parent ? parent._element : document).querySelector(query);
 86         if(!component) {
 87             return null;
 88         }
 89         return component.tag('component') || null;
 90     }
 91 
 92     /**
 93      * Search for the component by path in document or given parent
 94      * @param {string} componentPath path for searching
 95      * @param {Colibri.UI.Component} parent search component within given parent component
 96      * @returns {Colibri.UI.Component}
 97      * @static
 98      */
 99     static FindAll(componentPath, parent = null) {
100         const path = componentPath.split('/');
101         const query = '[data-object-name="' + path.join('"] [data-object-name="') + '"]';
102         const components = (parent ? parent._element : document).querySelectorAll(query);
103         const ret = [];
104         for(const component of components) {            
105             ret.push(component.tag('component'));
106         }
107         return ret;
108     }

111     /**
112      * Loads css and javascripts and fires a promise
113      * @param {Array<string>} css string array of css files
114      * @param {Array<string>} js string array of js files 
115      * @returns Promise
116      * @static
117      */
118     static Require(css, js) { 
119         return new Promise((resolve, reject) => {
120 
121             let loading = 0;
122 
123             css.forEach((c) => { 
124                 var res = hex_md5(c); 
125                 if(!document.querySelector('#res' + res)) { 
126                     loading++;
127                     const style = document.createElement('link');
128                     style.id = 'res' + res;
129                     style.type = 'text/css';
130                     style.href = c;
131                     style.rel = 'stylesheet';
132                     style.onload = () => {
133                         loading--;
134                     }
135                     document.querySelector('head').append(style);
136                 }; 
137             });
138 
139             js.forEach((j) => { 
140                 var res = hex_md5(j); 
141                 if(!document.querySelector('#res' + res)) { 
142                     loading++;
143                     const script = document.createElement('script');
144                     script.id = 'res' + res;
145                     script.src = j;
146                     script.async = true;
147                     script.onload = () => {
148                         loading--;
149                     }
150                     document.querySelector('head').append(script);
151                 }; 
152             }); 
153 
154             Colibri.Common.Wait(() => loading === 0).then(() => {
155                 resolve();
156             });
157         });
158 
159     }
