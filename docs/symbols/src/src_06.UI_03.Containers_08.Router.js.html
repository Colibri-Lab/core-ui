 24      * Route structure Object

 27     get structure() {
 28         return this._structure;

 30     /**
 31      * Route structure Object
 32      * @type {Object|Function}
 33      */
 34     set structure(value) {
 35         let struct = {};
 36         if(Object.isObject(value)) {
 37             struct = value;
 38         } else if(typeof value === 'string') {
 39             struct = eval(value);
 40         } else if(typeof value === 'function') {
 41             struct = value();
 42         }
 43 
 44         if(struct instanceof Promise) {
 45             struct.then(structure => {
 46                 this._structure = this.toPlain(structure, '');
 47                 this._initStructure();        
 48             });
 49         } else {
 50             this._structure = this.toPlain(struct, '/');
 51             this._initStructure();    
 52         }
 53 
 54     }
 55     _initStructure() {
 56         
 57         for(const pattern of Object.keys(this._structure)) {
 58             const route = this._structure[pattern];
 59             let component = null;
 60             if(route?.component ?? null) {
 61                 let componentObject = route.component;
 62                 if(typeof componentObject === 'string') {
 63                     componentObject = eval(componentObject);
 64                 }
 65                 if(componentObject) {
 66                     component = new componentObject(route?.name ?? ('component-' + Date.Mc()), this);
 67                 }
 68             } else {
 69                 component = new route('component-' + Date.Mc(), this);
 70             }
 71 
 72             if(component) {
 73                 component.routePattern = pattern.substring(this.basePattern.length); 
 74                 if(route?.attrs ?? null) {
 75                     Object.forEach(route?.attrs, (attrName, attrValue) => {
 76                         component[attrName] = attrValue;
 77                     });
 78                 }
 79     
 80                 component.Disconnect();
 81                 component.shown = true;    
 82             }
 83         }
 84 
 85 
 86     }

 88     toPlain(object, prefix = '') {
 89         let ret = {};
 90         object = Object.cloneRecursive(object, null, ['parent']);
 91         for(const name of Object.keys(object)) {
 92             const value = object[name];
 93             ret[prefix + name] = value;
 94             if(value?.childs ?? null) {
 95                 ret = Object.assign(ret, this.toPlain(value.childs, prefix + name + '/'));
 96                 delete value.childs;
 97             }
 98         }
 99         return ret;
100     }

117     __appRouteChanged(event, args) {
118         if(args.url.substring(0, this._current.length) === this._current) {
119             this.ForEach((name, component) => {
120                 let isPattern = component.routePattern === args.url;
121                 let match = args.url;
122                 if(component.routeIsRegExp) {
123                     const pattern = this._current + (component.routePattern ?? '').replace('#', '.+').replaceAll('?', '.*') + '$';
124                     const reg = new RegExp(pattern);
125                     isPattern = reg.test(args.url);
126                     match = reg.all(args.url);
127                 }
128                 if(isPattern) {
129                     if(!component.isConnected) {
130                         component.ConnectTo(this, null, true);
131                     }
132                     component.__processChangeOnRouteSwitch(match);
133                 } else {
134                     if(component.isConnected) {
135                         component.Disconnect();
136                     }
137                 }
138             });
139         }
