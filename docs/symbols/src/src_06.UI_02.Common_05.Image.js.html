1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Image = class extends Colibri.UI.Component {
  7     
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.create('span'));
 15         this.AddClass('app-component-image');
 16     }
 17 
 18     /**
 19      * Source string
 20      * @type {string}
 21      */
 22     get source() {
 23         return this._element.css('background-image');
 24     }
 25 
 26     /**
 27      * Source string
 28      * @type {string}
 29      */
 30     set source(value) {
 31         
 32         this._element.css('background-image', value);
 33         value = value.replaceAll(/\w+\(/, '').replaceAll(')', '');
 34         if(value.indexOf('data:') === -1) {
 35             const img = new Image();
 36             /**
 37              * @private
 38              * @param {Event} e 
 39              */
 40             img.onload = (e) => {
 41                 this.width = img.width;
 42                 this.height = img.height;
 43             }
 44             img.src = value;
 45         } 
 46 
 47     }
 48 
 49     /**
 50      * Repeat the background
 51      * @type {string}
 52      */
 53     get repeat() {
 54         return this._element.css('background-repeat');
 55     }
 56 
 57     /**
 58      * Repeat the background
 59      * @type {string}
 60      */
 61     set repeat(value) {
 62         this._element.css('background-repeat', value);
 63     }
 64 
 65     /**
 66      * Background size
 67      * @type {contain,cover,revert}
 68      */
 69     get size() {
 70         return this._element.css('background-size');
 71     }
 72     /**
 73      * Background size
 74      * @type {contain,cover,revert}
 75      */
 76     set size(value) {
 77         this._element.css('background-size', value);
 78     }
 79 
 80     /**
 81      * Background position
 82      * @type {String}
 83      */
 84     get position() {
 85         return this._element.css('background-position');
 86     }
 87     /**
 88      * Background position
 89      * @type {String}
 90      */
 91     set position(value) {
 92         this._element.css('background-position', value);
 93     }
 94     
 95     /**
 96      * Value of image
 97      * @type {string}
 98      */
 99     set image(value) {
100         let reader = new FileReader();
101         /**
102          * @private
103          * @param {Event} e 
104          */
105         reader.onload = (e) => {
106             this.source = 'url("' + reader.result + '")';
107         }
108         reader.readAsDataURL(value);
109         this._name = value.name;
110     }
111 
112     /**
113      * Value of image
114      * @type {string}
115      */
116     get image() {
117         const imageSrc = this.source.replaceAll('url("', '').replaceAll('")', '');
118         const data = imageSrc.split(';');
119         const mime = data[0].split(':')[1];
120         const base64 = data[1].split(',')[1];
121         return Base2File(base64, this._name, mime);
122     }
123 
124 }
