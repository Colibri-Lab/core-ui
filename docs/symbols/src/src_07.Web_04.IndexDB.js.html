1 /**
  2  * Manages IndexedDB operations.
  3  * @class 
  4  * @extends Colibri.Events.Dispatcher
  5  */
  6 Colibri.Web.IndexDB = class extends Colibri.Events.Dispatcher {
  7     
  8     _db = null;
  9     _lastTransaction = null;
 10     _name = null;
 11     _version = null;

 13     /**
 14      * Creates an IndexedDB manager.
 15      * @param {string} name - The name of the IndexedDB.
 16      * @param {number} version - The version of the IndexedDB.
 17      */
 18     constructor(name, version) {
 19         super();
 20 
 21         this._name = name;
 22         this._version = version;
 23 
 24         if (!('indexedDB' in window)) {
 25             throw new Error('IndexDB does not supported in your browser');
 26         }
 27 
 28         this.RegisterEvent('DatabaseDoesNotExists', false, 'When database is not exists');
 29         this.RegisterEvent('DatabaseVersionDoesNotExists', false, 'When the version is not found');
 30         this.RegisterEvent('DatabaseTransactionComplete', false, 'When transaction is successed');
 31         this.RegisterEvent('DatabaseTransactionError', false, 'When transaction is not success');
 32         this.RegisterEvent('DatabaseOpened', false, 'When database is opened');
 33         this.RegisterEvent('DatabaseOpenError', false, 'When database has open error');
 34 
 35     }
 36     
 37     /**
 38      * Opens the IndexedDB.
 39      */
 40     Open() {
 41         const request = window.indexedDB.open(this._name, this._version);
 42         request.onupgradeneeded = (event) => {
 43             console.log('version change');
 44             this._db = request.result;
 45             if(event.oldVersion) {
 46                 this.Dispatch('DatabaseVersionDoesNotExists', {version: event.oldVersion});
 47             }
 48             else {
 49                 // базы нет нужно создать
 50                 this.Dispatch('DatabaseDoesNotExists', {});
 51             }
 52         };
 53         
 54         request.onerror = () => {
 55             this.Dispatch('DatabaseOpenError', {error: request.error}); 
 56         };
 57         
 58         request.onsuccess = () => {
 59             this._db = request.result;
 60             this.Dispatch('DatabaseOpened', {});
 61         };
 62     }

 64     /**
 65      * Checks if a store exists in the database.
 66      * @param {string} name - The name of the store.
 67      * @returns {boolean} - True if the store exists, false otherwise.
 68      */
 69     StoreExists(name) {
 70         return this._db.objectStoreNames.contains(name);
 71     }

 73     /**
 74      * Creates a new store in the database.
 75      * @param {string} name - The name of the store.
 76      * @param {string} keyPath - The key path for the store.
 77      * @param {boolean} autoIncrement - Whether the store should auto increment keys.
 78      * @param {array} indices - Array of index objects to create for the store.
 79      * @returns {object} - The newly created store.
 80      */
 81     CreateStore(name, keyPath = 'id', autoIncrement = false, indices = []) {
 82         const store = this._db.createObjectStore(name, {keyPath: keyPath, autoIncrement: autoIncrement});
 83         for(const index of indices) {
 84             store.createIndex(index.name, index.keyPath, {unique: index.unique, multiEntry: index.multiEntry});
 85         }
 86         return store;
 87     }

 89     /**
 90      * Creates a new index for a store in the database.
 91      * @param {string} name - The name of the store.
 92      * @param {string} indexName - The name of the index.
 93      * @param {string} key - The key for the index.
 94      * @param {boolean} unique - Whether the index should be unique.
 95      * @param {boolean} multiEntry - Whether the index should allow multiple entries for a key.
 96      * @returns {object} - The newly created index.
 97      */
 98     CreateIndex(name, indexName, key, unique = false, multiEntry = false) {
 99         const store = this._db.objectStore(name);
100         const options = {unique: unique, multiEntry: multiEntry};
101         return store.createIndex(indexName, key, options);
102     }

112     /**
113      * Adds data to a specified store in the database.
114      * @param {string} storeName - The name of the store.
115      * @param {object} dataObject - The data object to add.
116      * @returns {Promise} - A promise that resolves with the result of the operation.
117      */
118     AddData(storeName, dataObject) {
119         return new Promise((resolve, reject) => {
120             let transaction = this._db.transaction(storeName, "readwrite"); 
121             let store = transaction.objectStore(storeName);
122             let request = store.add(dataObject); 
123             request.onsuccess = () => { 
124                 resolve(request.result);
125             };
126             request.onerror = () => {
127                 reject(request.error);
128             };
129             transaction.oncomplete = () => {
130                 this.Dispatch('DatabaseTransactionComplete', {transaction: transaction});
131             };
132             transaction.onerror = () => {
133                 this.Dispatch('DatabaseTransactionError', {transaction: transaction, error: transaction.error});
134             };
135         });
136     }

184     /**
185      * Retrieves data from a specified store within a specified range of keys.
186      * @param {string} storeName - The name of the store to retrieve data from.
187      * @param {*} [idFrom=null] - The lower bound of the key range. If null, starts from the first key.
188      * @param {*} [idTo=null] - The upper bound of the key range. If null, ends at the last key.
189      * @returns {Promise} A Promise that resolves with the retrieved data or rejects with an error.
190      */
191     GetDataByRange(storeName, idFrom = null, idTo = null) {
192         return new Promise((resolve, reject) => {
193             const transaction = this._db.transaction(storeName, "readwrite"); 
194             const store = transaction.objectStore(storeName);
195             let request = null;
196             if(idFrom === null) {
197                 request = store.getAll(IDBKeyRange.upperBound(idTo));
198             }
199             else if(idTo === null) {
200                 request = store.getAll(IDBKeyRange.lowerBound(idFrom));
201             }
202             else if(idFrom !== null && idTo !== null) {
203                 request = store.getAll(IDBKeyRange.bound(idFrom, idTo));
204             }
205             if(!request) {
206                 reject('Bad request');
207             }
208             
209             request.onsuccess = () => {
210                 resolve(request.result);
211             }
212             request.onerror = () => {
213                 reject(request.error);
214             }
215             transaction.oncomplete = () => {
216                 this.Dispatch('DatabaseTransactionComplete', {transaction: transaction});
217             };
218             transaction.onerror = () => {
219                 this.Dispatch('DatabaseTransactionError', {transaction: transaction, error: transaction.error});
220             };
221         });
222     }

388     /**
389      * Deletes data entries in a specified store using an index and a specified key.
390      * @param {string} storeName - The name of the store to delete data from.
391      * @param {string} indexName - The name of the index to use for deletion.
392      * @param {*} key - The key to use for data deletion.
393      * @returns {Promise} A Promise that resolves when the deletion is successful or rejects with an error.
394      */
395     DeleteByIndex(storeName, indexName, key) {
396         return new Promise((resolve, reject) => {
397             const transaction = this._db.transaction(storeName, "readwrite"); 
398             const store = transaction.objectStore(storeName);
399             const index = store.index(indexName);
400             const request = index.openCursor(key);
401             request.onsuccess = () => {
402                 let cursor = request.result;
403                 if(cursor) {
404                     cursor.delete();
405                     cursor.continue();
406                 }
407                 else {
408                     resolve();
409                 }
410             };
411         });
