1 Colibri.UI.LinkEditor = class extends Colibri.UI.Editor {
  2 
  3     constructor(name, container, element = null, root = null) {
  4         super(name, container, element || Element.create('span'), root);
  5         this.AddClass('app-link-editor-component');
  6 
  7         this._value = null;
  8         this._downloadlink = null;
  9 
 10         this._grid = this.parent.grid;
 11 
 12         this._icon = new Colibri.UI.Icon('icon', this);
 13         this._icon.shown = true;
 14         this._icon.value = Colibri.UI.FileLinkIcon;
 15 
 16         this._text = new Colibri.UI.TextSpan('span', this);
 17         this._text.shown = true;
 18 
 19         this.AddHandler('Clicked', (event, args) => {
 20 
 21             this._clickOnLink();
 22 
 23             args.domEvent.stopPropagation();
 24             args.domEvent.preventDefault();
 25             return false;
 26         });
 27 
 28     }
 29 
 30     _showValue() {
 31         this._text.value = this.value;
 32         if(this._grid) {
 33             const field = this._grid.header.columns.Children(this.parent.name.replaceAll(this.parent.parentRow.name + '-', '')).tag;
 34             if(field) {
 35                 this._text.value = field.params && field.params.view ? field.params.view : field.desc;
 36             }
 37         }
 38         else if(this.field) {
 39             this._text.value = this.field.desc + ' #{ui-editors-link-download}';
 40         }
 41         else {
 42             this._text.value = 'Скачать';
 43         }
 44 
 45     }
 46 
 47     _clickOnLink() {
 48         if(this.field && this._downloadlink) {
 49             window.open((window.rpchandler ? window.rpchandler : '') + this._downloadlink + '?name=' + this.field.name + '&data=' + this.field.params.data);
 50         }
 51     }
 52 
 53     
 54     get value() {
 55         return this._value;
 56     }
 57 
 58     set value(value) {
 59         this._value = value;
 60         this._showValue();
 61     }
 62 
 63     set downloadlink(value) {
 64         this._downloadlink = value;
 65     }
 66 
 67 
 68 }
