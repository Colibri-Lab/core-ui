1 /**
  2  * @classdesc Drag process worker
  3  * @class
  4  */
  5 Colibri.UI.Drag = class {
  6 
  7     /**
  8      * Element for dragging
  9      * @type {Element}
 10      */
 11     _element = null; 
 12 
 13     /**
 14      * Container in wich drag must be realized
 15      * @type {Element}
 16      */
 17     _container = null;
 18 
 19     /**
 20      * Element for handle drag start
 21      * @type {Element}
 22      */
 23     _moveHandler = null;
 24 
 25     /**
 26      * @constructor
 27      * @param {Element} element Element to drag
 28      * @param {Element} container Container for drag in
 29      * @param {Element} moveHandler Element for drag handle
 30      */
 31     constructor(element, container, moveHandler) {
 32 
 33         this._element = element;
 34         this._container = container;
 35         this._moveHandler = moveHandler;
 36 
 37         this.ЕndHandle = (e) => this.__end(e);
 38         this.MoveHandle = (e) => this.__move(e);
 39 
 40         this._element.addEventListener('mousedown', (e) => this.__start(e));
 41 
 42     }
 43 
 44     /**
 45      * @private
 46      */
 47     __start(e) {
 48         const bounds = this._container.bounds();
 49         this._element.tag = {state: true, point: [e.clientX - bounds.left, e.clientY - bounds.top]};
 50         document.addEventListener('mouseup', this.ЕndHandle, true);
 51         document.addEventListener('mousemove', this.MoveHandle, true);
 52     }
 53 
 54     /**
 55      * @private
 56      */
 57     __end(e) {
 58         this._element.tag = {state: false};
 59         document.removeEventListener('mouseup', this.EndHandle, true);
 60         document.removeEventListener('mousemove', this.MoveHandle, true);
 61     }
 62 
 63     /**
 64      * @private
 65      */
 66     __move(e) {
 67         if(this._element.tag.state) {
 68             // двигаем
 69             const containerBounds = this._container.bounds();
 70             const elementBounds = this._element.bounds();
 71             let newLeft = (e.clientX - containerBounds.left - elementBounds.outerWidth / 2);
 72             let newTop = (e.clientY - containerBounds.top - elementBounds.outerHeight / 2);
 73             
 74             if(newLeft < -1 * elementBounds.outerWidth / 2) { newLeft = -1 * elementBounds.outerWidth / 2; }
 75             if(newLeft > containerBounds.outerWidth - elementBounds.outerWidth / 2) { newLeft = containerBounds.outerWidth - elementBounds.outerWidth / 2; }
 76 
 77             if(newTop < -1 * elementBounds.outerHeight / 2) { newTop = -1 * elementBounds.outerHeight / 2; }
 78             if(newTop > containerBounds.outerHeight - elementBounds.outerHeight / 2) { newTop = containerBounds.outerHeight - elementBounds.outerHeight / 2;}
 79 
 80             this._moveHandler(newLeft, newTop);
 81        }
 82 
 83     }
 84 
 85 }
