497     /**
498      * @constructor
499      * @param {string} name name of component
500      * @param {Element|Colibri.UI.Component} container container of component
501      */    
502     constructor(name, container) {
503         super(name, container);

510         this._renderContent();
511         this._bind();

514     _renderContent() {
515 
516         this._element.html('');
517         let dateformat = App.DateFormat || 'ru-RU';
518         const formatter = new Intl.DateTimeFormat(dateformat, { day: '2-digit' });
519         const weekformatter = new Intl.DateTimeFormat(dateformat, { weekday: 'narrow' });
520 
521         const table = this._element.append(Element.fromHtml('<table cellspacing="2"><thead></thead><tbody></tbody><tfoot></tfoot></table>'));
522 
523         const thead = table.querySelector('thead');
524         const tbody = table.querySelector('tbody');
525         const tfoot = table.querySelector('tfoot');
526 
527         let tr = thead.append(Element.fromHtml('<tr></tr>'));
528 
529         let dt = this.parent.value.copy();
530         let checkedDate = dt.copy();
531 
532         let today = new Date();
533 
534         let weekday = dt.getDay();
535         if (weekday == 0) {
536             weekday = 7;
537         }
538 
539         dt.setTime(dt.getTime() - (weekday - 1) * 86400000);
540         for (let i = 0; i < 7; i++) {
541             tr.append(Element.fromHtml('<td>' + (weekformatter.format(dt)) + '</td>'));
542             dt.setTime(dt.getTime() + 86400000);
543         }
544 
545         dt = this.parent.value.copy();
546         let day = dt.getDate();
547 
548         dt.setTime(dt.getTime() - day * 86400000);
549         weekday = dt.getDay();
550 
551         // нашли начало календарика;
552         dt.setTime(dt.getTime() - (weekday - 1) * 86400000);
553 
554         for (let i = 0; i < 6; i++) {
555             // печатаем 6 строк
556             tr = tbody.append(Element.fromHtml('<tr></tr>'));
557             for (let i = 0; i < 7; i++) {
558                 // печатаем 7 дней
559                 let className = "";
560                 if (today.getDate() == dt.getDate() && today.getMonth() == dt.getMonth()) {
561                     className += ' today';
562                 }
563                 if (checkedDate.getDate() == dt.getDate() && checkedDate.getMonth() == dt.getMonth()) {
564                     className += ' current';
565                 }
566                 if (checkedDate.getMonth() != dt.getMonth()) {
567                     className += ' ntm';
568                 }
569                 tr.append(Element.fromHtml('<td class="' + className + '" data-value="' + (dt.getTime()) + '">' + formatter.format(dt) + '</td>'));
570                 dt.setTime(dt.getTime() + 86400000);
571             }
572         }
573 
574         tfoot.append(Element.fromHtml('<td colspan="7" data-today="today" data-value="' + (today.getTime()) + '">#{ui-dateselector-today}</td>'));
575     }
576 
577     _bind() {
578         this._element.querySelectorAll('td').forEach((td) => {
579             td.addEventListener('mousedown', (e) => {
580                 if (e.target.classList.contains('ntm')) {
581                     return false;
582                 }
583                 this.parent.value = Date.from(e.target.dataset.value);
584                 this.parent.parent.value = Date.from(e.target.dataset.value);
585                 this.parent.parent.Close();
586                 e.stopPropagation();
587                 e.preventDefault();
588                 return false;
589             });
590         })
591     }

650                 if (month == checkedMonth) {
651                     className = ' current';
652                 }
653 
654                 tr.append(Element.fromHtml('<td class="' + className + '" data-value="' + (dt.getTime()) + '">' + formatter.format(dt).substring(0, 3) + '</td>'));
