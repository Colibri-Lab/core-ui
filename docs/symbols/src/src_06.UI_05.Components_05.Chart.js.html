1 /**
  2  * @class
  3  * @namespace
  4  * @extends Colibri.UI.Component
  5  * @memberof Colibri.UI
  6  */
  7 Colibri.UI.Chart = class extends Colibri.UI.Component {
  8 
  9     /** Horizontal orientation */
 10     static OrientationHorizontal = 'horizontal';
 11     /** Vertical orientation */
 12     static OrientationVertical = 'vertical';
 13 
 14     /** @type {string} */
 15     _orientation = Colibri.UI.Split.OrientationHorizontal;
 16 
 17     /**
 18      * @constructor
 19      * @param {string} name name of component
 20      * @param {Element|Colibri.UI.Component} container container of component
 21      */
 22     constructor(name, container) {
 23         super(name, container, Element.create('div'));
 24         this.AddClass('app-component-chart');
 25     }
 26 
 27     /**
 28      * Orientation of chart
 29      * @type {string}
 30      */
 31     get orientation() {
 32         return this._orientation;
 33     }
 34     /**
 35      * Orientation of chart
 36      * @type {string}
 37      */
 38     set orientation(value) {
 39         this._orientation = value;
 40         this.AddClass('app-component-chart-' + this._orientation);
 41     }
 42 
 43     /**
 44      * Adds a bar chart
 45      * @param {string} name name of chart
 46      * @returns {Colibri.UI.Chart.Barchart}
 47      */
 48     AddBarchart(name) {
 49         let barchart = new Colibri.UI.Chart.Barchart(name, this);
 50         barchart._orientation = this._orientation;
 51         return barchart
 52     }
 53 }

 55 /**
 56  * @class
 57  * @extends Colibri.UI.Component
 58  * @memberof Colibri.UI.Chart
 59  */
 60 Colibri.UI.Chart.Barchart = class extends Colibri.UI.Component {
 61 
 62     /**
 63      * @constructor
 64      * @param {string} name name of component
 65      * @param {Element|Colibri.UI.Component} container container of component
 66      */
 67     constructor(name, container) {
 68         super(name, container, Element.create('div'));
 69 
 70         this.AddClass('app-component-barchart');
 71 
 72         this.shown = true;
 73 
 74         this._title = new Colibri.UI.Component('barchart-title', this);
 75         this._title.AddClass('barchart-title');
 76         this._title.shown = true;
 77 
 78         this._barchart = new Colibri.UI.Component('barchart', this);
 79         this._barchart.AddClass('barchart-body');
 80         this._barchart._element.css('width', '50%');
 81         this._barchart.shown = true;
 82 
 83         this._textValue = new Colibri.UI.Component('barchart-text-value', this);
 84         this._textValue.AddClass('barchart-text-value');
 85     }
 86 
 87     /**
 88      * Title of chart
 89      * @type {string}
 90      */
 91     get title() {
 92         return this._title;
 93     }
 94 
 95     /**
 96      * Title of chart
 97      * @type {string}
 98      */
 99     set title(value) {
100         this._title.value = value;
101     }
102 
103     /**
104      * Chart value
105      * @type {number}
106      */
107     set value(value) {
108         return this._barchart._element.css('width', value + '%');
109     }
110 
111     /**
112      * Chart value
113      * @type {number}
114      */
115     get value() {
116         return this._barchart._element.css('width');
117     }
118 
119     /**
120      * Text value
121      * @type {string}
122      */
123     get textValue() {
124         return this._textValue;
125     }
126 
127     /**
128      * Text value
129      * @type {string}
130      */
131     set textValue(value) {
132         this._textValue.value = value;
133     }

135 }
