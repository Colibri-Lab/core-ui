 33 Colibri.UI.Chart.Barchart = class extends Colibri.UI.Component {

 39         this.shown = true;
 40

 44 
 45         this._barchart = new Colibri.UI.Component('barchart', this);
 46         this._barchart.AddClass('barchart-body');

 49

 53

 56     }
 57

 64     }

 68     }
 69
