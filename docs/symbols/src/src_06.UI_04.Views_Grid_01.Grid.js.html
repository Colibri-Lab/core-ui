 97     /**
 98      * Возвращает сыделенные строки/ячейки
 99      * @type {Colibri.UI.Component[]}
100      */
101     get selected() {
102         let selected = [];
103         if (this.selectionMode === Colibri.UI.Grid.FullRow) {
104             Object.forEach(this.groups, (name, group) => {
105                 if(group.selected) {
106                     selected = selected.concat(group.selected);
107                 }
108             });
109         } else if (this.selectionMode === Colibri.UI.Grid.EveryCell) {
110             let selectedCell = [];
111             this.ForEveryRow((nameRow, row) => {
112                 row.ForEach((nameCell, cell) => {
113                     cell.selected ? selectedCell.push(cell) : '';
114                 });
115             });
116             selected = selectedCell;
117         }
118         if (this.multiple) {
119             return selected;
120         } else {
121             return selected[0];
122         }
123     }
124     /**
125      * Возвращает сыделенные строки/ячейки
126      * @type {Colibri.UI.Component[]}
127      */
128     set selected(value) {
129 
130         if(value === null) {
131             this.UnselectAllRows();
132             return;
133         }
134 
135         let cell = value instanceof Colibri.UI.Grid.Cell ? value : null;
136         let row = value instanceof Colibri.UI.Grid.Row ? value : null;
137 
138         if (!cell && !row) {
139             return;
140         }
141         
142         let args = {};
143         args.cell = cell;
144         args.row = row;
145 
146         switch(this.selectionMode) {
147             case Colibri.UI.Grid.EveryCell:
148                 if (cell) {
149                     this.DeactivateAllCells();
150                     if (!this.multiple) {
151                         this.DeselectAllCells();
152                     }
153                     cell.activated = !cell.activated;
154                     cell.selected = !cell.selected;
155                 }
156                 break;
157             case Colibri.UI.Grid.FullRow:
158                 if (row) {
159                     this.DeactivateAllRows();
160 
161                     if (!this.multiple) {
162                         this.UnselectAllRows();
163                     }
164                     row.activated = !row.activated;
165                     row.selected = !row.selected;
166                 }
167                 break;
168         }
169 
170         args.item = this.selected;
171         this.Dispatch('SelectionChanged', args);
172     }

471                 }
472                 args.cell.activated = !args.cell.activated;
473                 args.cell.selected = !args.cell.selected;
474             }
475             this.Dispatch('CellClicked', args);
476         });
477 
478         rows.AddHandler('CellDoubleClicked', (event, args) => {
479             this.Dispatch('CellDoubleClicked', args);
480         });
481

483             this.Dispatch('RowClicked', args);
484             this.Dispatch('SelectionChanged', Object.assign(args, {item: this.selected}));
485         });
486 
487         rows.AddHandler('RowSelected', (event, args) => {
488             if(rows.checked.length == 0) {
489                 rows.checkbox.checked = false;
490             }
491             this.Dispatch('RowSelected', args);
492         });
493 
494         rows.AddHandler('RowDisposed', (event, args) => {
495             this.RecalculateCellPositions();
496             this.Dispatch('RowDisposed', args);
497         });
498 
499         rows.AddHandler('StickyChanged', (event, args) => {
500             this.RecalculateCellPositions();
501         });
502 
503         rows.AddHandler('GridCellsChanged', (event, args) => {
504             this.RecalculateCellPositions();
505         });
506 
507         rows.AddHandler('RowAdded', (event, args) => {
508             this._norows.shown = false;
509             this._gridContent.shown = true;
510             Object.forEach(this.groups, (name, rows) => {
511                 rows.columns = this.header.columnsCount;
512             });
513             this.RecalculateCellPositions();
514             this.Dispatch('RowAdded', {row: args.row});
515         });
516 
517         rows.AddHandler('RowUpdated', (event, args) => {
518             this.RecalculateCellPositions();
519             this.Dispatch('RowUpdated', {row: args.row});
520         });
521 
522 
523         return rows;
524     }

538     /**
539      * Обработчик события нажатия кнопки на гриде
540      * @param {Object} sender 
541      * @param {Object} args 
542      * @returns 
543      */
544     /**
545      * @private
546      * @param {Colibri.Events.Event} event event object
547      * @param {*} args event arguments
548      */ 
549     __processKeydown(event, args) {
550 
551         const e = args.domEvent;
552 
553         if (['ArrowDown', 'ArrowUp', 'ArrowLeft', 'ArrowRight', 'Enter', 'NumpadEnter'].indexOf(e.code) !== -1) {
554 
555             let activeItem;
556             let newActiveItem;
557             let correctionCoefficient = 0;
558             let direction = 'all';
559 
560             if (this.selectionMode === Colibri.UI.Grid.FullRow) {
561                 direction = 'Y';
562                 let activeRow = this.activeRow;
563                 let activeGroup = this.activeGroup;
564                 if(!activeGroup) {
565                     activeGroup = this.firstGroup;
566                 }
567                 let lastGroup = this.lastGroup;
568 
569                 let newActiveRow = null;
570 
571                 if (activeRow === null) {
572                     if (['ArrowDown', 'ArrowRight'].indexOf(e.code) !==  -1) {
573                         newActiveRow = activeGroup.firstRow;
574                     } 
575                     else if (['ArrowUp', 'ArrowLeft'].indexOf(e.code) !== -1) {
576                         newActiveRow = lastGroup.lastRow;
577                     }
578                 } else {
579                     if (e.code !== 'Enter') {
580                         if (['ArrowDown', 'ArrowRight'].indexOf(e.code) !== -1) {
581                             if (activeRow.nextRow !== null) {
582                                 newActiveRow = activeRow.nextRow;
583                             }
584                             else {
585                                 newActiveRow = activeGroup.nextGroup ? activeGroup.nextGroup.firstRow : this.firstGroup.firstRow;
586                             }
587                         } else {
588                             if (activeRow.prevRow !== null) {
589                                 newActiveRow = activeRow.prevRow;
590                             }
591                             else {
592                                 newActiveRow = activeGroup.prevGroup ? activeGroup.prevGroup.lastRow : this.lastGroup.lastRow;
593                             }
594                         }
595                         if(newActiveRow) {
596                             correctionCoefficient = newActiveRow._heightPrevStickyRow;
597                         }
598                     } else if (e.code === 'Enter') {
599                         if (!this.multiple) {
600                             this.UnselectAllRows();
601                             activeRow.selected = true;
602                         } else {
603                             activeRow.selected = !activeRow.selected;
604                         }
605                         this.Dispatch('SelectionChanged', {item: activeRow});
606                     }
607                 }
608 
609                 activeItem = activeRow !== null ? activeRow : null;
610                 newActiveItem = newActiveRow;
611 
612             } else {
613 
614                 let activeCell = this.activeCell;
615                 let newActiveCell = null;
616 
617                 if (activeCell === null) {
618                     newActiveCell = this.firstGroup.firstRow.firstCell;
619 
620                 } else {
621 
622                     if (e.code === 'ArrowRight') {
623                         if (activeCell.nextCell !== null) {
624                             newActiveCell = activeCell.nextCell;
625                             correctionCoefficient = newActiveCell.parentColumn._widthPrevStickyCell;
626                         } else if (activeCell.parent.nextRow !== null) {
627                             if (this.cycleSelection) {
628                                 newActiveCell = activeCell.parent.nextRow.firstCell;
629                                 correctionCoefficient = newActiveCell.parentColumn._widthPrevStickyCell;
630                             }
631                         } else {
632                             if (this.cycleSelection) {
633                                 newActiveCell = activeCell.parent.parent.firstRow.firstCell;
634                                 correctionCoefficient = newActiveCell.parent._heightPrevStickyRow;
635                             }
636                         }
637                     } else if (e.code === 'ArrowLeft') {
638                         if (activeCell.prevCell !== null) {
639                             newActiveCell = activeCell.prevCell;
640                             correctionCoefficient = newActiveCell.parentColumn._widthPrevStickyCell;
641                         } else if (activeCell.parent.prevRow !== null) {
642                             if (this.cycleSelection) {
643                                 newActiveCell = activeCell.parent.prevRow.lastCell;
644                                 correctionCoefficient = newActiveCell.parentColumn._widthPrevStickyCell;
645                             }
646                         } else {
647                             if (this.cycleSelection) {
648                                 newActiveCell = activeCell.parent.parent.lastRow.lastCell;
649                                 correctionCoefficient = newActiveCell.parentColumn._widthPrevStickyCell;
650                             }
651                         }
652                     } else if (e.code === 'ArrowDown') {
653                         if (activeCell.parent.nextRow !== null) {
654                             let nameNewCell = 'app-ui-cell-' + activeCell.parent.nextRow.name + '-' + activeCell.parentColumn.name;
655                             newActiveCell = activeCell.parent.nextRow.Children(nameNewCell);
656                             correctionCoefficient = newActiveCell.parentRow._heightPrevStickyRow;
657                         } else {
658                             if (this.cycleSelection) {
659                                 let nameNewCell = 'app-ui-cell-' + activeCell.parent.parent.firstRow.name + '-' + activeCell.parentColumn.name;
660                                 newActiveCell = activeCell.parent.parent.firstRow.Children(nameNewCell);
661                                 correctionCoefficient = newActiveCell.parentRow._heightPrevStickyRow;
662                             }
663                         }
664                     } else if (e.code === 'ArrowUp') {
665                         if (activeCell.parent.prevRow !== null) {
666                             let nameNewCell = 'app-ui-cell-' + activeCell.parent.prevRow.name + '-' + activeCell.parentColumn.name;
667                             newActiveCell = activeCell.parent.prevRow.Children(nameNewCell);
668                             correctionCoefficient = newActiveCell.parentRow._heightPrevStickyRow;
669                         } else {
670                             if (this.cycleSelection) {
671                                 let nameNewCell = 'app-ui-cell-' + activeCell.parent.parent.lastRow.name + '-' + activeCell.parentColumn.name;
672                                 newActiveCell = activeCell.parent.parent.lastRow.Children(nameNewCell);
673                                 correctionCoefficient = 0;
674                             }
675                         }
676                     } else if (e.code === 'Enter') {
677                         if (!this.multiple) {
678                             this.DeselectAllCells();
679                         }
680                         activeCell.selected = !activeCell.selected;
681                         newActiveCell = activeCell;
682                         this.Dispatch('SelectionChanged', {item: newActiveCell});
683                     }
684 
685                 }
686 
687                 if (newActiveCell !== null) {
688                     if (activeCell !== null) {
689                         activeItem = activeCell;
690                     }
691                     newActiveItem = newActiveCell;
692                 }
693             }
694 
695             if (newActiveItem !== null && newActiveItem !== undefined) {
696                 if (activeItem !== null && activeItem !== undefined) {
697                     activeItem.activated = false;
698                 }
699                 newActiveItem.activated = true;
700                 this.__customScroll(newActiveItem._element, this.container, correctionCoefficient, direction);
701                 this.Dispatch('HighlightedItemChanged', {item: newActiveItem});
702             }
703             e.stopPropagation();
704             e.preventDefault();
705             return false;
706         }
707         else if(e.code === 'Space') {
708             this.selected = null;
709             this.activeRow.checked = !this.activeRow.checked;
710             this.Dispatch('CheckChanged');
711             e.stopPropagation();
712             e.preventDefault();
713             return false;
714         }
715         
716     }

718     /**
719      * @private
720      * @param {Colibri.Events.Event} event event object
721      * @param {*} args event arguments
722      */ 
723     __clickedProcessing(event, args) {
724         const   target = args.domEvent.target,
725                 cell = target.closest('.app-ui-row-cell')?.tag('component');
726 
727         if(!cell) {
728             return false;
729         }
730 
731         const row = target.closest('.app-ui-row').tag('component');
732 
733         args.cell = cell;
734         args.row = row;
735 
736         this._element.focus();
737 
738         switch(this.selectionMode) {
739             case Colibri.UI.Grid.EveryCell:
740                 this.DeactivateAllCells();
741                 if (!this.multiple) {
742                     this.DeselectAllCells();
743                 }
744                 cell.activated = !cell.activated;
745                 cell.selected = !cell.selected;
746                 break;
747             case Colibri.UI.Grid.FullRow:
748                 this.DeactivateAllRows();
749 
750                 if (!this.multiple) {
751                     this.UnselectAllRows();
752                 }
753                 row.activated = !row.activated;
754                 row.selected = !row.selected;
755                 break;
756         }
757 
758         this.Dispatch('CellClicked', args);
759         cell.EditValue && cell.EditValue();
760 
761         this.Dispatch('RowClicked', args);
762 
763         args.item = this.selected;
764         this.Dispatch('SelectionChanged', args);
765     }

767     /**
768      * Пересчитывает высоту строк
769      */
770     RecalculateCellPositions() {
771 
772         if (this._recalculateTimerCellPositions !== null) {

812     /**
813      * Скроллирует список
814      * @param {Element} element элемент
815      * @param {Element} container котейнер
816      * @param {number} correctionCoefficient коэфициент корректировки
817      * @param {bool} direction направление
818      */
819     __customScroll(element, container, correctionCoefficient, direction){
820 
821         if (!!!correctionCoefficient) {
822             correctionCoefficient = 0;
823         }
824 
825         let containerBoundingClientRect = container.getBoundingClientRect();
826         let cTop = containerBoundingClientRect.top;
827         let cBottom = containerBoundingClientRect.bottom;
828         let cLeft = containerBoundingClientRect.left;
829         let cRight = containerBoundingClientRect.right;
830 
831         let elementBoundingClientRect = element.getBoundingClientRect();
832         let eTop = elementBoundingClientRect.top;
833         let eBottom = elementBoundingClientRect.bottom;
834         let eLeft = elementBoundingClientRect.left;
835         let eRight = elementBoundingClientRect.right;
836 
837         let bottomCorrection = 0;
838         if (this.footer.shown) {
839             bottomCorrection = this.footer._element.clientHeight;
840         }
841 
842         if (direction === 'all' || direction === 'Y') {
843             if (eBottom > cBottom - bottomCorrection) {
844                 container.scrollTop += (eBottom - container.clientHeight + bottomCorrection);
845             } else if (eTop < (cTop + correctionCoefficient)) {
846                 container.scrollTop -= (cTop - eTop + correctionCoefficient);
847             }
848         }
849 
850         if (direction === 'all' || direction === 'X') {
851             if (eLeft < (cLeft + correctionCoefficient)) {
852                 container.scrollLeft -= (cLeft - eLeft + correctionCoefficient);
853             } else if (eRight > cRight) {
854                 container.scrollLeft += (eRight - cRight);
855             }
856         }
857     }
