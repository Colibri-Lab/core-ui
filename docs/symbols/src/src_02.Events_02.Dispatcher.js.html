1 /**
  2  * Represents a dispatcher for events.
  3  * @class
  4  * @extends Destructable
  5  */
  6 Colibri.Events.Dispatcher = class extends Destructable {
  7 
  8     /**
  9      * Creates a new instance of Colibri.Events.Dispatcher.
 10      */
 11     constructor() {
 12         super();
 13         /** @type {Object.<string, Array<Object.<handler: Function, respondent: Object>>>} - список обработчиков */
 14         this.__handlers = {};
 15         /** @type {Object.<string, Colibri.Events.Event>} - список зарегистрированых событий */
 16         this.__events = {};
 17     }
 18 
 19     destructor() {
 20         this.Dispose();
 21     }
 22 
 23     /**
 24      * Registers an event.
 25      * @param {string} eventName - The name of the event.
 26      * @param {boolean} bubbles - Indicates whether the event bubbles up the component tree.
 27      * @param {string} description - Description of the event.
 28      */
 29     RegisterEvent(eventName, bubbles, description) {
 30         this.__events[eventName] = { name: eventName, bubbles: bubbles, description: description };
 31     }
 32 
 33     /**
 34      * Adds an event handler.
 35      * @param {string|array} eventName - The event or array of events to add the handler to.
 36      * @param {Function} handler - The handler method.
 37      * @param {boolean} [prepend=false] - Whether to insert the handler at the beginning.
 38      * @param {Object} [respondent=this] - The object that responds to the event.
 39      * @returns {Colibri.Events.Dispatcher}
 40      */
 41     AddHandler(eventName, handler, prepend = false, respondent = this) {
 42 
 43         if (eventName instanceof Array) {
 44             eventName.forEach((en) => {
 45                 this.AddHandler(en, handler, prepend, respondent);
 46             });
 47         } else {
 48             
 49             this.RemoveHandler(eventName, handler);
 50 
 51             if (!this.__handlers[eventName]) {
 52                 this.__handlers[eventName] = [];
 53             }
 54 
 55             const handlerObject = { handler: handler, respondent: respondent };
 56             if (prepend) {
 57                 this.__handlers[eventName].splice(0, 0, handlerObject);
 58             } else {
 59                 this.__handlers[eventName].push(handlerObject);
 60             }
 61         }
 62 
 63         return this;
 64 
 65     }
 66 
 67     /**
 68      * Removes an event handler.
 69      * @param {string} eventName - The event to remove the handler from.
 70      * @param {Function} handler - The handler to remove.
 71      * @returns {Colibri.Events.Dispatcher}
 72      */
 73     RemoveHandler(eventName, handler) {
 74         
 75         if (!this.__handlers[eventName]) {
 76             this.__handlers[eventName] = [];
 77         }
 78 
 79         const handlerObject = { handler: handler, respondent: this };
 80         for (let i = 0; i < this.__handlers[eventName].length; i++) {
 81             const h = this.__handlers[eventName][i];
 82             if (h.handler == handlerObject.handler && h.respondent == handlerObject.respondent) {
 83                 this.__handlers[eventName].splice(i, 1);
 84                 break;
 85             }
 86         }
 87         return this;
 88     }
 89 
 90     /**
 91      * Checks if an event handler exists.
 92      * @param {string} eventName - The event to check for the handler.
 93      * @param {Function} handler - The handler to check for.
 94      * @returns {boolean} - True if the handler exists, false otherwise.
 95      */
 96     HandlerExists(eventName, handler) {
 97         const handlerObject = { handler: handler, respondent: this },
 98               handlerStr = handlerObject.handler.toString();
 99         for (let i = 0; i < this.__handlers[eventName].length; i++) {
100             const h = this.__handlers[eventName][i];
101             if (h.handler.toString() === handlerStr && h.respondent === handlerObject.respondent) {
102                 return true;
103             }
104         }
105         return false;
106     }
107 
108     /**
109      * Dispatches an event.
110      * @param {(Colibri.Events.Event|string)} event - The event object or its name.
111      * @param {object} args - Arguments for the event.
112      * @returns {Colibri.Events.Dispatcher}
113      */
114     Dispatch(event, args = null) {
115 
116         if (!(event instanceof Colibri.Events.Event)) {
117             // ищем сoбытие в стандартных
118             if(this.__events[event]) {
119                 event = this.__events[event];
120             } else if(!event.bubbles) {
121                 return true;
122             }
123         }
124 
125         if(!event.sender) {
126             event.sender = this;
127         }
128 
129         if(App && event.name !== 'Event') {
130             App.Dispatch('Event', {event: event, args: args});
131         }
132 
133         const eventHandlers = this.__handlers[event.name];
134         if (eventHandlers) {
135 
136             for (var j = 0; j < eventHandlers.length; j++) {
137                 const handlerObject = eventHandlers[j];
138                 if (handlerObject && handlerObject.handler.apply(handlerObject.respondent, [event, args]) === false) {
139                     return false;
140                 }
141             }
142         }
143 
144         if(event.bubbles && this.parent && this.parent.Dispatch(event, args) === false) {
145             return false;
146         }
147 
148         return true;
149     }
150 
151     /**
152      * Clears all event handlers.
153      */
154     ClearHandlers() {
155         this.__handlers = {};
156     }
157 
158     /**
159      * Disposes of the dispatcher.
160      */
161     Dispose() {
162         this.__events = {};
163         this.__handlers = {};
164     }
165

167 }
