1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.JsonViewer = class extends Colibri.UI.Pane {
  7     
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      */ 
 13     constructor(name, container) {
 14         /* создаем компонент и передаем шаблон */
 15         super(name, container);
 16         this.AddClass('colibri-ui-jsonviewer');
 17 
 18 
 19 
 20     }
 21 
 22     /**
 23      * Value as Object or array
 24      * @type {Object|Array}
 25      */
 26     get value() {
 27         return this._value;
 28     }
 29     /**
 30      * Value as Object or array
 31      * @type {Object|Array}
 32      */
 33     set value(value) {
 34         this._value = value;
 35         this._showValue();
 36     }
 37     _showValue() {
 38         const value = Object.cloneRecursive(this._value);
 39 
 40         if(Array.isArray(value)) {
 41             if(value.length > 0) {
 42                 for(let index=0; index<value.length;index++) {
 43                     const v = new Colibri.UI.JsonViewer(this.name + index, this);
 44                     v.shown = true;
 45                     v.value = value[index];
 46                 }
 47             }
 48         } else if(Object.isObject(value)) {
 49             const keys = Object.keys(value);
 50             if(keys.length > 0) {
 51                 const fieldsViewer = new Colibri.UI.FieldsViewer(this.name + 'viewer', this);
 52                 const fields = {};
 53                 for(const k of keys) {
 54                     fields[k] = {
 55                         desc: k,
 56                         component: Object.isObject(value[k]) ? 'Json' : 'Text'
 57                     };
 58                 }
 59                 fieldsViewer.shown = true;
 60                 fieldsViewer.fields = fields;
 61                 fieldsViewer.value = value;
 62             }
 63 
 64         }
 65         
 66 
 67 
 68     }
 69 
 70 
 71 }
 72 
 73 Colibri.UI.Viewer.Register('Colibri.UI.JsonViewer', '#{ui-viewers-json}');
