1 /**
  2  * Class representing Base64 encoding and decoding utility functions.
  3  * @class
  4  */
  5 Colibri.Common.Base64 = class {
  6 
  7     /**
  8      * Private property containing the Base64 key string.
  9      * @type {string}
 10      * @private
 11      */
 12     static _keyStr = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
 13 
 14     /**
 15      * Encode a string using Base64 encoding.
 16      * @param {string} _input - The input string to encode.
 17      * @returns {string} The Base64 encoded string.
 18      */
 19     static encode(_input) {
 20         var output = "";
 21         var chr1, chr2, chr3, enc1, enc2, enc3, enc4;
 22         var i = 0;
 23 
 24         _input = Colibri.Common.Base64._utf8_encode(_input);
 25 
 26         while (i < _input.length) {
 27 
 28             chr1 = _input.charCodeAt(i++);
 29             chr2 = _input.charCodeAt(i++);
 30             chr3 = _input.charCodeAt(i++);
 31 
 32             enc1 = chr1 >> 2;
 33             enc2 = ((chr1 & 3) << 4) | (chr2 >> 4);
 34             enc3 = ((chr2 & 15) << 2) | (chr3 >> 6);
 35             enc4 = chr3 & 63;
 36 
 37             if (isNaN(chr2)) {
 38                 enc3 = enc4 = 64;
 39             } else if (isNaN(chr3)) {
 40                 enc4 = 64;
 41             }
 42 
 43             output = output +
 44             Colibri.Common.Base64._keyStr.charAt(enc1) + Colibri.Common.Base64._keyStr.charAt(enc2) +
 45             Colibri.Common.Base64._keyStr.charAt(enc3) + Colibri.Common.Base64._keyStr.charAt(enc4);
 46         }
 47         return output;
 48     }
 49 
 50     /**
 51      * Decode a Base64 encoded string.
 52      * @param {string} _input - The Base64 encoded string to decode.
 53      * @returns {string} The decoded string.
 54      */
 55     static decode(_input) {
 56         var output = "";
 57         var chr1, chr2, chr3;
 58         var enc1, enc2, enc3, enc4;
 59         var i = 0;
 60 
 61         _input = _input.replace(/[^A-Za-z0-9\+\/\=]/g, "");
 62 
 63         while (i < _input.length) {
 64 
 65             enc1 = Colibri.Common.Base64._keyStr.indexOf(_input.charAt(i++));
 66             enc2 = Colibri.Common.Base64._keyStr.indexOf(_input.charAt(i++));
 67             enc3 = Colibri.Common.Base64._keyStr.indexOf(_input.charAt(i++));
 68             enc4 = Colibri.Common.Base64._keyStr.indexOf(_input.charAt(i++));
 69 
 70             chr1 = (enc1 << 2) | (enc2 >> 4);
 71             chr2 = ((enc2 & 15) << 4) | (enc3 >> 2);
 72             chr3 = ((enc3 & 3) << 6) | enc4;
 73 
 74             output = output + String.fromCharCode(chr1);
 75 
 76             if (enc3 != 64) {
 77                 output = output + String.fromCharCode(chr2);
 78             }
 79             if (enc4 != 64) {
 80                 output = output + String.fromCharCode(chr3);
 81             }
 82         }
 83 
 84         output = Colibri.Common.Base64._utf8_decode(output);
 85 
 86         return output;
 87     }
 88 
 89     /**
 90      * Private method for UTF-8 encoding.
 91      * @param {string} string - The input string to encode.
 92      * @returns {string} The UTF-8 encoded string.
 93      * @private
 94      */
 95     static _utf8_encode(string) {
 96         string = string.replace(/\r\n/g,"\n");
 97         var utftext = "";
 98 
 99         for (var n = 0; n < string.length; n++) {
100 
101             var c = string.charCodeAt(n);
102 
103             if (c < 128) {
104                 utftext += String.fromCharCode(c);
105             }
106             else if((c > 127) && (c < 2048)) {
107                 utftext += String.fromCharCode((c >> 6) | 192);
108                 utftext += String.fromCharCode((c & 63) | 128);
109             }
110             else {
111                 utftext += String.fromCharCode((c >> 12) | 224);
112                 utftext += String.fromCharCode(((c >> 6) & 63) | 128);
113                 utftext += String.fromCharCode((c & 63) | 128);
114             }
115         }
116         return utftext;
117     }
118 
119     /**
120      * Private method for UTF-8 decoding.
121      * @param {string} utftext - The UTF-8 encoded string to decode.
122      * @returns {string} The decoded string.
123      * @private
124      */
125     static _utf8_decode(utftext) {
126         var string = "";
127         var i = 0;
128         var c = 0; 
129         var c1 = 0;
130         var c2 = 0;
131         var c3 = 0;
132 
133         while ( i < utftext.length ) {
134 
135             c = utftext.charCodeAt(i);
136 
137             if (c < 128) {
138                 string += String.fromCharCode(c);
139                 i++;
140             }
141             else if((c > 191) && (c < 224)) {
142                 c2 = utftext.charCodeAt(i+1);
143                 string += String.fromCharCode(((c & 31) << 6) | (c2 & 63));
144                 i += 2;
145             }
146             else {
147                 c2 = utftext.charCodeAt(i+1);
148                 c3 = utftext.charCodeAt(i+2);
149                 string += String.fromCharCode(((c & 15) << 12) | ((c2 & 63) << 6) | (c3 & 63));
150                 i += 3;
151             }
152         }
153         return string;
154     }
