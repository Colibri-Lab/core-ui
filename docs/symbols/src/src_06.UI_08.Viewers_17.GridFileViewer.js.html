1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.GridFileViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-gridfile-viewer-component');
 18 
 19         this._value = null;
 20 
 21         this.AddHandler('ContextMenuIconClicked', (event, args) => this.__createContextMenu(event, args));
 22         this.AddHandler('ContextMenuItemClicked', (event, args) => this.__clickOnContextMenu(event, args));        
 23 
 24     }
 25 
 26     get value() {
 27         return this._value;
 28     }
 29 
 30     set value(value) {
 31         value = this._convertValue(value);
 32 
 33         this._value = value;
 34         if(this._value && this._value.length > 0) {
 35             this.hasContextMenu = true;
 36         }
 37         else {
 38             this.hasContextMenu = false;
 39         }
 40 
 41     }
 42 
 43     _createContextMenuButton() {
 44         if(!this._hasContextMenu || this.Children(this._name + '-contextmenu-icon-parent')) {
 45             return;
 46         }
 47 
 48         this.AddClass('app-component-hascontextmenu');
 49 
 50 
 51         const contextMenuParent = new Colibri.UI.Pane(this._name + '-contextmenu-icon-parent', this);
 52         contextMenuParent.AddClass('app-contextmenu-icon-component');
 53         contextMenuParent.shown = true;
 54         contextMenuParent.AddHandler('Clicked', (event, args) => this.Dispatch('ContextMenuIconClicked', args));
 55 
 56         let icons = [];
 57         this.value.forEach((v) => {
 58             let icon = Colibri.UI.FileDownloadIcon;
 59             if(v.name && v.name.indexOf('-signed') !== -1) {
 60                 icon = Colibri.UI.SignedFileDownloadIcon;
 61             }
 62 
 63             if(icons.indexOf(icon) == -1) {
 64                 icons.push(icon);
 65                 const contextMenuIcon = new Colibri.UI.Icon(this._name + '-contextmenu-icon', contextMenuParent);
 66                 contextMenuIcon.shown = true;
 67                 contextMenuIcon.value = icon;    
 68             }
 69 
 70         });
 71         
 72     }
 73 
 74     _removeContextMenuButton() {
 75         if(this._hasContextMenu && this.Children(this._name + '-contextmenu-icon-parent')) {
 76             this.Children(this._name + '-contextmenu-icon-parent').Dispose();
 77             this.RemoveClass('app-component-hascontextmenu');
 78         }
 79     }
 80 
 81     /**
 82      * @private
 83      * @param {Colibri.Events.Event} event event object
 84      * @param {*} args event arguments
 85      */ 
 86     __createContextMenu(event, args) {
 87         let contextmenu = [];
 88         this.value.forEach((v) => {
 89             let icon = Colibri.UI.ContextMenuDownloadFileIcon;
 90             if(v.name && v.name.indexOf('-signed') !== -1) {
 91                 icon = Colibri.UI.ContextMenuSignedDownloadIcon;
 92             }
 93             contextmenu.push({name: 'file' + v.guid ?? v, title: v.name ?? 'Загрузить файл', icon: icon, file: v});
 94         });
 95 
 96         this.contextmenu = contextmenu;
 97         this.ShowContextMenu(args.isContextMenuEvent ? [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.RB] : [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.LB], '', args.isContextMenuEvent ? {left: args.domEvent.clientX, top: args.domEvent.clientY} : null);
 98         
 99     }
100 
101     /**
102      * @private
103      * @param {Colibri.Events.Event} event event object
104      * @param {*} args event arguments
105      */ 
106     __clickOnContextMenu(event, args) {
107         if(!args.menuData) {
108             return;
109         }
110         const file = args.menuData.file;
111         window.open((window.rpchandler ? window.rpchandler : '') + this._download + '?guid=' + (file.guid ?? file));
112         
113     }
114 
115     set download(value) {
116         this._download = value;
117     }
118     get download() {
119         return this._download;
120     }
121 
122 }
123 Colibri.UI.Viewer.Register('Colibri.UI.GridFileViewer', '#{ui-viewers-gridfile}');
