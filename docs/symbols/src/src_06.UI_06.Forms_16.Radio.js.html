1 /**
  2  * @class
  3  * @memberof Colibri.UI.Forms
  4  * @extends Colibri.UI.Forms.Field
  5  */
  6 Colibri.UI.Forms.Radio = class extends Colibri.UI.Forms.Field {
  7 
  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-radio-field');
 14 
 15         this._validated = false;
 16         
 17         if(this._fieldData?.params?.readonly === undefined) {
 18             this.readonly = false;    
 19         }
 20         else {
 21             this.readonly = this._fieldData?.params?.readonly;
 22         }
 23         if(this._fieldData?.params?.enabled === undefined) {
 24             this.enabled = true;
 25         }
 26         else {
 27             this.enabled = this._fieldData.params.enabled;
 28         }
 29 
 30         this._loadValues();
 31 
 32     }
 33 
 34     /**
 35      * Focus on component
 36      */
 37     Focus() {
 38         if(this._element.querySelector('input')) {
 39             this._element.querySelector('input').focus();
 40         } else {
 41             this._focusOnFirst = true;
 42         }
 43     }
 44 
 45     /**
 46      * Value
 47      * @type {string}
 48      */
 49     get value() {
 50         return this._value;
 51     }
 52 
 53     /**
 54      * Value
 55      * @type {string}
 56      */
 57     set value(value) {
 58         this._value = value;
 59         this._showValue();
 60         this.Validate();
 61     }
 62 
 63     /** @private */
 64     _showValue() {
 65 
 66         let value = this._value;
 67         if(this._fieldData?.selector?.value && Object.isObject(this._value)) {
 68             value = this._value[this._fieldData?.selector?.value];
 69         }
 70 
 71         const input = this._element.querySelector('input[value="' + value + '"]');
 72         if(input) {
 73             input.attr('checked', 'checked');
 74         }
 75 
 76     }
 77 
 78     /**
 79      * Array of values
 80      * @type {Array}
 81      */
 82     get values() {
 83         return this._fieldData.values;
 84     }
 85     /**
 86      * Array of values
 87      * @type {Array}
 88      */
 89     set values(value) {
 90         value = this._convertProperty('Array', value);
 91         this._fieldData.values = value;
 92         this._loadValues();
 93     }
 94 
 95     /** @private */
 96     _loadValues() {
 97         const contentContainer = this.contentContainer;
 98         const ident = Date.Mc();
 99 
100         let promise = null;
101         let values = this._fieldData.values;
102         if(values instanceof Function) {
103             promise = values(this._fieldData, this);
104         } else {
105             promise = Promise.resolve(values);
106         }
107 
108         promise.then((values) => {
109             contentContainer.container.html('');
110             let selectedValue = null;
111             Object.values(values).forEach((value) => {
112                 if(value?.__selected) {
113                     selectedValue = value.value;
114                 }
115                 contentContainer.container.append(Element.fromHtml('<label><input type="radio" name="' + this.name + '" id="' + ident + '" value="' + value.value + '"' + (value?.__selected ? ' checked="checked"' : '') + ' /><span>' + (value.title[Lang.Current] ?? value.title) + '</span></label>'))
116             });
117     
118             contentContainer.container.querySelectorAll('input').forEach(input => input.addEventListener('click', e => {
119                 this._value = e.target.value;
120                 this.Dispatch('Changed', {domEvent: e, component: this});
121             }));
122 
123             if(selectedValue) {
124                 this._value = selectedValue;
125             }
126             
127             this._showValue();
128 
129             if(this._focusOnFirst) {
130                 this.Focus();
131             }
132 
133         });
134 
135 
136 
137     }
138 
139     /**
140      * Tab index
141      * @type {number}
142      */
143     get tabIndex() {
144         return this._input && this._input.attr('tabIndex');
145     }
146     /**
147      * Tab index
148      * @type {number}
149      */
150     set tabIndex(value) {
151         this._input && this._input.attr('tabIndex', value === true ? Colibri.UI.tabIndex++ : value);
152     }
153     
154 }
