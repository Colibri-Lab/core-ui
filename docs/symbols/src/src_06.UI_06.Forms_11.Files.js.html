  8     RenderFieldContainer() {
  9 
 10         this.AddClass('app-component-files-field');
 11 
 12         this._maxCount = null;
 13         this._allowedExtensions = null;
 14         this._maxFileSize = null;
 15 
 16         if(!Array.isArray(this._fieldData.params.allow)) {
 17             this._fieldData.params.allow = this._fieldData.params.allow.split(',');
 18         }
 19         
 20         if(typeof this._fieldData.params.size === 'string') {
 21             this._fieldData.params.size = parseInt(this._fieldData.params.size);
 22         }
 23         
 24         this._allowedExtensions = this._fieldData.params.allow ?? null;
 25         this._maxFileSize = this._fieldData.params.size ?? null;
 26         this._maxCount = this._fieldData.params.maxadd ?? null;
 27 
 28         this._validated = false;
 29         this._errorMessages = [];
 30 
 31         this._dropAreaEnabled = false;
 32         if (this._fieldData.params && this._fieldData.params.droparea) {
 33             this._dropAreaEnabled = this._fieldData.params.droparea;
 34         }
 35 
 36         if (this._fieldData.params && this._fieldData.params.reverse) {
 37             this.AddClass('-reverse');
 38         }
 39         else {
 40             this.RemoveClass('-reverse');
 41         }
 42 
 43         this._renderFilesList();
 44         this._renderInput();
 45         this._handleEvents();
 46 
 47         if(this._fieldData?.params?.readonly === undefined) {
 48             this.readonly = false;    
 49         }
 50         else {
 51             this.readonly = this._fieldData?.params?.readonly;
 52         }
 53         if(this._fieldData?.params?.enabled === undefined) {
 54             this.enabled = true;
 55         }
 56         else {
 57             this.enabled = this._fieldData.params.enabled;
 58         }
 59 
 60     }

 62     _handleEvents() {
 63 
 64         this._files.AddHandler('ItemClicked', (event, args) => this._itemClicked(args.item.value));
 65 
 66         if (this.dropAreaEnabled) {
 67             /* Валидация выбранных файлов, отображение файлов, прошедших проверку, вывод ошибок */
 68             this._input.AddHandler('InputFileChosen', (event, args) => {
 69                 if (args.files) {
 70                     let validatedFiles = this._validate(args.files);
 71 
 72                     if (validatedFiles.length > 0) {
 73                         this._addValue(validatedFiles);
 74                         this._clearInput();
 75 
 76                         if (this._fieldData.params && this._fieldData.params.more) {
 77                             this._input.title = this._fieldData.params.more;
 78                             this._input.AddClass('-full');
 79                         }
 80                     }
 81 
 82                     if (!this._validated) {
 83                         this._showError();
 84                     }
 85                     this.Dispatch('Changed', Object.assign(args, {component: this}));
 86                 }
 87             });
 88         }
 89         else {
 90 
 91             this._input.AddHandler('InputFileChanged', (event, args) => {
 92                 this._files.shown = true;
 93                 this.lastValue = this.value;
 94                 this._addValue(this._input.Files());
 95 
 96                 if (this._fieldData.params && this._fieldData.params.more) {
 97                     this._input.title = this._fieldData.params.more;
 98                     this._input.AddClass('-full');
 99                 }
100 
101                 this.Dispatch('Changed', Object.assign(args, {component: this}));
102 
103             });
104         }
105     }

107     /**
108      * Нарисовать список выбранных файлов
109      * @private
110      */
111     _renderFilesList() {
112         this._files = new Colibri.UI.List('list', this.contentContainer);
113         this._filesGroup = this._files.AddGroup('group', '');
114         this._files.__renderItemContent = (itemData, container) => {
115             const picture = new Colibri.UI.Image('picture', container);
116             const icon = new Colibri.UI.Icon('icon', container);
117             const filename = new Colibri.UI.TextSpan('span', container);
118             let sign = null;
119             if (this._fieldData?.params?.canbesigned) {
120                 sign = new Colibri.UI.TextSpan('sign', container);
121             }
122             const size = new Colibri.UI.TextSpan('size', container);
123             size.shown = true;
124             size.value = itemData.file.size.toSizeString(['bytes', 'Kb', 'Mb', 'Gb'], 1024);
125             const deleteIcon = new Colibri.UI.TextSpan('delete', container);
126 
127             filename.AddClass('files-file-name');
128             if ((itemData.file?.type ?? itemData.file?.mimetype).match('image.*')) {
129                 picture.shown = filename.shown = deleteIcon.shown = true;
130                 picture.image = itemData.file;
131                 picture.width = picture.height = 40;
132                 
133             } else {
134                 icon.shown = filename.shown = deleteIcon.shown = true;
135                 icon.value = Colibri.UI.FileLinkIcon;
136                 deleteIcon.value = Colibri.UI.ClearIcon;
137             }
138 
139             if (sign) {
140                 sign.shown = true;
141             }
142 
143             deleteIcon.value = Colibri.UI.ClearIcon;
144 
145             if (sign) {
146                 sign.value = '#{ui-files-sign}';
147             }
148 
149             deleteIcon.AddHandler('Clicked', (event, args) => {
150                 const value = this.value,
151                     delParams = this._delDialog,
152                     delAction = () => {
153                         this.lastValue = value;
154                         args.removed = itemData.file || null;
155                         event.sender.parent.Dispose();
156                         if (this._filesGroup.children == 0) {
157                             this._files.shown = false;
158                             if (this._fieldData.params && this._fieldData.params.button) {
159                                 this._input.title = this._fieldData.params.button;
160                                 this._input.RemoveClass('-full');
161                             }
162                         }
163                         this.Dispatch('Changed', Object.assign(args, {component: this}));
164                     };
165 
166                 if (delParams) {
167                     const dialog = new Colibri.UI.ConfirmDialog(this.name + '-confirm-delete-dialog', document.body);
168                     dialog.Show(delParams.title, delParams.message.replace(/{([^{}]+)}/g, (keyExpr, key) => {
169                         return itemData.file[key] || "";
170                     }), delParams.button ?? 'Удалить', (result) => {
171                         if (result === true) {
172                             delAction.call(this);
173                         }
174                     });
175                 } else {
176                     delAction.call(this);
177                 }
178 
179                 args.domEvent?.stopPropagation();
180                 return false;
181             })
182 
183             sign && sign.AddHandler('Clicked', (event, args) => {
184 
185                 const component = eval(this._fieldData?.params?.sign_component);
186                 component.Show(itemData.file, (signedFile) => {
187                     container.value.file = signedFile;
188                     container.value.title = signedFile.name;
189                     sign.value = 'Файл подписан';
190                     sign.shown = false;
191                     filename.value = signedFile.name;
192                 });
193 
194                 args.domEvent?.stopPropagation();
195                 return false;
196             });
197             filename.value = itemData.title;
198         }
199     }

251     /**
252      * Собрать список разрешенных расширений в строку нужного формата
253      * @return {string}
254      * @private
255      */
256     _extensionsToString() {
257         let extensionsString = [];
258 
259         if (Array.isArray(this._allowedExtensions)) {
260             let extensions = this._allowedExtensions.map((item) => String(item).toUpperCase());
261             extensionsString.push(extensions.join(', '));
262         }
263 
264         if (this._maxCount) {
265             extensionsString.push('#{ui-files-filescount}'.replaceAll('%s', this._maxCount));
266         }
267 
268         if (this._maxFileSize) {
269             extensionsString.push('#{ui-files-size}'.replaceAll('%s', (this._maxFileSize + 0).toSizeString(['байт', 'Кб', 'Мб'], 1024, true)));
270         }
271 
272 
273         return extensionsString.join(' ');
274     }

276     /**
277      * Валидация выбранного списка файлов
278      * @param {array} filesList
279      * @return {array} список файлов, прошедших проверку
280      * @private
281      */
282     _validate(filesList) {
283         let error = false;
284         let validatedList = filesList;
285 
286         if (this._maxCount) {
287             if ((this._getValue().length + filesList.length) > this._maxCount) {
288                 error = true;
289                 this._validated = false;
290 
291                 let countAvailable = this._maxCount - this._getValue().length;
292                 validatedList.splice(countAvailable);
293                 this._errorMessages.push('#{ui-files-filescount2}'.replaceAll('%s', this._maxCount));
294             }
295         }
296 
297         if (this._maxFileSize) {
298             validatedList.forEach((file, index) => {
299                 if (file.size > this._maxFileSize) {
300                     error = true;
301                     this._validated = false;
302 
303                     validatedList.splice(index, 1);
304                     this._errorMessages.push('#{ui-files-filehuge}'.replaceAll('%s', file.name));
305                 }
306             });
307         }
308 
309         if (this._allowedExtensions) {
310             let extensions = this._allowedExtensions.map((item) => item.toLowerCase());
311 
312             filesList.forEach((file, index) => {
313                 let fileName = file.name;
314                 let ext = '';
315 
316                 if (fileName.includes('.')) {
317                     ext = fileName.substring(fileName.lastIndexOf('.') + 1);
318                     ext = ext.toLowerCase();
319                 }
320 
321                 if (!extensions.includes(ext)) {
322                     error = true;
323                     this._validated = false;
324 
325                     validatedList.splice(index, 1);
326                     this._errorMessages.push('#{ui-files-format} ' + fileName);
327                 }
328             });
329         }
330 
331         if (!error) {
332             this._errorMessages = [];
333             this._validated = true;
334         }
335 
336         return validatedList;
337     }

339     _itemClicked(value) {
340         if (value.file instanceof File) {
341             value.file.download();
342         }
343         else if (this._fieldData.params?.download && value.file?.guid) {
344             window.open((window.rpchandler ?? '') + this._fieldData.params.download + '?guid=' + value.file.guid);
345         }
346     }

348     Rollback() {
349         if (this.lastValue !== undefined) {
350             this._filesGroup.Clear();
351             this.value = this.lastValue;
352             if (this._filesGroup.children && !(this._filesGroup.Children('lastChild').value instanceof File)) {
353                 this._input._input.value = null;
354                 this._input._inputFiles.pop();
355             }
356         }
357     }

399     _getValue() {

403         })
404         return ret;

406     _setValue(value) {
407         if(Array.isArray(value)) {
408             this._filesGroup.value = [];
409             for(const file of value) {
410                 this._filesGroup.AddItem({title: file.name, file: file});
411             }            
412         } else if( !value) {
413             this._filesGroup.value = [];
414         } else {
415             this._filesGroup.AddItem({title: value.name, file: value});
416         }
417         
418         this._files.shown = this._filesGroup.value.length > 0;
419 
420     }
421     _addValue(value) {
422         if(Array.isArray(value)) {
423             for(const file of value) {
424                 this._filesGroup.AddItem({title: file.name, file: file});
425             }            
426         } else {
427             this._filesGroup.AddItem({title: value.name, file: value});
428         }
429         this._files.shown = true;
430     }
