1 /**
  2  * Drop manager
  3  * @class
  4  * @extends Colibri.Events.Dispatcher
  5  * @memberof Colibri.UI
  6  */
  7 Colibri.UI.FileDropManager = class extends Colibri.Events.Dispatcher {
  8 
  9     /**
 10      * @constructor
 11      * @param {Colibri.UI.Component} container container to drop in
 12      * @param {string} message message to show when dragging 
 13      */
 14     constructor(container, message = '#{ui-filedropmanager-message}') {
 15         super();
 16 
 17         this._allowSize = 900000000;
 18         this._allowTypes = '*';
 19         this._enabled = true;
 20         
 21         this._dropContainer = container;this._dropHover = Element.create('div', {class: 'app-drop-over'});
 22         this._dropHover.append(Element.create('div'));
 23         container.append(this._dropHover);
 24 
 25         this.RegisterEvent('FileDropped', false, 'Когда перетащили файл в контейнер');
 26 
 27         this._initManager();
 28         this.message = message;
 29     }
 30 
 31     /**
 32      * Enable/Disable manager
 33      * @type {boolean}
 34      */
 35     get enabled() {
 36         return this._enabled;
 37     }
 38     /**
 39      * Enable/Disable manager
 40      * @type {boolean}
 41      */
 42     set enabled(value) {
 43         this._enabled = value;
 44     }
 45 
 46     /**
 47      * Message value
 48      * @type {string}
 49      */
 50     set message(value) {
 51         this._message = value;
 52         this._setMessage();
 53     }
 54 
 55     /**
 56      * Message value
 57      * @type {string}
 58      */
 59     get message() {
 60         return this._message;
 61     }
 62 
 63     /**
 64      * Allow types
 65      * @type {string|Array}
 66      */
 67     set allowTypes(value) {
 68         if(typeof value === 'string') {
 69             value = value.split(',');
 70         }
 71         this._allowTypes = value;
 72     }
 73 
 74     /**
 75      * Allow types
 76      * @type {string|Array}
 77      */
 78     get allowTypes() {
 79         return this._allowTypes;
 80     }
 81 
 82     /**
 83      * Allow size
 84      * @type {number}
 85      */
 86     set allowSize(value) {
 87         this._allowSize = value;
 88     }
 89 
 90     /**
 91      * Allow size
 92      * @type {number}
 93      */
 94     get allowSize() {
 95         return this._allowSize;
 96     }
 97 
 98     /**
 99      * @private
100      */
101     _setMessage() {
102         this._dropHover.querySelector('div').html(this._message);
103     }
104 
105     /**
106      * @private
107      */
108     _initManager() {
109         
110         this._dropContainer.addEventListener('dragover', (e) => {
111             if(!this._enabled) {
112                 e.stopPropagation();
113                 e.preventDefault();
114                 return false;                    
115             }
116             if(!this._dropContainer.classList.contains('-dragging')) {
117                 this._dropContainer.classList.add('-dragging');
118                 this._dropHover.css({zIndex: Colibri.UI.maxZIndex + 1});
119             }
120             e.stopPropagation();
121             e.preventDefault();
122             return false;
123 
124         });
125         this._dropHover.addEventListener('dragleave', (e) => {
126             if(!this._enabled) {
127                 e.stopPropagation();
128                 e.preventDefault();
129                 return false;                    
130             }
131             this._dropContainer.classList.remove('-dragging');
132             this._dropHover.css({zIndex: null});
133             e.stopPropagation();
134             e.preventDefault();
135             return false;
136         });
137         this._dropHover.addEventListener('drop', (e) => {
138             if(!this._enabled) {
139                 e.stopPropagation();
140                 e.preventDefault();
141                 return false;                    
142             }
143             this._dropContainer.classList.remove('-dragging');
144             this._dropHover.css({zIndex: null});
145 
146             let eventFiles;
147             if (e.dataTransfer.files) {
148                 eventFiles = e.dataTransfer.files;
149             }
150             else if (e.dataTransfer.items) {
151                 eventFiles = e.dataTransfer.items;
152             }
153 
154             this._checkFiles(eventFiles);
155 
156             e.stopPropagation();
157             e.preventDefault();
158             return false;
159 
160         });
161 
162     }
163 
164     /**
165      * @private
166      * @param {Colibri.Events.Event} event event object
167      * @param {*} args event arguments
168      */ 
169     __dragStartFromSources(event, args) {
170         this._current = args.domEvent.target.closest('[data-object-name]').tag('component');
171     }
172 
173     /**
174      * @private
175      * @param {Colibri.Events.Event} event event object
176      * @param {*} args event arguments
177      */ 
178     __dragEndFromSources(event, args) {
179         this._current = null;
180 
181         const dropComponent = document.querySelector('.app-drop-component');
182         const dropTargetComponent = document.querySelector('.app-drop-target-component');
183         dropComponent && dropComponent.classList.remove('app-drop-component');
184         dropTargetComponent && dropTargetComponent.classList.remove('app-drop-target-component');
185 
186     }
187 
188     /**
189      * @private
190      * @param {Colibri.Events.Event} event event object
191      * @param {*} args event arguments
192      */ 
193     __dragOverTheDestination(event, args) {
194         if(!this._current) {
195             return false;
196         }
197 
198         const dropElement = args.domEvent.target.closest('[data-object-name][dropable="true"]');
199         if(!dropElement) {
200             return false;
201         }
202 
203         const dropTarget = dropElement.tag('component');
204         if(this._current == dropTarget) {
205             return false;
206         }
207 
208         dropTarget.AddClass('app-drop-target-component');
209         event.sender.AddClass('app-drop-component');
210 
211         args.domEvent.dataTransfer.effectAllowed = 'all';
212         args.domEvent.dataTransfer.dropEffect = 'move';
213         args.domEvent.preventDefault();
214         return true;
215     }
216 
217     /**
218      * @private
219      * @param {Colibri.Events.Event} event event object
220      * @param {*} args event arguments
221      */ 
222     __dragLeaveTheDestination(event, args) {
223         if(!this._current) {
224             return false;
225         }
226 
227         const dropElement = args.domEvent.target.closest('[data-object-name][dropable="true"]');
228         if(!dropElement) {
229             return false;
230         }
231 
232         const dropTarget = dropElement.tag('component');
233         dropTarget.RemoveClass('app-drop-target-component');
234         event.sender.RemoveClass('app-drop-component');
235 
236         args.domEvent.preventDefault();
237     }
238 
239     /**
240      * @private
241      * @param {Colibri.Events.Event} event event object
242      * @param {*} args event arguments
243      */ 
244     __dragDropOnTheDestination(event, args) {
245 
246         if(!this._current) {
247             return false;
248         }
249 
250         const dropElement = args.domEvent.target.closest('[data-object-name][dropable="true"]');
251         if(!dropElement) {
252             return false;
253         }
254 
255         const dropTarget = dropElement.tag('component');
256         args.domEvent.preventDefault();
257 
258         this.Dispatch('DragDropComplete', {dragged: this._current, droppedTo: dropTarget});
259 
260     }
261 
262     /**
263      * @private
264      */
265     _checkFiles(files) {
266         let errors = [];
267         let success = [];
268         for(const file of files) {
269             const ext = file.name.extractExt();
270             if(file.size > this._allowSize || (this._allowTypes !== '*' && this._allowTypes.indexOf(ext) === -1)) {
271                 errors.push({file: file, error: '#{ui-filedropmanager-error1} ' + (this._allowTypes !== '*' ? '#{ui-renderer-allowed}'.replaceAll('%s', this._allowTypes.join(',')) + ', ' : '') + '#{ui-renderer-max}'.replaceAll('%s', this._allowSize.toSizeString(['bytes', 'Kb', 'Mb', 'Gb'], 1024, true))});
272             }
273             else {
274                 success.push(file);
275             }
276         }
277         this.Dispatch('FileDropped', {errors: errors, success: success});
278     }
