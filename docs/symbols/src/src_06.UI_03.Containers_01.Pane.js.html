1 Colibri.UI.Pane = class extends Colibri.UI.Component {
  2 
  3     static ResizeNone = 'none';
  4     static ResizeAll = 'all';
  5     static ResizeHorizontalOnly = 'horizontal';
  6     static ResizeVerticalOnly = 'vertical';
  7 
  8     _resizable = Colibri.UI.Pane.ResizeNone;
  9     _resizeHandler = null;
 10     _resizing = false;
 11     _resizeData = {
 12         horizontalSize: 0,
 13         verticalSize: 0,
 14     };
 15 
 16     /**
 17      * Конструктор
 18      * @param {string} name Наименование компонента
 19      * @param {(HTMLElement|Colibri.UI.Component))} container контейнер 
 20      * @param {(string|HTMLElement)} element название тэга, либо содержание компонента (шаблон) в виде строки или обьекта 
 21      * @param {int} resizable способ изменения размера 
 22      */
 23     constructor(name, container, element, resizable= 'none') {
 24         super(name, container, element || Element.create('div'));
 25 
 26         this.AddClass('app-component-pane');
 27 
 28         this._resizable = resizable || 'none';
 29         this.AddClass('app-component-resize-' + this._resizable);
 30     }
 31 
 32     _createResizeHandler() {
 33         this._resizeHandler = Element.create('div', {class: 'app-component-pane-resize'});
 34         this._element.prepend(this._resizeHandler);
 35 
 36         const stopClick = (e) => { e.preventDefault(); e.stopPropagation(); return false; };
 37 
 38         const startResize = (e) => {
 39             this._resizing = true;
 40             Colibri.UI.Resizing = true;
 41             this._resizeData = this._element.bounds();
 42 
 43             // ставим на документ, чтобы точно перехватить        
 44             document.addEventListener("touchend", stopResize, false);
 45             document.addEventListener("mouseup", stopResize, false);
 46 
 47             document.addEventListener("touchmove", doResize, false);
 48             document.addEventListener("mousemove", doResize, false);
 49 
 50             return false;
 51 
 52         };
 53 
 54         const stopResize = (e) => {
 55             e.preventDefault();
 56             e.stopPropagation();
 57         
 58             this._resizing = false;
 59             Colibri.UI.Resizing = false;
 60 
 61             document.removeEventListener("touchend", stopResize, false);
 62             document.removeEventListener("mouseup", stopResize, false);
 63     
 64             document.removeEventListener("touchmove", doResize, false);
 65             document.removeEventListener("mousemove", doResize, false);
 66 
 67             return false;
 68 
 69         };
 70 
 71         const doResize = (e) => {
 72             if (this._resizing) {
 73                 e.preventDefault();
 74     
 75                 const wdelta = (e.pageX - (this._resizeData.left + this._resizeData.outerWidth));
 76                 const hdelta = (e.pageY - (this._resizeData.top + this._resizeData.outerHeight));
 77 
 78                 if(this._resizable == Colibri.UI.Pane.ResizeHorizontalOnly || this._resizable == Colibri.UI.Pane.ResizeAll) {
 79                     this.width += wdelta + 10;
 80                 }
 81     
 82                 if(this._resizable == Colibri.UI.Pane.ResizeVerticalOnly || this._resizable == Colibri.UI.Pane.ResizeAll) {
 83                     this.height += hdelta + 10;
 84                 }
 85     
 86                 this._resizeData = this._element.bounds();
 87 
 88                 return false;
 89             }
 90         };
 91 
 92         this._resizeHandler.addEventListener("touchstart", startResize, false);
 93         this._resizeHandler.addEventListener("mousedown", startResize, false);
 94 
 95     }
 96 
 97     _removeResizeHandler() {
 98         this._resizeHandler && this._resizeHandler.remove();
 99     }
100 
101     get resizable() {
102         return this._resizable;
103     }
104     set resizable(value) {
105         this.RemoveClass('app-component-resize-' + this._resizable);
106         this._resizable = value;
107         this.AddClass('app-component-resize-' + this._resizable);
108 
109         if(this._resizable === true) {
110             this._createResizeHandler();
111         }
112         else {
113             this._removeResizeHandler();
114         }
115     }
116 
117 }
