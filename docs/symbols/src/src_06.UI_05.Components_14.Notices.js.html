1 Colibri.UI.Notices = class extends Colibri.UI.Pane {
  2     
  3     constructor(name, container) {
  4         super(name, container, Element.create('div'));
  5         this.AddClass('app-notices-component');
  6 
  7         this._list = new Colibri.UI.List('list', this);
  8         this._list.shown = true;
  9         this._group = this._list.AddGroup('group', '');
 10     }
 11 
 12     /**
 13      * Добавить сообщение
 14      * @param {Colibri.UI.Notice} noticeData сообщение 
 15      * @returns {Colibri.UI.Notice}
 16      */
 17     Add(noticeData) {
 18 
 19         if(!noticeData.title) {
 20             return;
 21         }
 22 
 23         console.log(noticeData); 
 24 
 25         this.shown = true;
 26         this.BringToFront();
 27 
 28         if(noticeData.severity === Colibri.UI.Notice.Error) {
 29             console.log(noticeData);
 30             console.trace();
 31             debugger;
 32         }
 33         const notice = this._group.AddItem(noticeData);
 34         
 35         const removeNotice = () => {
 36             notice.RemoveClass(noticeData.className);
 37             Colibri.Common.Delay(300).then(() => {
 38                 notice.height = 0;
 39                 return Colibri.Common.Delay(100);
 40             }).then(() => {
 41                 notice.Dispose();
 42                 if(this._group.children == 0) {
 43                     this.shown = false;
 44                 }
 45             });
 46         };
 47 
 48         Colibri.Common.Delay(10).then(() => {
 49 
 50             notice.AddClass(noticeData.className);
 51 
 52             const icon = new Colibri.UI.Icon('icon', notice);
 53             icon.shown = true;
 54             icon.value = Colibri.UI.CloseIcon;
 55             icon.AddClass('app-notice-icon-component');
 56             icon.AddHandler('Clicked', removeNotice);
 57             notice.AddHandler('Clicked', removeNotice);
 58 
 59             Colibri.Common.Delay(noticeData.timeout).then(removeNotice);
 60             
 61         });
 62         return notice;
 63     }
 64 
 65 }
 66 
 67 Colibri.UI.Notice = class {
 68 
 69     static Error = 'error';
 70     static Success = 'success';
 71     static Warning = 'warning';
 72     static White = 'white';
 73 
 74     _exception = null;
 75 
 76     constructor(title, severity = Colibri.UI.Notice.Error, timeout = 3000) {
 77         if(typeof title === 'object' && !!title.code && !!title.message) {
 78             // error object
 79             this._exception = title;
 80             title = this._exception.message;
 81         }
 82         this._title = title;
 83         this._severity = severity;
 84         this._timeout = timeout;
 85     }
 86 
 87     get title() {
 88         return this._title;
 89     }
 90 
 91     get severity() {
 92         return this._severity;
 93     }
 94 
 95     get timeout() {
 96         return this._timeout;
 97     }
 98 
 99     get className() {
100         return 'app-notice-' + this.severity + '-component';
101     }
102 
103 }
