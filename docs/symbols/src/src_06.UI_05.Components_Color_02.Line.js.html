 20         this.tabIndex = 0;
 21         this.AddHandler('KeyDown', (event, args) => this.__keyDown(event, args));
 22 
 23         new Colibri.UI.Drag(this._pointer.container, this.container, (left, top) => {
 24             this._pointer.styles = {left: left + 'px'};
 25             this._setNewValue(left);            
 26         });
 27 
 28         Colibri.Common.Delay(100).then(() => {
 29             this._renderGradient();
 30         });
 31         
 32     }
 33 
 34     __keyDown(event, args) {
 35         if(['ArrowLeft', 'ArrowRight'].indexOf(args.domEvent.code) !== -1) {
 36             if(args.domEvent.code == 'ArrowLeft') {
 37                 this.value += 0.01;
 38             }
 39             else if(args.domEvent.code == 'ArrowRight') {
 40                 this.value -= 0.01;
 41             }
 42         }
 43     }
 44 
 45     set width(value) {
 46         super.width = value;
 47         this._renderGradient();
 48     }
 49     set height(value) {
 50         super.height = value;
 51         this._renderGradient();
 52     }
 53 
 54     _setNewValue(left) {
 55         left += 5;
 56         const bounds = this._element.bounds();
 57         let t = Math.round(left / (bounds.outerWidth / 360));
 58         t = Math.abs(t - 360);
 59         t = (t == 360) ? 0 : t;
 60         t = Math.round(t * 100 / 360) / 100;
 61 
 62         this.value = t;
 63         this.Dispatch('Changed', {value: this.value}); 
 64     }
 65 
 66     _showValue() {
 67         let t = this.value;
 68         const bounds = this._element.bounds();
 69         const pointerBounds = this._pointer.container.bounds();
 70 
 71         // процент
 72         let left = bounds.outerWidth - ((t * 100) * bounds.outerWidth / 100);
 73         this._pointer.styles = {backgroundColor: Colibri.UI.Rgb.Create().fromHSV(t, 1, 1).hex, left: (left - pointerBounds.outerWidth / 2)  + 'px'};
 74 
 75     }
 76 
 77     _renderGradient() {
 78         const bounds = this._element.bounds(); 
 79 		const canva = this._canvas.container.getContext("2d");
 80 
 81 		const gradient = canva.createLinearGradient(bounds.outerWidth, bounds.outerHeight / 2, 0, bounds.outerHeight);
 82 	    const hue = [[255, 0, 0], [255, 255, 0], [0, 255, 0], [0, 255, 255], [0, 0, 255], [255, 0, 255], [255, 0, 0]];
 83 		for (var i = 0; i <= 6; i++) {
 84 			gradient.addColorStop(i * 1 / 6, 'rgb(' + hue[i][0] + ',' + hue[i][1] + ',' + hue[i][2] + ')');
 85 		}
 86 
 87         canva.fillStyle = gradient;
 88 		canva.fillRect(0, 0, bounds.outerWidth + 100, bounds.outerHeight * 100);
 89     }
 90 
 91     __lineClicked(event, args) {
 92         const bounds = this._element.bounds();
 93         const e = args.domEvent;
 94         let left = e.pageX - bounds.left - 5;
 95         this._pointer.styles = {left: left + 'px'};
 96         this._setNewValue(left);
 97     }

 99 
100     get value() {
101         return this._value;
102     }
103     set value(value) {
104         if(value <= 0) {
105             value = 0;
106         }
107         if(value >= 1) {
108             value = 1;
109         }
110         this._value = value;
111         this._renderGradient();
112         this._showValue();
