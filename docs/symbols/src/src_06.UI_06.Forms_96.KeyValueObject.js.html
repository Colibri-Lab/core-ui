  6 Colibri.UI.Forms.KeyValueObject = class extends Colibri.UI.Forms.Field {

  8     RenderFieldContainer() {
  9 
 10         this.AddClass('app-component-simplearray-field');
 11         this._enabled = true;
 12 
 13         const contentContainer = this.contentContainer;
 14 
 15         this._grid = new Colibri.UI.Grid('grid', contentContainer);
 16         this._grid.shown = true;
 17         this._grid.rows.title = '';
 18         this._grid.height = 200;
 19 
 20         this._link = new Colibri.UI.Link('link', contentContainer);
 21         this._link.shown = true;
 22         this._link.value = '#{ui-fields-keyvalueobject-add}'
 23 
 24         if(this._fieldData?.params?.readonly === undefined) {
 25             this.readonly = false;    
 26         }
 27         else {
 28             this.readonly = this._fieldData?.params?.readonly;
 29         }
 30         if(this._fieldData?.params?.enabled === undefined) {
 31             this.enabled = true;
 32         }
 33         else {
 34             this.enabled = this._fieldData.params.enabled;
 35         }
 36 
 37         this._fieldData.params.simplearraywidth = (this._fieldData.params?.simplearraywidth ?? 1);
 38         this._fieldData.params.simplearrayheight = (this._fieldData.params?.simplearrayheight ?? 1);
 39 
 40         const column1 = this._grid.header.columns.Add('key', '');
 41         column1.width = '50%';
 42         column1.align = 'left';
 43         column1.editor = Colibri.UI.TextEditor;
 44         column1.editorAllways = true;
 45         column1.resizable = true;
 46         column1.value = '#{ui-fields-keyvalueobject-key}';
 47 
 48         const column2 = this._grid.header.columns.Add('value', '');
 49         column2.width = '50%';
 50         column2.align = 'left';
 51         column2.editorAllways = true;
 52         column2.editor = Colibri.UI.TextEditor;
 53         column2.value = '#{ui-fields-keyvalueobject-value}';
 54 
 55         this._grid.AddHandler('CellEditorChanged', (event, args) => this.Dispatch('Changed', {component: this}));
 56         this._link.AddHandler('Clicked', (event, args) => this._grid.rows.Add('row' + Date.Mc(), {key: 'new-key-' + (this._grid.rows.children - 1), value: ''}));
 57 
 58     } 
 59 
 60     _registerEvents() {
 61         super._registerEvents();
 62         // если нужно добавить что то
 63     }
 64 
 65     Focus() {
 66         this.contentContainer.Children('firstChild').Focus();
 67     }
 68 
 69     get readonly() {
 70         return this._grid.readonly;
 71     }
 72 
 73     set readonly(value) {
 74         value = this._convertProperty('Boolean', value);
 75         this._grid.readonly = value;
 76     }
 77 
 78     get enabled() {
 79         return this._enabled ?? true;
 80     }
 81 
 82     set enabled(value) {
 83         value = this._convertProperty('Boolean', value);
 84         if(this._enabled != value) {
 85             this._enabled = value;
 86             this.contentContainer.ForEach((name, component) => {
 87                 component.enabled = this._enabled; 
 88             });
 89             this._link && (this._link.enabled = this._enabled);
 90         }
 91     }
 92 
 93     get value() {
 94 
 95         let val = {};
 96         this._grid.ForEveryRow((name, row) => {
 97             val[row.value.key] = row.value.value;
 98         });
 99         return val;
100     }
101 
102     set value(value) {
103         this._grid.ClearAllRows();
104         
105         Object.forEach(value, (key, value) => {
106             this._grid.rows.Add('row' + Date.Mc(), {key: key, value: value});
107         });
108 
109         if(this._grid.rows.children === 1) {
110             this._link.Dispatch('Clicked');
111         }
112 
113     }
114 
115 
116     set tabIndex(value) {
117         this._grid.tabIndex = value;
118     }
119 
120 
121     get tabIndex() {
122         return this._grid.tabIndex;
123     }
124

130         return this._grid.height;
131     }

136     set height(value) {
