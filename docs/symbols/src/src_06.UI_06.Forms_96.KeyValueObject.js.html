  6 Colibri.UI.Forms.KeyValueObject = class extends Colibri.UI.Forms.Field {
  7 
  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-simplearray-field');
 14         this._enabled = true;
 15 
 16         const contentContainer = this.contentContainer;
 17 
 18         this._grid = new Colibri.UI.Grid('grid', contentContainer);
 19         this._grid.shown = true;
 20         this._grid.rows.title = '';
 21         this._grid.height = 200;
 22 
 23         this._link = new Colibri.UI.Link('link', contentContainer);
 24         this._link.shown = true;
 25         this._link.value = '#{ui-fields-keyvalueobject-add}'
 26 
 27         if(this._fieldData?.params?.readonly === undefined) {
 28             this.readonly = false;    
 29         }
 30         else {
 31             this.readonly = this._fieldData?.params?.readonly;
 32         }
 33         if(this._fieldData?.params?.enabled === undefined) {
 34             this.enabled = true;
 35         }
 36         else {
 37             this.enabled = this._fieldData.params.enabled;
 38         }
 39 
 40         this._fieldData.params.simplearraywidth = (this._fieldData.params?.simplearraywidth ?? 1);
 41         this._fieldData.params.simplearrayheight = (this._fieldData.params?.simplearrayheight ?? 1);
 42 
 43         const column1 = this._grid.header.columns.Add('key', '');
 44         column1.width = '50%';
 45         column1.align = 'left';
 46         column1.editor = Colibri.UI.TextEditor;
 47         column1.editorAllways = true;
 48         column1.resizable = true;
 49         column1.value = '#{ui-fields-keyvalueobject-key}';
 50 
 51         const column2 = this._grid.header.columns.Add('value', '');
 52         column2.width = '50%';
 53         column2.align = 'left';
 54         column2.editorAllways = true;
 55         column2.editor = Colibri.UI.TextEditor;
 56         column2.value = '#{ui-fields-keyvalueobject-value}';
 57 
 58         this._grid.AddHandler('CellEditorChanged', (event, args) => this.Dispatch('Changed', {component: this}));
 59         this._link.AddHandler('Clicked', (event, args) => this._grid.rows.Add('row' + Date.Mc(), {key: 'new-key-' + (this._grid.rows.children - 1), value: ''}));
 60 
 61     } 
 62 
 63     /** @protected */
 64     _registerEvents() {
 65         super._registerEvents();
 66         // если нужно добавить что то
 67     }
 68 
 69     /**
 70      * Focus on component
 71      */
 72     Focus() {
 73         this.contentContainer.Children('firstChild').Focus();
 74     }
 75 
 76     /**
 77      * Readonly
 78      * @type {boolean}
 79      */
 80     get readonly() {
 81         return this._grid.readonly;
 82     }
 83 
 84     /**
 85      * Readonly
 86      * @type {boolean}
 87      */
 88     set readonly(value) {
 89         value = this._convertProperty('Boolean', value);
 90         this._grid.readonly = value;
 91     }
 92 
 93     /**
 94      * Enable/Disable
 95      * @type {boolean}
 96      */
 97     get enabled() {
 98         return this._enabled ?? true;
 99     }
100 
101     /**
102      * Enable/Disable
103      * @type {boolean}
104      */
105     set enabled(value) {
106         value = this._convertProperty('Boolean', value);
107         if(this._enabled != value) {
108             this._enabled = value;
109             this.contentContainer.ForEach((name, component) => {
110                 component.enabled = this._enabled; 
111             });
112             this._link && (this._link.enabled = this._enabled);
113         }
114     }
115 
116     /**
117      * Value
118      * @type {Array}
119      */
120     get value() {
121 
122         let val = {};
123         this._grid.ForEveryRow((name, row) => {
124             val[row.value.key] = row.value.value;
125         });
126         return val;
127     }
128 
129     /**
130      * Value
131      * @type {Array}
132      */
133     set value(value) {
134         this._grid.ClearAllRows();
135         
136         Object.forEach(value, (key, value) => {
137             this._grid.rows.Add('row' + Date.Mc(), {key: key, value: value});
138         });
139 
140         if(this._grid.rows.children === 1) {
141             this._link.Dispatch('Clicked');
142         }
143 
144     }
145 
146     /**
147      * Tab index
148      * @type {number|boolean}
149      */
150     set tabIndex(value) {
151         this._grid.tabIndex = value;
152     }
153 
154     /**
155      * Tab index
156      * @type {number|boolean}
157      */
158     get tabIndex() {
159         return this._grid.tabIndex;
160     }
161 
162     /**
163      * Height
164      * @type {number}
165      */
166     get height() {
167         return this._grid.height;
168     }
169     /**
170      * Height
171      * @type {number}
172      */
173     set height(value) {
174         this._grid.height = value;
175     }
