1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.FilesDragAndDrop = class extends Colibri.UI.Component {
  7     
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      * @param {boolean} multiple can drop multiple files
 13      */    
 14     constructor(name, container, multiple = false) {
 15         super(name, container, Element.create('div'));
 16         this.AddClass('app-component-files-drag-and-drop');
 17 
 18         // внутренний невидимый инпут для выбора файлов из проводника по клику
 19         this._uniqueString = Number.unique();
 20         this._innerInput = Element.create('input', {type:'file', id: 'component-' + name + '-' + this._uniqueString, class: 'app-files-drag-and-drop'});
 21         this._element.append(this._innerInput);
 22 
 23         this._inputFiles = [];
 24         this._multiple = multiple;
 25         this._errorMessage = '';
 26 
 27         this._renderDropContainer();
 28         this._handleEvents();
 29     }
 30 
 31     _registerEvents() {
 32         super._registerEvents();
 33         this.RegisterEvent('InputFileChosen', false, 'Выбран файл/файлы');
 34     }
 35 
 36     _handleEvents() {
 37         this._innerInput.addEventListener('change', (event) => {
 38             this._dropContainer.Dispatch('Drop', {domEvent: event, inputFiles: this._innerInput.files});
 39             this._innerInput.value = '';
 40         });
 41 
 42         this._dropContainer.AddHandler('Clicked', (event, args) => { this._innerInput.click(); });
 43         this._dropContainer.AddHandler('DragOver', (event, args) => {
 44             args.domEvent.preventDefault();
 45             this.AddClass('-active');
 46         });
 47         this._dropContainer.AddHandler('DragLeave', (event, args) => { this.RemoveClass('-active'); });
 48 
 49         /* выбрать файлы в зависимости от источника, очистить выбор после вызова события  */
 50         this._dropContainer.AddHandler('Drop', (event, args) => {
 51             args.domEvent.preventDefault();
 52             this.RemoveClass('-active');
 53 
 54             let eventFiles;
 55             if (args.inputFiles) {
 56                 eventFiles = args.inputFiles;
 57             }
 58             else if (args.domEvent.dataTransfer.files) {
 59                 eventFiles = args.domEvent.dataTransfer.files;
 60             }
 61             else if (args.domEvent.dataTransfer.items) {
 62                 eventFiles = args.domEvent.dataTransfer.items;
 63             }
 64 
 65             this._chooseFiles(eventFiles);
 66             this.Dispatch('InputFileChosen', {files: this._inputFiles});
 67 
 68             this._inputFiles = [];
 69         });
 70     }
 71 
 72     /**
 73      * Выбрать файлы
 74      * @param {FileList|DataTransferItemList} filesList
 75      * @private
 76      */
 77     _chooseFiles(filesList) {
 78         if (!this._multiple) { this._inputFiles = []; }
 79 
 80         for (let i = 0; i < filesList.length; i++) {
 81             let file = filesList[i];
 82             if (filesList[i].kind && filesList[i].kind === 'file') {
 83                 file = filesList[i].getAsFile();
 84             }
 85 
 86             this._inputFiles.push(file);
 87         }
 88     }
 89 
 90     /**
 91      * Нарисовать drag-and-drop область
 92      * @private
 93      */
 94     _renderDropContainer() {
 95         this._dropContainer = new Colibri.UI.Pane(this._name + '-drop-container', this);
 96         this._titleContainer = new Colibri.UI.Pane(this._name + '-title-container', this._dropContainer);
 97 
 98         this._uploadIcon = new Colibri.UI.Icon(this._name + '-upload-icon', this._titleContainer);
 99         this._uploadIcon.value = Colibri.UI.FileLoadIcon;
100 
101         this._title = new Colibri.UI.TextSpan(this._name + '-title', this._titleContainer);
102         this._setTitle();
103 
104         this._extensionsLabel = new Colibri.UI.TextSpan(this._name + '-extensions', this._titleContainer);
105 
106         this._dropContainer.shown = true;
107         this._titleContainer.shown = true;
108         this._uploadIcon.shown = true;
109         this._title.shown = true;
110         this._extensionsLabel.shown = true;
111     }
112 
113     /**
114      * drag-and-drop контейнер
115      */
116     get dropContainer() {
117         return this._dropContainer;
118     }
119 
120     /**
121      * Лэйбл с допустимыми расширениями
122      */
123     get extensionsLabel() {
124         return this._extensionsLabel;
125     }
126 
127     /**
128      * Отображаемый текст
129      * @return {string}
130      */
131     get title() {
132         return this._title.value;
133     }
134 
135     /**
136      * Отображаемый текст
137      * @param {string} value
138      */
139     set title(value) {
140         this._setTitle(value);
141     }
142 
143     /**
144      * Текст ошибки валидации
145      * @param {string} value
146      */
147     set errorMessage(value) {
148         this._errorMessage = value;
149     }
150 
151     /**
152      * Текст ошибки валидации
153      * @return {string}
154      */
155     get errorMessage() {
156         return this._errorMessage;
157     }
158 
159     /**
160      * Изменить отображаемый текст на новый, либо по умолчанию
161      * @param {string} value
162      * @private
163      */
164     _setTitle(value) {
165         this._title.value = value ?? (this._multiple ? '#{ui-filesdragdrop-choosefiles}' : '#{ui-filesdragdrop-choosefile}');
166     }
167 
168     /**
169      * Возможность загрузить больше одного файла
170      * @return {boolean}
171      */
172     get multiple() {
173         return this._multiple;
174     }
175 
176     /**
177      * Возможность загрузить больше одного файла
178      * @param {boolean} value
179      */
180     set multiple(value) {
181         if (value === true || value === 'true') {
182             this._innerInput.setAttribute('multiple', 'true');
183             this._multiple = true;
184         } else {
185             this._innerInput.removeAttribute('multiple');
186             this._multiple = false;
187         }
188 
189         this._setTitle();
190     }
191 
192     /**
193      * Показать элемент (обновить отображаемый текст и отобразить все вложенные элементы)
194      * @param {boolean|string} value
195      */
196     set shown(value) {
197         super.shown = value === 'true' || value === true;
198 
199         if(super.shown) {
200             if (this._errorMessage) {
201                 this._setTitle(this._errorMessage);
202             }
203             else {
204                 this._setTitle();
205             }
206 
207             this._dropContainer.shown = true;
208             this._titleContainer.shown = true;
209             this._uploadIcon.shown = true;
210             this._title.shown = true;
211             this._extensionsLabel.shown = true;
212         }
213     }
214 }
