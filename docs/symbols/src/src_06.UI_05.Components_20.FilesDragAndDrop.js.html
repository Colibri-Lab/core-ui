1 Colibri.UI.FilesDragAndDrop = class extends Colibri.UI.Component {
  2     constructor(name, container, multiple = false) {
  3         super(name, container, Element.create('div'));
  4         this.AddClass('app-component-files-drag-and-drop');
  5 
  6         // внутренний невидимый инпут для выбора файлов из проводника по клику
  7         this._uniqueString = Number.unique();
  8         this._innerInput = Element.create('input', {type:'file', id: 'component-' + name + '-' + this._uniqueString, class: 'app-files-drag-and-drop'});
  9         this._element.append(this._innerInput);
 10 
 11         this._inputFiles = [];
 12         this._multiple = multiple;
 13         this._errorMessage = '';
 14 
 15         this._renderDropContainer();
 16         this._handleEvents();
 17     }
 18 
 19     _registerEvents() {
 20         super._registerEvents();
 21         this.RegisterEvent('InputFileChosen', false, 'Выбран файл/файлы');
 22     }
 23 
 24     _handleEvents() {
 25         this._innerInput.addEventListener('change', (event) => {
 26             this._dropContainer.Dispatch('Drop', {domEvent: event, inputFiles: this._innerInput.files});
 27             this._innerInput.value = '';
 28         });
 29 
 30         this._dropContainer.AddHandler('Clicked', (event, args) => { this._innerInput.click(); });
 31         this._dropContainer.AddHandler('DragOver', (event, args) => {
 32             args.domEvent.preventDefault();
 33             this.AddClass('-active');
 34         });
 35         this._dropContainer.AddHandler('DragLeave', (event, args) => { this.RemoveClass('-active'); });
 36 
 37         /* выбрать файлы в зависимости от источника, очистить выбор после вызова события  */
 38         this._dropContainer.AddHandler('Drop', (event, args) => {
 39             args.domEvent.preventDefault();
 40             this.RemoveClass('-active');
 41 
 42             let eventFiles;
 43             if (args.inputFiles) {
 44                 eventFiles = args.inputFiles;
 45             }
 46             else if (args.domEvent.dataTransfer.files) {
 47                 eventFiles = args.domEvent.dataTransfer.files;
 48             }
 49             else if (args.domEvent.dataTransfer.items) {
 50                 eventFiles = args.domEvent.dataTransfer.items;
 51             }
 52 
 53             this._chooseFiles(eventFiles);
 54             this.Dispatch('InputFileChosen', {files: this._inputFiles});
 55 
 56             this._inputFiles = [];
 57         });
 58     }
 59 
 60     /**
 61      * Выбрать файлы
 62      * @param {FileList|DataTransferItemList} filesList
 63      * @private
 64      */
 65     _chooseFiles(filesList) {
 66         if (!this._multiple) { this._inputFiles = []; }
 67 
 68         for (let i = 0; i < filesList.length; i++) {
 69             let file = filesList[i];
 70             if (filesList[i].kind && filesList[i].kind === 'file') {
 71                 file = filesList[i].getAsFile();
 72             }
 73 
 74             this._inputFiles.push(file);
 75         }
 76     }
 77 
 78     /**
 79      * Нарисовать drag-and-drop область
 80      * @private
 81      */
 82     _renderDropContainer() {
 83         this._dropContainer = new Colibri.UI.Pane(this._name + '-drop-container', this);
 84         this._titleContainer = new Colibri.UI.Pane(this._name + '-title-container', this._dropContainer);
 85 
 86         this._uploadIcon = new Colibri.UI.Icon(this._name + '-upload-icon', this._titleContainer);
 87         this._uploadIcon.value = Colibri.UI.FileLoadIcon;
 88 
 89         this._title = new Colibri.UI.TextSpan(this._name + '-title', this._titleContainer);
 90         this._setTitle();
 91 
 92         this._extensionsLabel = new Colibri.UI.TextSpan(this._name + '-extensions', this._titleContainer);
 93 
 94         this._dropContainer.shown = true;
 95         this._titleContainer.shown = true;
 96         this._uploadIcon.shown = true;
 97         this._title.shown = true;
 98         this._extensionsLabel.shown = true;
 99     }
100 
101     /**
102      * drag-and-drop контейнер
103      */
104     get dropContainer() {
105         return this._dropContainer;
106     }
107 
108     /**
109      * Лэйбл с допустимыми расширениями
110      */
111     get extensionsLabel() {
112         return this._extensionsLabel;
113     }
114 
115     /**
116      * Отображаемый текст
117      * @return {string}
118      */
119     get title() {
120         return this._title.value;
121     }
122 
123     /**
124      * Отображаемый текст
125      * @param {string} value
126      */
127     set title(value) {
128         this._setTitle(value);
129     }
130 
131     /**
132      * Текст ошибки валидации
133      * @param {string} value
134      */
135     set errorMessage(value) {
136         this._errorMessage = value;
137     }
138 
139     /**
140      * Текст ошибки валидации
141      * @return {string}
142      */
143     get errorMessage() {
144         return this._errorMessage;
145     }
146 
147     /**
148      * Изменить отображаемый текст на новый, либо по умолчанию
149      * @param {string} value
150      * @private
151      */
152     _setTitle(value) {
153         this._title.value = value ?? (this._multiple ? '#{ui-filesdragdrop-choosefiles}' : '#{ui-filesdragdrop-choosefile}');
154     }
155 
156     /**
157      * Возможность загрузить больше одного файла
158      * @return {boolean}
159      */
160     get multiple() {
161         return this._multiple;
162     }
163 
164     /**
165      * Возможность загрузить больше одного файла
166      * @param {boolean} value
167      */
168     set multiple(value) {
169         if (value === true || value === 'true') {
170             this._innerInput.setAttribute('multiple', 'true');
171             this._multiple = true;
172         } else {
173             this._innerInput.removeAttribute('multiple');
174             this._multiple = false;
175         }
176 
177         this._setTitle();
178     }
179 
180     /**
181      * Показать элемент (обновить отображаемый текст и отобразить все вложенные элементы)
182      * @param {boolean|string} value
183      */
184     set shown(value) {
185         super.shown = value === 'true' || value === true;
186 
187         if(super.shown) {
188             if (this._errorMessage) {
189                 this._setTitle(this._errorMessage);
190             }
191             else {
192                 this._setTitle();
193             }
194 
195             this._dropContainer.shown = true;
196             this._titleContainer.shown = true;
197             this._uploadIcon.shown = true;
198             this._title.shown = true;
199             this._extensionsLabel.shown = true;
200         }
201     }
202 }
