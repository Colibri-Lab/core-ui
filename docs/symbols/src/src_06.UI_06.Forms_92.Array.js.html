  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-array-field');
 14         
 15         this._enabled = true;
 16 
 17         const contentContainer = this.contentContainer;
 18         this._itemsContainer = new Colibri.UI.Pane(this.name + '-items', contentContainer);
 19         this._itemsContainer.shown = true;
 20 
 21         if(this._fieldData?.params?.initempty === undefined || this._fieldData?.params?.initempty === true) {
 22             this._addNew();
 23         }
 24 
 25         this._createAddNewLink();
 26         
 27 
 28         if(this._fieldData?.params?.readonly === undefined) {
 29             this.readonly = false;    
 30         }
 31         else {
 32             this.readonly = this._fieldData?.params?.readonly;
 33         }
 34         if(this._fieldData?.params?.enabled === undefined) {
 35             this.enabled = true;
 36         }
 37         else {
 38             this.enabled = this._fieldData.params.enabled;
 39         }
 40 
 41         this.RegisterEvent('ObjectRemoved', false, 'Object in array removed');
 42 
 43     }

 45     /** @private */
 46     _createAddNewLink() {
 47         if(this._fieldData.params && this._fieldData.params.addlink === null) {
 48             return;
 49         }
 50 
 51         this.contentContainer.Children('add-new') && this.contentContainer.Children('add-new').Dispose();
 52         if(typeof Lang.Translate(this._fieldData.params.addlink) === 'object') {
 53             const linkData = this._fieldData.params.addlink;
 54             const c = eval(linkData.component);
 55             this._link = new c('add-new', this.contentContainer);
 56             Object.forEach(linkData.attrs, (name, value) => {
 57                 this._link[name] = value;
 58             });
 59 
 60         } else {
 61             this._fieldData.desc = this._fieldData.desc ? this._fieldData.desc[Lang.Current] ?? this._fieldData.desc : ''; 
 62             this._fieldData.params.addlink = this._fieldData?.params?.addlink ? this._fieldData.params.addlink[Lang.Current] ?? this._fieldData.params.addlink : '';
 63             this._link = new Colibri.UI.Link('add-new', this.contentContainer);
 64             this._link.value = this._fieldData.params && this._fieldData.params.addlink || '#{ui-array-add} «' + (this._fieldData.desc) + '»';    
 65         }
 66         
 67 
 68         this._link.shown = true;
 69         this._link.AddHandler('Clicked', (event, args) => {
 70             if(this.readonly || !this.enabled) {
 71                 return;
 72             }
 73             this.AddNew();
 74         });
 75         if(this.readonly || !this.enabled) {
 76             this._link.Hide();
 77         }
 78         return this._link;
 79     }

 86     /**
 87      * Adds new object to array
 88      * @param {Object} value add new row
 89      * @returns {Colibri.UI.Forms.Object}
 90      */
 91     AddNew(value) {
 92         if(this._link.ContainsClass('ui-disabled')) {
 93             return;
 94         }
 95         const object = this._addNew(value);
 96         this.Dispatch('Changed', {component: this});
 97         return object;
 98     }

100     /**
101      * Adds new object
102      * @private
103      * @param {object} value object for add
104      * @returns {Colibri.UI.Forms.Object}
105      */
106     _addNew(value = null) {
107         // const containerElement = this.contentContainer.container.querySelector('.array-component-container');
108         let fieldData = Object.cloneRecursive(this._fieldData);
109         delete fieldData?.note;
110         delete fieldData?.params?.validate;
111         delete fieldData?.params?.fieldgenerator;
112         if(fieldData?.params){
113             fieldData.params.validate = [{message: '', method: () => true}];
114         } 
115         fieldData = this.__updateObjectFields(fieldData);
116         const object = new Colibri.UI.Forms.Object('object-' + Date.Now().getTime(), this._itemsContainer, fieldData, this, this.root);
117         object.shown = true;
118         object.title = '';
119         object.enabled = this.enabled;
120         if(this._fieldData.params && this._fieldData.params.removelink !== false && !this.readonly && this.enabled) {
121             object.AddRemoveLink(() => {
122 
123                 if(this.readonly || !this.enabled) {
124                     return;
125                 }
126 
127                 Object.forEach(this.Fields(), (name, field) => {
128                     if(field instanceof Colibri.UI.Forms.Field) {
129                         field.Dispatch('Changed', {});
130                     }
131                 });
132                 this.Dispatch('Changed', {component: this});
133                 
134                 if(this._fieldData.params && !!this._fieldData.params.maxadd) {
135                     const count = Object.countKeys(this.Fields());
136                     if(count < parseInt(this._fieldData.params.maxadd)) {
137                         this._link.Show();
138                     }
139                 }
140 
141                 if(this._fieldData.params && this._fieldData.params?.showObjectCount) {
142                     if(typeof this._fieldData.params?.showObjectCount === 'function') {
143                         const f = this._fieldData.params?.showObjectCount;
144                         f(this.itemsContainer.children, this);    
145                     } else {
146                         this.title = this._fieldData.desc + ' (' + this.itemsContainer.children + ')';
147                     }
148                 }
149 
150                 this.Dispatch('ObjectRemoved', {component: this});
151 
152                 this._hideAndShow();
153 
154             });
155         }
156         if(this._fieldData.params && this._fieldData.params.updownlink !== false && !this.readonly && this.enabled) {
157             object.AddUpDownLink(() => {
158                 object.MoveUp();
159                 object.Focus();
160                 this.Dispatch('Changed', {component: this});
161             }, () => {
162                 object.MoveDown();
163                 object.Focus();
164                 this.Dispatch('Changed', {component: this});
165             });
166         }
167 
168         object.AddHandler('Changed', (event, args) => {
169             if(this._fieldData.params && this._fieldData.params.title !== null) {
170                 const f = eval(this._fieldData.params.title);
171                 f && f(object, this);
172             }
173             return this.Dispatch('Changed', Object.assign(args ?? {}, {component: this}));
174         });
175         this._itemsContainer.Children(object.name, object);
176         if(this._fieldData.params && this._fieldData.params.title !== null) {
177             const f = eval(this._fieldData.params.title);
178             f && f(object, this);
179         }
180 
181         if(this._fieldData.params && !!this._fieldData.params.maxadd) {
182             const count = Object.countKeys(this.Fields());
183             if(count >= parseInt(this._fieldData.params.maxadd)) {
184                 this._link && this._link.Hide();
185             }
186         }
187 
188         this.Dispatch('FieldsRendered');
189         if(value) {
190             object.value = value;
191         }
192 
193         if(this._fieldData.params && this._fieldData.params?.showObjectCount) {
194             if(typeof this._fieldData.params?.showObjectCount === 'function') {
195                 const f = this._fieldData.params?.showObjectCount;
196                 f(this.itemsContainer.children, this);    
197             } else {
198                 this.title = this._fieldData.desc + ' (' + this.itemsContainer.children + ')';
199             }
200         }
201 
202         // object.Focus();
203         return object;
204     }

268      * Value
269      * @type {Array}
270      */
271     get value() {
272 
273         let data = [];
274         this._itemsContainer.ForEach((name, component) => {
275             if(component instanceof Colibri.UI.Forms.Object) {
276                 data.push(component.value);

278         });
279

284     /**
285      * Value
286      * @type {Array}
287      */
288     set value(value) {
289         
290         value = eval_default_values(value);
291         if(!value || !Array.isArray(value)) {
292             return;
293         }
294 
295         this._itemsContainer.Clear();
296         if(value.length > 0) {
297             for(const v of value) {
298                 this._addNew(v);
299             }
300         } else if(this._fieldData?.params?.initempty) {
301             this._addNew();            
302         }
303 
304 
305     }

307     /**
308      * Searches for objects in array
309      * @param {string} name field to search for
310      * @returns {Array<Colibri.UI.Forms.Object>}
311      */
312     Fields(name) {
313         if(!this._itemsContainer) {
314             return [];
315         }
316 
317         if(name) {
318             return this._itemsContainer.Children(name);
319         }
320 
321         let ret = {};
322         this._itemsContainer.ForEach((name, component) => {
323             if(component instanceof Colibri.UI.Forms.Field) {
324                 ret[name] = component;
325             }
326         });
327 
328         return ret;
329     }
