1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.MassActionsMenu = class extends Colibri.UI.Component {
  7     
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */    
 13     constructor(name, container) {
 14         super(name, container, Element.create('div'));
 15         this.AddClass('app-mass-actions-menu-component');
 16 
 17         this._actions = [];
 18         this._selectedItems = [];
 19 
 20         this._renderMenuPanel();
 21     }
 22 
 23     /**
 24      * Регистрация событий
 25      * @protected
 26      */
 27     _registerEvents() {
 28         super._registerEvents();
 29         this.RegisterEvent('ActionClicked', false, 'Когда кликнули по кнопке в меню');
 30     }
 31 
 32     /**
 33      * Нарисовать меню с контейнером для кнопок
 34      * @private
 35      */
 36     _renderMenuPanel() {
 37         this._actionsContainer = new Colibri.UI.Pane(this.name + '-actions-container', this);
 38         this._selectedItemsCounter = new Colibri.UI.Pane(this.name + '-selected-counter', this);
 39         this._selectedItemsCounter.value = 'Выбрано ' + this._selectedItems.length;
 40 
 41         this._actionsContainer.shown = true;
 42         this._selectedItemsCounter.shown = true;
 43     }
 44 
 45     /**
 46      * Нарисовать кнопки
 47      * @private
 48      */
 49     _renderActions() {
 50         this._actionsContainer.Clear();
 51         this._actions.forEach((action) => {
 52             let actionButton = new Colibri.UI.GrayButton(action.name, this._actionsContainer);
 53             actionButton.AddClass('app-mass-actions-menu-action-component');
 54             actionButton.value = action.title;
 55             if(action.icon) {
 56                 actionButton.icon = action.icon;
 57             }
 58             if(action.contextmenu && action.contextmenu.length > 0) {
 59                 actionButton.contextmenu = action.contextmenu;
 60             }
 61             actionButton.tag = action;
 62             actionButton.AddHandler('ContextMenuItemClicked', (event, args) => {
 63                 this.Dispatch('ActionClicked', args);
 64             });
 65             actionButton.AddHandler('Clicked', (event, args) => {
 66                 if(event.sender.contextmenu.length > 0) {
 67                     event.sender.ShowContextMenu([Colibri.UI.ContextMenu.LT, Colibri.UI.ContextMenu.RT], '', null);
 68                 } else {
 69                     this.Dispatch('ActionClicked', Object.assign({menu: event.sender, menuData: event.sender.tag}, args));
 70                 }
 71             });
 72             actionButton.shown = true
 73         });
 74     }
 75 
 76     /**
 77      * Список экшенов (кнопок) в меню
 78      * @type {array}
 79      */
 80     get actions() {
 81         return this._actions;
 82     }
 83     set actions(value) {
 84         this._actions = value;
 85         this._renderActions();
 86     }
 87 
 88     /**
 89      * Список выбранных объектов
 90      * @type {array}
 91      */
 92     get selectedItems() {
 93         return this._selectedItems;
 94     }
 95     set selectedItems(value) {
 96         this._selectedItems = value;
 97         this._selectedItemsCounter.value = '#{ui-massactions-choosed}'.replaceAll('%s', this._selectedItems.length);
 98         if(this._selectedItems.length == 0) {
 99             this.Hide();            
100         } else {
101             this.Show();
102         }
103     }
104 
105     Dispose() {
106         this.shown = false;
107         Colibri.Common.Delay(300).then(() => super.Dispose());
108     }
109 
110 }
