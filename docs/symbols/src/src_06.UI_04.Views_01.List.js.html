 82      * Search for item
 83      * @param {Function|string} searchingItemIdOrCompareMethod item id or compare method
 84      * @returns {Colibri.UI.List.Item}
 85      */
 86     FindItem(searchingItemIdOrCompareMethod) {
 87         
 88         let found = null;
 89         this.ForEach((name, group) => {
 90             group.ForEach((n, item) => {
 91 
 92                 let condition = false;
 93                 if(searchingItemIdOrCompareMethod instanceof Function) {
 94                     condition = searchingItemIdOrCompareMethod(item);
 95                 }
 96                 else {
 97                    condition = searchingItemIdOrCompareMethod == n; 
 98                 }
 99 
100                 if(condition) {
101                     found = item;
102                     return null;
103                 }
104 
105             });
106             if(found) {
107                 return null;
108             }
109         });
110 
111         return found;
112     }

114     /**
115      * Unselect items
116      * @param {Colibri.UI.List.Item[]} selected items to unselect
117      */
118     UnselectItem(selected) {
119         if(!this._multiple) {
120             this.ClearSelection(false);
121         }
122         else if(Array.isArray(selected)) {
123             selected.forEach((sel) => {
124                 this.UnselectItem(sel);
125             });
126         }
127         else {
128             const index = this._selected.findIndex(i => i.name == selected.name);
129             if(index >= 0) {
130                 // уже выбран, надо снять выбор
131                 this._selected.splice(index, 1);
132                 selected.selected = false;
133             }
134         }

137     /**
138      * Select the item
139      * @param {Colibri.UI.List.Item} selected
140      */
141     SelectItem(selected) {
142         if(!this._canSelect || !selected) {
143             return;
144         }
145         if(!this._multiple) {
146             this.ClearSelection(false);
147             selected = Array.isArray(selected) && selected.length ? selected.shift() : selected;
148             selected.selected = true;
149             this._selected.push(selected);
150         }
151         else if(Array.isArray(selected)) {
152             // если multiple то может быть массив
153             selected.forEach((sel) => {
154                 this.SelectItem(sel);
155             });
156         }
157         else {
158             const index = this._selected.findIndex(i => i.name == selected.name);
159             if(index >= 0) {
160                 // уже выбран, надо снять выбор
161                 this._selected.splice(index, 1);
162                 selected.selected = false;
163             }
164             else {
165                 selected.selected = true;
166                 this._selected.push(selected);
167             }
168         }
169     }

223         }

714     AddItem(itemData, id = null, selected = false, index = null) {
715 
716         const newKey = Colibri.UI.List.Group.CreateKey(itemData); 
717         const foundItem = this.FindByKey(newKey);
718 
719         let control;
720         if(foundItem !== -1) {
721             control = this._div.Children(foundItem);
722             control.value = itemData;
723             if(index) {
724                 this._div.Children(control.name, control, index);
725             }
726         } else {
727             const name = (id || itemData?.id || '_' + Number.unique());
728             control = new Colibri.UI.List.Item('item-' + name, this._div);
729             control.shown = true;
730             control.selected = selected;
731             control.hasContextMenu = this.hasContextMenu;
732             control.key = newKey;
733             control.value = itemData;
734 
735             if(this.parent?.tag && this.parent?.tag?.params && this.parent?.tag?.params?.sort) {
736                 const foundIndex = this.parent.tag.params.sort(control, this);
737                 this._div.Children(name, control, foundIndex);
738             }
739 
740         }
741 
742         if(selected) {
743             this.parent.SelectItem(control);
744         }
745 
746         return control;
747

839     set value(value) {
840 
841         this.parent.ClearSelection(false);
842         if(!(Symbol.iterator in Object(value))) {
843             return;
844         }
845 
846         this.KeepInMind();
847 
848         const oldKeys = [];
849         const oldValues = this.value;
850         for(const item of oldValues) {
851             const key = Colibri.UI.List.Group.CreateKey(item);
852             oldKeys.push(key);
853         }
854 
855         const newKeys = [];
856         let index = 0;
857         for(const item of value) {
858             newKeys.push(Colibri.UI.List.Group.CreateKey(item));
859             this.AddItem(item, null, item?.__selected, index++);
860         }
861 
862         for(const key of oldKeys) {
863             if(newKeys.indexOf(key) === -1) {
864                 const foundIndex = this.FindByKey(key);
865                 if(foundIndex !== -1) {
866                     const item = this._div.Children(foundIndex);
867                     item.Dispose();
868                 }
869             }
870         }
