1 /**
  2  * @class
  3  * @extends Colibri.UI.Forms.Field
  4  * @memberof Colibri.UI.Forms
  5  */
  6 Colibri.UI.Forms.CheckboxList = class extends Colibri.UI.Forms.Field {
  7 
  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-checkboxlist-field');
 14 
 15 
 16         if(this._fieldData?.params?.readonly === undefined) {
 17             this.readonly = false;    
 18         }
 19         else {
 20             this.readonly = this._fieldData?.params?.readonly;
 21         }
 22         if(this._fieldData?.params?.enabled === undefined) {
 23             this.enabled = true;
 24         }
 25         else {
 26             this.enabled = this._fieldData.params.enabled;
 27         }
 28         this.value = this._fieldData?.default ?? false;
 29 
 30         this.values = this._fieldData?.values ?? [];
 31 
 32     }
 33 
 34     /** @protected */
 35     _handleEvents(input) {
 36         input.AddHandler('Changed', (event, args) => this.Dispatch('Changed', Object.assign(args || {}, {component: this})));
 37         input.AddHandler('Clicked', (event, args) => {
 38             this.Dispatch('Clicked', args)
 39             args.domEvent.stopPropagation();
 40             args.domEvent.preventDefault();
 41             return false;
 42         });
 43         input.AddHandler('ReceiveFocus', (event, args) => this.Dispatch('ReceiveFocus', args));
 44         input.AddHandler('LoosedFocus', (event, args) => this.Dispatch('LoosedFocus', args));
 45     }
 46 
 47     /**
 48      * Focus on component
 49      */
 50     Focus() {
 51         this.contentContainer.Children('firstChild')?.Focus();
 52     }
 53 
 54     /**
 55      * Readonly
 56      * @type {boolean}
 57      */
 58     get readonly() {
 59         return this.contentContainer.Children('firstChild')?.readonly ?? false;
 60     }
 61 
 62     /**
 63      * Readonly
 64      * @type {boolean}
 65      */
 66     set readonly(value) {
 67         value = this._convertProperty('Boolean', value);
 68         this.contentContainer.ForEach((name, component) => {
 69             component.readonly = value;
 70         });
 71     }
 72 
 73     /**
 74      * Value
 75      * @type {Array}
 76      */
 77     get value() {
 78         const contentContainer = this.contentContainer;
 79         const ret = [];
 80         contentContainer.ForEach((name, component) => {
 81             if(component.checked) {
 82                 ret.push(component.tag);
 83             }
 84         });
 85         return ret;
 86     }
 87 
 88     /**
 89      * Value
 90      * @type {Array}
 91      */
 92     set value(value) {
 93 
 94         if(!Array.isArray(value)) {
 95             return;
 96         }
 97 
 98         const contentContainer = this.contentContainer;
 99         for(const v of value) {
100             contentContainer.Children(this._name + '-input-' + String.MD5(v.value + v.title)).checked = true;
101         }
102 
103     }
104 
105     /**
106      * Enable/Disable
107      * @type {boolean}
108      */
109     get enabled() {
110         return this.contentContainer.Children('firstChild')?.enabled ?? true;
111     }
112 
113     /**
114      * Enable/Disable
115      * @type {boolean}
116      */
117     set enabled(value) {
118         value = this._convertProperty('Boolean', value);
119         this.contentContainer.ForEach((name, component) => {
120             component.enabled = value;
121         });
122     }
123 
124     /**
125      * Tab index
126      * @type {number}
127      */
128     get tabIndex() {
129         return this.contentContainer.Children('firstChild').tabIndex;
130     }
131     /**
132      * Tab index
133      * @type {number}
134      */
135     set tabIndex(value) {
136         this.contentContainer.ForEach((name, component) => {
137             component.tabIndex = (value++);
138         });
139     }
140 
141     /**
142      * Array of values
143      * @type {Array}
144      */
145     get values() {
146         return this._values;
147     }
148     /**
149      * Array of values
150      * @type {Array}
151      */
152     set values(value) {
153         this._values = value;
154         this._showValues();
155     }
156     /** @private */
157     _showValues() {
158         const contentContainer = this.contentContainer;
159         contentContainer.Clear();
160         for(const v of this._values) {
161 
162             const input = new Colibri.UI.Checkbox(this._name + '-input-' + String.MD5(v.value + v.title), contentContainer);
163             input.shown = true;
164             input.placeholder = v.title;
165             input.tag = v;
166             input.tabIndex = true;
167             this._handleEvents(input);
168 
169         }
170 
171 
172     }
173 
174 }
