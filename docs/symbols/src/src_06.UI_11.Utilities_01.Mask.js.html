  8     static DIGIT = "9";
  9     static ALPHA = "A";
 10     static ALPHANUM = "S";
 11     static BY_PASS_KEYS = [9, 16, 17, 18, 36, 37, 38, 39, 40, 91, 92, 93];

 13 
 14     constructor(elements) {
 15         super();
 16         this.elements = elements;
 17     }
 18 
 19     _isAllowedKeyCode(keyCode) {
 20         for (var i = 0, len = Colibri.UI.Utilities.Mask.BY_PASS_KEYS.length; i < len; i++) {
 21             if (keyCode == Colibri.UI.Utilities.Mask.BY_PASS_KEYS[i]) {
 22                 return false;
 23             }
 24         }
 25         return true;
 26     }
 27 
 28     _mergeMoneyOptions(opts) {
 29         opts = opts || {};
 30         opts = {
 31             delimiter: opts.delimiter || ".",
 32             lastOutput: opts.lastOutput,
 33             precision: opts.hasOwnProperty("precision") ? opts.precision : 2,
 34             separator: opts.separator || ",",
 35             showSignal: opts.showSignal,
 36             suffixUnit: opts.suffixUnit && (" " + opts.suffixUnit.replace(/[\s]/g, '')) || "",
 37             unit: opts.unit && (opts.unit.replace(/[\s]/g, '') + " ") || "",
 38             zeroCents: opts.zeroCents
 39         };
 40         opts.moneyPrecision = opts.zeroCents ? 0 : opts.precision;
 41         return opts;
 42     }
 43 
 44     _unbindElementToMask() {
 45         for (var i = 0, len = this.elements.length; i < len; i++) {
 46             this.elements[i].lastOutput = "";
 47             this.elements[i].onkeyup = false;
 48             this.elements[i].onkeydown = false;
 49 
 50             if (this.elements[i].value.length) {
 51                 this.elements[i].value = this.elements[i].value.replace(/\D/g, '');
 52             }
 53         }
 54     }
 55 
 56     _bindElementToMask(maskFunction) {
 57         const onType = (e) => {
 58             e = e || window.event;
 59             var source = e.target || e.srcElement;
 60 
 61             if (this._isAllowedKeyCode(e.keyCode)) {
 62                 setTimeout(() => {
 63                     this.opts.lastOutput = source.lastOutput;
 64                     source.value = Colibri.UI.Utilities.Mask[maskFunction](source.value, this.opts);
 65                     source.lastOutput = source.value;
 66                     if (source.setSelectionRange && this.opts.suffixUnit) {
 67                         source.setSelectionRange(source.value.length, (source.value.length - this.opts.suffixUnit.length));
 68                     }
 69                 }, 0);
 70             }
 71         };
 72         for (var i = 0, len = this.elements.length; i < len; i++) {
 73             this.elements[i].lastOutput = "";
 74             this.elements[i].onkeyup = onType;
 75             if (this.elements[i].value.length) {
 76                 this.elements[i].value = Colibri.UI.Utilities.Mask[maskFunction](this.elements[i].value, this.opts);
 77             }
 78         }
 79     }
 80 
 81     maskMoney(opts) {
 82         this.opts = this._mergeMoneyOptions(opts);
 83         this._bindElementToMask("toMoney");
 84     }
 85 
 86     maskNumber() {
 87         this.opts = {};
 88         this._bindElementToMask("toNumber");
 89     }
 90 
 91     maskAlphaNum() {
 92         this.opts = {};
 93         this._bindElementToMask("toAlphaNumeric");
 94     }
 95 
 96     maskPattern(pattern) {
 97         this.opts = { pattern: pattern };
 98         this._bindElementToMask("toPattern");
 99     }
100 
101     unMask() {
102         this._unbindElementToMask();
103     }

105     // Fill wildcards past index in output with placeholder
106     static _addPlaceholdersToOutput(output, index, placeholder) {
107         for (; index < output.length; index++) {
108             if (output[index] === Colibri.UI.Utilities.Mask.DIGIT || output[index] === Colibri.UI.Utilities.Mask.ALPHA || output[index] === Colibri.UI.Utilities.Mask.ALPHANUM) {
109                 output[index] = placeholder;
110             }
111         }
112         return output;
113     }

116     static toMoney(value, opts) {
117         opts = this._mergeMoneyOptions(opts);
118         if (opts.zeroCents) {
119             opts.lastOutput = opts.lastOutput || "";
120             var zeroMatcher = ("(" + opts.separator + "[0]{0," + opts.precision + "})"),
121                 zeroRegExp = new RegExp(zeroMatcher, "g"),
122                 digitsLength = value.toString().replace(/[\D]/g, "").length || 0,
123                 lastDigitLength = opts.lastOutput.toString().replace(/[\D]/g, "").length || 0
124                 ;
125             value = value.toString().replace(zeroRegExp, "");
126             if (digitsLength < lastDigitLength) {
127                 value = value.slice(0, value.length - 1);
128             }
129         }
130 
131         var number = value.toString();
132         // if separator is in string, make sure we zero-pad to respect it
133         var separatorIndex = number.indexOf(opts.separator),
134             missingZeros = (opts.precision - (number.length - separatorIndex - 1));
135 
136         if (separatorIndex !== -1 && (missingZeros > 0)) {
137             number = number + ('0' * missingZeros);
138         }
139 
140         number = number.replace(/[\D]/g, "");
141 
142         var clearDelimiter = new RegExp("^(0|\\" + opts.delimiter + ")"),
143             clearSeparator = new RegExp("(\\" + opts.separator + ")$"),
144             money = number.substr(0, number.length - opts.moneyPrecision),
145             masked = money.substr(0, money.length % 3),
146             cents = new Array(opts.precision + 1).join("0")
147             ;
148 
149         money = money.substr(money.length % 3, money.length);
150         for (var i = 0, len = money.length; i < len; i++) {
151             if (i % 3 === 0) {
152                 masked += opts.delimiter;
153             }
154             masked += money[i];
155         }
156         masked = masked.replace(clearDelimiter, "");
157         masked = masked.length ? masked : "0";
158         var signal = "";
159         if (opts.showSignal === true) {
160             signal = value < 0 || (value.startsWith && value.startsWith('-')) ? "-" : "";
161         }
162         if (!opts.zeroCents) {
163             var beginCents = Math.max(0, number.length - opts.precision),
164                 centsValue = number.substr(beginCents, opts.precision),
165                 centsLength = centsValue.length,
166                 centsSliced = (opts.precision > centsLength) ? opts.precision : centsLength
167                 ;
168             cents = (cents + centsValue).slice(-centsSliced);
169         }
170         var output = opts.unit + signal + masked + opts.separator + cents;
171         return output.replace(clearSeparator, "") + opts.suffixUnit;
172     };

174     static toPattern(value, opts) {
175 
176         var pattern = (typeof opts === 'object' ? opts.pattern : opts),
177             patternChars = pattern.replace(/\W/g, ''),
178             output = pattern.split(""),
179             values = value.toString().replace(/\W/g, ""),
180             charsValues = values.replace(/\W/g, ''),
181             index = 0,
182             i,
183             outputLength = output.length,
184             placeholder = (typeof opts === 'object' ? opts.placeholder : undefined)
185             ;
186 
187 
188         for (i = 0; i < outputLength; i++) {
189             // Reached the end of input
190             if (index >= values.length) {
191                 if (patternChars.length == charsValues.length) {
192                     return output.join("");
193                 }
194                 else if ((placeholder !== undefined) && (patternChars.length > charsValues.length)) {
195                     return Colibri.UI.Utilities.Mask._addPlaceholdersToOutput(output, i, placeholder).join("");
196                 }
197                 else {
198                     break;
199                 }
200             }
201             // Remaining chars in input
202             else {
203                 if ((output[i] === Colibri.UI.Utilities.Mask.DIGIT && values[index].match(/[0-9]/)) ||
204                     (output[i] === Colibri.UI.Utilities.Mask.ALPHA && values[index].match(/[a-zA-Z]/)) ||
205                     (output[i] === Colibri.UI.Utilities.Mask.ALPHANUM && values[index].match(/[0-9a-zA-Z]/))) {
206                     output[i] = values[index++];
207                 } else if (output[i] === Colibri.UI.Utilities.Mask.DIGIT || output[i] === Colibri.UI.Utilities.Mask.ALPHA || output[i] === Colibri.UI.Utilities.Mask.ALPHANUM) {
208                     if (placeholder !== undefined) {
209                         return Colibri.UI.Utilities.Mask._addPlaceholdersToOutput(output, i, placeholder).join("");
210                     }
211                     else {
212                         return output.slice(0, i).join("");
213                     }
214                     // exact match for a non-magic character
215                 } else if (output[i] === values[index]) {
216                     index++;
217                 }
218 
219             }
220         }
221         return output.join("").substr(0, i);
222     };

224     static toNumber(value) {
225         return value.toString().replace(/(?!^-)[^0-9]/g, "");
226     }
227 
228     static toAlphaNumeric(value) {
229         return value.toString().replace(/[^a-z0-9 ]+/i, "");
230     }
231 
232     static Check(value, pattern) {
233         const v = Colibri.UI.Utilities.Mask.toPattern(value, {pattern: pattern, placeholder: "_"});
234         return v === value;
235     }
