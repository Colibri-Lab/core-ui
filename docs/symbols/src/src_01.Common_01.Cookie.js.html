1 /**
  2  * Represents a cookie utility for managing cookies in the browser.
  3  * @class
  4  */
  5 Colibri.Common.Cookie = class {
  6     
  7     /**
  8      * Sets a cookie.
  9      * @param {string} c_name - The name of the cookie.
 10      * @param {*} value - The value to set for the cookie.
 11      * @param {number} exdays - The number of days until the cookie expires.
 12      * @param {string} [path] - The path for which the cookie is valid.
 13      * @param {string} [domain] - The domain for which the cookie is valid.
 14      * @param {boolean} [secure] - Whether the cookie should only be sent over secure connections.
 15      * @param {string} [samesite] - The SameSite attribute for the cookie.
 16      */
 17     static Set(c_name, value, exdays, path, domain, secure, samesite) {
 18         var exdate = new Date();
 19         exdate.setDate(exdate.getDate() + exdays);
 20         var c_value = encodeURIComponent(value) + 
 21             ((exdays==null) ? "" : "; expires=" + exdate.toUTCString()) + 
 22             (path == null ? '; path=/' : '; path=' + path) + 
 23             (domain ? '; domain=' + domain : '') + 
 24             (secure !== undefined && secure ? '; secure' : '') + 
 25             (samesite !== undefined ? '; SameSite=' + samesite : '');
 26         document.cookie = c_name + "=" + c_value;
 27     }
 28 
 29     /**
 30      * Gets the value of a cookie.
 31      * @param {string} c_name - The name of the cookie to retrieve.
 32      * @returns {*} - The value of the cookie.
 33      */
 34     static Get (c_name) {
 35         var i,x,y,ARRcookies=document.cookie.split(";");
 36         for( i=0; i < ARRcookies.length; i++) {
 37             x = ARRcookies[i].substring(0, ARRcookies[i].indexOf("="));
 38             y = ARRcookies[i].substring(ARRcookies[i].indexOf("=") + 1);
 39             x = x.replace(/^\s+|\s+$/g,"");
 40             if(x == c_name)
 41                 return decodeURIComponent(y);
 42         }
 43         return null;
 44     }
 45 
 46     /**
 47      * Deletes a cookie.
 48      * @param {string} c_name - The name of the cookie to delete.
 49      * @param {string} [path] - The path of the cookie to delete.
 50      * @param {string} [domain] - The domain of the cookie to delete.
 51      */
 52     static Delete(c_name, path, domain) {
 53         document.cookie = c_name + '=;max-age=0;' + (path == null ? '; path=/' : '; path=' + path) + (domain ? '; domain=' + domain : '');
 54     }
 55     
 56 }
