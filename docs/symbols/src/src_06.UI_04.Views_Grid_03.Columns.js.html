1 /**
  2  * Класс списка колонок
  3  */
  4 Colibri.UI.Grid.Columns = class extends Colibri.UI.Component {
  5 
  6     constructor(name, container) {
  7         super(name, container, Element.create('tr'));
  8         this.AddClass('app-ui-header-columns');
  9         this.AddHandler('ChildAdded', (event, args) => this.__thisChildAdded(event, args));
 10     }
 11 
 12     __thisChildAdded(event, args) {
 13         args.component.AddHandler('ColumnStickyChange', (event, args) => {
 14             this.Dispatch('ColumnStickyChange', args);
 15         });
 16 
 17         args.component.AddHandler('ColumnClicked', (event, args) => {
 18             this.Dispatch('ColumnClicked', args);
 19         });
 20 
 21         args.component.AddHandler('ColumnDisposed', (event, args) => {
 22             this.Dispatch('ColumnDisposed', args);
 23         });
 24     }
 25 
 26     _registerEvents() {
 27         super._registerEvents();
 28         this.RegisterEvent('ColumnAdded', false, 'Поднимается, когда добавляется колонка');
 29         this.RegisterEvent('ColumnMoved', false, 'Поднимается, когда колонка передвинута');
 30         this.RegisterEvent('ColumnStickyChange', false, 'Поднимается, когда колонка меняет липкость');
 31         this.RegisterEvent('ColumnClicked', false, 'Поднимается, когда щелкнули по колонке в заголовке');
 32         this.RegisterEvent('ColumnDisposed', false, 'Поднимается, когда удалили колонку');
 33     }
 34 
 35     Add(name, title, attrs = {}) {
 36         let newColumn = new Colibri.UI.Grid.Column(name, this);
 37         newColumn.value = title;
 38         newColumn.shown = true;
 39 
 40         Object.forEach(attrs, (name, attr) => {
 41             newColumn[name] = attr;
 42         });
 43 
 44         newColumn.AddHandler('ComponentMoved', (event, args) => this.Dispatch('ColumnMoved', Object.assign(args, {column: newColumn})));
 45         this.Dispatch('ColumnAdded', {column: newColumn});
 46         
 47         return newColumn;
 48     }
 49 
 50     Remove(name) {
 51         const col = this.Children(name);
 52         if(col) {
 53             col.Dispose();
 54         }
 55     }
 56 
 57     Column(name) {
 58         if(name === 'firstChild') {
 59             return this.Children(1);
 60         } 
 61         return this.Children(name);
 62     }
 63 
 64     get grid() {
 65         return this.parent.grid;
 66     }
 67 
 68     get count() {
 69         return this.children - 1;
 70     }
 71 }
