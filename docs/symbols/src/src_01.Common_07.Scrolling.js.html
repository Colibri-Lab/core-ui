1 /**
  2  * Class representing scrolling utility functions.
  3  * @class
  4  */
  5 Colibri.Common.Scrolling = class {
  6 
  7     /**
  8      * Creates an instance of Colibri.Common.Scrolling.
  9      * @param {HTMLElement} element - The HTML element to apply scrolling behavior.
 10      */
 11     constructor(element) {
 12         
 13         this._element = element;
 14 
 15         this.supportsPassive = false;
 16         this.wheelOpt = this.supportsPassive ? { passive: false } : false;
 17         this.wheelEvent = 'onwheel' in document.createElement('div') ? 'wheel' : 'mousewheel';
 18     
 19     }
 20 
 21     /**
 22      * Creates an instance of Colibri.Common.Scrolling.
 23      * @param {HTMLElement} element - The HTML element to apply scrolling behavior.
 24      * @returns {Colibri.Common.Scrolling} A new instance of Colibri.Common.Scrolling.
 25      * @static
 26      */
 27     Create(element) {
 28         return new Colibri.Common.Scrolling(element);
 29     }
 30 
 31     /**
 32      * Prevents default behavior for the event.
 33      * @param {Event} e - The event object.
 34      * @private
 35      */
 36     __preventDefault(e) {
 37         try { e.preventDefault(); } catch(e) {}
 38     }
 39 
 40     /**
 41      * Prevents default behavior for scroll keys.
 42      * @param {KeyboardEvent} e - The keyboard event object.
 43      * @returns {boolean} True if default behavior is prevented, otherwise false.
 44      * @private
 45      */
 46     __preventDefaultForScrollKeys(e) {
 47         const keys = [37,38,39,40];
 48         if (keys.indexOf(e.keyCode) !== -1) {
 49             this.__preventDefault(e);
 50             return false;
 51         }
 52         return true;
 53     }
 54     
 55     /**
 56      * Disables scrolling behavior on the element.
 57      */
 58     Disable() {
 59         this._element.addEventListener('DOMMouseScroll', this.__preventDefault, false); // older FF
 60         this._element.addEventListener(this.wheelEvent, this.__preventDefault, this.wheelOpt); // modern desktop
 61         this._element.addEventListener('touchmove', this.__preventDefault, this.wheelOpt); // mobile
 62         this._element.addEventListener('keydown', this.__preventDefaultForScrollKeys, false);
 63     }
 64 
 65     /**
 66      * Enables scrolling behavior on the element.
 67      */
 68     Enable() {
 69         this._element.removeEventListener('DOMMouseScroll', this.__preventDefault, false);
 70         this._element.removeEventListener(this.wheelEvent, this.__preventDefault, this.wheelOpt); 
 71         this._element.removeEventListener('touchmove', this.__preventDefault, this.wheelOpt);
 72         this._element.removeEventListener('keydown', this.__preventDefaultForScrollKeys, false);
 73     }
 74 
 75 }
