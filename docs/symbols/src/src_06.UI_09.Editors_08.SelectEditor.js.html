 89     _showField() {
 90         this._input = this._createSelector();
 91         this._input.shown = true;
 92         this._input.AddHandler('Changed', (event, args) => this.Dispatch('Changed', args));
 93         this._input.AddHandler('ReceiveFocus', (event, args) => {
 94             this.parent.parent.AddClass('-focused');
 95         });
 96         this._input.AddHandler('LoosedFocus', (event, args) => this.parent.parent.RemoveClass('-focused'));
 97         this._initializeValues();
 98         if(this.field?.params?.readonly === undefined) {
 99             this.readonly = false;    
100         }
101         else {
102             this.readonly = this.field?.params?.readonly;
103         }
104         if(this.field?.params?.searchable === undefined) {
105             this.searchable = false;    
106         }
107         else {
108             this.searchable = this.field?.params?.searchable;
109         }
110         if(this.field?.params?.enabled === undefined) {
111             this.enabled = true;
112         }
113         else {
114             this.enabled = this.field.params.enabled;
115         }
116 
117         if(this.field?.selector?.ondemand) {
118             this._input.__BeforeFilled = () => {
119                 return new Promise((resolve, reject) => {
120                     if (this.field.lookup) {
121                         this.loading = true;
122                         this.AddClass('app-select-loading');
123                         this._setLookup(this.field.lookup).then((response) => {
124                             this.values = response.result || response;
125                         }).finally(() => {
126                             this.loading = false;                        
127                             this.RemoveClass('app-select-loading');
128                             this._setEnabled();
129                             resolve(true);
130                         });
131                     }
132                 });
133             };
134         }
135     }

147             this.field.selector?.title ?? 'title',

149             this.field.selector?.group ?? null,
150             this.field.selector?.__render ?? null,

156     /**
157      * Заново загрузить значения из хранилища
158      */
159     ReloadValues() {
160         this.values = this.field.values;
161         if (this.field.lookup) {
162             this.loading = true;
163             this.AddClass('app-select-loading');
164             this._setLookup(this.field.lookup).then((response) => {
165                 this.values = response.result || response;
166             }).finally(() => {
167                 this.loading = false;
168 
169                 this.value = this.value ? this._input._findValue(this.value) : (this.field.default ?? null);
170 
171                 this.RemoveClass('app-select-loading');
172                 this._setEnabled();
173                 this.Dispatch('Changed');
174             });
175         }
176     }

178     _getDependsValue(type = null) {
179         if (this.root && this.field?.lookup) {
180 
181             if((type && !this.field.lookup[type]['depends']) || (!type && !this.field.lookup['depends'])) {
182                 return;
183             }
184 
185             let dependsField = type ? this.field.lookup[type]['depends'] : this.field.lookup['depends'];
186             if (dependsField) {
187                 const rootValues = this.root?.value;
188                 if(eval(`typeof rootValues?.${dependsField}`) !== 'undefined') {
189                     return eval(`rootValues.${dependsField}`);
190                 }
191                 return null;
192             }
193         }
194     }

196     /**
197      * Установить новое значение свойству lookup
198      * Загрузить значения селектора альтернативным способом, указанным в lookup
199      * @param {(Object|function)} value
200      */
201     _setLookup(value) {
202         let lookupPromise;
203         this._lookup = value;
204 
205         if (typeof this._lookup == 'function' || typeof this._lookup == 'string') {
206             if(typeof this._lookup == 'string') {
207                 this._lookup = eval(this._lookup);
208             }
209 
210             let dependsValue = this._getDependsValue();
211             let dependsField = this._lookup.depends ?? null;
212     
213             const lookupMethodRun = this._lookup(dependsValue, dependsField);
214             lookupPromise = lookupMethodRun instanceof Promise ? lookupMethodRun : new Promise((resolve, reject) => {
215                 resolve({
216                     result: this._lookup()
217                 });
218             });
219         }
220         else if (typeof this._lookup == 'object') {
221 
222             if(this._lookup?.method) {
223                 let lookupMethod = this._lookup.method;
224                 if (typeof lookupMethod == 'string') {
225                     lookupMethod = eval(this._lookup.method);
226                 }
227 
228                 if(typeof lookupMethod !== 'function') {
229                     lookupPromise = new Promise((resolve, reject) => { resolve({result: ''}); })
230                 }
231                 else {
232                     let dependsValue = this._getDependsValue();
233                     let dependsField = this._lookup.depends ?? null;   
234                     lookupPromise = lookupMethod(this._input._input.value, dependsValue, dependsField);
235                 }
236             }
237             else if(this._lookup?.binding) {
238                 let binding = this._lookup.binding;
239                 if (typeof binding == 'string') {
240                     let dependsValue = this._getDependsValue('binding');
241                     lookupPromise = App.Store.AsyncQuery(binding, dependsValue);
242                 }
243             }
244             else if(this._lookup?.controller) {
245                 let controller = this._lookup.controller;
246                 let module = eval(controller.module);
247                 let dependsValue = this._getDependsValue('controller');
248                 let dependsField = this._lookup.controller.depends ?? null;
249                 lookupPromise = module.Call(controller.class, controller.method, {term: this._input._input.value, param: dependsValue, depends: dependsField, lookup: this._lookup});
250             }
251             else if(this._lookup?.storage) {
252                 let controller = this._lookup?.storage?.controller;
253                 let module = eval(controller?.module);
254                 let dependsValue = this._getDependsValue('storage');
255                 let dependsField = this._lookup?.storage?.depends ?? null;
256                 lookupPromise = module.Call(controller.class, controller.method, {term: this._input._input.value, param: dependsValue, depends: dependsField, lookup: this._lookup});
257             }
258             else {
259                 lookupPromise = new Promise((resolve, reject) => { resolve({result: ''}); })
260             }
261         }
262 
263         // каждый метод должен возвращать промис
264         return lookupPromise;
265     }

267     /**
268      * Значения селектора
269      * @param {array} value
270      * */
271     set values(value) {
272         let required = this.field?.params?.required;
273         if(required === undefined) {
274             required = false;
275         }
276         if(!required) {
277             const o = {};
278             o[this.field?.selector?.title] = '---';
279             o[this.field?.selector?.value] = 0;
280             value = isIterable(value) ? [o, ...value] : [o];
281         }
282         this._input.values = value;
283     }

293     /**
294      * Если необходимо инициализировать данные из lookup
295      * @private
296      */
297     _initializeValues() {
298         if (this.field.lookup) {
299             this.loading = true;
300             this.AddClass('app-select-loading');
301 
302             //delay для того чтобы в форму были загружены значения зависимых полей
303             Colibri.Common.Delay(this.root ? 100 : 0).then(() => {
304                 this._setLookup(this.field.lookup).then((response) => {
305                     this.values = response.result || response;
306                 }).finally(() => {
307                     this.loading = false;
308                     if(this._lastValue) {
309                         this.value = this._lastValue;
310                         this._lastValue = null;
311                     } else {
312                         this._input._renderValue(false);
313                     }
314                     this.RemoveClass('app-select-loading');
315                     this._setEnabled();
316                 });
317             });
318         } else {
319             this._setEnabled();
320         }
321     }
322 
323     _setEnabled() {
324         if(!this.value && this.field.default) {
325             this.value = this.field.default;
326         }
327     }
