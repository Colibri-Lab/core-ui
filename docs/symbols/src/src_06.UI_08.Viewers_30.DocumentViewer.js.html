1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.DocumentViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('div'), root);
 17         this.AddClass('app-document-viewer-component');
 18     }
 19 
 20     /**
 21      * Метод для скачивания данных
 22      * @type {Function}
 23      */
 24     get downloadHandler() {
 25         return this._downloadHandler;
 26     }
 27     /**
 28      * Метод для скачивания данных
 29      * @type {Function}
 30      */
 31     set downloadHandler(value) {
 32         this._downloadHandler = value;
 33     }
 34     
 35     /**
 36      * Неизвестный формат, сообщение
 37      * @type {string}
 38      */
 39     get unknownFormatText() {
 40         return this._unknownFormatText;
 41     }
 42     /**
 43      * Неизвестный формат, сообщение
 44      * @type {string}
 45      */
 46     set unknownFormatText(value) {
 47         this._unknownFormatText = value;
 48     }
 49 
 50     _isFileViewable(mimetype) {
 51         let found = false;
 52         ['application/pdf', 'image/'].forEach((m) => {
 53             if(mimetype.indexOf(m) === 0) {
 54                 found = true;
 55                 return false;
 56             }
 57         });
 58         return found;
 59     }
 60 
 61     set value(value) {
 62         value = this._convertValue(value);
 63 
 64         this._value = value;
 65         
 66         
 67         if(Object.isObject(value)) {
 68             // remote file
 69             this._downloadHandler(value.guid).then((data) => {
 70                 this._element.html('');
 71                 const uri = window.URL.createObjectURL(Base2File(data.data, value.name, value.mimetype, true), {type: value.mimetype});
 72                 if(this._isFileViewable(value.mimetype)) {
 73                     const ext = Colibri.Common.MimeType.type2ext(value.mimetype);
 74                     if(Colibri.Common.MimeType.isImage(ext)) {
 75                         this._element.append(Element.fromHtml('<div style="background-image: url(' + uri + ')"></div>'));
 76                     }
 77                     else {
 78                         this._element.append(Element.fromHtml('<iframe src="' + uri + '" />'));
 79                     }
 80                 }
 81                 else {
 82                     this._element.append(Element.fromHtml('<div class="download"><div>' + (this._unknownFormatText ? this._unknownFormatText : '#{ui-viewers-document-unknownformat}') + '<a href="' + uri + '" class="app-component-button app-extended-button-component app-success-button-component" download="' + value.name + '">' + value.name +  '</a></div></div>'));
 83                }
 84             });
 85 
 86         }
 87         else if(typeof value === 'string') {
 88             this._element.html('');
 89             const pathinfo = value.pathinfo();
 90             const mimetype = Colibri.Common.MimeType.ext2type(pathinfo.ext);
 91             if(this._isFileViewable(mimetype)) {
 92                 if(Colibri.Common.MimeType.isImage(pathinfo.ext)) {
 93                     this._element.append(Element.fromHtml('<div style="background-image: url(' + uri + ')"></div>'));
 94                 }
 95                 else {
 96                     this._element.append(Element.fromHtml('<iframe src="' + uri + '" />'));
 97                 }
 98             }
 99             else {
100                 this._element.append(Element.fromHtml('<div class="download"><div>' + (this._unknownFormatText ? this._unknownFormatText : '#{ui-viewers-document-unknownformat}') + '<a href="' + value + '" class="app-extended-button-component app-success-button-component" download="' + pathinfo.filename + '">Download ' + pathinfo.filename +  '</a></div></div>'));
101             }
102 
103         }
104 
105     }
106 
107 
108 }
109 Colibri.UI.Viewer.Register('Colibri.UI.DocumentViewer', '#{ui-viewers-document}');
