 41     _renderFields() {
 42 
 43         // this._fieldData.vertical && this.AddClass('app-field-vertical');
 44         // this._fieldData.params && this._fieldData.params.merged && this.AddClass('app-merged-object-component');
 45 
 46         Object.forEach(this._fieldData.fields, (name, fieldData) => {
 47 
 48             const field = Object.cloneRecursive(fieldData);
 49             const tabTitle = field.desc[Lang.Current] ?? field.desc ?? '';
 50 
 51             delete field.desc;
 52             
 53             const component = Colibri.UI.Forms.Field.Create(name, this._tabs.container, field, this, this.root);
 54             component.message = false;
 55             component.shown = true;
 56             component.AddHandler('Changed', (event, args) => {
 57                 this.Dispatch('Changed', {component: this});
 58             });
 59 
 60             const tabButton = new Colibri.UI.Button(component.name + '-button', this._tabs.header);
 61             tabButton.value = tabTitle;
 62             tabButton.shown = true;
 63 
 64             this._tabs.AddTab(tabButton, component);
 65             
 66             
 67         });
 68 
 69         this.Dispatch('FieldsRendered');
 70     }

111     set value(value) {
112         value = eval_default_values(value);
113         
114         Object.forEach(this._tabs.components, (name, component) => {
115             if(name == '_adds') {
116                 // если наткнулись на _adds
117                 component.ForEveryField((name, field) => {
118                     if(!this._value) {
119                         field.value = component.field.default ?? null;    
120                     }
121                     else {
122                         field.value = this._value[name] ?? field?.field?.default ?? null;
123                     }
124                 });
125             }
126             else {
127                 if(!value) {
128                     component.value = component.field.default ?? null;    
129                 }
130                 else {
131                     component.value = value[name] ?? component.field.default ?? null;
132                 }
133             }
134         });
135 
136         
137         if(value) {
138             const oneof = this._tabs.components['_oneof'];
139             if(oneof) {
140                 const keys = Object.keys(value);
141                 oneof.value = keys[0];
142             }

146     _hideAndShow() {
147 
148         const data = this.value;
149         const formData = this.root.value;
150 
151         Object.forEach(this._fieldData.fields, (name, fieldData) => {
152             if(!fieldData || !this.contentContainer) {
153                 return true;
154             }
155             
156             const fieldComponent = this.contentContainer.Children(name);
157             if(fieldData?.params?.fieldgenerator) {
158                 const gen = eval(fieldData.params.fieldgenerator);
159                 gen(fieldData, fieldComponent, this);
160             } 
161             if(fieldComponent && fieldData.params && fieldData.params.condition) {
162                 const condition = fieldData.params.condition;
163                 if(condition.field) {        
164                     const type = condition?.type == 'disable' ? 'enabled' : 'shown';            
165                     const empty = condition?.empty || false;
166                     const inverse = condition?.inverse || false;
167                     let fieldValue = eval('data?.' + condition.field.split('.').join('?.'));
168                     if(!fieldValue) {
169                         fieldValue = eval('formData?.' + condition.field.split('.').join('?.'));
170                     }
171                     fieldValue = fieldValue?.value ?? fieldValue;
172                     let conditionResult = true;
173                     if((condition?.value ?? null) !== null) {
174                         if(Array.isArray(condition.value)) {
175                             conditionResult = fieldValue === undefined || (fieldValue !== undefined && condition.value.indexOf(fieldValue) !== -1);
176                         }
177                         else {
178                             conditionResult = fieldValue === undefined || (fieldValue !== undefined && fieldValue === condition.value);
179                         }
180                     } else if(condition?.method) {
181                         if(typeof condition.method === 'string') {
182                             conditionResult = eval(condition.method);
183                         } else {
184                             conditionResult = condition.method(fieldValue, data, type, empty, inverse, fieldData);
185                         }
186                     }
187                     if(inverse) {
188                         conditionResult = !conditionResult;
189                     }
190                     fieldComponent[type] = conditionResult;
191                     if(!conditionResult && empty) {
192                         fieldComponent.value = null;  
193                     }
194                 }
195                 else {
196                     fieldComponent.shown = true;
197                     fieldComponent.enable = true;
198                 }
199             }
200             else if(fieldData.params && fieldData.params.hidden) {
201                 fieldComponent.shown = false;
202             }
203         });
204 
205     }

234     _calcRuntimeValues(rootValue = null) {
235         Object.forEach(this._fieldData.fields, (name, fieldData) => {
236             const fieldComponent = this.Fields(name);         
237             if(fieldComponent instanceof Colibri.UI.Forms.Object || fieldComponent instanceof Colibri.UI.Forms.Array || fieldComponent instanceof Colibri.UI.Forms.Tabs) {
238                 fieldComponent._calcRuntimeValues();
239             } else {
240                 if(fieldData?.params?.valuegenerator) {
241                     const f = eval(fieldData?.params?.valuegenerator);
242                     const v = f(this.value, this.root?.value, fieldComponent, this.root);
243                     fieldComponent.value = v;
244                 }
245             }
