  6 Colibri.UI.LinkViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('a'), root);
 17         this.AddClass('app-link-viewer-component');
 18     }
 19 
 20     get value() {
 21         return this._value;
 22     }
 23     set value(value) {
 24         value = this._convertValue(value);
 25         this._value = value;
 26         this._element.attr('target', this._field?.params?.target ?? '_self');
 27         this._element.attr('href', value);
 28         this._element.html(value);
 29     }
 30 
 31 }
