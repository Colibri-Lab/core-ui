 28     _handleEvents() {
 29         /** Открыть окно с новым объектом */
 30         this._addObjectButton.AddHandler('Clicked', (event, args) => this.__newObject(event, args));
 31 
 32         /** Открыть окно с выбранным объектом */
 33         this._objectsGrid.AddHandler('SelectionChanged', (event, args) => {
 34             // не открываем окно, если на строке вызывают контекстное меню
 35             if (args.isContextMenuEvent !== true) {
 36                 this.__showObject(event, Object.assign({object_row: this._getSelected()}, args))
 37             }
 38         });
 39 
 40         /** Добавить строкам таблицы контекстное меню */
 41         this._objectsGrid.AddHandler('ContextMenuIconClicked', (event, args) => this.__showContextMenu(event, args));
 42 
 43         /** Вызвать нужный обработчик контекстного меню */
 44         this._objectsGrid.AddHandler('ContextMenuItemClicked', (event, args) => this.__processContextMenuAction(event, args));
 45 
 46         this._objectsGrid.AddHandler('RowSelected', (event, args) => this.__toggleMassActionsMenu(event, args));
 47 
 48          this._objectsGrid.AddHandler('MassActionsMenuActionClicked', (event, args) => this.__processMassAction(event, args));

 74             newColumn.viewer = column.viewer || null;
 75             newColumn.editor = column.editor || null;
 76         });

 80     /**
 81      * Открыть окно с объектом
 82      * @param {Object|null} value //содержимое полей объекта/формы
 83      * @private
 84      */
 85     _openObjectWindow(value) {
 86         if (!this._objectWindow) {
 87             this._objectWindow = new Colibri.UI.Forms.ArrayGrid.ObjectWindow(this._name + '-object-window', document.body);
 88             this._objectWindow.parent = this;
 89 
 90             this._objectWindow.fields = this._fieldData.fields;
 91             this._objectWindow.title = this._fieldData.desc;
 92             this._objectWindow.stickyX = "center";
 93             this._objectWindow.stickyY = "center";
 94             this._objectWindow.root = this.root;
 95 
 96             if (this._fieldData.params?.window) { this._objectWindow.setParams(this._fieldData.params.window); }
 97 
 98             /** Добавить или обновить объект после отправки формы */
 99             this._objectWindow.AddHandler('FormSubmitted', (event, args) => {
100                 let newArgs = Object.assign({object_row: this._getSelected()}, args);
101                 this._objectWindow.containsNewObject ? this.__addObjectRow(event, newArgs) : this.__updateObjectRow(event, newArgs);
102 
103                 this._objectsGrid.UnselectAllRows();
104             });
105             /** Обновить объект, когда содержимое формы изменилось */
106             this._objectWindow.AddHandler('Changed', (event, args) => {
107                 if (!this._objectWindow.containsNewObject) {
108                     this.__updateObjectRow(event, Object.assign({object_row: this._getSelected()}, args));
109                 }
110             });
111             /** Удалить окно из свойства, если оно было удалено из DOM */
112             this._objectWindow.AddHandler('WindowClosed', (event, args) => {
113                 if (this._objectWindow.disposeOnClose) { this._objectWindow = null; }
114             });
115         }
116 
117         this._objectWindow.containsNewObject = !!!value;
118         this._objectWindow.value = value || null;
119         this._objectWindow.shown = true;
120     }

195         args.domEvent.preventDefault();

197 
198         let contextmenu = [
199             {
200                 name: 'show-object',

207                 icon: Colibri.UI.ContextMenuRemoveIcon
208             },
209         ];
210 
211         args.item.contextmenu = contextmenu;
212         args.item.ShowContextMenu(args.isContextMenuEvent ? [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.RB] : [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.LB], '', args.isContextMenuEvent ? {left: args.domEvent.clientX, top: args.domEvent.clientY} : null);
213     }

215     /**
216      * Обработать клик по одному из пунктов контекстного меню
217      * @param event
218      * @param args
219      * @private
220      */
221     __processContextMenuAction(event, args) {
222         if (args?.menuData && args?.item) {
223             switch (args.menuData.name) {
224                 case 'show-object':
225                     this._objectsGrid.UnselectAllRows();
226                     args.item.selected = true;
227 
228                     this.__showObject(event, Object.assign({object_row: args.item}, args));
229                     break;
230 
231                 case 'remove-object-row':
232                     this.__removeObjectRow(event, Object.assign({object_row: args.item}, args));
233                     break;
234             }
235         }
236     }
237 
238     __toggleMassActionsMenu(event, args) {
239         if (this._objectsGrid.checked.length > 1) {
240             this._objectsGrid.massActionsMenu = [
241                 {
242                     name: 'remove-object-rows',
243                     title: '#{ui-arraygrid-massactions-remove}'
244                 }
245             ];
246 
247             this._objectsGrid.ShowMassActionsMenu(document.body);
248         } else {
249             this._objectsGrid.massActionsMenu = null;
250         }
251     }
252 
253     __processMassAction(event, args) {
254         if (args?.menuData && args?.items) {
255             switch (args.menuData.name) {
256                 case 'remove-object-rows':
257                     args.items.forEach((item) => {
258                         this.__removeObjectRow(event, Object.assign({object_row: item}, args));
259                     })
260                     this._objectsGrid.massActionsMenu = null;
261                     break;
262             }
263         }
264     }

288     /**
289      * Список отображаемых колонок (если не указан, отображаются все)
290      * @private
291      */
292     _setDisplayedColumns() {
293         this._displayedColumns = [];
294         let paramColumns = this._fieldData.params?.displayed_columns;
295 
296         Object.forEach(this._fieldData.fields, (fieldName, fieldData) => {
297             if (paramColumns !== undefined && !(fieldName in paramColumns)) {
298                 return;
299             }
300 
301             let column = {};
302             column['name'] = fieldName;
303             column['title'] = fieldData.desc || '';
304             column['viewer'] = paramColumns ? (paramColumns[fieldName]?.viewer || null) : null;
305             column['editor'] = paramColumns ? (paramColumns[fieldName]?.editor || null) : null;
306 
307             this._displayedColumns.push(column);
308         });
309     }
