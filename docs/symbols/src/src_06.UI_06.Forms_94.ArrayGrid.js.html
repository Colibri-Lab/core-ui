  9     /**
 10      * Render field component
 11      */
 12     RenderFieldContainer() {
 13         this.AddClass('app-component-array-grid-field');
 14 
 15         this._setDisplayedColumns();
 16         this._renderAddObjectButton();
 17         this._renderObjectsGrid();
 18 
 19         this._objectWindow = null;
 20 
 21         this._handleEvents();
 22 
 23         if(this._fieldData?.params?.readonly === undefined) {
 24             this.readonly = false;    
 25         }
 26         else {
 27             this.readonly = this._fieldData?.params?.readonly;
 28         }
 29         if(this._fieldData?.params?.enabled === undefined) {
 30             this.enabled = true;
 31         }
 32         else {
 33             this.enabled = this._fieldData.params.enabled;
 34         }
 35 
 36     }

 38     /** @protected */
 39     _handleEvents() {
 40         /** Открыть окно с новым объектом */
 41         this._addObjectButton.AddHandler('Clicked', (event, args) => this.__newObject(event, args));
 42 
 43         /** Открыть окно с выбранным объектом */
 44         this._objectsGrid.AddHandler('SelectionChanged', (event, args) => {
 45             // не открываем окно, если на строке вызывают контекстное меню
 46             if (args.isContextMenuEvent !== true) {
 47                 this.__showObject(event, Object.assign({object_row: this._getSelected()}, args))
 48             }
 49         });
 50 
 51         /** Добавить строкам таблицы контекстное меню */
 52         this._objectsGrid.AddHandler('ContextMenuIconClicked', (event, args) => this.__showContextMenu(event, args));
 53 
 54         /** Вызвать нужный обработчик контекстного меню */
 55         this._objectsGrid.AddHandler('ContextMenuItemClicked', (event, args) => this.__processContextMenuAction(event, args));
 56 
 57         this._objectsGrid.AddHandler('RowSelected', (event, args) => this.__toggleMassActionsMenu(event, args));
 58 
 59          this._objectsGrid.AddHandler('MassActionsMenuActionClicked', (event, args) => this.__processMassAction(event, args));
 60     }
 61 
 62     /** @private */
 63     _renderAddObjectButton() {
 64         this._addObjectButton = new Colibri.UI.OutlineBlueButton(this._name + '-add-object-button', this.contentContainer);
 65         this._addObjectButton.value = this._fieldData.params?.addlink || '#{ui-arraygrid-add} «' + (this._fieldData.desc) + '»';
 66         this._addObjectButton.shown = true;
 67     }
 68 
 69     /** @private */
 70     _renderObjectsGrid() {
 71         this._objectsGrid = new Colibri.UI.Grid(this._name + '-container-grid', this.contentContainer);
 72         this._objectsGrid.selectionMode = Colibri.UI.Grid.FullRow;
 73         this._objectsGrid.hasContextMenu = true;
 74         this._objectsGrid.hasMassActionsMenu = true;
 75         this._objectsGrid.shown = true;
 76 
 77         this._displayedColumns.forEach((column) => {
 78             let newColumn = this._objectsGrid.header.columns.Add(column.name, column.title);
 79             newColumn.viewer = column.viewer || null;
 80             newColumn.editor = column.editor || null;
 81         });
 82         this._objectsGrid.header.columns.Add('controls-column-' + Date.Now().getTime(), '');
 83     }

 85     /**
 86      * Opens window for clicked object
 87      * @param {Object|null} value
 88      * @private
 89      */
 90     _openObjectWindow(value) {
 91         if (!this._objectWindow) {
 92             this._objectWindow = new Colibri.UI.Forms.ArrayGrid.ObjectWindow(this._name + '-object-window', document.body);
 93             this._objectWindow.parent = this;
 94 
 95             this._objectWindow.fields = this._fieldData.fields;
 96             this._objectWindow.title = this._fieldData.desc;
 97             this._objectWindow.stickyX = "center";
 98             this._objectWindow.stickyY = "center";
 99             this._objectWindow.root = this.root;
100 
101             if (this._fieldData.params?.window) { this._objectWindow.setParams(this._fieldData.params.window); }
102 
103             /** Добавить или обновить объект после отправки формы */
104             this._objectWindow.AddHandler('FormSubmitted', (event, args) => {
105                 let newArgs = Object.assign({object_row: this._getSelected()}, args);
106                 this._objectWindow.containsNewObject ? this.__addObjectRow(event, newArgs) : this.__updateObjectRow(event, newArgs);
107 
108                 this._objectsGrid.UnselectAllRows();
109             });
110             /** Обновить объект, когда содержимое формы изменилось */
111             this._objectWindow.AddHandler('Changed', (event, args) => {
112                 if (!this._objectWindow.containsNewObject) {
113                     this.__updateObjectRow(event, Object.assign({object_row: this._getSelected()}, args));
114                 }
115             });
116             /** Удалить окно из свойства, если оно было удалено из DOM */
117             this._objectWindow.AddHandler('WindowClosed', (event, args) => {
118                 if (this._objectWindow.disposeOnClose) { this._objectWindow = null; }
119             });
120         }
121 
122         this._objectWindow.containsNewObject = !!!value;
123         this._objectWindow.value = value || null;
124         this._objectWindow.shown = true;
125     }

136     /**

143         if (row) { this._openObjectWindow(row.tag._objectValue || null); }
144     }

146     /**
147      * @private
148      * @param {Colibri.Events.Event} event event object
149      * @param {*} args event arguments
150      */ 
151     __addObjectRow(event, args) {
152         let value = this._objectWindow?.value || args.formValues || {};
153         this._addRow(value);
154     }

156     /**
157      * @private
158      * @param {Colibri.Events.Event} event event object
159      * @param {*} args event arguments
160      */ 
161     __updateObjectRow(event, args) {
162         let row = this._getRow(args.object_row);
163         let value = this._objectWindow?.value || args.formValues || {};
164 
165         if (row) { row.value = row.tag._objectValue = value; }
166     }

168     /**
169      * @private
170      * @param {Colibri.Events.Event} event event object
171      * @param {*} args event arguments
172      */ 
173     __removeObjectRow(event, args) {
174         let row = this._getRow(args.object_row);
175         if (row) { row.Dispose(); }
176     }
177 
178     /**
179      * Adds a new row
180      * @param {Object} value
181      * @private
182      */
183     _addRow(value) {
184         let row = this._objectsGrid.rows.Add('row-' + Date.Now().getTime(), value || {});
185         row.tag._objectValue = value;
186     }

188     /**
189      * @private
190      * @param {Colibri.Events.Event} event event object
191      * @param {*} args event arguments
192      */ 
193     __showContextMenu(event, args) {
194         args.domEvent.preventDefault();
195         args.domEvent.stopPropagation();
196 
197         let contextmenu = [
198             {
199                 name: 'show-object',
200                 title: '#{ui-arraygrid-contextmenu-edit}',
201                 icon: Colibri.UI.ContextMenuEditIcon
202             },
203             {
204                 name: 'remove-object-row',
205                 title: '#{ui-arraygrid-contextmenu-remove}',
206                 icon: Colibri.UI.ContextMenuRemoveIcon
207             },
208         ];
209 
210         args.item.contextmenu = contextmenu;
211         args.item.ShowContextMenu(args.isContextMenuEvent ? [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.RB] : [Colibri.UI.ContextMenu.RB, Colibri.UI.ContextMenu.LB], '', args.isContextMenuEvent ? {left: args.domEvent.clientX, top: args.domEvent.clientY} : null);
212     }

214     /**
215      * @private
216      * @param {Colibri.Events.Event} event event object
217      * @param {*} args event arguments
218      */ 
219     __processContextMenuAction(event, args) {
220         if (args?.menuData && args?.item) {
221             switch (args.menuData.name) {
222                 case 'show-object':
223                     this._objectsGrid.UnselectAllRows();
224                     args.item.selected = true;
225 
226                     this.__showObject(event, Object.assign({object_row: args.item}, args));
227                     break;
228 
229                 case 'remove-object-row':
230                     this.__removeObjectRow(event, Object.assign({object_row: args.item}, args));
231                     break;
232             }
233         }
234     }
235 
236     /**
237      * @private
238      * @param {Colibri.Events.Event} event event object
239      * @param {*} args event arguments
240      */ 
241     __toggleMassActionsMenu(event, args) {
242         if (this._objectsGrid.checked.length > 1) {
243             this._objectsGrid.massActionsMenu = [
244                 {
245                     name: 'remove-object-rows',
246                     title: '#{ui-arraygrid-massactions-remove}'
247                 }
248             ];
249 
250             this._objectsGrid.ShowMassActionsMenu(document.body);
251         } else {
252             this._objectsGrid.massActionsMenu = null;
253         }
254     }
255 
256     /**
257      * @private
258      * @param {Colibri.Events.Event} event event object
259      * @param {*} args event arguments
260      */ 
261     __processMassAction(event, args) {
262         if (args?.menuData && args?.items) {
263             switch (args.menuData.name) {
264                 case 'remove-object-rows':
265                     args.items.forEach((item) => {
266                         this.__removeObjectRow(event, Object.assign({object_row: item}, args));
267                     })
268                     this._objectsGrid.massActionsMenu = null;
269                     break;
270             }
271         }
272     }

296     /**
297      * Shows/hides columns
298      * @private
299      */
300     _setDisplayedColumns() {
301         this._displayedColumns = [];
302         let paramColumns = this._fieldData.params?.displayed_columns;
303 
304         Object.forEach(this._fieldData.fields, (fieldName, fieldData) => {
305             if (paramColumns !== undefined && !(fieldName in paramColumns)) {
306                 return;
307             }
308 
309             let column = {};
310             column['name'] = fieldName;
311             column['title'] = fieldData.desc || '';
312             column['viewer'] = paramColumns ? (paramColumns[fieldName]?.viewer || null) : null;
313             column['editor'] = paramColumns ? (paramColumns[fieldName]?.editor || null) : null;
314 
315             this._displayedColumns.push(column);
316         });
317     }

351     /**
352      * Value
353      * @type {Object[]}
354      */
355     set value(value) {
356         value = eval_default_values(value);
357         if(value && !Array.isArray(value)) { 
358             throw new Error('#{ui-arraygrid-errors}'); 
359         }
360 
361         this._objectsGrid.rows.Clear();
362         value && value.forEach((item) => this._addRow(item));
363         
364     }

367      * Fiels of object
368      * @param {string|null} name имя поля, (если не указано, возвращает сразу все)
369      * @return {Object|Colibri.UI.Forms.Field}
370      * @constructor
371      */
372     Fields(name) {
373         if (name) { return this._objectWindow?.form?.Children(name); }
374 
375         let fields = {};
376         this._objectWindow?.form?.ForEach((name, field) => {
377             if (field instanceof Colibri.UI.Forms.Field) { fields[name] = field; }
378         });
379 
380         return fields;

408     /**
409      * @constructor
410      * @param {string} name name of component
411      * @param {Element|Colibri.UI.Component} container container of component
412      */
413     constructor(name, container) {
414         super(name, container);
415         this.AddClass('app-component-array-grid-object-window');
416 
417         this._containsNewObject = true;
418 
419         this._form = new Colibri.UI.Forms.Form(this._name + '-form', this.container);
420         this._saveButton = new Colibri.UI.SuccessButton(this._name + '-save-button', this.footer);
421         this._saveButton.value = '#{ui-arraygrid-save}';
422 
423         this._form.AddHandler('Changed', () => this.Dispatch('Changed', {formValues: this._form.value, component: this}));
424         this._saveButton.AddHandler('Clicked', (event, args) => {
425             this.Dispatch('FormSubmitted', {formValues: this._form.value});
426             this.close();
427         });
428 
429         this._form.shown = true;
430         this._saveButton.shown = true;
431     }

433     /** @protected */
434     _registerEvents() {
435         super._registerEvents();
436         this.RegisterEvent('Changed', 'Когда содержимое формы внутри окна изменилось');
437         this.RegisterEvent('FormSubmitted', 'Когда форма внутри окна отправлена');
438     }
