1 /**
  2  * Represents a storage class that extends Colibri.Events.Dispatcher.
  3  * Manages storage of data and provides methods for querying and updating data.
  4  * @class 
  5  * @extends Colibri.Events.Dispatcher
  6  * @memberof Colibri.Storages
  7  */
  8 Colibri.Storages.Store = class extends Colibri.Events.Dispatcher {

 10     /**
 11      * Creates an instance of Colibri.Storages.Store.
 12      * @param {string} name - The name of the store.
 13      * @param {Object} [data={}] - The initial data for the store.
 14      * @param {Object} [parent=null] - The parent store, if any.
 15      * @param {boolean} [permanent=false] - Indicates whether the store data should be kept permanently.
 16      */
 17     constructor(name, data = {}, parent = null, permanent = false) {
 18         super('Store');
 19 
 20         this._name = name;
 21         this._parent = parent;
 22         this._data = data;
 23 
 24         this._pathHandlers = {};
 25         this._pathLoaders = {};
 26 
 27         this.RegisterEvents();
 28         this.RegisterEventHandlers();
 29 
 30         this._permanent = permanent ?? false;
 31         if(this._permanent) {
 32             App.Db.AddHandler('DatabaseDoesNotExists', (event, args) => {
 33                 App.Db.CreateStore(this._name, '__domain');
 34             });
 35             App.Db.AddHandler('DatabaseOpened', (event, args) => {
 36                 this.RetreiveFromPermanentStore();
 37             });
 38             App.Db.Open();
 39         }
 40 
 41     }

 43     /**
 44      * Saves the store data in the permanent storage.
 45      */
 46     KeepInPermanentStore() {
 47         const __domain = location.hostname;
 48         let savingData = this.ExportData();
 49         savingData = Object.assign(savingData, {__domain: __domain});
 50         App.Db.GetDataById(this._name, __domain).then(() => {
 51             App.Db.UpdateData(this._name, savingData);
 52         }).catch(() => {
 53             App.Db.AddData(this._name, savingData);
 54         }).finally(() => {
 55             this.Dispatch('StoreKeeped', {});
 56         });
 57     }

 59     /**
 60      * Retrieves store data from the permanent storage.
 61      */
 62     RetreiveFromPermanentStore() {
 63         if(App.Db.StoreExists(this._name)) {
 64             this._data = App.Db.GetDataById(this._name, location.hostname);
 65             Colibri.Common.StartTimer(this._name + '-store-dump', 15000, () => {
 66                 this.KeepInPermanentStore();
 67             });
 68             this.Dispatch('StoreRetreived', {});
 69             this.Dispatch('StoreUpdated', {});
 70         }
 71     }

 73     /**
 74      * Exports store data.
 75      * @param {boolean} [fullData=false] - Indicates whether to export full data recursively.
 76      * @returns {Object} The exported data.
 77      */
 78     ExportData(fullData = false) {
 79 
 80         const newData = {};
 81 
 82         Object.forEach(this._data, (name, value) => {
 83             if(value instanceof Colibri.Storages.Store) {
 84                 if(fullData) {
 85                     newData[name] = value.ExportData();
 86                 }
 87             } else {
 88                 newData[name] = Object.cloneRecursive(value);
 89             }
 90         });
 91 
 92         return newData;
 93     }

178     AddChild(path, data = {}, owner = null, permanent = false) {
179         let paths = path.split('.');
180         const newStore = new Colibri.Storages.Store(paths[paths.length - 1], data, this, permanent);
181         newStore.owner = owner;
182         this.Set(path, newStore);
183         return newStore;

186     /**
187      * Retrieves the child store at the specified path.
188      * @param {string} path - The path to the child store.
189      * @returns {object} The child store object along with its path.
190      */
191     GetChild(path) {
192         let p = path.split('.');
193         let first = p.shift();
194         if (first !== this._name) {
195             throw new Error('path must start with the name of store "' + this._name + '" you queried "' + path + '"');
196         }
197 
198         if (p.length === 0) {
199             return null;
200         }
201 
202         let data = this._data;
203         if(!data) {
204             return null;
205         }
206 
207         while(p.length > 0) {
208             first = p.shift();
209             if((data[first] ?? undefined) !== undefined) {
210                 data = data[first];
211                 if(data instanceof Colibri.Storages.Store) {
212                     return {child: data, path: (first + (p.length > 0 ? '.' + p.join('.') : ''))};
213                 }
214             }
215             else {
216                 break;
217             }
218         }
219 
220         return null;
221     }

223     /**
224      * Adds a path loader.
225      * @param {string} path - The path to add the loader.
226      * @param {function|string} loader - The loader function or string indicating the method.
227      * @param {object} [params={}] - Additional parameters for the loader.
228      * @returns {Colibri.Storages.Store} The store object.
229      */
230     AddPathLoader(path, loader, params = {}) {
231         if(this._pathLoaders[path]) {
232             throw new Error('Path loader is Registered')
233         }
234 
235         if(typeof loader == 'string') {
236             // значит это Module:Controller.Method
237             const parts = loader.split(':');
238             const module = eval(parts[0]);
239             loader = () => module.Call(parts[1].split('.')[0], parts[1].split('.')[1], Object.assign(params, {__defered: true}));
240         }
241 
242         this._pathLoaders[path] = {loader: loader, loading: false, loaded: false};
243         return this;
244     }

246     /**
247      * Adds a path handler for the specified path.
248      * @param {string|string[]} path - The path or an array of paths for which the handler is registered.
249      * @param {Function|Array} handler - The handler function or an array containing the respondent and handler function.
250      * @param {boolean} prepend - Indicates whether the handler should be added at the beginning (true) or end (false) of the handlers list.
251      * @returns {Colibri.Storages.Store} The store instance.
252      */
253     AddPathHandler(path, handler, prepend) {
254 
255         if(Array.isArray(path)) {
256             for(const p of path) {
257                 this.AddPathHandler(p, handler, prepend);
258             }
259             return this;
260         }
261 
262         // let data = this._parsePathIfHasParam(path);
263         // path = data[0];
264 
265         const childStoreData = this.GetChild(path);
266         if(childStoreData) {
267             return childStoreData.child.AddPathHandler(childStoreData.path, handler, prepend);
268         }
269 
270         let respondent = this;
271         if(Array.isArray(handler)) {
272             respondent = handler[0];
273             handler = handler[1];
274         }
275 
276         if(respondent instanceof Colibri.UI.Component) {
277             respondent.AddHandler('ComponentDisposed', (event, args) => {
278                 this.RemovePathHandler(path, respondent, handler);
279             });
280         }
281 
282         if (path instanceof Array) {
283             path.forEach((p) => {
284                 this.AddPathHandler(p, handler, prepend);
285             });
286         } else {
287 
288             if (!this._pathHandlers[path]) {
289                 this._pathHandlers[path] = [];
290             }
291 
292             for (let i = 0; i < this._pathHandlers[path].length; i++) {
293                 const h = this._pathHandlers[path][i];
294                 if (h === handler) {
295                     this._pathHandlers[path].splice(i, 1);
296                 }
297             }
298 
299             const handlerObject = { handler: handler, respondent: respondent };
300             if (prepend) {
301                 this._pathHandlers[path].splice(0, 0, handlerObject);
302             } else {
303                 this._pathHandlers[path].push(handlerObject);
304             }
305         }
306 
307         return this;
308     }

310     /**
311      * Removes a path handler for the specified path.
312      * @param {string} path - The path for which the handler should be removed.
313      * @param {Object} respondent - The respondent object associated with the handler.
314      * @param {Function} handler - The handler function to be removed.
315      * @returns {Colibri.Storages.Store} The store instance.
316      */
317     RemovePathHandler(path, respondent, handler) {
318         for (let i = 0; i < this._pathHandlers[path].length; i++) {
319             const h = this._pathHandlers[path][i];
320             if (h.handler == handler && h.respondent == respondent) {
321                 this._pathHandlers[path].splice(i, 1);
322                 break;
323             }
324         }
325         return this;
326     }

328     /**
329      * Dispatches events along the specified path.
330      * @param {string} path - The path along which events should be dispatched.
331      * @returns {boolean} True if all event handlers were executed successfully, otherwise false.
332      */
333     DispatchPath(path) {
334 
335         const childStoreData = this.GetChild(path);
336         if(childStoreData) {
337             return childStoreData.child.DispatchPath(childStoreData.path);
338         }
339 
340         const keys = Object.keys(this._pathHandlers);
341         for(let j=0; j<keys.length; j++) {
342             if(path.indexOf(keys[j]) === 0 || keys[j].indexOf(path) === 0) {
343                 const pathHandlers = this._pathHandlers[keys[j]];
344 
345                 let queryPath = keys[j];
346                 let queryParam = null;
347 
348                 let queryData = this._parsePathIfHasParam(queryPath);
349                 if(queryData[1]) {
350                     queryPath = queryData[0];
351                     queryParam = queryData[1];
352                 }
353 
354                 const data = this.Query(queryPath, queryParam);
355 
356                 for(let i=0; i<pathHandlers.length; i++) {
357 
358                     const handlerObject = pathHandlers[i];
359                     if (handlerObject && handlerObject.handler.apply(handlerObject.respondent, [data, queryPath]) === false) {
360                         return false;
361                     }
362         
363                 }        
364 
365             }
366         }
367 
368         return true;
369 
370     }

373      * Checks if the loader associated with the specified path has been executed.
374      * @param {string} path - The path for which to check if the loader has been executed.
375      * @returns {boolean} True if the loader has been executed or if the path doesn't have a loader, otherwise false.

377     IsLoaderExecuted(path) {
378         
379         const childStore = this.GetChild(path);
380         if(childStore) {
381             return childStore.child.IsLoaderExecuted(childStore.path, nodispatch, param);
382         }
383         const loader = this._pathLoaders[path];
384         return !loader || loader.loaded;

388     /**
389      * Reloads the data associated with the specified path, optionally dispatching events after reloading.
390      * @param {string} path - The path to reload data for.
391      * @param {boolean} [nodispatch=true] - Whether to dispatch events after reloading (default: true).
392      * @param {string|int|null} [param=null] - Additional parameter for the reload operation (default: null).
393      * @returns {Promise} A promise that resolves when the reload operation is complete.
394      */
395     async Reload(path, nodispatch = true, param = null) {
396         
397         const childStore = this.GetChild(path);
398         if(childStore) {
399             return childStore.child.Reload(childStore.path, nodispatch, param);
400         }
401 
402         const loader = this._pathLoaders[path];
403         if(!loader) {
404             return this.Query(path + (param && param.indexOf('=') === -1 ? '.' + param : ''), param && param.indexOf('=') !== -1 ? param : null);
405         }
406 
407         if(loader.loading) {
408             //Уже загружается, ждем пока завершится
409             await Colibri.Common.Wait(() => !loader.loading);
410             return this.Query(path + (param && param.indexOf('=') === -1 ? '.' + param : ''), param && param.indexOf('=') !== -1 ? param : null);
411         }
412 
413         loader.loading = true;
414         try {
415             let response = await loader.loader(param);
416             if(response && response.result) {
417                 this.Set(path, response.result, nodispatch);
418                 return param ? response.result[param] : response.result;    
419             } else {
420                 return null;
421             }
422         }
423         finally {
424             loader.loading = false;
425             loader.loaded = true;
426         }
427         
428         
429     }

431     /**
432      * Asynchronously retrieves data from the storage using PathLoaders.
433      * @param {string} path - The path to the object.
434      * @param {string|int|null} [param=null] - Additional path.
435      * @param {boolean} [reload=false] - Whether to force reload the data (default: false).
436      * @returns {Promise<object>} A promise that resolves with the retrieved data.
437      */
438     async AsyncQuery(path, param = null, reload = false) {
439 
440         let data = this._parsePathIfHasParam(path);
441         if(data[1]) {
442             path = data[0];
443             param = data[1];
444         }
445 
446         const res = this.Query(path + (param && param.indexOf('=') === -1 ? '.' + param : ''), param && param.indexOf('=') !== -1 ? param : null);
447         if(!reload && ((Object.isObject(res) && Object.countKeys(res) > 0) || (Array.isArray(res) && res.length > 0))) {
448             return res;
449         }
450 
451         return this.Reload(path, false, param);
452     }

454     /**
455      * Retrieves data from the storage based on the provided path and optional query parameters.
456      * @param {string} path - The path to the object.
457      * @param {string|null} [queryList=null] - Optional query parameters in the format "field=value".
458      * @returns {object} The retrieved data.
459      */
460     Query(path, queryList = null) {
461 
462         let p = path.split('.');
463         let first = p.shift();
464         if (first !== this._name) {
465             throw new Error('path must start with the name of store "' + this._name + '" you queried "' + path + '"');
466         }
467 
468         let data = this._data;
469         if (!data || p.length === 0) {
470             return data;
471         }
472 
473         while(p.length > 0) {
474             first = p.shift();
475             if((data[first] ?? undefined) !== undefined) {
476                 data = data[first];
477                 if(data instanceof Colibri.Storages.Store) {
478                     return data.Query(first + '.' + p.join('.'), queryList);
479                 }
480             }
481             else {
482                 data = {};
483                 break;
484             }
485 
486         }
487 
488         if(queryList) {
489             // queryList = field=value
490             const queryParts = queryList.split('=');
491             data = Array.findObject(data, queryParts[0], queryParts[1]);
492         }
493 
494         return data;
495 
496     }

498     /**
499      * Sets data at the specified path in the storage.
500      * @param {string} path - The path where the data will be set.
501      * @param {any} d - The data to be set.
502      * @param {boolean} [nodispatch=false] - Whether to dispatch events after setting the data.
503      * @returns {object} The updated storage object.
504      */
505     Set(path, d, nodispatch = false) {
506 
507         let p = path.split('.');
508         let first = p.shift();
509         if (first !== this._name) {
510             throw new Error('path must start with the name of store "' + this._name + '" you queried "' + path + '"');
511         }
512 
513         if (p.length === 0) {
514             this._data = d;
515             return this;
516         }
517 
518         let realpath = 'this._data';
519         let data = this._data;
520         for(let i=0; i<p.length; i++) {
521             if((data[p[i]] ?? undefined) === undefined) {
522                 data[p[i]] = {};
523             }
524             realpath += '["' + p[i] + '"]';
525             data = data[p[i]];
526             if(data instanceof Colibri.Storages.Store) {
527                 return data.Set(first + '.' + p.join('.'));
528             }
529         }
530 
531         if(d !== null) {
532             eval(realpath + ' = d;');
533         }
534         else {
535             eval('delete ' + realpath + ';');
536         }
537 
538         if(!nodispatch) {
539             this.DispatchPath(path);
540             this.Dispatch('StoreUpdated', {path: path, data: d});
541         }
542         return this;
543 
544     }

546     /**
547      * Updates a list in the storage based on search criteria and sorting options.
548      * @param {string} path - The path of the list in the storage.
549      * @param {string} searchField - The field used for searching within the list.
550      * @param {any} [searchValue=null] - The value to search for within the list.
551      * @param {any} [newData=null] - The new data to replace the existing data that matches the search criteria.
552      * @param {string} [sortField=null] - The field used for sorting the list.
553      * @param {string} [sortOrder='asc'] - The order in which the list should be sorted ('asc' for ascending or 'desc' for descending).
554      * @param {boolean} [insertIfNotExists=true] - Whether to insert the new data if no matching entry is found in the list.
555      * @param {string} [incrementIfInserted=''] - The path where the length of the list should be incremented if a new entry is inserted.
556      * @returns {array} The updated list.
557      */
558     UpdateList(path, searchField, searchValue = null, newData = null, sortField = null, sortOrder = 'asc', insertIfNotExists = true, incrementIfInserted = '') {
559         let list = this.Query(path);
560         if(!Array.isArray(list)) {
561             list = [];
562         }
563         const oldLength = list.length;
564         list = Array.replaceObject(list, searchField, searchValue, newData, insertIfNotExists);
565         if(sortField) {
566             list.sort((a, b) => {
567                 if(a[sortField] > b[sortField]) {
568                     return sortOrder === 'asc' ? 1 : -1;
569                 }
570                 else if(a[sortField] < b[sortField]) {
571                     return sortOrder === 'asc' ? -1 : 1;
572                 }
573                 return 0;
574             });
575         }
576         if(oldLength != list.length && incrementIfInserted) {
577             this.Set(incrementIfInserted, list.length, true);
578         }
579         this.Set(path, list);
580         return list;
581     }

583     /**
584      * Updates the list in the storage by intersecting it with the provided values based on the specified search field.
585      * @param {string} path - The path of the list in the storage.
586      * @param {string} searchField - The field used for searching within the list.
587      * @param {Array} values - The array of values to intersect with the list.
588      */
589     IntersectList(path, searchField, values) {
590         let list = this.Query(path);
591         if(!Array.isArray(list)) {
592             list = [];
593         }
594 
595         for(const item of values) {
596             list = Array.replaceObject(list, searchField, item[searchField], item, true);
597         }
598 
599         this.Set(path, list);
600 
601     }

621      * Sorts the list in the storage based on the specified field and sort order.
622      * @param {string} path - The path of the list in the storage.
623      * @param {string} sortField - The field used for sorting the list.
624      * @param {string} [sortOrder='asc'] - The order in which the list should be sorted ('asc' for ascending or 'desc' for descending).
625      * @returns {Array} The sorted list.
626      */
627     SortList(path, sortField, sortOrder = 'asc') {
628         let list = this.Query(path);
629         if(!Array.isArray(list)) {
630             list = [];
631         }
632         list.sort((a, b) => {
633             if(a[sortField] > b[sortField]) {
634                 return sortOrder === 'asc' ? 1 : -1;
635             }
636             else if(a[sortField] < b[sortField]) {
637                 return sortOrder === 'asc' ? -1 : 1;
638             }
639             return 0;
640         });
641         this.Set(path, list);
642         return list;
643     }

661     /**
662      * Parses the path if it contains a parameter enclosed in parentheses.
663      * @param {string} path - The path to parse.
664      * @returns {[string, string|null]} An array containing the parsed path and the parameter, or null if no parameter is found.
665      * @private
666      */
667     _parsePathIfHasParam(path) {
668         if(path.indexOf('(') === -1) {
669             return [path, null];
670         }
671         path = path.replaceAll(')', '');
672         return path.split('(');
673     }
