1 /**
  2  * Represents a remote procedure call (RPC) request.
  3  * @class 
  4  * @extends Colibri.Events.Dispatcher
  5  */
  6 Colibri.IO.RpcRequest = class extends Colibri.Events.Dispatcher {
  7 
  8     /**
  9      * Creates an instance of RpcRequest.
 10      * @param {string} moduleEntry - The name of the module.
 11      * @param {string} [type] - The type of data.
 12      * @param {string|null} [remoteDomain=null] - The remote domain.
 13      * @param {Function|null} [urlResolver=null] - The URL resolver function.
 14      */
 15     constructor(moduleEntry, type, remoteDomain = null, urlResolver = null) {
 16         super();
 17 
 18         this._moduleEntry = moduleEntry;
 19         this._requestType = type || 'json';
 20         this._remoteDomain = remoteDomain || null;
 21         this._urlResolver = urlResolver || null;
 22         this._requestsCache = {};
 23 
 24         this._workingRequests = {};
 25 
 26         this.RegisterEvent('CallCompleted', false, 'Обработка запроса завершена');
 27         this.RegisterEvent('CallProgress', false, 'Событие прогресса');
 28         this.RegisterEvent('CallError', false, 'Ошибка обработки запроса');
 29         this.RegisterEvent('ResultsProcessed', false, 'Обработка результатов завершена');
 30     }
 31 
 32     /**
 33      * Gets or sets the remote domain for RPC requests.
 34      * @type {string|null}
 35      */
 36     get remoteDomain() {
 37         return this._remoteDomain;
 38     }
 39     /**
 40      * Gets or sets the remote domain for RPC requests.
 41      * @type {string|null}
 42      */
 43     set remoteDomain(value) {
 44         this._remoteDomain = value;
 45     }
 46 
 47     /**
 48      * Gets or sets the request type for RPC requests.
 49      * @type {string}
 50      */
 51     get requestType() {
 52         return this._requestType;
 53     }
 54     /**
 55      * Gets or sets the request type for RPC requests.
 56      * @type {string}
 57      */
 58     set requestType(value) {
 59         this._requestType = value;
 60     }
 61 
 62     /**
 63      * Url resolver function
 64      * @type {Function}
 65      */
 66     get urlResolver() {
 67         return this._urlResolver;
 68     }
 69     /**
 70      * Url resolver function
 71      * @type {Function}
 72      */
 73     set urlResolver(value) {
 74         this._urlResolver = value;
 75     }
 76 
 77     /**
 78      * Prepares a string for use.
 79      * @param {string} string - The string to prepare.
 80      * @returns {string} The prepared string.
 81      * @private
 82      */
 83     _prepareStrings(string) {
 84         return string.split('\\').map((v) => v.fromCamelCase()).join('/');
 85     }
 86 
 87     /**
 88      * Gets the requests being processed.
 89      * @returns {Object} The working requests.
 90      */
 91     Requests() {
 92         return this._workingRequests;
 93     }
 94 
 95     /**
 96      * Clears the request cache.
 97      */
 98     ClearCache() {
 99         this._requestsCache = {};
100     }
101 
102     /**
103      * Makes a remote procedure call.
104      * @param {string} controller - The controller name.
105      * @param {string} method - The method to execute.
106      * @param {Object} [params=null] - The parameters to pass.
107      * @param {Object} [headers={}] - The headers.
108      * @param {boolean} [withCredentials=true] - Whether to include credentials.
109      * @param {string} [requestKeyword=Date.Mc()] - The request keyword.
110      * @returns {Promise} The result of the RPC call.
111      */
112     Call(controller, method, params = null, headers = {}, withCredentials = true, requestKeyword = Date.Mc()) {
113 
114         const request = new Colibri.IO.Request();
115         this._workingRequests[requestKeyword] = request;
116         
117         const requestMethod = params && params._requestMethod && params._requestMethod === 'get' ? 'Get' : 'Post'; 
118         const requestType = params && params._requestType ? params._requestType : this._requestType;
119         const requestCache = params && params._requestCache ? params._requestCache : false;
120         const uploadProgress = params && params._uploadProgress ? params._uploadProgress : false;
121         params && delete params._requestMethod;
122         params && delete params._requestCache;
123         headers.requester = location.hostname;
124 
125         return new Promise((resolve, reject) => {
126             
127             let url = null;
128             
129             // if url resolver is not set
130             if(!this._urlResolver || typeof this._urlResolver !== 'function') {
131                 url = this._prepareStrings((this._moduleEntry ? '\\Modules\\' + this._moduleEntry : '') + '\\' +  controller + '\\' + method + '.' + requestType);
132             } else {
133                 url = this._urlResolver(this._moduleEntry, controller, method, requestType, params, headers, withCredentials);
134             }
135 
136             if(this._remoteDomain) {
137                 url = this._remoteDomain + url;
138             }
139 
140             const requestUnique = String.MD5(JSON.stringify(Object.sortPropertiesRecursive({url: url, params: params, headers: headers})));
141             if(requestCache && this._requestsCache['cache' + requestUnique]) {
142                 Colibri.Common.Wait(() => this._requestsCache['cache' + requestUnique].working !== true).then(() => {
143                     const data = this._requestsCache['cache' + requestUnique];
144                     delete this._workingRequests[requestKeyword];
145                     this.Dispatch('CallCompleted', {result: data.result, request: requestKeyword});
146                     this.Dispatch('ResultsProcessed', {result: data.result});
147                     resolve(data);    
148                 });
149                 return;
150             } else {
151                 this._requestsCache['cache' + requestUnique] = {working: true};
152             }
153 
154             request.AddHeaders(headers);
155             request[requestMethod](url, params, withCredentials, (progressEvent) => {
156                 if(uploadProgress) {
157                     uploadProgress(progressEvent, requestKeyword);
158                 }
159                 this.Dispatch('CallProgress', {event: progressEvent, request: requestKeyword});
160             }).then((data) => {
161                 
162                 delete this._workingRequests[requestKeyword];
163 
164                 this.Dispatch('CallCompleted', {result: data.result, request: requestKeyword});
165 
166                 try {
167                     if(requestType == 'json') {
168                         data.result = JSON.parse(data.result);
169                     } else if(requestType == 'xml') {
170                         data.result = new DOMParser().parseFromString(data.result, "text/xml");
171                     } else if(requestType == 'stream') {
172                         const disposition = data.headers['content-disposition'];
173                         let name = '';
174                         let ext = '';
175                         if(disposition) {
176                             name = disposition.split('filename=')[1].replaceAll('"', '');
177                             ext = name.split('.').pop();
178                         }
179                         data.result = {
180                             name: name,
181                             ext: ext,
182                             mimetype: Colibri.Common.MimeType.ext2type(ext),
183                             content: data.result
184                         };
185                     }
186                 }
187                 catch(e) {
188                     console.log(url, params, withCredentials, e);
189                     data.result = {};
190                 }
191 
192                 // ! для конвертации 
193                 // data.result = Object.cloneRecursive(data.result, Object.convertToExtended);
194 
195                 this.Dispatch('ResultsProcessed', {result: data.result});
196 
197                 if(requestCache) {
198                     this._requestsCache['cache' + requestUnique] = Object.cloneRecursive(data);
199                 }
200 
201                 resolve(data); 
202 
203             }).catch((data) => {
204                 delete this._workingRequests[requestKeyword];
205                 try {
206                     data.result = JSON.parse(data?.result || '{}');
207                 } catch(e) {}
208                 this.Dispatch('CallError', {error: data.result, status: data.status});
209                 reject(data);
210 
211             });
212         });
213 
214     }
