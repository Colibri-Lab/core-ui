  2 Colibri.UI.Shortcuts = {
  3 
  4     keys: {},
  5 
  6     Add: function (keys, handler) {
  7         if(!this.keys[keys]) {
  8             this.keys[keys] = [];
  9         }
 10         this.keys[keys].push(handler);
 11     },
 12 
 13     Bind: function() {
 14 
 15         window.addEventListener('keydown', (e) => {
 16     
 17             let key = [];
 18             if(e.ctrlKey) {
 19                 key.push('Control');
 20             }
 21             if(e.altKey) {
 22                 key.push('Alt');
 23             }
 24             if(e.shiftKey) {
 25                 key.push('Shift');
 26             }
 27 
 28             if(e.key) {
 29                 key.push(e.key.toUpperCase());
 30             }
 31     
 32             key = key.join('+');
 33     
 34             if(this.keys[key]) {
 35                 let ret = true;
 36                 this.keys[key].forEach((handler) => {
 37                     ret &= handler(key);
 38                 });
 39                 if(!ret) {
 40                     e.preventDefault();
 41                     e.stopPropagation();
 42                 }
 43                 return ret;
 44             }
 45     
 46         });
 47     
 48     }
 49 
 50 };
