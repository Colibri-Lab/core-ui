1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  */
  5 Colibri.UI.Shortcuts = {
  6 
  7     /**
  8      * Keys added for shortcuts
  9      * @type {object}
 10      */
 11     keys: {},
 12 
 13     /**
 14      * Add handler to shortcut
 15      * @public
 16      * @static
 17      * @param {Array} keys array of keys to add
 18      * @param {Function} handler method to execute when shortcut is raised
 19      */
 20     Add: function (keys, handler) {
 21         if(!this.keys[keys]) {
 22             this.keys[keys] = [];
 23         }
 24         this.keys[keys].push(handler);
 25     },
 26 
 27     /**
 28      * Bund all shortcuts added before
 29      * @public
 30      * @static
 31      */
 32     Bind: function() {
 33 
 34         window.addEventListener('keydown', (e) => {
 35     
 36             let key = [];
 37             if(e.ctrlKey) {
 38                 key.push('Control');
 39             }
 40             if(e.altKey) {
 41                 key.push('Alt');
 42             }
 43             if(e.shiftKey) {
 44                 key.push('Shift');
 45             }
 46 
 47             if(e.key) {
 48                 key.push(e.key.toUpperCase());
 49             }
 50     
 51             key = key.join('+');
 52     
 53             if(this.keys[key]) {
 54                 let ret = true;
 55                 this.keys[key].forEach((handler) => {
 56                     ret &= handler(key);
 57                 });
 58                 if(!ret) {
 59                     e.preventDefault();
 60                     e.stopPropagation();
 61                 }
 62                 return ret;
 63             }
 64     
 65         });
 66     
 67     }
 68 
 69 };
 70 
 71 Colibri.UI.Shortcuts.Bind();
