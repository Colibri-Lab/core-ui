205     /**
206      * Returns the MIME type associated with the given file extension.
207      * @param {string} ext - The file extension.
208      * @returns {string|undefined} The MIME type, or undefined if not found.
209      */
210     static ext2type(ext) {
211         return Colibri.Common.MimeType.types[ext];
212     }
213 
214     /**
215      * Returns the file extension associated with the given MIME type.
216      * @param {string} type - The MIME type.
217      * @returns {string|false} The file extension, or false if not found.
218      */
219     static type2ext(type) {
220         var ret = false;
221         Object.keys(Colibri.Common.MimeType.types).forEach(key => {
222             if(type == Colibri.Common.MimeType.types[key]) { ret = key; return false; }
223             return true;
224         });
225         return ret;
226     }
227 
228     /**
229      * Returns the file extension associated with the given base64 encoded data.
230      * @param {string} base - The base64 encoded data.
231      * @returns {string|false} The file extension, or false if not found.
232      */
233     static base2type(base) {
234         var ret = false;
235         var type = base.split('data:')[1];
236         type = type.split(';')[0];
237         Object.keys(Colibri.Common.MimeType.types).forEach(key => {
238             if(type == Colibri.Common.MimeType.types[key]) { ret = key; return false; }
239             return true;
240         });
241         return ret;
242     }

293     /**
294      * Get the CodeMirror mode associated with a given file extension.
295      * @param {string} ext - The file extension.
296      * @returns {string} The CodeMirror mode.
297      */
298     static ext2mode(ext) {
299         
300         if(ext == 'js') return 'javascript';
301         else if(ext == 'css') return 'css';
302         else if(ext == 'less') return 'text/x-less';
303         else if(ext == 'scss') return 'text/x-scss';
304         else if(ext == 'html' || ext == 'htm') return 'htmlmixed';
305         else if(ext == 'php' || ext == 'layout' || ext == 'release' || ext == 'service') return 'php';
306         else if(ext == 'xml') return 'xml';
307         else if(ext == 'yaml') return 'yaml';
308         else return Colibri.Common.MimeType.ext2type(ext);
309         
310     }
