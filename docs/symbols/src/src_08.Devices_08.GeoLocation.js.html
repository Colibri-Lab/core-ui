  2 /**
  3  * Represents a utility for accessing geolocation information.
  4  * @class
  5  * @extends Destructable
  6  * @memberof Colibri.Devices
  7  */
  8 Colibri.Devices.GeoLocation = class extends Destructable {
  9 
 10     /**
 11      * Instance variable representing the device.
 12      * @type {Colibri.UI.Device}
 13      * @private
 14      */
 15     _device = null;
 16     /**
 17      * Instance variable representing the geolocation plugin.
 18      * @type {object}
 19      * @private
 20      */
 21     _plugin = null;
 22     /**
 23      * Flag indicating whether permission is granted.
 24      * @type {boolean}
 25      * @private
 26      */
 27     _permited = false;
 28 
 29     /**
 30      * Creates an instance of GeoLocation.
 31      * @param {*} device - The device object.
 32      */
 33     constructor(device) {
 34         super();
 35         this._device = device;
 36     }
 37 
 38     /**
 39      * Detects the current position.
 40      * @param {object} options - The options for geolocation detection.
 41      * @returns {Promise} - Promise resolving with the current position.
 42      */
 43     Detect(options = { maximumAge: 3000, timeout: 5000, enableHighAccuracy: true }) {
 44         return new Promise((resolve, reject) => {
 45             navigator.geolocation.getCurrentPosition((position) => {
 46                 resolve(position);
 47             }, (error) => {
 48                 reject(error);
 49             }, options);    
 50         });
 51     }
 52 
 53     /**
 54      * Watches for changes in position.
 55      * @param {function} callback - The callback function to handle position changes.
 56      * @param {object} options - The options for watching position changes.
 57      * @returns {Promise} - Promise resolving when position watching is started.
 58      */
 59     Watch(callback, options = { maximumAge: 3000, timeout: 5000, enableHighAccuracy: true }) {
 60         return new Promise((resolve, reject) => {
 61             navigator.geolocation.watchPosition(callback, (error) => {
 62                 reject(error);
 63             }, options);
 64         });
 65     }
 66 
 67 }
