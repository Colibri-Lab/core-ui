1 /**
  2  * @class
  3  * @extends Colibri.UI.Editor
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.FilesEditor = class extends Colibri.UI.Editor {
  7     
  8     constructor(name, container) {
  9         super(name, container, Element.create('div'));
 10         this.AddClass('app-files-editor-component');
 11 
 12         this._value = null;
 13 
 14         this._maxCount = null;
 15         this._allowedExtensions = null;
 16         this._maxFileSize = null;
 17 
 18         this._files = new Colibri.UI.List('files-list', this);
 19         this._filesGroup = this._files.AddGroup('group', '');
 20         this._files.__renderItemContent = (itemData, container) => {
 21 
 22             const icon = new Colibri.UI.Icon('icon', container);
 23             const filename = new Colibri.UI.TextSpan('span', container);
 24             let sign = null;
 25             if(this._fieldData?.params?.canbesigned && itemData.title.indexOf('-signed') === -1) {
 26                 sign = new Colibri.UI.TextSpan('sign', container);
 27                 sign.value = 'Подписать';
 28             }
 29             const deleteIcon = new Colibri.UI.TextSpan('delete', container);
 30             icon.shown = filename.shown = deleteIcon.shown = true;
 31             if(sign) {
 32                 sign.shown = true;
 33             }
 34             icon.value = Colibri.UI.FileLinkIcon;
 35             deleteIcon.value = Colibri.UI.ClearIcon;
 36             deleteIcon.AddHandler('Clicked', (event, args) => {
 37                 event.sender.parent.Dispose();
 38                 if(this._filesGroup.children == 0) {
 39                     this._files.shown = false;
 40                 }
 41                 this.Dispatch('Changed');
 42             });
 43             sign && sign.AddHandler('Clicked', (event, args) => {
 44                 const component = eval(this._fieldData?.params?.sign_component);
 45                 component.Show(itemData.file, (signedFile) => {
 46                     const idata = container.value;
 47                     idata.file = signedFile;
 48                     idata.title = signedFile.name;
 49                     container.value = idata;
 50                 });
 51                 args.domEvent?.stopPropagation();
 52                 return false;
 53             });
 54 
 55             filename.value = itemData.title;
 56 
 57         }
 58         this._input = new Colibri.UI.Input.File('input', this);
 59         this._input.shown = true;
 60         this._input.multiple = true;
 61 
 62         this._input.AddHandler('InputFileChanged', (event, args) => {
 63 
 64             let validatedFiles = this._validate(this._input.Files());
 65 
 66             this._files.shown = true;
 67             validatedFiles.forEach((file) => {
 68                 const item = {title: file.name, file: file};
 69                 this._filesGroup.AddItem(item);    
 70             });
 71 
 72             
 73             if (!this._validated) {
 74                 this._showError();
 75             }
 76 
 77             this.Dispatch('Changed');
 78         });
 79 
 80     }
 81 
 82     _registerEvents() {
 83         super._registerEvents();
 84         this.RegisterEvent('Changed', false, 'Когда файлы выбраны');
 85     }
 86 
 87     
 88     /**
 89      * Валидация выбранного списка файлов
 90      * @param {array} filesList
 91      * @return {array} список файлов, прошедших проверку
 92      * @private
 93      */
 94     _validate(filesList) {
 95         let error = false;
 96         let validatedList = filesList;
 97 
 98         if (this._maxCount) {
 99             if ((this.value.length + filesList.length) > this._maxCount) {
100                 error = true;
101                 this._validated = false;
102 
103                 let countAvailable = this._maxCount - this.value.length;
104                 validatedList.splice(countAvailable);
105                 this._errorMessages.push('#{ui-editors-files-message}'.replaceAll('%s', this._maxCount));
106             }
107         }
108 
109         if (this._maxFileSize) {
110             validatedList.forEach((file, index) => {
111                 if (file.size > this._maxFileSize) {
112                     error = true;
113                     this._validated = false;
114 
115                     validatedList.splice(index, 1);
116                     this._errorMessages.push('#{ui-editors-files-message3}'.replaceAll('%s', file.name));
117                 }
118             });
119         }
120 
121         if (this._allowedExtensions) {
122             let extensions = this._allowedExtensions.map((item) => item.toLowerCase());
123 
124             filesList.forEach((file, index) => {
125                 let fileName = file.name;
126                 let ext = '';
127 
128                 if (fileName.includes('.')) {
129                     ext = fileName.substring(fileName.lastIndexOf('.') + 1);
130                     ext = ext.toLowerCase();
131                 }
132 
133                 if (!extensions.includes(ext)) {
134                     error = true;
135                     this._validated = false;
136 
137                     validatedList.splice(index, 1);
138                     this._errorMessages.push('#{ui-editors-files-message5} ' + fileName);
139                 }
140             });
141         }
142 
143         if (!error) {
144             this._errorMessages = [];
145             this._validated = true;
146         }
147 
148         return validatedList;
149     }
150 
151     /**
152      * Показать ошибки валидации
153      * @private
154      */
155      _showError() {
156         this._errorMessages.forEach((message) => {
157             App.Notices.Add({
158                 severity: 'error',
159                 title: message,
160                 timeout: 5000
161             });
162         });
163 
164         this._errorMessages = [];
165     }
166 
167     
168     Validate() {
169         
170     }
171 
172     get value() {
173         let ret = [];
174         this._filesGroup.ForEach((name, item) => {
175             ret.push(item.value.file);
176         })
177         return ret;
178     }
179 
180     set value(value) {
181         this._filesGroup.Clear();
182         value && value.forEach((file) => {
183             const item = {title: file.name, file: file};
184             this._filesGroup.AddItem(item);
185         });
186         
187         if(value && value.length > 0) {
188             this._setFilled();
189         } else {
190             this._unsetFilled();
191         }
192 
193     }
194     
195     
196     set field(value) {
197         this._fieldData = value;
198 
199         if (this._fieldData?.params && this._fieldData?.params.allow) {
200             this._allowedExtensions = this._fieldData.params.allow.exts ?? null;
201             this._maxFileSize = this._fieldData.params.allow.size ?? null;
202             this._maxCount = this._fieldData.params.allow.count ?? null;
203         }
204 
205     }
206 
207     get field() {
208         return this._fieldData;
209     }
