1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer

  6 Colibri.UI.SelectViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-select-viewer-component');
 18     }
 19 
 20     get value() {
 21         return this._value;
 22     }
 23 
 24     set value(value) {
 25         this._value = value;
 26         if(value !== null && value !== undefined && value !== '') {
 27             if((this._field?.multiple ?? this._field?.params?.multiple)) {
 28                 // надо обработать вариант с мультизначениями
 29                 if(typeof value == 'string') {
 30                     value = value.split(',');
 31                 }
 32 
 33                 if(Array.isArray(value)) {
 34                     let r = [];
 35                     
 36                     let isObject = false;
 37                     for(let vv of value) {
 38                         if(Object.isObject(vv)) {
 39                             r.push(vv[this._field?.selector?.title ?? 'title']);
 40                             isObject = true;
 41                         }
 42                     }
 43 
 44                     if(isObject) {
 45                         if(r.length > 1) {
 46                             let v1 = r.splice(0, 1);
 47                             let v2 = r.join('<br />');
 48                             super.value = v1;
 49                             const icon1 = new Colibri.UI.Icon(this.name + '-hover', this);
 50                             icon1.shown = true;
 51                             icon1.value = '<em>+' + r.length + '</em> ' + (this._field.params?.infoIcon ?? Colibri.UI.InfoIcon);
 52                             icon1.toolTip = v2;
 53                         }
 54                         else {
 55                             super.value = r.pop();
 56                         }
 57                     }
 58                     else if(this._field?.lookup) {
 59                         this._setLookup(this._field.lookup).then((response) => {
 60                             const _result = (response.result ?? response);
 61                             if(_result?.length) {
 62                                 for (let val of _result) {
 63                                     if (value.indexOf(val[this._field?.selector?.value || 'value']) !== -1) {
 64                                         r.push(val[this._field?.selector?.title || 'title']);
 65                                     }
 66                                 }
 67                             }
 68                         }).finally(() => {
 69                             this.RemoveClass('app-viewer-loading')
 70 
 71                             if(r.length > 1) {
 72                                 let v1 = '<dd>' + r[0] + '</dd>';
 73                                 let v2 = r.join('<br />');
 74                                 super.value = v1;
 75                                 const icon1 = new Colibri.UI.Icon(this.name + '-hover', this);
 76                                 icon1.shown = true;
 77                                 icon1.value = '<em>' + r.length + '</em> ' + (this._field.params?.infoIcon ?? Colibri.UI.InfoIcon);
 78                                 icon1.toolTip = v2;
 79                             }
 80                             else {
 81                                 super.value = r.pop();
 82                             }
 83     
 84                         });
 85                     }
 86                     else if(this._field.values) {
 87                         for(const v of this._field.values) {
 88                             if(vv == (v.value ?? v.title ?? v)) {
 89                                 r.push(v.title);
 90                             }
 91                         }
 92                         if(r.length > 1) {
 93                             let v1 = '<dd>' + r[0] + '</dd>';
 94                             let v2 = r.join('<br />');
 95                             super.value = v1;
 96                             const icon1 = new Colibri.UI.Icon(this.name + '-hover', this);
 97                             icon1.shown = true;
 98                             icon1.value = '<em>' + r.length + '</em> ' + (this._field.params?.infoIcon ?? Colibri.UI.InfoIcon);
 99                             icon1.toolTip = v2;
100                         }
101                         else {
102                             super.value = r.pop();
103                         }
104                     } 
105 
106                     
107                     
108                 }
109             }
110             else {
111                 if(Object.isObject(value)) {
112                     try {
113                         super.value = value[this._field?.selector?.title ?? 'title'][Lang.Current] ?? value[this._field?.selector?.title ?? 'title'];
114                     } catch(e) {}
115                 }
116                 else if (this._field?.lookup) {
117                     let selected = false;
118                     value = value.value ?? value;
119                     this.AddClass('app-viewer-loading');
120                     this._setLookup(this._field.lookup).then((response) => {
121                         const _result = (response.result ?? response);
122                         if(_result?.length) {
123                             for (let val of _result) {
124                                 if (val[this._field?.selector?.value || 'value'] == value) {
125                                     super.value = val[this._field?.selector?.title || 'title'][Lang.Current] ?? val[this._field?.selector?.title || 'title'];
126                                     selected = true;
127                                     break;
128                                 }
129                             }
130                         }
131                     }).finally(() => {
132                         this.RemoveClass('app-viewer-loading')
133                         if(!selected) {
134                             try {
135                                 super.value = value[this._field?.selector?.title ?? 'title'][Lang.Current] ?? value[this._field?.selector?.title ?? 'title'];
136                             } catch(e) {}
137                         }
138                     });
139                 }
140                 else if(this._field.values) {
141                     for(const v of this._field.values) {
142                         if((v.value ?? v.title) == (value.value ?? value.title ?? value)) {
143                             super.value = v.title[Lang.Current] ?? v.title;
144                         }
145                     }
146                 } 
147             }
148 
149                
150             
151         }
152         else {
153             super.value = '';
154         }
155 
156         if(!super.value) {
157             super.value = '—';
158         }
159 
160     }
161 
162     _getDependsValue() {
163         if (this.root && this._field?.lookup &&
164             this._field.lookup['depends']) {
165 
166             let dependsField = this._field.lookup['depends'],
167                 rootValues = this.root.value;
168             if (dependsField) {
169                 if(eval(`typeof rootValues.${dependsField}`) !== 'undefined') {
170                     return eval(`rootValues.${dependsField}`);
171                 }
172                 return null;
173             }
174         }
175     }
176 
177     /**
178      * Установить новое значение свойству lookup
179      * Загрузить значения селектора альтернативным способом, указанным в lookup
180      */
181      _setLookup(value) {
182         let lookupPromise;
183         let dependsValue = this._getDependsValue();
184 
185         if(dependsValue !== undefined && !dependsValue) {
186             return new Promise((resolve, reject) => {
187                 resolve({});
188             });
189         }
190 
191         if (typeof this._field.lookup == 'function' || typeof this._field.lookup == 'string') {
192             if(typeof this._field.lookup == 'string') {
193                 this._field.lookup = eval(this._field.lookup);
194             }
195             const lookupMethodRun = this._field.lookup();
196             lookupPromise = lookupMethodRun instanceof Promise ? lookupMethodRun : new Promise((resolve, reject) => {
197                 resolve({
198                     result: this._field.lookup()
199                 });
200             });
201         }
202         else if (typeof this._field.lookup == 'object') {
203 
204             if(this._field.lookup?.method) {
205                 let lookupMethod = this._field.lookup.method;
206                 if (typeof lookupMethod == 'string') {
207                     lookupMethod = eval(this._field.lookup.method);
208                 }
209                 lookupPromise = lookupMethod('', dependsValue);
210             }
211             else if(this._field.lookup?.binding) {
212                 let binding = this._field.lookup.binding;
213                 if (typeof binding == 'string') {
214                     lookupPromise = App.Store.AsyncQuery(binding, dependsValue);
215                 }
216             }
217             else if(this._field.lookup?.controller) {
218                 let controller = this._field.lookup.controller;
219                 let module = eval(controller.module);
220                 lookupPromise = module.Call(controller.class, controller.method, {term: '', param: dependsValue, lookup: this._field.lookup});
221             }
222             else if(this._field.lookup?.storage) {
223                 let controller = this._field.lookup?.storage?.controller;
224                 let module = eval(controller?.module);
225                 lookupPromise = module.Call(controller.class, controller.method, {term: '', param: dependsValue, lookup: this._field.lookup});
226             }
227             else {
228                 lookupPromise = new Promise((resolve, reject) => { resolve({result: ''}); })
229             }
230         }
231 
232         // каждый метод должен возвращать промис
233         return lookupPromise;
234     }
