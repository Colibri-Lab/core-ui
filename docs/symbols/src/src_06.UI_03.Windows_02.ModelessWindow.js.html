  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {HTMLElement|Colibri.UI.Component} container container of component

 17     constructor(name, container, element, title, width, height) {
 18         super(name, container, Colibri.UI.Templates['Colibri.UI.ModelessWindow']);
 19         this.AddClass('app-component-modeless-window');
 20         this.styles = {position: 'absolute'};
 21 
 22         this.GenerateChildren(element);
 23 
 24         this._setTitle(title);
 25 
 26         if (width) { this._setWidth(width); }
 27         if (height) { this._setHeight(height); }
 28 
 29         this._closable = true;
 30         this._isDragged = false;
 31         this._disposeOnClose = false;
 32 
 33         this._sticky = false;
 34         this._stickyX = null;
 35         this._stickyY = null;
 36 
 37         this._startPointX = null;
 38         this._startPointY = null;
 39 
 40         this._getCloseButton().shown = this._closable;
 41 
 42         this.Dispatch('WindowContentRendered');
 43         this._handleEvents();
 44     }

 46     /** @protected */
 47     _registerEvents() {
 48         super._registerEvents();
 49         this.RegisterEvent('WindowClosed', false, 'Поднимается, когда окно закрылось');
 50         this.RegisterEvent('WindowContentRendered', false, 'Когда содержание окна отрисовалось');
 51     }
 52 
 53     /** @protected */
 54     _handleEvents() {
 55         this._getCloseButton().AddHandler('Clicked', (event, args) => this.__close(event, args));
 56 
 57         this._element.querySelector('.modeless-window-header-container').addEventListener('mousedown', this.__dragStart.bind(this));
 58         this._container.addEventListener('mousemove', this.__move.bind(this));
 59         document.addEventListener('mouseup', this.__dragStop.bind(this));
 60 
 61         this.AddHandler('Resize', (event, args) => {
 62             this._toggleBodyScroll(false);
 63             this._updateStyleVariables();
 64         });
 65 
 66         this.AddHandler('Resized', (event, args) => { this._toggleBodyScroll(true); });
 67     }

 72     /**
 73      * @private

 77     __close(event, args) {
 78         if (this._closable) {
 79             this.shown = false;
 80             if (this._disposeOnClose) { this.Dispose(); }
 81 
 82             this.Dispatch('WindowClosed', {});
 83         }
 84     }

 86     /**
 87      * @private
 88      */
 89     __dragStart(event) {
 90         if (!this._sticky) {
 91             let previousClientX = event.clientX,
 92                 previousClientY = event.clientY;
 93 
 94             let previousLeft = this.styles.left ? parseInt(this.styles.left) : 0,
 95                 previousTop = this.styles.top ? parseInt(this.styles.top) : 0;
 96 
 97             this.tag._diffX = previousClientX - previousLeft;
 98             this.tag._diffY = previousClientY - previousTop;
 99             this._isDragged = true;
100         }
101     }
102 
103     /**
104      * @private
105      */
106     __move(event) {
107         if (!this._sticky && this._isDragged) {
108             let newClientX = event.clientX,
109                 newClientY = event.clientY,
110                 newLeft = newClientX - this.tag._diffX,
111                 newTop = newClientY - this.tag._diffY;
112 
113             this._element.style.top = newTop + 'px';
114             this._element.style.left = newLeft + 'px';
115         }
116     }
117 
118     /**
119      * @private
120      */
121     __dragStop() {
122          if (!this._sticky && this._isDragged) {
123              this._container.removeEventListener('mousemove', this.__move);
124              document.removeEventListener('mouseup', this.__dragStop);
125 
126              this._isDragged = false;
127          }
128     }

130     /**
131      * @private
132      */
133     _updateStyleVariables() {
134         let bounds = this._element.bounds();
135         this._element.style.setProperty('--windowWidth', bounds.width + 'px');
136         this._element.style.setProperty('--windowHeight', bounds.height + 'px');
137     }

139     /**
140      * @private
141      */
142     _toggleBodyScroll(value) {
143         const scrolling = new Colibri.Common.Scrolling(document.body);
144         if (value === false) {
145             scrolling.Disable();
146         } else {
147             scrolling.Enable();
148         }
149     }

172     /** @private */
173     _setSticky(value) {
174         this._sticky = (value === true || value === 'true');
175 
176         if (this._sticky && !this._stickyX && !this._stickyY) {
177             this._stickToX('center');
178             this._stickToY('center');
179         }
180         if(value) {
181             this._updateStyleVariables();
182         }

199     /** @private */
200     _stickToX(value) {
201         this._stickyX = value;
202 
203         switch (value) {
204             case 'left':
205             case 'right':
206             case 'center':
207                 this.RemoveClass('-stick-to-x-left');
208                 this.RemoveClass('-stick-to-x-right');
209                 this.RemoveClass('-stick-to-x-center');
210 
211                 this.AddClass('-stick-to-x-' + value);
212                 break;
213             default:
214                 this._stickyX = null;
215                 this.RemoveClass('-stick-to-x-left');
216                 this.RemoveClass('-stick-to-x-right');
217                 this.RemoveClass('-stick-to-x-center');
218         }
219     }

265     /**
266      * The point from which the window will occupy the entire available width depending on stickyX 
267      * if stickyX is the center, it will occupy such a width so that the window is centered
268      * @type {string|number|null}
269      */
270     set startPointX(value) {
271         this._setStartPointX(value);
272     }
273     /** @private */
274     _setStartPointX(value) {
275         if (![null, false, undefined].includes(value)) {
276 
277             this._startPointX = parseInt(value);
278             let measure = '';
279             if (typeof value === 'string' && value.indexOf('%') !== -1) {
280                 measure = '%';
281             }
282             else {
283                 measure = 'px';
284             }
285 
286             switch (this._stickyX) {
287                 case 'right':
288                 case 'left':
289                     this._setWidth('calc(100vw - ' + this._startPointX + measure + ')');
290                     break;
291                 case 'center':
292                     this._setWidth('calc(100vw - '+ (this._startPointX * 2) + measure +')');
293                     break;
294                 default:
295                     this._startPointX = null;
296             }
297 
298             this._startPointX += (this._startPointX && measure === '%') ? '%' : '';
299         }
300         else {
301             this._startPointX = null;
302         }
303     }

514         super.shown = value;
515         if(value) {
