  3 /**
  4  * Button with upload support
  5  */
  6 Colibri.UI.UploadButton = class extends Colibri.UI.ExtendedButton {
  7     
  8     constructor(name, container) {
  9         super(name, container);
 10         this.AddClass('app-success-button-component');
 11         this.AddClass('app-upload-button-component'); 
 12 
 13         this._allowSize = 900000000;
 14         this._allowTypes = '*';
 15         
 16         this._uniqueString = Number.unique();
 17         this._input = Element.create('input', {type:'file', id: 'component-' + name + '-' + this._uniqueString});
 18         this._element.append(this._input);
 19 
 20         this.AddHandler('Clicked', (event, args) => this.__clicked(event, args));
 21         this._input.addEventListener('change', (event) => {
 22             this._checkChoosedFiles(this._input.files);
 23             this._input.value = '';
 24             event.preventDefault();
 25             event.stopPropagation();
 26             return false;
 27         });
 28 
 29     }
 30 
 31     _registerEvents() {
 32         super._registerEvents();
 33 
 34         this.RegisterEvent('Changed', false, 'Когда выбраны файлы');
 35 
 36     }
 37 
 38     set multiple(value) {
 39         if(value) {
 40             this._input.attr('multiple', 'multiple');
 41         }
 42         else {
 43             this._input.attr('multiple', null);
 44         }
 45     }
 46 
 47     get multiple() {
 48         return this._input.attr('multiple') === 'multiple';
 49     }
 50 
 51     set allowTypes(value) {
 52         if(typeof value === 'string') {
 53             value = value.split(',');
 54         }
 55         this._allowTypes = value.map(v => v.toLowerCase());
 56         this._setAccept();
 57     }
 58 
 59     get allowTypes() {
 60         return this._allowTypes;
 61     }
 62 
 63     set allowSize(value) {
 64         this._allowSize = parseInt(value);
 65     }
 66 
 67     get allowSize() {
 68         return this._allowSize;
 69     }
 70 
 71     __clicked(event, args) {
 72         this.ClickOnButton();
 73     }
 74     
 75     ClickOnButton() {
 76         this._input.click();
 77     }
 78 
 79     _checkChoosedFiles(files) {
 80         let errors = [];
 81         let success = [];
 82         for(const file of files) {
 83             const ext = file.name.extractExt();
 84             if(file.size > this._allowSize || (this._allowTypes !== '*' && this._allowTypes.indexOf(ext) === -1)) {
 85                 errors.push({file: file, error: '#{ui-uploadbutton-error1} ' + (this._allowTypes !== '*' ? '#{ui-uploadbutton-allow}'.replaceAll('%s', this._allowTypes.join(',')) + ', ' : '') + '#{ui-uploadbutton-max}'.replaceAll('%s', this._allowSize.toSizeString(['bytes', 'Kb', 'Mb', 'Gb'], 1024, true))});
 86             }
 87             else {
 88                 success.push(file);
 89             }
 90         }
 91         this.Dispatch('Changed', {errors: errors, success: success});
 92     }
 93     
 94     _setAccept() {
 95         let ret = [];
 96         for(const ext of this._allowTypes) {
 97             ret.push(Colibri.Common.MimeType.ext2type(ext));
 98         }    
 99         this._input.attr('accept', ret.join(', '));
100     }
101 
102 }
