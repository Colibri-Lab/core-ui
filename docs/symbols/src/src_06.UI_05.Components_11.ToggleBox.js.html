  2 Colibri.UI.ToggleBox = class extends Colibri.UI.Component {
  3 
  4     constructor(name, container) {
  5         super(name, container, Element.create('div'));
  6 
  7         this.AddClass('app-togglebox-component');
  8 
  9         this._element.append(Element.create('input', {type: 'hidden', value: 'on'}, {}));
 10         const span = Element.create('span', {class: 'toggle'});
 11         span.append(Element.create('span', {class: 'box'}));
 12         this._element.append(span);
 13         this._element.append(Element.create('span', {class: 'text'}, {}));
 14 
 15         
 16         this._input = this._element.querySelector('input');
 17         this._showState();    
 18 
 19     }
 20 
 21     _registerEvents() {
 22         super._registerEvents();
 23         this.RegisterEvent('Changed', false, 'Когда изменилось состояние переключателя');
 24     }
 25 
 26     _showState() {
 27         if(this._input.checked) {
 28             this.AddClass('-selected');
 29         }
 30         else {
 31             this.RemoveClass('-selected');
 32         }
 33     }
 34 
 35     _registerEventHandlers() {
 36         super._registerEventHandlers();
 37  
 38         this.AddHandler('Clicked', () => {
 39             this.Toggle();
 40         });
 41 
 42     }
 43 
 44     Toggle() {
 45         if(!this.enabled) {
 46             return;
 47         }
 48         this._input.checked = !this._input.checked;
 49         this._showState(); 
 50         this.Dispatch('Changed', {state: this._input.checked});
 51     }
 52 
 53     get checked() {
 54         return this._input.checked;
 55     }
 56 
 57     set checked(value) {
 58         this._input.checked = value === 'true' || value === true;
 59         this._showState();
 60     }
 61 
 62     get label() {
 63         return this._element.querySelector('.text').html();
 64     } 
 65 
 66     set label(value) {
 67         this._element.querySelector('.text').html(value);
 68     }

 70 }
