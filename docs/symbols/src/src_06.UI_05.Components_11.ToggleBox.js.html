1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.ToggleBox = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.create('div'));
 15 
 16         this.AddClass('app-togglebox-component');
 17 
 18         this._element.append(Element.create('input', {type: 'hidden', value: 'on'}, {}));
 19         const span = Element.create('span', {class: 'toggle'});
 20         span.append(Element.create('span', {class: 'box'}));
 21         this._element.append(span);
 22         this._element.append(Element.create('span', {class: 'text'}, {}));
 23 
 24         
 25         this._input = this._element.querySelector('input');
 26         this._showState();    
 27 
 28     }
 29 
 30     /** @protected */
 31     _registerEvents() {
 32         super._registerEvents();
 33         this.RegisterEvent('Changed', false, 'Когда изменилось состояние переключателя');
 34     }
 35 
 36     /** @private */
 37     _showState() {
 38         if(this._input.checked) {
 39             this.AddClass('-selected');
 40         }
 41         else {
 42             this.RemoveClass('-selected');
 43         }
 44     }
 45 
 46     /** @protected */
 47     _registerEventHandlers() {
 48         super._registerEventHandlers();
 49  
 50         this.AddHandler('Clicked', () => {
 51             this.Toggle();
 52         });
 53 
 54     }
 55 
 56     /**
 57      * Toggle the box
 58      */
 59     Toggle() {
 60         if(!this.enabled) {
 61             return;
 62         }
 63         this._input.checked = !this._input.checked;
 64         this._showState(); 
 65         this.Dispatch('Changed', {state: this._input.checked});
 66     }
 67 
 68     /**
 69      * Checked
 70      * @type {boolean}
 71      */
 72     get checked() {
 73         return this._input.checked;
 74     }
 75 
 76     /**
 77      * Checked
 78      * @type {boolean}
 79      */
 80     set checked(value) {
 81         this._input.checked = value === 'true' || value === true;
 82         this._showState();
 83     }
 84 
 85     /**
 86      * Label
 87      * @type {string}
 88      */
 89     get label() {
 90         return this._element.querySelector('.text').html();
 91     } 
 92 
 93     /**
 94      * Label
 95      * @type {string}
 96      */
 97     set label(value) {
 98         this._element.querySelector('.text').html(value);
 99     }
100 
101 }
