1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Switcher = class extends Colibri.UI.Component {
  7     /**
  8      * @constructor
  9      * @param {string} name name of component
 10      * @param {Element|Colibri.UI.Component} container container of component
 11      * @param {*} data data to swith
 12      */    
 13     constructor(name, container, data) {
 14         super(name, container, Element.create('div'));
 15 
 16         this.AddClass('app-component-switcher');
 17 
 18         const d = data ? data : [{title: '—', data: {}}];
 19 
 20         this.infinitySwitching = false;
 21 
 22         this._text = new Colibri.UI.TextSpan('text', this, d[0].title);
 23         this._text.AddClass('text');
 24         this._text.shown = true;
 25 
 26         this._prevButton = new Colibri.UI.Icon('prev-button', this);
 27         this._prevButton.AddClass('prev-button');
 28         this._prevButton.value = Colibri.UI.LeftArrowIcon;
 29         this._prevButton.shown = true;
 30 
 31         this._nextButton = new Colibri.UI.Icon('next-button', this);
 32         this._nextButton.AddClass('next-button');
 33         this._nextButton.value = Colibri.UI.RightArrowIcon;
 34         this._nextButton.shown = true;
 35 
 36         this.value = d;
 37         this.itemIndex = 0;
 38 
 39         this._prevButton.AddHandler('Clicked', () => {
 40             if(this._prevButton.ContainsClass('disabled')) {
 41                 return false;
 42             }
 43             this.itemIndex--;
 44             this.Dispatch('ButtonClicked', {button: 'prevButton', itemIndex: this.itemIndex});
 45             this.Dispatch('Changed', this.value[this._itemIndex]);
 46         });
 47 
 48         this._nextButton.AddHandler('Clicked', () => {
 49             if(this._nextButton.ContainsClass('disabled')) {
 50                 return false;
 51             }
 52             this.itemIndex++;
 53             this.Dispatch('ButtonClicked', {button: 'nextButton', itemIndex: this.itemIndex});
 54             this.Dispatch('Changed', this.value[this._itemIndex]);
 55         });
 56 
 57     }
 58 
 59     /** @protected */
 60     _registerEvents() {
 61         this.RegisterEvent('Changed', false, 'Поднимается когда происходи изменение');
 62         this.RegisterEvent('ButtonClicked', false, 'Поднимается когда происходит нажатие на кнопку пред/след');
 63     }
 64 
 65     /**
 66      * Selected item value
 67      * @type {any}
 68      */
 69     get itemValue() {
 70         try {
 71             return this.value[this._itemIndex].value;
 72         } catch(e) {
 73             return null;
 74         }
 75     }
 76     /**
 77      * Selected item value
 78      * @type {any}
 79      */
 80     set itemValue(value) {
 81         let itemIndex = Array.findIndex(this._items, v => v?.value === value);
 82         if(itemIndex === -1) {
 83             itemIndex = 0;
 84         }
 85         this.itemIndex = itemIndex;
 86     }
 87 
 88 
 89     /**
 90      * Selected item index
 91      * @type {Number}
 92      */
 93     get itemIndex() {
 94         return this._itemIndex;
 95     }
 96 
 97     /**
 98      * Selected item index
 99      * @type {Number}
100      */
101     set itemIndex(value) {
102         if (value >= 0 && value <= (this._items.length - 1)) {
103             this._itemIndex = value;
104         } else {
105             this._itemIndex = 0;
106         }
107         this._showItemIndex();
108     }
109 
110     /** @private */
111     _showItemIndex() {
112         if(!this._items) {
113             return ;
114         }
115         this._prevButton.RemoveClass('disabled');
116         this._nextButton.RemoveClass('disabled');
117         this._text.value = this.value[this._itemIndex].title;
118         if (!this._infinitySwitching) {
119             if (this._itemIndex === 0) {
120                 this._prevButton.AddClass('disabled');
121             } else if (this._itemIndex === (this._items.length - 1)) {
122                 this._nextButton.AddClass('disabled');
123             } else {
124                 this._prevButton.RemoveClass('disabled');
125                 this._nextButton.RemoveClass('disabled');
126             }
127         }
128 
129     }
130 
131     /**
132      * @deprecated
133      */
134     get items() {
135         return this.value;
136     }
137 
138     /**
139      * @deprecated
140      */
141     set items(value) {
142         this.value = value;
143     }
144 
145     /**
146      * Value Array
147      * @type {Array}
148      */
149     get value() {
150         return this._items;
151     }
152     /**
153      * Value Array
154      * @type {Array}
155      */
156     set value(value) {
157         this._items = value;
158         this._showValue();
159     }
160     /** @private */
161     _showValue() {
162         this.itemIndex = 0;
163     }
164 
165     /**
166      * Indicates when the swithing process is cycled
167      * @type {Boolean}
168      */
169     get infinitySwitching() {
170         return this._infinitySwitching;
171     }
172     /**
173      * Indicates when the swithing process is cycled
174      * @type {Boolean}
175      */
176     set infinitySwitching(value) {
177         this._infinitySwitching = value === 'true' || value === true;
178         if (value) {
179             this._prevButton.RemoveClass('disabled');
180             this._nextButton.RemoveClass('disabled');
181         }
182     }
183 
184     /**
185      * Left arrow icon
186      * @type {String}
187      */
188     get leftArrowIcon() {
189         return this._prevButton.iconSVG;
190     }
191     /**
192      * Left arrow icon
193      * @type {String}
194      */
195     set leftArrowIcon(value) {
196         this._prevButton.iconSVG = value;
197     }
198 
199     /**
200      * Right arrow icon
201      * @type {String}
202      */
203     get rightArrowIcon() {
204         return this._nextButton.iconSVG;
205     }
206     /**
207      * Right arrow icon
208      * @type {String}
209      */
210     set rightArrowIcon(value) {
211         this._nextButton.iconSVG = value;
212     }
213     
214 
215 }
