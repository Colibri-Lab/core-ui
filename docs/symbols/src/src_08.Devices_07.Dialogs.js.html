1 /**
  2  * Represents a utility for displaying dialogs and notifications.
  3  * @class
  4  * @extends Destructable
  5  */
  6 Colibri.Devices.Dialogs = class extends Destructable {
  7     
  8     /**
  9      * Instance variable representing the device.
 10      * @type {Colibri.UI.Device}
 11      * @private
 12      */
 13     _device = null;
 14     /**
 15      * Instance variable representing the notification plugin.
 16      * @type {object}
 17      * @private
 18      */
 19     _plugin = null;
 20     /**
 21      * Flag indicating whether permission is granted.
 22      * @type {boolean}
 23      * @private
 24      */
 25     _permited = false;
 26 
 27     /**
 28      * Creates an instance of Dialogs.
 29      * @param {*} device - The device object.
 30      */
 31     constructor(device) {
 32         super();
 33         this._device = device;
 34         this._plugin = this._device.Plugin('notification');
 35     }
 36 
 37     /**
 38      * Displays an alert dialog.
 39      * @param {string} message - The message to display.
 40      * @param {string} title - The title of the alert dialog.
 41      * @param {string} buttonName - The label for the button.
 42      * @returns {Promise} - Promise resolving when the alert is dismissed.
 43      */
 44     Alert(message, title, buttonName) {
 45         return new Promise((resolve, reject) => {
 46             this._plugin.alert(
 47                 message,
 48                 () => {
 49                     resolve();
 50                 },
 51                 title,
 52                 buttonName
 53             );
 54         });
 55     }
 56 
 57     /**
 58      * Displays a confirmation dialog.
 59      * @param {string} message - The message to display.
 60      * @param {string} title - The title of the confirmation dialog.
 61      * @param {string[]} buttonLabels - Labels for the buttons.
 62      * @returns {Promise} - Promise resolving with the index of the selected button.
 63      */
 64     Confirm(message, title, buttonLabels = ['Ok', 'Cancel']) {
 65         return new Promise((resolve, reject) => {
 66             this._plugin.confirm(
 67                 message,
 68                 (buttonIndex) => {
 69                     resolve(buttonIndex);
 70                 },
 71                 title,
 72                 buttonLabels
 73             );
 74         });
 75     }
 76  
 77     /**
 78      * Displays a prompt dialog.
 79      * @param {string} message - The message to display.
 80      * @param {string} title - The title of the prompt dialog.
 81      * @param {string[]} buttonLabels - Labels for the buttons.
 82      * @param {string} defaultText - The default text in the input field.
 83      * @returns {Promise} - Promise resolving with the entered text and button index.
 84      */
 85     Prompt(message, title, buttonLabels = ['Ok', 'Cancel'], defaultText = '') {
 86         return new Promise((resolve, reject) => {
 87             this._plugin.confirm(
 88                 message,
 89                 (results) => {
 90                     resolve(results);
 91                 },
 92                 title,
 93                 buttonLabels
 94             );
 95         });
 96     }
 97 
 98     /**
 99      * Emits a beep sound.
100      * @param {number} times - The number of times to beep.
101      */
102     Beep(times) {
103         this._plugin.beep(times);
104     }
105 
106     /**
107      * Schedules a local notification.
108      * @param {string} title - The title of the notification.
109      * @param {string} message - The message of the notification.
110      * @param {object} trigger - The trigger object defining when to fire the notification.
111      * @param {boolean} isForeground - Whether the app is in the foreground when notification fires.
112      * @param {boolean} isLaunch - Whether the app launches when notification is tapped.
113      * @param {number} priority - The priority of the notification.
114      */
115     Schedule(title, message, trigger, isForeground = true, isLaunch = true, priority = 2) {
116         // trigger = { in: 1, unit: 'second' }, { in: 15, unit: 'minutes' }
117         this._plugin.local.schedule({
118             title: title,
119             text: message,
120             trigger: trigger,
121             foreground: isForeground,
122             launch: isLaunch,
123             priority: priority,
124         });
125     }
126 
127 }
