1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.DropDown = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {HTMLElement|Colibri.UI.Component} container container of component 
 12      */
 13     constructor(name, container) {
 14         super(name, container);
 15         this.AddClass('app-dropdown-component');
 16 
 17         this.handleClickedOut = true;
 18 
 19         this._search = new Colibri.UI.Input('search', this);
 20         this._list = new Colibri.UI.List('list', this);
 21         this._list.__renderItemContent = this.__renderItemContent;
 22 
 23         this._search.placeholder = "Поиск по списку компаний";
 24         this._search.loading = false;
 25         this._search.iconSVG = Colibri.UI.SearchIcon;
 26 
 27         this._list.shown = true;
 28         this._search.shown = true;
 29 
 30         this._list.AddHandler('ItemClicked', (event, args) => this.Dispatch('ItemClicked', {item: args.item, domEvent: args.domEvent}));
 31 
 32         this.AddHandler('Shown', (event, args) => {
 33             this._search.Focus();
 34         });
 35 
 36         this.AddHandler('ClickedOut', (event, args) => {
 37             this.Hide();
 38         });
 39 
 40 
 41     }
 42 
 43     _registerEvents() {
 44         super._registerEvents();
 45         this.RegisterEvent('ItemClicked', false, 'Выбран пунт меню');
 46     }
 47 
 48     set search(value) {
 49         this._search.shown = value;
 50     }
 51     get search() {
 52         return this._search.shown;
 53     }
 54 
 55     set shown(value) {
 56         super.shown = value;
 57         if(value) {
 58             this.BringToFront();
 59         }
 60         else {
 61             this.SendToBack();
 62         }
 63     }
 64 
 65 
 66 }
