156 /**
157  * Represents a utility for handling local notifications.
158  * @class
159  * @extends Destructable
160  */
161 Colibri.Devices.LocalNotifications = class extends Destructable {
162     
163     /**
164      * Instance variable representing the device.
165      * @type {Colibri.UI.Device}
166      * @private
167      */
168     _device = null;
169     /**
170      * Instance variable representing the plugin.
171      * @type {object}
172      * @private
173      */
174     _plugin = null;
175     /**
176      * Flag indicating whether permission is granted.
177      * @type {boolean}
178      * @private
179      */
180     _permited = false;
181 
182     /**
183      * Creates an instance of LocalNotifications.
184      * @param {*} device - The device object.
185      */
186     constructor(device) {
187         super();
188         this._device = device;
189         if(this._device.isWeb) {
190             this._plugin = {local: new Colibri.Devices.LocalNotificationsEmulator()};
191         } else {
192             this._plugin = this._device.Plugin('plugins.notification');
193         }
194         this._plugin.local.setDummyNotifications();
195         this._plugin.local.fireQueuedEvents();
196 
197     }
198 
199     /**
200      * Checks if permission is granted.
201      * @returns {Promise} - Promise resolving when permission is granted.
202      */
203     HasPermission() {
204         return new Promise((resolve, reject) => {
205             if(this._granted) {
206                 resolve();
207             } else {
208                 this._plugin.local.hasPermission((granted) => {
209                     this._granted = granted;
210                     if(granted) {
211                         resolve();
212                     } else {
213                         reject();
214                     }
215                 });
216             }
217         });
218     }
219 
220     /**
221      * Requests permission for notifications.
222      * @returns {Promise} - Promise resolving when permission is granted.
223      */
224     RequestPermission() {
225         return new Promise((resolve, reject) => {
226             if(this._granted) {
227                 resolve();
228             } else {
229                 this.HasPermission().catch(() => {
230                     this._plugin.local.requestPermission(function (granted) {
231                         this._granted = granted;
232                         if(granted) {
233                             resolve();
234                         } else {
235                             reject();
236                         }
237                     });    
238                 }).then(() => resolve());
239             }
240         });
241     }
242 
243     /**
244      * Adds actions to a notification group.
245      * @param {string} groupName - The name of the notification group.
246      * @param {*} actions - Actions to add.
247      */
248     AddActions(groupName, actions) {
249         this._plugin.local.addActions(groupName, actions);
250     }
251     /**
252      * Removes actions from a notification group.
253      * @param {string} groupName - The name of the notification group.
254      */
255     RemoveActions(groupName) {
256         this._plugin.local.removeActions(groupName);
257     }
258 
259     /**
260      * Schedules a notification with given parameters.
261      * @param {string} title - The title of the notification.
262      * @param {string} message - The message of the notification.
263      * @param {*} actions - Actions to attach to the notification.
264      * @param {*} trigger - Trigger for the notification.
265      * @param {boolean} isForeground - Flag indicating whether the notification should appear in foreground.
266      * @param {boolean} isLaunch - Flag indicating whether the notification should launch the app.
267      * @param {number} priority - Priority of the notification.
268      * @param {number} id - The id of the notification.
269      * @param {*} progressBar - Progress bar configuration.
270      * @param {boolean} sound - Flag indicating whether to include sound with the notification.
271      */
272     Schedule(title, message, actions = null, trigger = null, isForeground = true, isLaunch = true, priority = 2, id = null, progressBar = null, sound = true) {
273         // trigger = { in: 1, unit: 'second' }, { in: 15, unit: 'minutes' }
274         this.RequestPermission().then(() => {
275             const params = {
276                 title: title,
277                 text: message,
278                 foreground: isForeground,
279                 launch: isLaunch,
280                 priority: priority,
281                 sticky: true,
282                 sound: sound
283             }
284             if(id) {
285                 params.id = id;
286             }
287             if(trigger) {
288                 params.trigger = trigger;
289             }
290             if(actions && actions.length > 0) {
291                 params.actions = actions;
292             }
293             if(progressBar) {
294                 params.progressBar = progressBar;
295             }
296             
297             this._plugin.local.schedule(params);
298 
299         });
300     }
301 
302     /**
303      * Cancels a scheduled notification with given id.
304      * @param {number} id - The id of the notification to cancel.
305      */
306     Cancel(id) {
307         this.RequestPermission().then(() => {
308             this._plugin.local.cancel(id);    
309         });
310     }
311 
312     /**
313      * Registers an event listener.
314      * @param {string} event - The event to listen for.
315      * @param {function} callback - The callback function.
316      * @param {*} scope - The scope of the callback.
317      */
318     On(event, callback, scope) {
319         this._plugin.local.on(event, callback, scope);
320     }
321 
322     /**
323      * Unregisters an event listener.
324      * @param {string} event - The event to stop listening for.
325      * @param {function} callback - The callback function.
326      * @param {*} scope - The scope of the callback.
327      */
328     Off(event, callback, scope) {
329         this._plugin.local.un(event, callback, scope);
330     }
