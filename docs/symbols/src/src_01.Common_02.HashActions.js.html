1 /**
  2  * Represents a utility for handling hash actions in the browser.
  3  * @class
  4  * @memberof Colibri.Common
  5  */
  6 Colibri.Common.HashActions = class {
  7     
  8     
  9     /** @constructor */
 10     constructor() {
 11         this.handlers = {};
 12         this.init();
 13     }
 14     
 15     init() {
 16         this.__clickEvent = (e) => {
 17             this._handleAction(e.target.data('action').substring(1));
 18             e.preventDefault();
 19             e.stopPropagation();
 20             return false;
 21         };
 22     }
 23     
 24     /**
 25      * Handles the DOM ready event.
 26      */
 27     HandleDomReady() {
 28         this.InitDOMHandlers();
 29         this._handleAction(location.hash.substring(1));
 30     }
 31     
 32     /**
 33      * Initializes DOM event handlers.
 34      */
 35     InitDOMHandlers() {
 36 
 37         Colibri.Common.StartTimer('actions-timer', 500, () => {
 38 
 39             document.querySelectorAll('a[href*="#action"]').forEach((a) => {
 40                 a.data('action', a.attr('href'))
 41                 a.attr('href', 'javascript:void(0)');
 42             });
 43 
 44             document.querySelectorAll('[data-action]').forEach((element) => { 
 45                 element.removeEventListener('mousedown', this.__clickEvent);
 46                 element.addEventListener('mousedown', this.__clickEvent);
 47             }); 
 48     
 49         });
 50         
 51     }
 52     
 53     /**
 54      * Adds a handler for a hash action.
 55      * @param {string} action - The hash action.
 56      * @param {Function} handler - The handler function.
 57      */
 58     AddHandler(action, handler) {
 59         if(this.handlers[action] === undefined)
 60             this.handlers[action] = [];
 61         this.handlers[action].push(handler);
 62     }
 63 
 64     /**
 65      * Removes a handler for a hash action.
 66      * @param {string} action - The hash action.
 67      * @param {Function} handler - The handler function.
 68      */
 69     RemoveHandler(action, handler) {
 70         if(this.handlers[action] === undefined)
 71             this.handlers[action] = [];
 72         let index = this.handlers[action].indexOf(handler);
 73         if(index !== -1) {
 74             delete this.handlers[action][index];
 75         }
 76     }
 77     
 78     /**
 79      * Raises a hash action.
 80      * @param {string} action - The hash action.
 81      * @param {Object} [args={}] - Additional arguments.
 82      * @returns {boolean} - Indicates if the action was raised successfully.
 83      */
 84     Raise(action, args = {}) {
 85         try {
 86             if(this.handlers[action] === undefined)
 87                 return false;
 88             var handlers = this.handlers[action];
 89             for(const handler of handlers) {
 90                 if(!handler.apply(this, [action, args])) {
 91                     return false;
 92                 }
 93                 return true;
 94             }
 95         }
 96         catch(e) { console.log('no action handler ' + action + ', exception: ' + e); }
 97         
 98     }
 99     
100     /**
101      * Handles a hash action.
102      * @param {string} actionString - The hash action string.
103      * @private
104      */
105     _handleAction(actionString) {
106         
107         var queryString = actionString.toObject('&=');
108         if(queryString.action == undefined)
109             return  false;
110         
111         history.replaceState ? 
112             history.replaceState("", document.title, window.location.pathname + window.location.search) 
113                 :
114             history.pushState("", document.title, window.location.pathname + window.location.search);
115         
116         this.Raise(queryString.action, queryString);
117         
118     }
119     
120 }
