1 /**
  2  * Represents a utility for handling SMS operations.
  3  * @class
  4  * @extends Destructable
  5  */
  6 Colibri.Devices.Sms = class extends Destructable {
  7     
  8     /**
  9      * Instance variable representing the device.
 10      * @type {Colibri.UI.Device}
 11      * @private
 12      */
 13     _device = null;
 14     /**
 15      * Instance variable representing the notification plugin.
 16      * @type {object}
 17      * @private
 18      */
 19     _plugin = null;
 20     /**
 21      * Instance variable representing the SMS plugin for receiving messages.
 22      * @type {null}
 23      * @private
 24      */
 25     _permitedSend = false;
 26 
 27     /**
 28      * Flag indicating whether permission to send SMS is granted.
 29      * @type {boolean}
 30      * @private
 31      */
 32     constructor(device) {
 33         super();
 34         this._device = device;
 35         this._pluginSend = this._device.Plugin('sms');
 36         if(this._device.isAndroid) {
 37             this._pluginRead = this._device.Plugin('SMSReceive');
 38         } else {
 39             this._pluginRead = null;    
 40         }
 41         this.CheckPermissionForSend();
 42     }
 43 
 44     /**
 45      * Checks permission to send SMS.
 46      * @returns {Promise} - Promise resolving when permission is checked.
 47      */
 48     CheckPermissionForSend() {
 49         return new Promise((resolve, reject) => {
 50             if(this._device.isIOs) {
 51                 this._permitedSend = true;
 52                 resolve();
 53             } else {
 54                 this._pluginSend.hasPermission((hasPermission) => {
 55                     this._permitedSend = hasPermission;
 56                     if(this._permitedSend) {
 57                         resolve();
 58                     } else {
 59                         reject({error: 'Not set'});
 60                     }
 61                 }, (e) => { 
 62                     this._permitedSend = false; 
 63                     reject({error: e});
 64                 });
 65             }
 66         });
 67     }
 68 
 69     /**
 70      * Requests permission to send SMS.
 71      * @returns {Promise} - Promise resolving when permission is requested.
 72      */
 73     RequestPermissionForSend() {
 74         return new Promise((resolve, reject) => {
 75             this.CheckPermissionForSend().then(() => {
 76                 this._permitedSend = true;
 77                 resolve();
 78             }).catch((response) => {
 79                 if(response.error === 'Not set') {
 80                     this._pluginSend.requestPermission(() => {
 81                         this._permitedSend = true;
 82                         resolve();
 83                     }, (error) => {
 84                         this._permitedSend = false;
 85                         reject({error: error});
 86                     });
 87                 }
 88             });
 89         });
 90     }
 91 
 92     /**
 93      * Sends an SMS message.
 94      * @param {string} number - The recipient's phone number.
 95      * @param {string} message - The message content.
 96      * @param {string} intent - The intent for sending the message.
 97      * @returns {Promise} - Promise resolving when message is sent.
 98      */
 99     Send(number, message, intent = 'INTENT') {
100         return new Promise((resolve, reject) => {
101             this.RequestPermissionForSend().then(() => {
102                 this._pluginSend.send(number, message, {
103                     replaceLineBreaks: true, // true to replace \n by a new line, false by default
104                     android: {
105                         intent: intent
106                         //intent: '' // send SMS without opening any other app, require : android.permission.SEND_SMS and android.permission.READ_PHONE_STATE
107                     }
108                 }, () => {
109                     resolve();
110                 }, () => {
111                     reject();
112                 });        
113             }).catch(e => {
114                 // do nothing
115             });
116         });
117     }
118 
119     /**
120      * Callback function for SMS arrival.
121      * @param {*} message - The incoming SMS message.
122      */
123     _smsReceiverCallback(message) {
124         this._arriveCallback(message);
125     }
126 
127     /**
128      * Registers an event listener for incoming SMS messages.
129      * @param {function} listener - The listener function.
130      */
131     RegisterArriveListener(listener) {
132         document.addEventListener('onSMSArrive', (e) => {
133             listener(e);
134         });
135     }
136 
137     /**
138      * Starts watching for incoming SMS messages.
139      * @returns {Promise} - Promise resolving when watching is started.
140      */
141     Watch() {
142         if(!this._pluginRead) {
143             return;
144         }
145         return new Promise((resolve, reject) => {
146             this._pluginRead.startWatch((strSuccess) => {
147                 resolve(strSuccess);
148             }, (strError) => {
149                 reject(strError);
150             });    
151         });
152     }
153 
154     /**
155      * Stops watching for incoming SMS messages.
156      * @returns {Promise} - Promise resolving when watching is stopped.
157      */
158     Stop() {
159         if(!this._pluginRead) {
160             return;
161         }
162         return new Promise((resolve, reject) => {
163             this._pluginRead.stopWatch((strSuccess) => {
164                 resolve();
165             }, (strError) => {
166                 reject(strError);
167             });
168         });
169     }
170 
171 }
