1 /**
  2  * Represents a browser storage utility.
  3  * @class
  4  * @memberof Colibri.Common
  5  */
  6 Colibri.Common.BrowserStorage = class {
  7     /** @constructor */
  8     constructor() {}
  9 
 10     /**
 11      * Sets a value in the browser storage.
 12      * @param {string} name - The name of the item.
 13      * @param {*} [value=true] - The value to set. Defaults to true if not provided.
 14      */
 15     Set(name, value = true) {
 16         window.localStorage.setItem(name, value);
 17     }
 18 
 19     /**
 20      * Gets a value from the browser storage.
 21      * @param {string} name - The name of the item.
 22      * @returns {*} - The value retrieved from the storage.
 23      */
 24     Get(name) {
 25         return window.localStorage.getItem(name);
 26     }
 27 
 28     /**
 29      * Deletes an item from the browser storage.
 30      * @param {string} name - The name of the item to delete.
 31      */
 32     Delete(name) {
 33         window.localStorage.removeItem(name);
 34     }
 35 
 36     /**
 37      * Checks if an item exists in the browser storage.
 38      * @param {string} name - The name of the item to check.
 39      * @returns {boolean} - True if the item exists, false otherwise.
 40      */
 41     Has(name) {
 42         return !!this.Get(name);
 43     }
 44 };
