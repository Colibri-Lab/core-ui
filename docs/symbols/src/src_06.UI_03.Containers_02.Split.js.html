1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Split = class extends Colibri.UI.Component {
  7 
  8     static OrientationHorizontal = 'horizontal';
  9     static OrientationVertical = 'vertical';
 10 
 11     /** @type {string} */
 12     _orientation = Colibri.UI.Split.OrientationHorizontal;
 13 
 14     /**
 15      * @constructor
 16      * @param {string} name name of component
 17      * @param {HTMLElement|Colibri.UI.Component} container container of component 
 18      * @param {string|HTMLElement} element element to generate in
 19      * @param {orientation} orientation orientation of split (horizontal|vertical) 
 20      */
 21     constructor(name, container, element, orientation) {
 22         super(name, container, element || Element.create('div'));
 23 
 24         this.AddClass('app-component-split');
 25 
 26         this._left = Element.create('div', {class: 'app-component-split-left'});
 27         this._handler = Element.create('div', {class: 'app-component-split-handler'});
 28         this._right = Element.create('div', {class: 'app-component-split-right'});
 29         this._element.append(this._left);
 30         this._element.append(this._handler);
 31         this._element.append(this._right);
 32 
 33         this._hasHandle = true;
 34 
 35         this._orientation = orientation || Colibri.UI.Split.OrientationHorizontal;
 36         this.AddClass('app-component-split-' + this._orientation);
 37 
 38         
 39         const startResize = (e) => {
 40             this._resizing = true;
 41             Colibri.UI.Resizing = true;
 42             this._resizeData = this._left.bounds();
 43             this.AddClass('-resizing');
 44 
 45             // ставим на документ, чтобы точно перехватить        
 46             document.addEventListener("touchend", stopResize, {capture: true});
 47             document.addEventListener("mouseup", stopResize, {capture: true});
 48 
 49             document.addEventListener("touchmove", doResize, {capture: true});
 50             document.addEventListener("mousemove", doResize, {capture: true});
 51 
 52             this.Dispatch('SplitResizeStart');
 53 
 54             return false;
 55 
 56         };
 57 
 58         const stopResize = (e) => {
 59             e.preventDefault();
 60             e.stopPropagation();
 61             this.RemoveClass('-resizing');
 62         
 63             this._resizing = false;
 64             Colibri.UI.Resizing = false;
 65 
 66             document.removeEventListener("touchend", this.__resizeEnd, {capture: true});
 67             document.removeEventListener("mouseup", this.__resizeEnd, {capture: true});
 68     
 69             document.removeEventListener("touchmove", doResize, {capture: true});
 70             document.removeEventListener("mousemove", doResize, {capture: true});
 71 
 72             this.Dispatch('SplitResizeStop');
 73 
 74             return false;
 75 
 76         };
 77 
 78         const doResize = (e) => {
 79             if (this._resizing) {
 80                 e.preventDefault();
 81     
 82                 if(this._orientation == Colibri.UI.Split.OrientationHorizontal) {
 83                     this._left.css('flex', '0 0 ' + (e.pageX - this._resizeData.left) + 'px');
 84                 }
 85     
 86                 if(this._orientation == Colibri.UI.Split.OrientationVertical) {
 87                     this._left.css('flex', '0 0 ' + (e.pageY - this._resizeData.top) + 'px');
 88                 }
 89     
 90                 this.Dispatch('SplitResizing');
 91 
 92                 return false;
 93             }
 94         };
 95 
 96         this._handler.addEventListener("touchstart", startResize, false);
 97         this._handler.addEventListener("mousedown", startResize, false);
 98 
 99     }
100 
101     /** @protected */
102     _registerEvents() {
103         super._registerEvents();
104         this.RegisterEvent('SplitResizeStart', false, 'Когда начинаем изменять размер');
105         this.RegisterEvent('SplitResizing', false, 'В процессе изменения размера');
106         this.RegisterEvent('SplitResizeStop', false, 'Когда закончили изменять размер');
107     }
108 
109     /**
110      * Orienatation of split
111      * @type {vertical,horizontal}
112      */
113     get orientation() {
114         return this._orientation;
115     }
116     /**
117      * Orienatation of split
118      * @type {vertical,horizontal}
119      */
120     set orientation(value) {
121         this._orientation = value;
122         this.AddClass('app-component-split-' + this._orientation);
123     }
124 
125     /**
126      * Container element
127      * @type {Element}
128      */
129     get container() {
130         if(this.children == 0) {
131             return this._element.querySelector(':scope > .app-component-split-left');
132         }
133         else {
134             return this._element.querySelector(':scope > .app-component-split-right');
135         }
136     }
137 
138     /**
139      * Is split has resize handle
140      * @type {Boolean}
141      */
142     get hasHandle() {
143         return this._hasHandle;
144     }
145     /**
146      * Is split has resize handle
147      * @type {Boolean}
148      */
149     set hasHandle(value) {
150         this._hasHandle = value;
151         if (value === true || value === 'true') {
152             this._handler.style.display = '';
153         } else {
154             this._handler.style.display = 'none';
155         }
156     }
157     
158     /**
159      * Processes the children object
160      * @param {string|Element|Element[]} children children to generate
161      * @param {Element|Colibri.UI.Component} parent parent component or element
162      */
163     ProcessChildren(children, parent) {
164 
165         for (let i = 0; i < children.length; i++) {
166             const element = children[i];
167             const componentClass = this.CreateComponentClass(element);
168             if(componentClass) {
169                 let component = this.CreateComponent(componentClass, element, this);
170                 component && component.ProcessChildren(element.childNodes, component.container);
171             }
172         }
173 
174     }
175 
176     /**
177      * Right width
178      * @type {Number}
179      */
180     get rightWidth() {
181         return parseInt(this._right.css('width'));
182     }
183     /**
184      * Left width
185      * @type {Number}
186      */
187     get leftWidth() {
188         return parseInt(this._left.css('width'));
189     }
190 
191     /**
192      * Left width
193      * @type {Number}
194      */
195     set leftWidth(value) {
196         this._left.css('flex', '0 0 ' + value);
197     }
198 
199 }
