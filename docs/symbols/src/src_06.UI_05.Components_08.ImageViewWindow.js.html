  8     constructor(name, container, title) {
  9         super(name, container, '<div><div class="content-container"><div class="text-description-before"></div><img class="image-container" src="#" alt="#"></img><div class="text-description-after"></div></div><div class="buttons-container"><button class="app-component-image-view-window-close-button">Закрыть</button></div></div>', title);
 10         this.AddClass('app-component-image-view-window');

 29         });
 30     }
