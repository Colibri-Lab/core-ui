1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Component
  5  */
  6 Colibri.UI.Viewer = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element, root) {
 16         super(name, container, element || Element.create('span'));
 17         this.AddClass('app-viewer-component');
 18         this.root = root;
 19     }
 20 
 21     /** @private */
 22     _injectParams() {
 23         if(this._field?.params && this._field?.params?.className) {
 24             let className = this._field?.params?.className;
 25             if(typeof className === 'function') {
 26                 className = className(this, this.value);
 27             }
 28             this.AddClass(className);
 29         }
 30         if(this._field?.params?.fieldgenerator) {
 31             const f = eval(this._field?.params?.fieldgenerator);
 32             f(this._field);
 33         }
 34 
 35         if(this._field?.params?.copy) {
 36             this.copy = this._field?.params?.copy;
 37         }        
 38 
 39     }
 40 
 41     /**
 42      * Value object
 43      * @type {object}
 44      */
 45     get viewedObject() {
 46         return this._object;
 47     }
 48 
 49     /**
 50      * Value object
 51      * @type {object}
 52      */
 53     set viewedObject(value) {
 54         this._object = value;
 55     }
 56 
 57     /**
 58      * Field object
 59      * @type {object}
 60      */
 61     set field(value) {
 62         this._field = value;
 63         this._injectParams();
 64     }
 65 
 66     /**
 67      * Field object
 68      * @type {object}
 69      */
 70     get field() {
 71         return this._field;
 72     }
 73 
 74     /**
 75      * Value object
 76      * @type {object}
 77      */
 78     set value(value) {
 79         if(this.field?.params?.pre) {
 80             value = this.field?.params?.pre + ' ' + value;
 81         }
 82         if(this.field?.params?.post) {
 83             value = value + ' ' + this.field?.params?.post;
 84         }
 85         super.value = this._convertValue(value);
 86     }
 87 
 88     /**
 89      * Value object
 90      * @type {object}
 91      */
 92     get value() {
 93         return super.value;
 94     }
 95 
 96     /**
 97      * Root component
 98      * @type {Colibri.UI.Component}
 99      */
100     get root() {
101         return this._root;
102     }
103 
104     /**
105      * Root component
106      * @type {Colibri.UI.Component}
107      */
108     set root(value) {
109         this._root = value;
110     }
111 
112     /** @private */
113     _convertValue(value) {
114         
115         if(this.field?.params?.converter) {
116             const f = this.field.params?.converter;
117             value = f(value, this);
118         }
119 
120         return value;
121     }
122 
123     /**
124      * Registered viewers
125      * @private
126      * @static
127      * @type {Array}
128      */
129     static _registered = [];
130     /**
131      * Registers the viewer
132      * @static
133      * @param {string} name name of viewer
134      * @param {string} desc description of viewer
135      */
136     static Register(name, desc) {
137         Colibri.UI.Viewer._registered.push({value: name, title: desc});
138     }
139     /**
140      * Enums the registered viewers
141      * @static
142      * @returns Array
143      */
144     static Enum() {
145         return Colibri.UI.Viewer._registered;
146     }
147 }
