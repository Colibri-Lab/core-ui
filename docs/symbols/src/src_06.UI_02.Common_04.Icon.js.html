1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Icon = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.create('span'));
 15         this.AddClass('app-component-icon');
 16         
 17     }
 18 
 19     /**
 20      * Icon value
 21      * @type {string}
 22      */
 23     get icon() {
 24         return this._element.css('background-image');
 25     }
 26 
 27     /**
 28      * Icon value
 29      * @type {string}
 30      */
 31     set icon(value) {
 32         this._element.css('background-image', value);
 33     }
 34 
 35     /**
 36      * Icon as SVG value
 37      * @type {string}
 38      */
 39     set iconSVG(value) {
 40         if(!value) {
 41             return;
 42         }
 43 
 44         this._iconSVG = value;
 45         
 46         if(this._element.querySelector('svg')) {
 47             this._element.querySelector('svg').remove();
 48         }
 49         try {
 50             const iconc = eval(this._iconSVG);
 51             this._element.append(Element.fromHtml(iconc));
 52         } catch(e) {
 53             console.log('Unknown icon: ' + value); console.trace();
 54         }
 55     }
 56 
 57     /**
 58      * Icon as SVG value
 59      * @type {string}
 60      */
 61     get iconSVG() {
 62         return this._iconSVG;
 63     }
 64 
 65     /**
 66      * Fill color
 67      * @type {string}
 68      */
 69     set fill(value) {
 70         for(const el of this._element.querySelectorAll('path,circle,rect,polygon')) {
 71             el.attr('fill', value);
 72         }
 73     }
 74 
 75     /**
 76      * Fill color
 77      * @type {string}
 78      */
 79     get fill() {
 80         const el = this._element.querySelector('path,circle,rect,polygon');
 81         if(!el) {
 82             return null;
 83         }
 84         return el.attr('fill');
 85     }
 86 
 87     /**
 88      * Stroke color
 89      * @type {string}
 90      */
 91     set stroke(value) {
 92         for(const el of this._element.querySelectorAll('path,circle,rect,polygon')) {
 93             el.attr('stroke', value);
 94         }
 95     }
 96 
 97     /**
 98      * Stroke color
 99      * @type {string}
100      */
101     get stroke() {
102         const el = this._element.querySelector('path,circle,rect,polygon');
103         if(!el) {
104             return null;
105         }
106         return el.attr('stroke');
107     }
108 
109     /**
110      * Is dot icon
111      * @type {boolean}
112      */
113     get dot() {
114         return this._dot;
115     }
116 
117     /**
118      * Is dot icon
119      * @type {boolean}
120      */
121     set dot(value) {
122         this._dot = value === 'true' || value === true;
123         this._setDot();
124     }
125     /**
126      * @private
127      */
128     _setDot() {
129         if(this._dot) {
130             this._dotElement = Element.create('em', {class: 'app-component-icon-dot'});
131             this._element.append(this._dotElement);
132         }
133         else {
134             this._dotElement.remove();
135         }
136     }
137     
138 }
