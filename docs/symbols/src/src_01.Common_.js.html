1 /**
  2  * Represents common utility functions.
  3  * @namespace Colibri.Common
  4  */
  5 Colibri.Common = class {
  6 
  7     /** @constructor */
  8     constructor() {
  9         // Do nothing
 10     } 
 11 
 12     
 13     /**
 14      * Delays execution for a specified time.
 15      * @param {number} timeout - The time to wait in milliseconds.
 16      * @return {Promise} - A promise that resolves after the specified time.
 17      */
 18     static Delay(timeout) {
 19         return new Promise((resolve, reject) => setTimeout(() => resolve(), timeout));
 20     }
 21 
 22     /** @type {Object.<string, number>} */
 23     static _timers = {};
 24 
 25     /**
 26      * Starts a timer with the specified name.
 27      * @param {string} name - The name of the timer.
 28      * @param {number} timeout - The interval of the timer in milliseconds.
 29      * @param {Function} tickFunction - The function to execute on each tick of the timer.
 30      */
 31     static StartTimer(name, timeout, tickFunction) {
 32         if(Colibri.Common._timers[name]) {
 33             clearTimeout(Colibri.Common._timers[name]);
 34         }
 35         Colibri.Common._timers[name] = setInterval(tickFunction, timeout);
 36     }
 37 
 38     /**
 39      * Stops the timer with the specified name.
 40      * @param {string} name - The name of the timer to stop.
 41      */
 42     static StopTimer(name) {
 43         clearInterval(Colibri.Common._timers[name]);
 44         delete Colibri.Common._timers[name];
 45     }
 46 
 47     /**
 48      * Waits for a condition to be true.
 49      * @param {Function} action - The function to check the condition.
 50      * @param {number} [maxTimeout=0] - The maximum time to wait in milliseconds.
 51      * @param {number} [interval=100] - The interval to check the condition in milliseconds.
 52      * @returns {Promise} - A promise that resolves when the condition is true or the timeout occurs.
 53      */
 54     static Wait(action, maxTimeout = 0, interval = 100) {
 55         let waiting = 0;
 56         const _checkAction = (a, h) => {
 57             try {
 58                 if(a() || (maxTimeout && waiting >= maxTimeout)) {
 59                     h();
 60                 }
 61                 else {
 62                     Colibri.Common.Delay(interval).then(() => _checkAction(a, h));
 63                 }
 64             }
 65             catch(e) {
 66                 Colibri.Common.Delay(interval).then(() => _checkAction(a, h));
 67             } finally {
 68                 waiting += interval;
 69             }
 70         }
 71         
 72         return new Promise((resolve, reject) => {
 73 
 74             _checkAction(action, () => {
 75                 resolve();
 76             })
 77 
 78         });
 79         
 80     }
 81 
 82     /**
 83      * Waits for the document to be ready.
 84      * @return {Promise} - A promise that resolves when the document is ready.
 85      */
 86     static WaitForDocumentReady() {
 87         return Colibri.Common.Wait(() => document.readyState === 'complete');
 88     }
 89 
 90     /**
 91      * Waits for the body element to be available.
 92      * @return {Promise} - A promise that resolves when the body element exists.
 93      */
 94     static WaitForBody() {
 95         return Colibri.Common.Wait(() => document.body != null);
 96     }
 97 
 98     /**
 99      * Loads a script from a URL.
100      * @param {string} url - The URL of the script.
101      * @param {string} [id=null] - The ID to assign to the script element.
102      * @returns {Promise} - A promise that resolves with the ID of the loaded script.
103      */
104     static LoadScript(url, id = null) {
105 
106         if(document.getElementById(id)) {
107             return new Promise((resolve, reject) => {
108                 resolve(id);
109             });
110         }
111 
112         return new Promise((resolve, reject) => {
113             var script = document.createElement('script');
114             script.async = true;
115             script.defer = true;
116             script.src = url;
117             script.onload = () => {
118                 resolve(script.id);
119             };
120             script.onerror = (e) => {
121                 reject(e);
122             };
123             script.id = id ? id : 'script_' + (new Date()).getTime();
124             document.getElementsByTagName('head')[0].appendChild(script);
125         });
126 
127     }
128 
129     /**
130      * Loads styles from a URL.
131      * @param {string} url - The URL of the styles.
132      * @param {string} [id=null] - The ID to assign to the link element.
133      * @returns {Promise} - A promise that resolves with the ID of the loaded styles.
134      */
135      static LoadStyles(url, id = null) {
136 
137         if(document.getElementById(id)) {
138             return new Promise((resolve, reject) => {
139                 resolve(id);
140             });
141         }
142 
143         return new Promise((resolve, reject) => {
144             var link = document.createElement('link');
145             link.async = true;
146             link.defer = true;
147             link.rel = 'stylesheet';
148             link.href = url;
149             link.onload = () => {
150                 resolve(link.id);
151             };
152             link.onerror = (e) => {
153                 reject(e);
154             };
155             link.id = id ? id : 'link_' + (new Date()).getTime();
156             document.getElementsByTagName('head')[0].appendChild(link);
157         });
158 
159     }
160 
161     /**
162      * Executes a callback function for each item in an array with a delay between each execution.
163      * @param {Array} array - The array of items.
164      * @param {number} timeout - The delay between executions in milliseconds.
165      * @param {Function} callback - The callback function to execute for each item.
166      */
167     static Tick(array, timeout, callback) {
168         array.forEach((v, i) => {
169             Colibri.Common.Delay(i * timeout).then(() => {
170                 callback(v);
171             });
172         }); 
173     }
174 
175 }
