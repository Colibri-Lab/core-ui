1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.ObjectViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-object-viewer-component');
 18     }
 19 
 20     /** @private */ 
 21     _showValue() {
 22         let ret = [];
 23         if(Object.countKeys(this._field.fields) > 0) {
 24             Object.forEach(this._value, (name, value) => {
 25                 if(Object.isObject(value) && value.value && value.title) {
 26                     value = value.title;
 27                 }
 28     
 29                 if(this._field.fields[name]) {
 30                     const desc = Lang ? Lang.Translate(this._field.fields[name].desc) : this._field.fields[name].desc;
 31                     ret.push(desc + ': ' + value);
 32                 }
 33             });
 34             this._element.html(ret.join(' '));
 35         } else {
 36             this._element.html(JSON.stringify(this._value));
 37         }
 38     }
 39 
 40     /**
 41      * Value
 42      * @type {object}
 43      */
 44     get value() {
 45         return this._value;
 46     }
 47 
 48     /**
 49      * Value
 50      * @type {object}
 51      */
 52     set value(value) {
 53         value = this._convertValue(value);
 54         this._value = value;
 55         this._showValue();
 56     }
 57 
 58 
 59 }
 60 Colibri.UI.Viewer.Register('Colibri.UI.ObjectViewer', '#{ui-viewers-object}');
