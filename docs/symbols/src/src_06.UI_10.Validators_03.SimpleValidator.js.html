 14     _init(fields) {
 15         if(fields.length == 0) {
 16             return;
 17         }
 18         Object.forEach(fields, (name, field) => {
 19             field.field.params && (field.field.params.validated = 'not-validated-yet');
 20             field.AddHandler('Changed', (event, args) => this.__validateField(field));
 21             field.AddHandler('FieldsRendered', (event, args) => this._init(event.sender.Fields()));
 22             this._init(field.Fields ? field.Fields() : []);
 23         });

 26     __validateField(field) {
 27 
 28         field.field.params && (field.field.params.validated = 'success');
 29         field.message = '';
 30         field.RemoveClass('app-validate-error');
 31 
 32         if(!field.shown) {
 33             return;
 34         }
 35 
 36         const validate = field.field?.params?.validate;
 37         if(!validate) {
 38             return;
 39         }
 40 
 41         for(const v of validate) {
 42             const message = v.message instanceof Function ? v.message(field, this) : v.message;
 43             const method = typeof v.method !== 'function' ? eval(v.method) : v.method;
 44             if(!method(field, this)) {
 45                 field.field.params && (field.field.params.validated = 'error');
 46                 field.message = message;
 47                 field.AddClass('app-validate-error');
 48                 break;
 49             } else {
 50 
 51             }
 52         }

 56     Status(fields = null) {
 57 
 58         if(!fields) {
 59             fields = this._form.Fields();
 60         }
 61 
 62         if(!Array.isArray(fields)) {
 63             fields = Object.values(fields);
 64         }
 65 
 66         if(fields.length == 0) {
 67             return true;
 68         }
 69 
 70         for(const field of fields) {
 71             if(field.field?.params?.validate) {    
 72                 if(field.field.params && field.field.params.validated !== 'success') {
 73                     return false;
 74                 }
 75                 if(!this.Status(field.Fields ? field.Fields() : [])) {
 76                     return false;
 77                 }
 78             }
 79         }
 80 
 81         return true;

 83     }

125             return;

130             this.ValidateAll(field.Fields ? field.Fields() : []);

140             return;
