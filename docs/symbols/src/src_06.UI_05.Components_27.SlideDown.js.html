1 /**
  2  * @class
  3  * @extends Colibri.UI.Pane
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.SlideDown = class extends Colibri.UI.Pane {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      * @param {*} element element to create in
 13      */    
 14     constructor(name, container, element) {
 15         super(name, container, element);
 16         this.AddClass('app-component-slidedown-component');
 17         
 18         this.tabIndex = null;
 19         this.AddHandler('ShadowClicked', (event, args) => {
 20             this.Hide();
 21         });
 22 
 23     }
 24 
 25     Toggle() {
 26         this.__handlerClicked();    
 27     }
 28 
 29     __handlerClicked(event, args) {
 30         if(this.shown) {
 31             this.shown = false;
 32             this.handler.state = 'collapsed';
 33         } else {
 34             this.shown = true;
 35             this.handler.state = 'expanded';
 36         }
 37     }
 38 
 39 
 40     /**
 41      * Name of object with is the handler
 42      * @type {string}
 43      */
 44     get handler() {
 45         return this._handler;
 46     }
 47     /**
 48      * Name of object with is the handler
 49      * @type {string}
 50      */
 51     set handler(value) {
 52         if(!value && this._handler) {
 53             this._handler.ClearHandlers();
 54         }
 55         this._handler = value;
 56         if(typeof this._handler === 'string') {
 57             Colibri.Common.Wait(() => Colibri.UI.Find(this._handler)).then(() => {
 58                 this._handler = Colibri.UI.Find(this._handler);
 59                 if(this._handler) {
 60                     this._handler.AddHandler('Clicked', (event, args) => this.__handlerClicked(event, args));
 61                 }
 62             });
 63         } else {
 64             this._handler.AddHandler('Clicked', (event, args) => this.__handlerClicked(event, args));
 65         }
 66     }
 67 
 68 
 69 
 70 }
