1 /**
  2  * Button group
  3  * @class
  4  * @extends Colibri.UI.Component
  5  * @memberof Colibri.UI
  6  */
  7 Colibri.UI.ButtonGroup = class extends Colibri.UI.Component {
  8 
  9     /**
 10      * @constructor
 11      * @param {string} name name of component
 12      * @param {Element|Colibri.UI.Component} container container element and component
 13      * @param {Element|string} element element to generate childs 
 14      */
 15     constructor(name, container, element) {
 16         super(name, container, element || Element.create('div'));
 17         this.AddClass('app-buttongroup-component');
 18 
 19         this._selectedButton = null;
 20 
 21         this.AddHandler('Clicked', (event, args) => this.__thisClicked(event, args));
 22     }
 23 
 24     /**
 25      * @private
 26      * @param {Colibri.UI.Event} event event object 
 27      * @param {*} args event args 
 28      */
 29     __thisClicked(event, args) {
 30         const button = args.domEvent.target.tag('component').Closest(component => component.parent instanceof Colibri.UI.ButtonGroup);
 31         this.SelectButton(button);
 32     }
 33 
 34     /**
 35      * @protected
 36      */
 37     _registerEvents() {
 38         super._registerEvents();
 39         this.RegisterEvent('Changed', false, 'Поднимается, когда изменилась выбранная кнопка');
 40     }
 41 
 42     /**
 43      * Select the button by index or name
 44      * @param {string|number} button button index or name
 45      */
 46     SelectButton(button) {
 47         if(typeof button == 'string' || typeof button == 'number') {
 48             button = this.Children(button);
 49         }
 50         if(!button) {
 51             return;
 52         }
 53 
 54         const isSelected = button.ContainsClass('-selected');
 55         this.ForEach((name, button) => {
 56             button.RemoveClass('-selected');
 57         });
 58         
 59         this._selectedButton = button;
 60         this._selectedButton.AddClass('-selected');
 61 
 62         if(!isSelected) {
 63             this.Dispatch('Changed', {button: this._selectedButton, index: this.selectedIndex});
 64         }
 65 
 66     }
 67     
 68     /**
 69      * Adds a button to group
 70      * @param {string} name name of button
 71      * @param {string} title title of button
 72      * @returns {Colibri.UI.Button}
 73      */
 74     AddButton(name, title) {
 75         if(this.Children(name)) {
 76             return this.Children(name);
 77         }
 78         const button = new Colibri.UI.Button(name, this);
 79         button.value = (title[Lang.Current] ?? title);
 80         button.shown = true;
 81         return button;
 82     }
 83 
 84     /**
 85      * Selected button
 86      * @type {Colibri.UI.Button}
 87      */
 88     get selected() {
 89         return this._selectedButton;
 90     }
 91 
 92     /**
 93      * Selected button
 94      * @type {Colibri.UI.Button}
 95      */
 96     set selected(value) {
 97         this.SelectButton(value);
 98     }
 99 
100     /**
101      * Selected button index
102      * @type {number}
103      */
104     set selectedIndex(index) {
105         const button = this.Children(index);
106         if(button) {
107             this.selected = button;
108         }
109     }
110 
111     /**
112      * Selected button index
113      * @type {number}
114      */
115     get selectedIndex() {
116         return this._selectedButton ? this._selectedButton.childIndex : null;
117     }
118 
119     /**
120      * Disables all buttons
121      */
122     DisableAllButtons() {
123         const childs = this.Children();
124         for(const child of childs) {
125             child.enabled = false;
126         }
127     }
128 
129     /**
130      * Enables button by index or name
131      * @param {string|number} name button index or number
132      */
133     EnableButton(name) {
134         this.Children(name).enabled = true;
135     }
136 
137 }
