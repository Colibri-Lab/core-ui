 16     RenderFieldContainer() {
 17 
 18         this.AddClass('app-component-password-field');
 19         this._original = null;
 20 
 21         const contentContainer = this.contentContainer;
 22         const params = {type: 'password', name: (this.form && this.form.shuffleFieldNames ? 'field-' + Date.Mc() : this._name + '-input')};
 23         if(this.form && this.form.shuffleFieldNames) {
 24             params.autocomplete = 'new-password';
 25         }
 26         this._input = contentContainer.container.append(Element.create('input', params));
 27 
 28         this.RegisterEvent('PasswordValidated', false, 'Когда сила пароля проверена');
 29         this.RegisterEvent('PasswordGenerated', false, 'Когда пароль сгенерирован');
 30 
 31         this._input.addEventListener('change', (e) => {
 32             if(this._original != this._input.value) {
 33                 this.Dispatch('Changed', {domEvent: e, component: this});
 34             }
 35             this._original = this._input.value;
 36         });
 37         this._input.addEventListener('keyup', (e) => {
 38             this.Dispatch('KeyUp', {domEvent: e});
 39             if( (this._fieldData?.params?.changeOnKeyPress ?? false) ) {
 40                 if(this._keyUpChangeTimer !== -1) {
 41                     clearTimeout(this._keyUpChangeTimer);
 42                 }
 43                 this._keyUpChangeTimer = setTimeout(() => {
 44                     this._input.emitHtmlEvents('change');
 45                 }, 500);
 46             }
 47         });
 48         this._input.addEventListener('paste', (e) => Colibri.Common.Delay(100).then(() => {
 49             this._input.emitHtmlEvents('change');
 50             this._original = this._input.value;
 51             this.Dispatch('Pasted', { domEvent: e });
 52         }));
 53         this._input.addEventListener('keydown', (e) => this.Dispatch('KeyDown', {domEvent: e}));
 54         this._input.addEventListener('focus', (e) => this.Dispatch('ReceiveFocus', {domEvent: e}));
 55         this._input.addEventListener('blur', (e) => this.Dispatch('LoosedFocus', {domEvent: e}));
 56         this._input.addEventListener('click', (e) => {
 57             this.Focus();
 58             e.stopPropagation();
 59             return false;
 60         });
 61 
 62         if(this._fieldData?.params?.readonly === undefined) {
 63             this.readonly = false;    
 64         }
 65         else {
 66             this.readonly = this._fieldData?.params?.readonly;
 67         }
 68         if(this._fieldData?.params?.enabled === undefined) {
 69             this.enabled = true;
 70         }
 71         else {
 72             this.enabled = this._fieldData.params.enabled;
 73         }
 74 
 75         if(this._fieldData?.params?.icon) {
 76             this.icon = this._fieldData?.params?.icon;
 77         }
 78         if(this._fieldData?.params?.eyeicon) {
 79             this.eyeIcon = this._fieldData?.params?.eyeicon;
 80         }
 81 
 82         this.AddHandler(['Changed', 'KeyUp', 'Paste', 'ReceiveFocus'], (event, args) => {
 83             const strength = this.CalcPasswordStrength();
 84             this._showPasswordTip(strength);
 85             this.Dispatch('PasswordValidated', {strength: strength});
 86         });
 87

100     /** @private */
101     _showPasswordTip(strength) {
102         
103         if(this._fieldData?.params?.tip) {
104             const tipData = this._fieldData?.params?.tip;
105             if(!this._passwordTip) {
106                 this._passwordTip = new Colibri.UI.ToolTip(
107                     this.name + '_tip', document.body, 
108                     tipData.orientation ? tipData.orientation : [Colibri.UI.ToolTip.RB, Colibri.UI.ToolTip.LB
109                 ]);
110             }
111 
112             if(tipData.className) {
113                 this._passwordTip.AddClass(tipData.className);
114             }
115 
116             let cls = 'bad';
117             if (strength > 80) {
118                 cls = "strong";
119             }
120             else if (strength > 60) {
121                 cls = "good";
122             }
123             else if (strength >= 30) {
124                 cls = "weak";
125             }
126             const requirements = this._fieldData?.params?.requirements || {digits: 8, strength: 40};
127             let tipText = '<p>' + (Array.isArray(tipData.text) ? tipData.text.join('</p><p>') : tipData.text) + '</p>' + 
128                 '<ul><li>' + (Array.isArray(tipData.digits) ? requirements.digits.formatSequence(tipData.digits, true) : tipData.digits.replaceAll('%s', requirements.digits)) + '</li>' + tipData.additional.map(f => '<li>' + f + '</li>').join('') + '</ul>' + 
129                 '<div class="password-progress ' + cls + '"><span style="width: ' + strength + '%"></span></div>' +  
130                 '<p>' + (strength < requirements.strength ? tipData.error : tipData.success) + '</p>' + 
131                 '<a href="#">' + tipData.generate + '</a>';
132             
133             
134             this._passwordTip.value = tipText;
135             this._passwordTip.Show(this.contentContainer);
136             const a = this._passwordTip.container.querySelector('a');
137             if(a) {
138                 a.addEventListener('click', (e) => this._generatePassword(e));
139             }
140 
141             if(cls != 'weak' && cls != 'bad') {
142                 this._hidePasswordTip();
143             }            
144             
145         }
146     }

148     /** @private */
149     _generatePassword(e) {
150         const tipData = this._fieldData?.params?.tip;
151         this.value = String.Password(16);
152         const strength = this.CalcPasswordStrength();
153         this._showPasswordTip(strength);
154         this.value.copyToClipboard().then(() => {
155             App.Notices.Add(new Colibri.UI.Notice(tipData.copied, Colibri.UI.Notice.Success));
156         });
157         this.Dispatch('Changed', {component: this});
158         this.Dispatch('PasswordValidated', {value: this.value});
159         e.preventDefault();
160         e.stopPropagation();
161         return false;
162     }

164     /**
165      * Calculates password strength
166      * @returns {number}
167      */
168     CalcPasswordStrength() {
169         const pass = this.value;
170         const requirements = this._fieldData?.params?.requirements || {digits: 8, strength: 40};
171 
172         if (!pass || pass.length < requirements.digits) {
173             return 0;
174         }
175 
176         let score = 0;
177         // award every unique letter until 5 repetitions
178         let letters = {};
179         for (let i=0; i<pass.length; i++) {
180             letters[pass[i]] = (letters[pass[i]] || 0) + 1;
181             score += 5.0 / letters[pass[i]];
182         }
183 
184         // bonus points for mixing it up
185         let variations = {
186             digits: /\d/.test(pass),
187             lower: /[a-z]/.test(pass),
188             upper: /[A-Z]/.test(pass),
189             nonWords: /\W/.test(pass),
190         }
191 
192         let variationCount = 0;
193         for (let check in variations) {
194             variationCount += (variations[check] == true) ? 1 : 0;
195         }
196         score += (variationCount - 1) * 10;
197 
198         if(score > 100) {
199             score = 100;
200         }
201 
202         score = parseInt(score);
203 
204         return score;
205 
206     }

364         icon.value = Colibri.UI.Forms.Password.EyeIconOpen;
365         icon.shown = true;
366         icon.AddHandler('Clicked', (event, args) => {
367             if(this._input.attr('type') === 'password') {
368                 icon.value = Colibri.UI.Forms.Password.EyeIconClose;
369                 this._input.attr('type', 'text');
370             }
371             else {
372                 icon.value = Colibri.UI.Forms.Password.EyeIconOpen;

374             } 
375         })

377         this.AddClass('-has-eye-icon');
