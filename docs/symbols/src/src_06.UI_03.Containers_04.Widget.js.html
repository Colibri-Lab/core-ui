1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Widget = class extends Colibri.UI.FlexBox {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {HTMLElement|Colibri.UI.Component} container container of component 
 12      * @param {string|HTMLElement} element element to generate in
 13      */
 14     constructor(name, container, element) {
 15         super(name, container, Element.fromHtml('<div><div class="widget-header"></div><div class="widget-container"></div><div class="widget-footer"></div></div>')[0]);
 16         this.AddClass('app-component-widget');
 17         // default
 18 
 19         this.namespace = typeof element === 'string' ? element : 'Colibri.UI.Widget';
 20         if(typeof element === 'string') {
 21             element = Colibri.UI.Templates[element];
 22         }
 23         
 24         this.GenerateChildren(element, this.container);
 25         
 26         this._header = this._element.querySelector('.widget-header');
 27         this._footer = this._element.querySelector('.widget-footer');
 28 
 29         this._headerTitle = new Colibri.UI.TextSpan('title', this._header);
 30         this._headerCloseToggle = new Colibri.UI.Button('toggle-button', this._header);
 31         this._headerCloseButton = new Colibri.UI.Button('close-button', this._header);
 32 
 33         if(App.Browser.Get(this.name + '-minimized') == 1) {
 34             this.AddClass('-minimized');
 35         }
 36 
 37         this._handlerEvents();
 38     }
 39 
 40     _handlerEvents() {
 41         this._headerCloseToggle.AddHandler('Clicked', (event, args) => {
 42             if(this.ContainsClass('-minimized')) {
 43                 this.RemoveClass('-minimized');
 44                 App.Browser.Delete(this.name + '-minimized');
 45             } else {
 46                 this.AddClass('-minimized');
 47                 App.Browser.Set(this.name + '-minimized', 1);
 48             }
 49         });
 50         this._headerTitle.AddHandler('Clicked', (event, args) => {
 51             if(this._toggleOnTitle) {
 52                 this._headerCloseToggle.Dispatch('Clicked');
 53             }
 54         });
 55     }
 56 
 57     get header() {
 58         return this._header.querySelector('span');
 59     }
 60 
 61     get container() {
 62         return this._element.querySelector('.widget-container');
 63     }
 64 
 65     get footer() {
 66         return this._footer;
 67     }
 68 
 69     
 70     /**
 71      * Title of widget
 72      * @type {string}
 73      */
 74     get title() {
 75         return this._title;
 76     }
 77     /**
 78      * Title of widget
 79      * @type {string}
 80      */
 81     set title(value) {
 82         this._title = value;
 83         this._showTitle();
 84     }
 85     _showTitle() {
 86         this._headerTitle.value = this._title;
 87         if(this._title) {
 88             this._headerTitle.shown = true;
 89         } else {
 90             this._headerTitle.shown = false;
 91         }
 92     }
 93 
 94     /**
 95      * Is widget closable
 96      * @type {Boolean}
 97      */
 98     get closable() {
 99         return this._closable;
100     }
101     /**
102      * Is widget closable
103      * @type {Boolean}
104      */
105     set closable(value) {
106         this._closable = value === true || value === 'true';
107         this._showClosable();
108     }
109     _showClosable() {        
110         this._headerCloseButton.shown = this._closable;
111     }
112 
113     /**
114      * Is widget can toggle
115      * @type {Boolean}
116      */
117     get togglable() {
118         return this._togglable;
119     }
120     /**
121      * Is widget can toggle
122      * @type {Boolean}
123      */
124     set togglable(value) {
125         this._togglable = value === true || value === 'true';
126         this._showTogglable();
127     }
128     _showTogglable() {
129         this._headerCloseToggle.shown = this._togglable;
130     }
131 
132     /**
133      * Toggle when click the title
134      * @type {Boolean}
135      */
136     get toggleOnTitle() {
137         return this._toggleOnTitle;
138     }
139     /**
140      * Toggle when click the title
141      * @type {Boolean}
142      */
143     set toggleOnTitle(value) {
144         this._toggleOnTitle = value;
145     }
146 
147     /**
148      * Icon for toggle
149      * @type {String}
150      */
151     get toggleIcon() {
152         return this._toggleIcon.iconSVG;
153     }
154     /**
155      * Icon for toggle
156      * @type {String}
157      */
158     set toggleIcon(value) {
159         if(value) {
160             this._headerCloseToggle.AddClass('-custom');
161         } else {
162             this._headerCloseToggle.RemoveClass('-custom');
163         }
164         this._toggleIcon = this._headerCloseToggle.Children('icon');
165         if(!this._toggleIcon) {
166             this._toggleIcon = new Colibri.UI.Icon('icon', this._headerCloseToggle);
167             this._toggleIcon.shown = true;
168         }
169         this._toggleIcon.iconSVG = value;
170     }
171 
172     /**
173      * Toggle string
174      * @type {Array}
175      */
176     get toggleText() {
177         return this._toggleText;
178     }
179     /**
180      * Toggle string
181      * @type {Array}
182      */
183     set toggleText(value) {
184         value = this._convertProperty('Array', value);
185         if(value) {
186             this._headerCloseToggle.AddClass('-custom');
187         } else {
188             this._headerCloseToggle.RemoveClass('-custom');
189         }
190         this._toggleText1 = this._headerCloseToggle.Children('text1');
191         this._toggleText2 = this._headerCloseToggle.Children('text2');
192         if(!this._toggleText1) {
193             this._toggleText1 = new Colibri.UI.TextSpan('text1', this._headerCloseToggle);
194             this._toggleText1.shown = true;
195         }
196         if(!this._toggleText2) {
197             this._toggleText2 = new Colibri.UI.TextSpan('text2', this._headerCloseToggle);
198             this._toggleText2.shown = true;
199         }
200         this._toggleText1.value = value[0];
201         this._toggleText2.value = value[1];
202     }
203 
204     
205 }
