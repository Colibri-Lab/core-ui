1 /**
  2  * 
  3  * Tab component
  4  * @class
  5  * @extends Colibri.UI.Component
  6  * @memberof Colibri.UI
  7  * @example Using in component html
  8  * <component Component="Colibri.UI.Tabs">
  9  * 
 10  *      <component-header>
 11  *          <component Component="Colibri.UI.Button" name="tab1-button" value="Запросы" />
 12  *          <component Component="Colibri.UI.Button" name="tab2-button" value="Личные кабинеты" />
 13  *      </component-header>     
 14  * 
 15  *      <component-container>
 16  *          <component Component="Colibri.UI.Pane" name="tab1-content">
 17  *              ...
 18  *          </component>
 19  *          <component Component="Colibri.UI.Pane" name="tab2-content">
 20  *              ...
 21  *          </component>
 22  *      </component-container>
 23  * 
 24  * </component>
 25  * 
 26  * @example Using as Class in js
 27  * const tabs = new Colibri.UI.Tabs('name', document.body);
 28  * tabs.AddTab(new Colibri.UI.Button('tab1-button', this.header), new Colibri.UI.Pane('tab1-content', this.container));
 29  * tabs.AddTab(new Colibri.UI.Button('tab2-button', this.header), new Colibri.UI.Pane('tab2-content', this.container));
 30  * 
 31  */
 32 Colibri.UI.Tabs = class extends Colibri.UI.Component {
 33 
 34     /**
 35      * @constructor
 36      * @param {string} name name of component
 37      * @param {HTMLElement|Colibri.UI.Component} container container of component 
 38      * @param {string|HTMLElement} element element to generate in
 39      */
 40     constructor(name, container, element) {
 41         super(name, container, element || Element.create('div'));
 42         this.AddClass('app-tabs-container-component');
 43 
 44         this._element.append(Element.fromHtml('<div class="tabs-header-container"><div class="tabs-header"></div><div class="tabs-links"></div></div>'));
 45         this._element.append(Element.fromHtml('<div class="tabs-container"></div>'));
 46 
 47         this.AddHandler('ChildsProcessed', (event, args) => {
 48             this._processEvents();
 49             this._selectTab(0);
 50         });
 51 
 52         this.AddHandler('TabClicked', (event, args) => {
 53             let newIndex = args.tab.container.index();
 54             this._selectTab(newIndex);
 55         });
 56     }
 57 
 58     /** @protected */
 59     _registerEvents() {
 60         super._registerEvents();
 61         this.RegisterEvent('TabClicked', false, 'Когда кликнули на вкладку');
 62         this.RegisterEvent('SelectionChanged', false, 'Deprecated event, must handle Changed');
 63         this.RegisterEvent('Changed', false, 'Когда выбранная вкладка изменилась');
 64     }
 65 
 66     /** @protected */
 67     _processEvents() {
 68         const buttons = this.header.querySelectorAll(':scope > .app-ui-component');
 69         buttons.forEach((button) => button.tag('component').AddHandler('Clicked', (event, args) => this.Dispatch('TabClicked', {domEvent: args.domEvent, tab: event.sender})));
 70     }
 71 
 72     /** @private */
 73     _unselectAllTabs() {
 74         const buttons = this.header.querySelectorAll(':scope > .app-ui-component');
 75         const containers = this.container.querySelectorAll(':scope > .app-ui-component');
 76 
 77         buttons.forEach((button) => {
 78             const bc = button.tag('component');
 79             bc.RemoveClass('tab-selected');
 80         });
 81 
 82         containers.forEach((container) => {
 83             const cc = container.tag('component');
 84             cc.RemoveClass('tab-selected');
 85             cc.shown = false;
 86         });
 87 
 88     }
 89 
 90     /** @private */
 91     _selectTab(index) {
 92 
 93         const currentSelection = this.selectedIndex;
 94         const newIndex = index;
 95 
 96         index ++;
 97 
 98         this._unselectAllTabs();
 99 
100         const foundButton = this.header.querySelector(':scope > .app-ui-component:nth-child(' + index + ')');
101 
102         if(!foundButton) {
103             return;
104         }
105 
106         const button = foundButton.tag('component');
107         let container = button.contentContainer;
108         if(!container) {
109             const foundContainer = this.container.querySelector(':scope > .app-ui-component:nth-child(' + index + ')');
110             container = foundContainer?.tag('component');
111         }
112 
113         button.AddClass('tab-selected');
114         if(container) {        
115             container.shown = true;
116             container.AddClass('tab-selected');
117         }
118 
119         if(currentSelection != newIndex) {
120             
121             this.Dispatch('Changed', {newIndex: newIndex, oldIndex: currentSelection, tab: button, container: container});
122             // @deprecation warning !!
123             this.Dispatch('SelectionChanged', {newIndex: newIndex, oldIndex: currentSelection, tab: button, container: container});
124             
125         }
126 
127     }
128 
129     /**
130      * Header container
131      * @type {Element}
132      */
133     get headerContainer() {
134         return this._element.querySelector(':scope > .tabs-header-container');
135     }
136 
137     /**
138      * Header container
139      * @type {Element}
140      */
141     set headerContainer(value) {
142         if(value) {
143             this.RemoveClass('-header-hidden');
144         }
145         else {
146             this.AddClass('-header-hidden');
147         }
148     }
149 
150     /**
151      * Header container element
152      * @type {Element}
153      * @readonly
154      */
155     get header() {
156         return this._element.querySelector(':scope > .tabs-header-container > .tabs-header');
157     }
158 
159     /**
160      * Links container element
161      * @type {Element}
162      * @readonly
163      */
164     get links() {
165         return this._element.querySelector(':scope > .tabs-header-container > .tabs-links');
166     }
167 
168     /**
169      * Content container element
170      * @type {Element}
171      * @readonly
172      */
173     get container() {
174         return this._element.querySelector(':scope > .tabs-container');
175     }
176 
177     /**
178      * Selected tab index
179      * @type {Number}
180      */
181     get selectedIndex() {
182         return this.header.querySelector('.tab-selected') ? this.header.querySelector('.tab-selected').index() : null;
183     }
184 
185     /**
186      * Selected tab index
187      * @type {Number}
188      */
189     set selectedIndex(value) {
190         this._selectTab(value);
191     }
192 
193     /**
194      * Selected tab button name
195      * @type {String}
196      */
197     get selectedTab() {
198         return this.header.querySelector('.tab-selected') ? this.header.querySelector('.tab-selected').dataset.objectName : null;
199     }
200 
201     /**
202      * Selected tab button name
203      * @type {String}
204      */
205     set selectedTab(value) {
206         const selectedTabIndex = this.header.querySelector('[data-object-name="' + value + '"]') ? this.header.querySelector('[data-object-name="' + value + '"]').index() : null;
207         this.selectedIndex = selectedTabIndex;
208     }
209 
210     /**
211      * Selected container component
212      * @type {Colibri.UI.Component}
213      * @readonly
214      */
215     get selectedContainer() {
216         const components = this.components;
217         return components[Object.keys(components)[this.selectedIndex]];
218     }
219 
220     /**
221      * Count of tabs
222      * @type {Number}
223      * @readonly
224      */
225     get tabsCount() {
226         return this._element.querySelectorAll(':scope > .tabs-header-container > .tabs-header > *').length;
227     }
228 
229     /**
230      * Adds tab button and container
231      * @param {Colibri.UI.button} componentHeaderButton tab button
232      * @param {Colibri.UI.Pane} componentContainer tab content component
233      * @returns Colibri.UI.Pane
234      */
235     AddTab(componentHeaderButton, componentContainer) {
236 
237         componentHeaderButton.contentContainer = componentContainer;
238 
239         this.Children(componentHeaderButton.name, componentHeaderButton);
240         componentContainer && this.Children(componentContainer.name, componentContainer);
241 
242         componentHeaderButton.AddHandler('Clicked', (event, args) => this.Dispatch('TabClicked', {domEvent: args.domEvent, tab: event.sender}));
243         return componentContainer;
244     }
245 
246 
247     /**
248      * Array of tab components
249      * @type {Array}
250      * @readonly
251      */
252     get components() {
253 
254         let ret = {};
255         this.ForEach((name, component) => {
256             if(this.container && this.container.contains(component.container)) {
257                 ret[name] = component;
258             }
259         });
260         return ret;
261 
262     }
263 
264     /**
265      * Array of tab buttons
266      * @type {Array}
267      * @readonly
268      */
269     get buttons() {
270 
271         let ret = {};
272         this.ForEach((name, component) => {
273             if(this.container && this.header.contains(component.container)) {
274                 ret[name] = component;
275             }
276         });
277         return ret;
278 
279     }
280 
281     /**
282      * Clear tab buttons and containers
283      */
284     Clear() {
285         Object.forEach(this.buttons, (name, button) => {
286             button.Dispose();
287         });
288         Object.forEach(this.components, (name, component) => {
289             component.Dispose();
290         });
291     }
292 
293 }
