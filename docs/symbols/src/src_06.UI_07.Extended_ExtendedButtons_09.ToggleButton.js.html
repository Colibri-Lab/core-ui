1 /**
  2  * Button viewed as link
  3  */
  4 Colibri.UI.ToggleButton = class extends Colibri.UI.ExtendedButton {
  5     constructor(name, container) {
  6         super(name, container);
  7         this.AddClass('app-toggle-button-component');
  8 
  9         this.iconPosition = 'right';
 10         this.icon = this.icon || 'Colibri.UI.SelectArrowIcon';
 11         
 12         this.state = 'expanded';
 13 
 14     }
 15 
 16     /**
 17      * 
 18      * @type {expanded|collapsed}
 19      */
 20     get state() {
 21         return this._state;
 22     }
 23     /**
 24      * 
 25      * @type {expanded|collapsed}
 26      */
 27     set state(value) {
 28         this._state = value;
 29         this._showState();
 30     }
 31     _showState() {
 32         if(this._state === 'expanded') {
 33             this.AddClass('-expanded').RemoveClass('-collapsed');
 34             this.value = '#{ui-toggle-button-collapse}';
 35         } else {
 36             this.RemoveClass('-expanded').AddClass('-collapsed');
 37             this.value = '#{ui-toggle-button-expand}';
 38         }
 39         
 40     }
 41 
 42 }
