1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Color = class extends Colibri.UI.Component {
  7 
  8     constructor(name, container) {
  9         super(name, container, Element.create('div'));
 10         this.AddClass('app-color-component');
 11 
 12         this.RegisterEvent('Changed', false, 'Когда значение изменилось');
 13 
 14         this._colorGrad = new Colibri.UI.Color.Line(this.name + '_grad', this);
 15 
 16         this._colorSelectedColorGrad = new Colibri.UI.Color.Block(this.name + '_block', this);
 17 
 18         this._colorOpacityGrad = new Colibri.UI.Color.Alpha(this.name + '_alpha', this);
 19 
 20         this._colorHex = Element.create('input', {class: 'app-color-hex-component'});
 21         this._colorSelected = Element.create('div', {class: 'app-color-selected-component'});
 22 
 23         this._element.append(this._colorHex);
 24         this._element.append(this._colorSelected);
 25 
 26         this._colorGrad.shown = true;
 27         this._colorSelectedColorGrad.shown = true;
 28         this._colorOpacityGrad.shown = true;
 29 
 30         this._colorGrad.AddHandler('Changed', (event, args) => this.__lineValueChanged(event, args));
 31         this._colorSelectedColorGrad.AddHandler('Changed', (event, args) => this.__blockValueChanged(event, args));
 32         this._colorOpacityGrad.AddHandler('Changed', (event, args) => this.__opacityValueChanged(event, args));
 33 
 34         this._colorHex.addEventListener('change', (e) => {
 35             this.value = this._colorHex.value;
 36         });
 37 
 38         this.handleVisibilityChange = true;
 39         this.AddHandler('VisibilityChanged', (event, args) => {
 40             const bounds = this.parent.container.bounds(true, true);
 41             if(!args.state) {
 42                 this.top = null;
 43                 this.bottom = (window.innerHeight - bounds.top);
 44             }
 45         });
 46 
 47     }
 48 
 49     _updateUIComponents() {
 50         this._colorGrad.value = this._value.hue;
 51         this._colorSelectedColorGrad.value = this._value;
 52         this._colorOpacityGrad.value = this._value.alpha;
 53     }
 54 
 55     _showValue() {
 56         this._colorHex.value = this._value.hex;
 57         this._colorSelected.css('background-color', this._colorHex.value);
 58         this.Dispatch('Changed');
 59     }
 60 
 61     __lineValueChanged(event, args) {
 62         const alpha = this._value.alpha;
 63         this._colorSelectedColorGrad.color = this._colorGrad.value;
 64         this._value = this._colorSelectedColorGrad.value;
 65         this._value.alpha = alpha;
 66         this._showValue();
 67     }
 68 
 69     __blockValueChanged(event, args) {
 70         const alpha = this._value.alpha;
 71         this._value = this._colorSelectedColorGrad.value;
 72         this._value.alpha = alpha;
 73         this._showValue();
 74     }
 75 
 76     __opacityValueChanged(event, args) {
 77         this._value.alpha = this._colorOpacityGrad.value;
 78         this._showValue();
 79     }
 80 
 81     Focus() {
 82         this._colorHex.focus();
 83     }
 84     
 85     get readonly() {
 86         return this._colorHex.attr('readonly') === 'readonly';
 87     }
 88 
 89     set readonly(value) {
 90         if(value === true || value === 'true') {
 91             this._colorHex.attr('readonly', 'readonly');
 92         }
 93         else {
 94             this._colorHex.attr('readonly', null);
 95         }
 96     }
 97 
 98     get placeholder() {
 99         return this._colorHex.attr('placeholder');
100     }
101 
102     set placeholder(value) {
103         this._colorHex.attr('placeholder', value ? value[Lang.Current] ?? value : '');
104     }
105 
106     get value() {
107         let value = this._value;
108         if(this._fieldData?.params?.emptyAsNull && !value) {
109             value = null;
110         }
111         if(this._fieldData?.params?.eval) {
112             value = eval(this._fieldData?.params?.eval);
113         }
114         return value;
115     }
116 
117     set value(value) {
118         if(typeof value == 'string') {
119             this._value = Colibri.UI.Rgb.Create().fromHex(value);
120         }
121         else if(typeof value == 'number') {
122             this._value = Colibri.UI.Rgb.Create().fromHue(value);
123         }
124         else if(value instanceof Colibri.UI.Rgb) {
125             this._value = value;
126         }
127         else if(Object.isObject(value)) {
128             this._value = Colibri.UI.Rgb.Create().fromObject(value);
129         }
130         this._updateUIComponents();
131         this._showValue();
132     }
133 
134     
135     get enabled() {
136         return this._colorHex.attr('disabled') != 'disabled';
137     }
138 
139     set enabled(value) {
140         if(value) {
141             this.RemoveClass('app-component-disabled');
142             this._colorHex.attr('disabled', null);
143         }
144         else {
145             this.AddClass('app-component-disabled');
146             this._colorHex.attr('disabled', 'disabled');
147         }
148     }
149 
150     /**
151      * Индекс табуляции
152      * @todo проверить правильно ли получаю tabIndex и исправить
153      * @type {number}
154      */
155     get tabIndex() {
156         return this._colorHex && this._colorHex.attr('tabIndex');
157     }
158     set tabIndex(value) {
159         this._colorHex && this._colorHex.attr('tabIndex', value === true ? Colibri.UI.tabIndex++ : value);
160     }

163 }
