1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Color.Block = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.fromHtml('<div><div class="grad"></div></div>')[0]);
 15         this.AddClass('app-color-block-component');
 16         
 17         this._pointer = new Colibri.UI.Pane(this.name + '_pointer', this);
 18         this._pointer.AddClass('app-color-block-pointer-component');
 19         this._pointer.shown = true;
 20 
 21         this.RegisterEvent('Changed', false, 'Когда значение изменилось');
 22 
 23         new Colibri.UI.Drag(this._pointer.container, this.container, (left, top) => {
 24             this._setNewColor(left, top);     
 25         });
 26 
 27         this.AddHandler('Clicked', (event, args) => this.__blockClicked(event, args));
 28 
 29     }
 30 
 31     _setNewColor(left, top) {
 32         this._setPoint(left, top);
 33 
 34         const bounds = this._element.bounds();
 35 
 36         left = left + this._pointer.container.bounds().outerWidth / 2;
 37         top = top + this._pointer.container.bounds().outerHeight / 2;
 38 
 39         let S = Math.ceil(left / (bounds.outerWidth / 100));
 40         let V = Math.ceil(Math.abs(top / (bounds.outerHeight / 100) - 100));
 41 
 42         this._S = S / 100;
 43         this._V = V / 100;
 44 
 45         this.Dispatch('Changed', {value: this.value});
 46 
 47     }
 48 
 49     __blockClicked(event, args) {
 50         const bounds = this._element.bounds();
 51         const pointBounds = this._pointer.container.bounds();
 52         const e = args.domEvent;
 53         let top = e.pageY - bounds.top - pointBounds.outerHeight / 2;
 54         let left = e.pageX - bounds.left - pointBounds.outerWidth / 2;
 55         this._setNewColor(left, top);     
 56     }
 57 
 58     _setPoint(left = null, top = null) {
 59         const bounds = this._element.bounds();
 60         const pointBounds = this._pointer.container.bounds();
 61         let S = this._S * 100;
 62         let V = this._V * 100;
 63 
 64         left = left ? left : bounds.outerWidth * S / 100 - pointBounds.outerWidth / 2;
 65         top = top ? top : bounds.outerHeight - bounds.outerHeight * V / 100 - pointBounds.outerHeight / 2;
 66         this._pointer.styles = {left: (left) + 'px', top: (top) + 'px'};
 67     }
 68 
 69     
 70     _showColor() {
 71         this.styles = {backgroundImage: 'linear-gradient(to right, #ffffff, ' + this._color + ')'};
 72     }
 73     set color(value) {
 74         if(value instanceof Colibri.UI.Rgb) {
 75             value = value.hex;
 76         }
 77         else if(Object.isObject(value)) {
 78             value = Colibri.UI.Rgb.Create().fromObject(value).hex;
 79         }
 80         else if(typeof value == 'number') {
 81             // hue
 82             value = Colibri.UI.Rgb.Create().fromHue(value).hex;
 83         }
 84         this._color = value;
 85         this._hue = Colibri.UI.Rgb.Create().fromHex(this._color).hue;
 86         this._showColor();
 87     }
 88 
 89     set value(value) {
 90         if(typeof value == 'string') {
 91             value = Colibri.UI.Rgb.Create().fromHex(value);
 92         }
 93         else if(typeof value == 'number') {
 94             value = Colibri.UI.Rgb.Create().fromHue(value);
 95         }
 96         
 97         this._S = value.hsv.s;
 98         this._V = value.hsv.v;
 99         this._hue = value.hsv.h;
100         this.color = value.hsv.h;
101         this._setPoint();
102     }
103 
104     get value() {
105         return Colibri.UI.Rgb.Create().fromHSV(this._hue, this._S, this._V);
106     }
107 
108 }
