 94     get value() {
 95         let value1 = this._input1.value;
 96         let value2 = this._input2.value;
 97         let timeValue1 = this._time1.value;
 98         let timeValue2 = this._time2.value;
 99         
100         value1 = (value1?.toString() === 'Invalid Date' ? null : value1.toShortDateString());
101         value2 = (value2?.toString() === 'Invalid Date' ? null : value2.toShortDateString());
102         return [value1 ? value1 + ' ' + timeValue1 : null, value2 ? value2 + ' ' + timeValue2 : null];
103     }

105     set value(value) {
106         if(!Array.isArray(value)) {
107             value = [value ?? '', ''];
108         }
109         if(typeof value[0] == 'string') {
110             value[0] = new Date(value[0]);
111         }
112         if(typeof value[1] == 'string') {
113             value[1] = new Date(value[1]);
114         }
115         this._input1.value = value[0];
116         this._input2.value = value[1];
117         this._time1.value = value[0] instanceof Date ? value[0].toTimeString() : '';
118         this._time1.value = value[1] instanceof Date ? value[1].toTimeString() : '';
119 
120     }
