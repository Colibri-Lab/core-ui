 66     /**
 67      * Initializes the application with the specified configuration.
 68      * @method
 69      * @param {string} [name='app'] - The name of the application.
 70      * @param {number} [version=1] - The version of the application.
 71      * @param {string} [routerType=Colibri.Web.Router.RouteOnHash] - The type of router to use.
 72      * @param {string} [requestType=Colibri.IO.Request.RequestEncodeTypeEncrypted] - The type of request encoding to use.
 73      * @param {boolean} [initComet=false] - Whether to initialize Comet.
 74      * @param {boolean} [showLoader=true] - Whether to show the loader.
 75      * @param {string} [remoteDomain=null] - The remote domain for requests.
 76      * @param {string} [dateformat=null] - The date format.
 77      * @param {string} [numberformat=null] - The number format.
 78      * @param {string} [currency=null] - The currency format.
 79      * @param {string} [loadingIcon=null] - The loading icon.
 80      * @param {string} [csrfToken=null] - The CSRF token.
 81      * @returns {Promise} A promise that resolves when the application is initialized.
 82      */
 83     InitializeApplication(
 84         name = 'app',
 85         version = 1,
 86         routerType = Colibri.Web.Router.RouteOnHash, 
 87         requestType = Colibri.IO.Request.RequestEncodeTypeEncrypted,
 88         initComet = false,
 89         showLoader = true,
 90         remoteDomain = null,
 91         dateformat = null,
 92         numberformat = null,
 93         currency = null,
 94         loadingIcon = null,
 95         csrfToken = null
 96     ) {
 97 
 98         return new Promise((resolve, reject) => {
 99 
100             if(this._initialized) {
101                 resolve();
102                 return;
103             }
104     
105             this._name = name;
106             this._version = version;
107     
108             Colibri.IO.Request.type = requestType;
109             if(remoteDomain) {
110                 this._remoteDomain = remoteDomain;
111             }
112             else {
113                 this._remoteDomain = '';
114             }
115     
116             this._actions = new Colibri.Common.HashActions(); 
117             this._store = new Colibri.Storages.Store('app', {});
118             this._request = new Colibri.Web.Request();
119             this._router = new Colibri.Web.Router(routerType);
120             this._device = new Colibri.Devices.Device();
121             this._browser = new Colibri.Common.BrowserStorage();
122             this._session = new Colibri.Common.SessionStorage();
123             this._db = new Colibri.Web.IndexDB(this._name, this._version);
124             this._dateformat = dateformat;
125             this._numberformat = numberformat;
126             this._currency = currency;
127             this._csrfToken = csrfToken;
128     
129             Colibri.Common.WaitForBody().then(() => {
130                 this.InitializeModules();
131                 if(showLoader) {
132                     if(loadingIcon) {
133                         this.Loader.icon = loadingIcon
134                         this.Loader.showIcon = true;
135                     }
136                     this.Loader.Show();
137                     this.Loader.StartProgress(200, 1.5);
138                 }
139                 this._loadingBox = new Colibri.UI.Loading('app-loading-box', document.body, Element.create('div'), true);
140                 this._confirmDialog = new Colibri.UI.ConfirmDialog('confirm', document.body, 600);
141                 this._promptDialog = new Colibri.UI.PromptDialog('prompt', document.body, 600);
142                 this._alertDialog = new Colibri.UI.AlertDialog('alert', document.body, 600);
143                 this._customToolTip = new Colibri.UI.ToolTip('tooltip', document.body);
144                 this._loadingBallun = new Colibri.UI.LoadingBallun('ballun', document.body);
145             });
146                 
147             
148             // Делаем всякое после того, как DOM загрузился окончательно
149             Colibri.Common.WaitForDocumentReady().then(() => { 
150                 this.Dispatch('DocumentReady');
151     
152                 const headers = {};
153                 if(this._csrfToken) {
154                     headers['X-CSRF-TOKEN'] = this._csrfToken;
155                 }
156                 Colibri.IO.Request.Post(this._remoteDomain + '/settings', {}, headers).then((response) => {
157                     if(response.status != 200) {
158                         App.Notices.Add(new Colibri.UI.Notice('#{ui-messages-cannotgetsettings}'));
159                     }
160                     else {
161                         const settings = (typeof response.result === 'string' ? JSON.parse(response.result) : response.result);
162                         this._store.Set('app.settings', settings);
163     
164                         if(initComet && settings.comet && settings.comet.host) {
165                             this._comet = new Colibri.Web.Comet(settings.comet);
166                             this._comet.AddHandler(['MessageReceived', 'MessagesMarkedAsRead', 'MessageRemoved'], (event, args) => {
167                                 if(!document.hasFocus()) {
168                                     this.StartFlashTitle();
169                                 }
170                                 else {
171                                     this.StopFlashTitle();
172                                 }
173                             });
174                         } 
175                         document.addEventListener('visibilitychange', (e) => {
176                             
177                             if(initComet && settings.comet && settings.comet.host) {
178                                 this.StopFlashTitle();
179                             }
180     
181                             if(document.hidden) {
182                                 this.Dispatch('DocumentHidden', {});
183                             }
184                             else {
185                                 this.Dispatch('DocumentShown', {});
186                             }
187                         });
188     
189                         if(settings?.screen?.theme === 'follow-device') {
190                             this._device.AddHandler('ThemeChanged', (event, args) => {
191                                 this.SetTheme(args.current);
192                             });
193                             this.SetTheme(this._device.Theme);
194                         }    
195                         else {
196                             this.SetTheme(settings?.screen?.theme ?? 'light');
197                         }
198     
199                     }
200     
201                     // запускаем обработку экшенов в документе
202                     this._router.HandleDomReady();
203                     this._actions.HandleDomReady();
204     
205                     if(showLoader) {
206                         Colibri.Common.Delay(1500).then(() => {
207                             this.Loader.StopProgress();  
208                             this.Loader.Hide();
209                         });   
210                     }
211                      
212                     Colibri.UI.UpdateMaxZIndex();
213                     this.Dispatch('ApplicationReady');
214                     resolve();
215      
216                 }).catch(response => {
217                     console.log(response);
218                     App.Notices.Add(new Colibri.UI.Notice('Невозможно получить настройки!'));
219                     reject();
220                 });
221     
222     
223                 this._notices = new Colibri.UI.Notices('notices', document.body);
224     
225                 
226                 
227             });
228     
229             this._initialized = true;
230         });
231

236     /**
237      * Sets the theme for the application.
238      * @param {string} theme - The theme to set.
239      */
240     SetTheme(theme) {
241         document.body.classList.remove('prefer-dark');
242         document.body.classList.remove('prefer-light');
243         document.body.classList.add('prefer-' + theme);
244     }

246     /**
247      * Initializes the modules.
248      */
249     InitializeModules() {
250         Object.keys(App.Modules).forEach((module) => {
251             try {
252                 eval(module + '.InitializeModule();');
253             }
254             catch(e) { }
255         })
256     }

258     /**
259      * Starts flashing the title of the document by appending "(*)" every second.
260      */
261     StartFlashTitle() {
262         this.StopFlashTitle();
263 
264         this._title = document.title;
265         this._flashInterval = setInterval(() => {
266             document.title = document.title.indexOf('(*)') > -1 ? this._title : '(*) ' + this._title;
267         }, 1000);
268     }
269 
270     /**
271      * Stops flashing the title of the document and restores the original title.
272      */
273     StopFlashTitle() {
274         if(this._flashInterval) {
275             if(this._title) {
276                 document.title = this._title;
277             }
278             clearInterval(this._flashInterval);
279             this._flashInterval = -1;
280         }
281     }

299             // do nothing
300         }
