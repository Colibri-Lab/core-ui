1 /**
  2  * @class 
  3  * @extends Colibri.Events.Dispatcher
  4  */
  5 Colibri.Storages.Models.Table = class extends Colibri.Events.Dispatcher {
  6  
  7     constructor(controller, params) {
  8         super();
  9 
 10         this._controller = controller;
 11         this._params = params;
 12         this._loaded = false;
 13 
 14         this._setControllerData();
 15 
 16         this.bind();
 17         this.Load();
 18     }
 19     
 20     _setControllerData() {
 21 
 22         // значит это Module:Controller
 23         const parts = this._controller.split(':');
 24         const module = eval(parts[0]);
 25         const controller = parts[1].split('.')[0];
 26 
 27         this._loader = () => module.Call(controller, 'Load', this._params);
 28         this._saver = () => module.Call(controller, 'Save', this._params);
 29         this._deleter = () => module.Call(controller, 'Delete', this._params);
 30     
 31     }
 32 
 33     _registerEvents() {
 34         this.RegisterEvent('DataChanged', false, 'Когда данные изменились');
 35     }
 36 
 37     _convertData(data) {
 38         this._data = [];
 39         for(const d of data) {
 40             this._data.push(new Colibri.Storages.Models.Model(d, this));
 41         }
 42     }
 43 
 44     __cometEventReveived(event, args) {
 45         if(args.message.action == 'data-changed' && args.message.binding == this._binding) {
 46             this.Load();
 47         }
 48     }
 49 
 50     bind() {
 51         if(App.Comet) {
 52             App.Comet.Addhandler('EventReceived', (event, args) => this.__cometEventReveived(event, args));
 53         }
 54     }
 55 
 56     Load() {
 57         this._loader().then(data => {
 58             this._loaded = true;
 59             this._data = this._convertData(data);
 60             this.Dispatch('DataChanged', {});
 61         });
 62     }
 63 
 64     ForEach(rowHandler) {
 65         for(const row of this._data) {
 66             rowHandler(row);
 67         }
 68     }
 69 
 70     SaveRow(row) {
 71         this._saver().then((data) => {
 72             row.data = data;
 73             resolve();
 74         }).catch(error => reject(error));
 75     }
 76 
 77     DeleteRow(row) {
 78         this._deleter().then((data) => {
 79             row.data = data;
 80             resolve();
 81         }).catch(error => reject(error));
 82     }
 83 
 84 }
