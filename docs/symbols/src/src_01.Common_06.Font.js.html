1 /**
  2  * Class representing font utility functions.
  3  * @class
  4  * @memberof Colibri.Common
  5  */
  6 Colibri.Common.Font = class {
  7 
  8     /**
  9      * Array containing the loaded fonts.
 10      * @type {Array}
 11      * @private
 12      */
 13     _list = [];
 14 
 15     /**
 16      * Creates an instance of Colibri.Common.Font.
 17      */
 18     constructor() {
 19         this._load();
 20     }
 21 
 22     /**
 23      * Loads the fonts from the document.
 24      * @private
 25      */
 26     _load() {
 27         let { fonts } = document;
 28         const it = fonts.entries();
 29         
 30         let arr = [];
 31         let done = false;
 32         
 33         while (!done) {
 34             const font = it.next();
 35             if (!font.done) {
 36                 arr.push(font.value[0]);
 37             } else {
 38                 done = font.done;
 39             }
 40         }
 41             
 42         this._list = arr;
 43     }
 44 
 45     /**
 46      * Creates an instance of Colibri.Common.Font.
 47      * @returns {Colibri.Common.Font} A new instance of Colibri.Common.Font.
 48      * @static
 49      */
 50     static Create() {
 51         return new Colibri.Common.Font();
 52     }
 53 
 54     /**
 55      * Get an array of unique font families.
 56      * @returns {Array} Array of unique font families.
 57      */
 58     get families() {
 59         let ret = [];
 60         for(const font of this._list) {
 61             ret.push(font.family);
 62         }
 63         return Array.unique(ret);
 64     }
 65 
 66     /**
 67      * Get an array of font lookup objects containing value and title properties.
 68      * @returns {Array} Array of font lookup objects.
 69      */
 70     get lookup() {
 71         return this.families.map(v => { return {value: v, title: v}; });
 72     }
 73 
 74     /**
 75      * Get the list of loaded fonts.
 76      * @returns {Array} List of loaded fonts.
 77      */
 78     get list() {
 79         return this._list;
 80     }
 81 
 82 }
