1 Colibri.UI.TextArea = class extends Colibri.UI.Component {
  2 
  3     constructor(name, container) {
  4         super(name, container, Element.create('div', {class: 'app-ui-component'}));
  5 
  6         this.AddClass('app-textarea-component');
  7 
  8         this._input = Element.create('textarea', { placeholder: this.placeholder || '' });
  9         this._element.append(this._input);
 10 
 11         new Colibri.UI.Pane('clear', this);
 12 
 13         this.Children('clear').html = Colibri.UI.ClearIcon;
 14 
 15         this._input.addEventListener('keyup', (e) => {
 16             this.Children('clear').shown = this._input.value.length > 0;
 17             this.Dispatch('KeyUp', { value: this.value, domEvent: e });
 18             if (this._input.value.length == 0) {
 19                 this.Dispatch('Cleared');
 20             }
 21         });
 22 
 23         this._input.addEventListener('change', (e) => {
 24             this.Dispatch('Changed', { value: this.value, domEvent: e });
 25         });
 26 
 27         this._input.addEventListener('keydown', (e) => {
 28             this.Dispatch('KeyDown', { value: this.value, domEvent: e });
 29         });
 30 
 31         this._input.addEventListener('mousedown', (e) => {
 32             e.target.focus();
 33             //e.target.select();
 34             return false;
 35         });
 36 
 37         this.Children('clear').AddHandler('Clicked', (event, args) => {
 38             this._input.value = '';
 39             this._input.focus();
 40             this._input.select();
 41             this.Children('clear').shown = false;
 42             this.Dispatch('Cleared');
 43         });
 44 
 45     }
 46 
 47     _registerEvents() {
 48         super._registerEvents();
 49         this.RegisterEvent('KeyUp', false, 'Поднимается, когда клавиша поднята');
 50         this.RegisterEvent('KeyDown', false, 'Поднимается, когда клавиша нажата');
 51         this.RegisterEvent('Changed', false, 'Поднимается, когда содержимое поля обновлено');
 52         this.RegisterEvent('Cleared', false, 'Поднимается, когда содержимое поля очищено с помощью крестика');
 53     }
 54 
 55     /**
 56      * Ставит фокус на компоменту
 57      * @returns {Colibri.UI.Component}
 58      */
 59      Focus() {
 60         this._input.focus();
 61         //this._input.select();
 62         return this;
 63     }
 64 
 65     /** @type {integer} */
 66     get maxlength() {
 67         return this._input.attr('maxlength');
 68     }
 69     set maxlength(value) {
 70         this._input.attr('maxlength', value);
 71     }
 72 
 73     /** @type {string} */
 74     get placeholder() {
 75         try {
 76             return this._input.attr('placeholder');
 77         }
 78         catch(e) {
 79             return '';
 80         }
 81     }
 82     set placeholder(value) {
 83         this._input.attr('placeholder', value ? value[Lang.Current] ?? value : '');
 84     }
 85 
 86     /** @type {string} */
 87     get value() {
 88         return this._input.value;
 89     }
 90     set value(value) {
 91         this._input.value = value;
 92         this.Children('clear').shown = this._input.value.length > 0;
 93     }
 94 
 95     /**
 96      * Элемент только для чтения
 97      * @type {boolean}
 98      */
 99     get readonly() {
100         return this._input.is(':scope[readonly]');
101     }
102     set readonly(value) {
103         this._input.attr('readonly', value);
104         this.Dispatch('ReadonlyStateChanged');
105     }
106 
107 }
