1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.TextArea = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.create('div', {class: 'app-ui-component'}));
 15 
 16         this.AddClass('app-textarea-component');
 17 
 18         this._input = Element.create('textarea', { placeholder: this.placeholder || '' });
 19         this._element.append(this._input);
 20 
 21         new Colibri.UI.Pane('clear', this);
 22 
 23         this.Children('clear').html = Colibri.UI.ClearIcon;
 24 
 25         this._input.addEventListener('keyup', (e) => {
 26             this.Children('clear').shown = this._input.value.length > 0;
 27             this.Dispatch('KeyUp', { value: this.value, domEvent: e });
 28             if (this._input.value.length == 0) {
 29                 this.Dispatch('Cleared');
 30             }
 31         });
 32 
 33         this._input.addEventListener('change', (e) => {
 34             this.Dispatch('Changed', { value: this.value, domEvent: e });
 35         });
 36 
 37         this._input.addEventListener('keydown', (e) => {
 38             this.Dispatch('KeyDown', { value: this.value, domEvent: e });
 39         });
 40 
 41         this._input.addEventListener('mousedown', (e) => {
 42             e.target.focus();
 43             //e.target.select();
 44             return false;
 45         });
 46 
 47         this.Children('clear').AddHandler('Clicked', (event, args) => {
 48             this._input.value = '';
 49             this._input.focus();
 50             this._input.select();
 51             this.Children('clear').shown = false;
 52             this.Dispatch('Cleared');
 53         });
 54 
 55     }
 56 
 57     /** @protected */
 58     _registerEvents() {
 59         super._registerEvents();
 60         this.RegisterEvent('KeyUp', false, 'Поднимается, когда клавиша поднята');
 61         this.RegisterEvent('KeyDown', false, 'Поднимается, когда клавиша нажата');
 62         this.RegisterEvent('Changed', false, 'Поднимается, когда содержимое поля обновлено');
 63         this.RegisterEvent('Cleared', false, 'Поднимается, когда содержимое поля очищено с помощью крестика');
 64     }
 65 
 66     /**
 67      * Focus on component
 68      * @returns this
 69      */
 70      Focus() {
 71         this._input.focus();
 72         return this;
 73     }
 74 
 75     /**
 76      * Maximum length in chars of component 
 77      * @type {number} 
 78      */
 79     get maxlength() {
 80         return this._input.attr('maxlength');
 81     }
 82     /**
 83      * Maximum length in chars of component 
 84      * @type {number} 
 85      */
 86     set maxlength(value) {
 87         this._input.attr('maxlength', value);
 88     }
 89 
 90     /**
 91      * Textarea placeholder 
 92      * @type {string} 
 93      */
 94     get placeholder() {
 95         try {
 96             return this._input.attr('placeholder');
 97         }
 98         catch(e) {
 99             return '';
100         }
101     }
102     /**
103      * Textarea placeholder 
104      * @type {string} 
105      */
106     set placeholder(value) {
107         this._input.attr('placeholder', value ? value[Lang.Current] ?? value : '');
108     }
109 
110     /**
111      * Value string 
112      * @type {string} 
113      */
114     get value() {
115         return this._input.value;
116     }
117     /**
118      * Value string 
119      * @type {string} 
120      */
121     set value(value) {
122         this._input.value = value;
123         this.Children('clear').shown = this._input.value.length > 0;
124     }
125 
126     /**
127      * Is textarea readonly
128      * @type {boolean}
129      */
130     get readonly() {
131         return this._input.is(':scope[readonly]');
132     }
133     /**
134      * Is textarea readonly
135      * @type {boolean}
136      */
137     set readonly(value) {
138         this._input.attr('readonly', value);
139         this.Dispatch('ReadonlyStateChanged');
140     }
141 
142 }
