  6 Colibri.UI.FileSizeViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-filesize-viewer-component');
 18 
 19         this._size = new Colibri.UI.TextSpan('date', this);
 20         this._size.shown = true;
 21 
 22         this._value = null;
 23 
 24     }
 25 
 26     /**
 27      * Value
 28      * @type {number}
 29      */
 30     get value() {
 31         return this._value;
 32     }
 33 
 34     /**
 35      * Value
 36      * @type {number}
 37      */
 38     set value(value) {
 39         value = this._convertValue(value);
 40         if(typeof value !== 'number') {
 41             value = value.toInt();
 42         }
 43         this._value = value;
 44         this._size.value = this._value.toSizeString(['bytes', 'Kb', 'Mb', 'Gb'], 1024, true);
 45 
 46     }
 47 
 48 }
