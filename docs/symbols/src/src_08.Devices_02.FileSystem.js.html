180     /**
181      * Generates a Blob object.
182      * @param {*} name - The name of the file.
183      * @param {*} content - The content of the file.
184      * @returns {*} - Blob object.
185      */
186     Blob(name, content) {
187         if(content instanceof Blob) {
188             return content;
189         }
190         const mimetype = Colibri.Common.MimeType.ext2type(name.pathinfo().ext);
191         return new Blob([content], { type: mimetype });
192     }
193 
194     /**
195      * Converts a base64 string to a Blob object.
196      * @param {string} b64Data - The base64 data.
197      * @param {string} contentType - The content type.
198      * @param {number} sliceSize - The slice size.
199      * @returns {*} - Blob object.
200      */
201     B64ToBlob(b64Data, contentType, sliceSize = 512) {
202         
203         contentType = contentType || '';
204         sliceSize = sliceSize || 512;
205 
206         var byteCharacters = atob(b64Data);
207         var byteArrays = [];
208 
209         for (var offset = 0; offset < byteCharacters.length; offset += sliceSize) {
210             var slice = byteCharacters.slice(offset, offset + sliceSize);
211 
212             var byteNumbers = new Array(slice.length);
213             for (var i = 0; i < slice.length; i++) {
214                 byteNumbers[i] = slice.charCodeAt(i);
215             }
216 
217             var byteArray = new Uint8Array(byteNumbers);
218 
219             byteArrays.push(byteArray);
220         }
221 
222         return new Blob(byteArrays, {type: contentType});
223     
224     }

226     /**
227      * Gets the file system.
228      * @param {*} type - The type of the file system.
229      * @param {number} quota - The quota for the file system.
230      * @returns {Promise} - Promise resolving to the file system root.
231      */
232     Get(type = LocalFileSystem.PERSISTENT, quota = 0) {
233         return new Promise((resolve, reject) => {
234             window.requestFileSystem(type, quota, (fs) => {
235                 resolve(fs.root);
236             }, (e) => reject(e));
237         });
238     }

253     /**
254      * Creates a file.
255      * @param {*} fsOrDir - The file system or directory.
256      * @param {string} fileName - The name of the file.
257      * @param {*} options - Options for file creation.
258      * @returns {Promise} - Promise resolving to the file entry.
259      */
260     File(fsOrDir, fileName, options) {
261         options = Object.assign({ create: true, exclusive: false }, options);
262         return new Promise((resolve, reject) => {
263             fsOrDir.getFile(fileName, options, (fileEntry) => {
264                 resolve(fileEntry);
265             }, e => reject(e));
266         });
267     }

269     /**
270      * Creates a directory.
271      * @param {*} fsOrDir - The file system or directory.
272      * @param {string} dirName - The name of the directory.
273      * @param {*} options - Options for directory creation.
274      * @returns {Promise} - Promise resolving to the directory entry.
275      */
276     Directory(fsOrDir, dirName, options) {
277         return new Promise((resolve, reject) => {
278             dirName = dirName.split('/');
279             options = Object.assign({ create: true, exclusive: false }, options);
280             fsOrDir.getDirectory(dirName.shift(), options, (dirEntry) => {
281                 if(dirName.length === 0) {
282                     resolve(dirEntry);
283                 }
284                 else {
285                     resolve(this.Directory(dirEntry, dirName.join('/'), options));
286                 }
287             }, e => reject(e));
288         });
289     }

291     /**
292      * Writes content to a file.
293      * @param {*} fileEntry - The file entry.
294      * @param {*} content - The content to write.
295      * @param {boolean} isAppend - Indicates whether to append to the file.
296      * @returns {Promise} - Promise resolving to the file entry.
297      */
298     Write(fileEntry, content, isAppend) {
299         return new Promise((resolve, reject) => {
300             fileEntry.createWriter((fileWriter) => {
301                 fileWriter.onwriteend = () => {
302                     resolve(fileEntry);
303                 };
304                 fileWriter.onerror = (e) => {
305                     reject(e);
306                 };
307                 if (isAppend) {
308                     try {
309                         fileWriter.seek(fileWriter.length);
310                     }
311                     catch (e) {
312                         reject(e);
313                     }
314                 }
315                 fileWriter.write(this.Blob(fileEntry.name, content));
316             });
317     
318         });
319     }

321     /**
322      * Reads content from a file.
323      * @param {*} fileEntry - The file entry.
324      * @returns {Promise} - Promise resolving to the file content.
325      */
326     Read(fileEntry) {
327         return new Promise((resolve, reject) => {
328             fileEntry.file((file) => {
329                 const reader = new FileReader();        
330                 reader.onloadend = () => {
331                     resolve(reader.result);
332                 };
333                 reader.onerror = (e) => {
334                     reject(e);
335                 }        
336                 reader.readAsText(file);
337         
338             }, e => reject(e));
339         });
340     }

342     /**
343      * Requests a file.
344      * @param {*} type - The type of the file system.
345      * @param {number} quota - The quota for the file system.
346      * @param {string} path - The file path.
347      * @param {string} fileName - The file name.
348      * @param {*} content - The content of the file.
349      * @param {boolean} isAppend - Indicates whether to append to the file.
350      * @param {*} options - Options for file request.
351      * @returns {Promise} - Promise resolving to the file entry.
352      */
353     RequestFile(type = LocalFileSystem.PERSISTENT, quota = 0, path, fileName, content, isAppend, options = {}) {
354         return new Promise((resolve, reject) => {
355             this.Get(type, quota)
356                 .then((fs) => this.Directory(fs, path, options))
357                 .then((dirEntry) => this.File(dirEntry, fileName, options))
358                 .then((fileEntry) => this.Write(fileEntry, content, isAppend))
359                 .then((fileEntry) => resolve(fileEntry))
360                 .catch(e => reject(e));
361         });
362     }

419     /**
420      * Reads entries from a directory.
421      * @param {string} rootPath - The root path.
422      * @param {string} path - The directory path.
423      * @returns {Promise} - Promise resolving to the directory entries.
424      */
425     ReadDirectory(rootPath, path) {
426         return new Promise((resolve, reject) => {
427             this.Local(rootPath)
428                 .then((fs) => this.Directory(fs, path))
429                 .then((dirEntry) => new Promise((resolve, reject) => resolve(dirEntry.createReader())))
430                 .then((reader) => new Promise((resolve, reject) => reader.readEntries((entries) => resolve(entries), e => reject(e))))
431                 .then((entries) => resolve(entries))
432                 .catch(e => reject(e));
433         });
