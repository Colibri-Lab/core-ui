1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.FilesViewer = class extends Colibri.UI.Viewer {
  7     
  8     constructor(name, container, element = null, root = null) {
  9         super(name, container, element || Element.create('div'), root);
 10         this.AddClass('app-files-viewer-component');
 11 
 12         this._value = null;
 13         this._download = null;
 14         this._fileNotExistText = '—';
 15 
 16         this._list = new Colibri.UI.List('viewer-list', this);
 17         this._list.AddGroup('group', '');
 18         this._list.__renderItemContent = (itemData, container) => {
 19 
 20             if(itemData.file !== null) {
 21                 const icon = new Colibri.UI.Icon('icon', container);
 22                 icon.shown = true;
 23                 icon.value = Colibri.UI.FileLinkIcon;    
 24             }
 25 
 26             const filename = new Colibri.UI.TextSpan('span', container);
 27             filename.value = itemData.title;
 28             filename.shown = true;
 29             if(itemData.file === null) {
 30                 filename.AddClass('gray');
 31             }
 32         }
 33 
 34         this._list.AddHandler('ItemClicked', (event, args) => {
 35 
 36             const value = args.item.value;
 37             if(value && value.file && this._download) {
 38                 window.open((window.rpchandler ? window.rpchandler : '') + this._download + '?guid=' + value.file.guid);
 39             }
 40 
 41         });
 42 
 43 
 44     }
 45 
 46     _showValue() {
 47 
 48         const group = this._list.Children('group');
 49         group.Clear();
 50 
 51         this._value && this._value.forEach((file) => {
 52             try {
 53                 group.AddItem({title: file.name, file: file});
 54             }
 55             catch(e) {}
 56         });
 57         this._list.shown = true;
 58         if(!this._value || Object.countKeys(this._value) == 0) {
 59             group.AddItem({title: this._field && this._field.params && this._field.params.text || this._fileNotExistText, file: null});
 60 
 61         }
 62         
 63     }
 64     
 65     get value() {
 66         return this._value;
 67     }
 68 
 69     set value(value) {
 70         this._value = this._convertValue(Object.isObject(value) ? Object.values(value) : value);
 71         this._showValue();
 72     }
 73 
 74     get field() {
 75         return this._field;
 76     }
 77 
 78     set field(field) {
 79         this._field = field;
 80         this._showValue();
 81     }
 82 
 83     set download(value) {
 84         this._download = value;
 85     }
 86 
 87     get fileNotExistText() {
 88         return this._fileNotExistText
 89     }
 90 
 91     set fileNotExistText(value) {
 92         this._fileNotExistText = value
 93     }
 94 
 95 
 96 }
 97 Colibri.UI.Viewer.Register('Colibri.UI.FilesViewer', '#{ui-viewers-files}');
