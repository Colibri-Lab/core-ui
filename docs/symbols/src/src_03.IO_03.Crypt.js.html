1 /**
  2  * Provides encryption and decryption functionality.
  3  * @class
  4  */
  5 Colibri.IO.Crypt = class {
  6 
  7     /**
  8      * Encrypts the given data using the specified key.
  9      * @param {string} key The encryption key.
 10      * @param {string} data The data to encrypt.
 11      * @returns {Promise<string>} A promise that resolves to the encrypted data.
 12      */
 13     static Encrypt(key, data) {
 14         return new Promise((resolve, reject) => {
 15             key.sha256().then((sha256) => {
 16                 const encoded = data.rc4(sha256);
 17                 resolve(encoded);
 18             }).catch(error => reject(error));
 19         });
 20     }
 21 
 22     /**
 23      * Decrypts the given data using the specified key.
 24      * @param {string} key The decryption key.
 25      * @param {string} data The data to decrypt.
 26      * @returns {Promise<string>} A promise that resolves to the decrypted data.
 27      */
 28     static Decrypt(key, data) {
 29         return Colibri.IO.Crypt.Encrypt(key, data);
 30     }
 31 
 32 }
