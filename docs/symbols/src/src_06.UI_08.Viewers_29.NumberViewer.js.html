1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.NumberViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-number-viewer-component');
 18     }
 19 
 20     /**
 21      * Value
 22      * @type {number}
 23      */
 24     get value() {
 25         return this._value;
 26     }
 27 
 28     /**
 29      * Value
 30      * @type {number}
 31      */
 32     set value(value) {
 33         this._value = value;
 34         value = this._convertValue(value);
 35         const emptyMessage = this.field?.params?.empty ?? this._emptyMessage ?? '';
 36         if((!value || !isFinite(value)) && !!emptyMessage) {
 37             super.value = emptyMessage;
 38             return;
 39         }
 40 
 41         let v = !value || !isFinite(value) ? '0' : value; 
 42         if(this.field?.params?.format === 'money') {
 43             const formatter = new Intl.NumberFormat(App.NumberFormat, {style: 'currency', currency: this._field?.params?.currency ?? App.Currency?.code ?? '', maximumFractionDigits: this.field?.params?.decimal ?? 2});
 44             v = formatter.format(parseFloat(v));
 45         }
 46         else if(this.field?.params?.format === 'percent') {
 47             const isShare = this.field?.params?.isShare ?? false;
 48             const formatter = new Intl.NumberFormat(App.NumberFormat, {style: 'percent', maximumFractionDigits: this.field?.params?.decimal ?? 2, minimumFractionDigits: this.field?.params?.decimal ?? 2});
 49             if(!isShare) {
 50                 v = v / 100;
 51             }
 52             v = formatter.format(parseFloat(v));
 53             // v = parseFloat(v).toMoney(this.field?.params?.decimal ?? 2);
 54         }
 55         else if(this.field?.params?.format === 'bytes') {
 56             v = parseFloat(v).toSizeString(['bytes', 'Kb', 'Mb', 'Gb'], 1024);            
 57         }
 58         else {
 59             
 60             if(this.field?.params?.decimal && !this.field?.params?.forceformat) {
 61                 v = parseFloat(v).toMoney(this.field?.params?.decimal ?? 2);
 62             }
 63 
 64             if(this.field?.params?.unit) {
 65                 v = v + ' ' + (Array.isArray(this.field?.params?.unit) ? parseFloat(v).formatSequence(this.field?.params?.unit, false) : this.field?.params?.unit);
 66             } else if(this.field?.params?.forceformat) {
 67                 const options = {};
 68                 if(this.field?.params?.decimal) {
 69                     options.maximumFractionDigits = this.field?.params?.decimal;
 70                 }
 71                 v = new Intl.NumberFormat(App.NumberFormat, options).format(v);
 72             }
 73         }
 74 
 75         super.value = v;
 76     }
 77 
 78 
 79 }
 80 Colibri.UI.Viewer.Register('Colibri.UI.NumberViewer', '#{ui-viewers-number}');
