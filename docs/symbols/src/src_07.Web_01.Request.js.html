1 /**
  2  * Represents a utility for handling web requests.
  3  * @class
  4  * @extends Destructable
  5  * @memberof Colibri.Web
  6  */
  7 Colibri.Web.Request = class extends Destructable {
  8 
  9     /**
 10      * Creates an instance of Request.
 11      */
 12     constructor() {
 13         super();
 14     }
 15 
 16     /**
 17      * Retrieves the URI of the request, prioritizing location.hash.
 18      * @returns {string} - The URI of the request.
 19      */
 20     get uri() {
 21         let h = App.Router.type === Colibri.Web.Router.RouteOnHash ? location.hash.substring(1) : location.pathname;
 22         if(h.indexOf('?') !== -1) {
 23             h = h.split('?')[0];
 24         }
 25         return h;
 26     }
 27 
 28     /**
 29      * Retrieves the path segments of the request.
 30      * @returns {Array} - The path segments of the request.
 31      */
 32     get path() {
 33         return this.uri.split('/').filter(v => v != '');
 34     }
 35 
 36     /**
 37      * Retrieves the query parameters of the request, prioritizing location.hash.
 38      * @returns {Object} - The query parameters of the request.
 39      */
 40     get query() {
 41         let h = location.hash ? location.hash.substring(1) : location.search;
 42         if(h.indexOf('?') !== -1) {
 43             h = h.split('?')[1];
 44             if(h) {
 45                 return h.toObject(['&', '='], (v) => decodeURI(v));
 46             }
 47         }
 48         return {};
 49     }
 50 
 51     /**
 52      * Retrieves the path in the hash.
 53      * @returns {string} - The path in the hash.
 54      */
 55     get hash() {
 56         let h = location.hash ? location.hash.substring(1) : '/';
 57         if(h.indexOf('?') !== -1) {
 58             h = h.split('?')[0];
 59         }
 60         return h;
 61     }
 62 
 63     /**
 64      * Retrieves the root path.
 65      * @returns {string} - The root path.
 66      */
 67     get rootPath() {
 68         return location.pathname;
 69     }
 70 
 71     /**
 72      * Sets the root path.
 73      * @param {string} [value='/'] - The value to set as the root path.
 74      */
 75     set rootPath(value = '/') {
 76         history.pushState({url: value}, '', value);
 77     }
 78 
 79 }
