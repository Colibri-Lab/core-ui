1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.ToolTip = class extends Colibri.UI.Component {
  7 
  8     static LB = 'lb';
  9     static LT = 'lt';
 10     static LM = 'lm';
 11     static RB = 'rb';
 12     static RT = 'rt';
 13     static RM = 'rm';
 14 
 15     /**
 16      * 
 17      * @param {string} name Coponent name
 18      * @param {*} container Component container|parenbt
 19      * @param {Array} orientation coords on container to point to, and orientation around the container, example: rt, rb; 
 20      *                rt - container coords, rb - orientation
 21      */
 22     constructor(name, container, orientation = [Colibri.UI.ToolTip.RT, Colibri.UI.ToolTip.RB], point = null, permanent = false) {
 23         super(name, container, Element.create('div'));
 24 
 25         this._arrow = new Colibri.UI.TextSpan('arrow', this._element);
 26         this._arrow.shown = true;
 27 
 28         this._contentContainer = new Colibri.UI.Pane('content', this._element);
 29         this._contentContainer.shown = true;
 30 
 31         this.AddClass('app-tooltip-component');
 32 
 33         this._orientation = orientation;
 34         this._element.data('orientation', orientation);
 35         this._point = point;
 36         this._permanent = permanent ?? false;
 37 
 38         this.AddHandler('ShadowClicked', (event, args) => this.__thisShadowClicked(event, args));
 39 
 40     }
 41 
 42     __thisShadowClicked(event, args) {
 43         if(!this._permanent) {
 44             this.shown = false;
 45         }
 46     }
 47 
 48     get container() {
 49         return this._contentContainer.container;
 50     } 
 51 
 52     get orientation() {
 53         return this._orientation;
 54     }
 55 
 56     set orientation(value) {
 57         this._orientation = value;
 58         this._element.data('orientation', value);
 59     }
 60 
 61     _findParent() {
 62         return this.parent ?? null;
 63     }
 64 
 65     _findPointOnParent() {
 66         const parent = this._findParent();
 67         const ori = this._orientation[0];
 68         const parentBounds = parent.container.bounds(true, true);
 69         if(parent.container.contains(this._element)) {
 70             parentBounds.left = 0;
 71             parentBounds.top = 0;
 72         }
 73         switch(ori) {
 74             default:
 75             case Colibri.UI.ToolTip.RB: {
 76                 return {
 77                     left: parentBounds.left + parentBounds.outerWidth, 
 78                     top: parentBounds.top + parentBounds.outerHeight
 79                 };
 80             }
 81             case Colibri.UI.ToolTip.LB: {
 82                 return {
 83                     left: parentBounds.left, 
 84                     top: parentBounds.top + parentBounds.outerHeight
 85                 };
 86             }
 87             case Colibri.UI.ToolTip.LM: {
 88                 return {
 89                     left: parentBounds.left, 
 90                     top: parentBounds.top + (parentBounds.outerHeight / 2)
 91                 };
 92             }
 93             case Colibri.UI.ToolTip.LT: {
 94                 return {
 95                     left: parentBounds.left, 
 96                     top: parentBounds.top
 97                 };
 98             }
 99             case Colibri.UI.ToolTip.RT: {
100                 return {
101                     left: parentBounds.left + parentBounds.outerWidth, 
102                     top: parentBounds.top
103                 };
104             }
105             case Colibri.UI.ToolTip.RM: {
106                 return {
107                     left: parentBounds.left + parentBounds.outerWidth, 
108                     top: parentBounds.top + (parentBounds.outerHeight / 2)
109                 };
110             }
111         }
112     }
113 
114     _getOrientationPoint(pointOnParent) {
115         const ori = this._orientation[1];
116         const thisBounds = this._element.bounds(true, true);
117         switch(ori) {
118             default:
119             case Colibri.UI.ToolTip.RB: {
120                 return {
121                     left: pointOnParent.left, 
122                     top: pointOnParent.top
123                 };
124             }
125             case Colibri.UI.ToolTip.LB: {
126                 return {
127                     left: pointOnParent.left - thisBounds.outerWidth, 
128                     top: pointOnParent.top
129                 };
130             }
131             case Colibri.UI.ToolTip.LM: {
132                 return {
133                     left: pointOnParent.left - thisBounds.outerWidth, 
134                     top: pointOnParent.top - (thisBounds.outerHeight / 2)
135                 };
136             }
137             case Colibri.UI.ToolTip.LT: {
138                 return {
139                     left: pointOnParent.left, 
140                     top: pointOnParent.top 
141                 };
142             }
143             case Colibri.UI.ToolTip.RT: {
144                 return {
145                     left: pointOnParent.left - thisBounds.outerWidth, 
146                     top: pointOnParent.top - thisBounds.outerHeight
147                 };
148             }
149             case Colibri.UI.ToolTip.RM: {
150                 return {
151                     left: pointOnParent.left, 
152                     top: pointOnParent.top - (thisBounds.outerHeight / 2)
153                 };
154             }
155         }
156     }
157 
158     _setPosition() {
159 
160         const pointOnParent = this._point || this._findPointOnParent();
161         const point = this._getOrientationPoint(pointOnParent);
162         this.styles = {left: point.left + 'px', top: point.top + 'px'};
163 
164     }
165 
166     _checkPosition() {
167         const thisBounds = this._element.bounds(true, true);
168 
169         let orientation = [].concat(this._orientation);
170         if(thisBounds.top + thisBounds.outerHeight > window.innerHeight) {
171             // надо двинуть точку на паренте и относительную ориентацию
172             // справа на лево, или слева на право
173             orientation[0] = orientation[0].replaceAll('t', 'b');
174             orientation[1] = orientation[1].replaceAll('b', 't');
175             
176         }
177         if(thisBounds.left + thisBounds.outerWidth > window.innerWidth) {
178             // надо двинуть точку на паренте и относительную ориентацию
179             // справа на лево, или слева на право
180             orientation[0] = orientation[0].replaceAll('r', 'l');
181             orientation[1] = orientation[1].replaceAll('r', 'l');
182         }
183         
184         if(this._orientation != orientation) {
185             this._orientation = orientation;
186             this._setPosition();
187         }
188 
189     }
190 
191     get shown() {
192         return super.shown;
193     }
194     set shown(value) {
195         super.shown = value;
196         
197         this.BringToFront();
198         this.hasShadow = value && !this._permanent;
199         if(value) {
200             this._element.css('visibility', 'hidden');
201             Colibri.Common.Delay(10).then(() => {
202                 this._setPosition();
203                 this._element.css('visibility', 'visible');
204             });    
205         }
206     }
207 
208     Show(parent = null) {
209         if(parent) {
210             this.parent = parent;
211         }
212         if(!this.shown) {
213             this.shown = true;
214         }
215     }

228     set value(value) {
229         this._contentContainer.value = value;

233      * Is the tooltip permenantly shown
234      * @type {Boolean}
235      */
236     get permanent() {
237         return this._permanent;
238     }
239     /**
240      * Is the tooltip permenantly shown
241      * @type {Boolean}
242      */
243     set permanent(value) {
244         this._permanent = value;
245         this._showPermanent();
246     }
247     _showPermanent() {
248         if(this._permanent && !this.shown) {
249             this.shown = true;
250             this.hasShadow = false;
251         } 
252     }
