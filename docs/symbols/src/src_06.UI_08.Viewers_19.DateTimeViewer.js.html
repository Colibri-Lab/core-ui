1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.DateTimeViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-datetime-viewer-component');
 18 
 19         let dateformat = App.DateFormat || 'ru-RU';
 20         this._format = new Intl.DateTimeFormat(dateformat, {day: '2-digit', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit'});
 21         
 22         this._value = null;
 23 
 24     }
 25 
 26     /**
 27      * Value
 28      * @type {Date}
 29      */
 30     get value() {
 31         return this._value;
 32     }
 33 
 34     /**
 35      * Value
 36      * @type {Date}
 37      */
 38     set value(value) {
 39         value = this._convertValue(value);
 40         if(typeof value === 'string') {
 41             value = value.toDate();
 42         }
 43         else if(typeof value === 'number') {
 44             value = value.toDateFromUnixTime();
 45         }
 46         this._value = value;
 47     
 48         try {
 49             super.value = this._value && this._format.format(this._value);
 50         } catch(e) {
 51             super.value = '';
 52         }
 53 
 54     }
 55 
 56 }
 57 Colibri.UI.Viewer.Register('Colibri.UI.DateTimeViewer', '#{ui-viewers-datetime}');
