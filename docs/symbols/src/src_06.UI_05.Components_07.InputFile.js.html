1 Colibri.UI.Input.File = class extends Colibri.UI.Component {

  3     constructor(name, container) {

  5         this.AddClass('app-component-input-type-file');

  9 
 10         this._title = '';
 11 
 12         this._element.querySelector('.app-component-input-type-file-text').html(this.__title());
 13     }
 14 
 15     _registerEvents() {
 16         super._registerEvents();
 17         this.RegisterEvent('InputFileChanged', false, 'Изменён выбранный файл/файлы');
 18     }
 19 
 20     _handleEvents() {
 21         this.AddHandler('Clicked', (sender, args) => {
 22             this._input.click();
 23         });
 24 
 25         this._input.addEventListener('change', (event) => {
 26             this._inputFiles = [];
 27             for (let i = 0; i < this._input.files.length; ++i) {
 28                 this._inputFiles.push(this._input.files[i]);
 29             }
 30             this._input.value = '';
 31             this.Dispatch('InputFileChanged', {files: this._inputFiles});
 32         });
 33     }
 34 
 35     __title() {
 36         return this.multiple ? '#{ui-files-choosefiles}' : '#{ui-files-choosefile}';
 37     }
 38 
 39     get container() {
 40         return this._element.querySelector('.app-component-input-type-file-text');
 41     }
 42 
 43     get title() {
 44         return this._element.querySelector('.app-component-input-type-file-text').html();
 45     }
 46 
 47     set title(value) {
 48         this._title = value;
 49         this._element.querySelector('.app-component-input-type-file-text').html(value || this.__title());
 50     }
 51 
 52     get multiple() {
 53         return this._input.getAttribute('multiple');
 54     }
 55 
 56     set multiple(value) {
 57         if (value) {
 58             this._input.setAttribute('multiple', 'true');
 59         } else {
 60             this._input.removeAttribute('multiple');
 61         }
 62         if(!this._title) {
 63             this._element.querySelector('.app-component-input-type-file-text').html(this.__title());
 64         }
 65     }
 66 
 67     Files() {
 68         if (this._input.hasAttribute('multiple')) {
 69             return this._inputFiles;
 70         }
 71         return this._inputFiles[0];
 72 
 73     }
 74 
 75     ClearData() {
 76         this._input.value = null;
 77         this._inputFiles = [];
 78     }

 80 }
