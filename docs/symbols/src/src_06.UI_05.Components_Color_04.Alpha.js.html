1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Color.Alpha = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         super(name, container, Element.create('div'));
 15         this.AddClass('app-color-alpha-component');
 16 
 17         this.RegisterEvent('Changed', false, 'Когда значение изменилось');
 18 
 19         this._pointer = new Colibri.UI.Pane(this.name + '_pointer', this);
 20         this._pointer.AddClass('app-color-alpha-pointer-component');
 21         this._pointer.shown = true;
 22 
 23         this.AddHandler('Clicked', (event, args) => this.__lineClicked(event, args));
 24         this.tabIndex = 0;
 25         this.AddHandler('KeyDown', (event, args) => this.__keyDown(event, args));
 26 
 27         new Colibri.UI.Drag(this._pointer.container, this.container, (left, top) => {
 28             this._pointer.styles = {top: top + 'px'};
 29             this._setNewValue(top);
 30         });
 31         
 32     }
 33 
 34     _setNewValue(top) {
 35         const bounds = this._element.bounds();
 36         const trackbounds = this._pointer.container.bounds();
 37 
 38         top = top + trackbounds.outerHeight / 2;
 39         const percent = top * 100 / bounds.outerHeight;
 40         const alpha = percent * 255 / 100;
 41         
 42         this._value = Math.round(alpha);
 43         this.Dispatch('Changed');          
 44     }
 45 
 46     __keyDown(event, args) {
 47         if(['ArrowUp', 'ArrowDown'].indexOf(args.domEvent.code) !== -1) {
 48             if(args.domEvent.code == 'ArrowUp') {
 49                 this._value -= 1;
 50                 this._setTrackPosition();
 51                 this.Dispatch('Changed'); 
 52             }
 53             else if(args.domEvent.code == 'ArrowDown') {
 54                 this._value += 1;
 55                 this._setTrackPosition();
 56                 this.Dispatch('Changed'); 
 57             }
 58         }
 59     }
 60 
 61     __lineClicked(event, args) {
 62         const bounds = this._element.bounds();
 63         const trackbounds = this._pointer.container.bounds();
 64         const e = args.domEvent;
 65         let top = e.pageY - bounds.top - trackbounds.outerHeight / 2;
 66         this._pointer.styles = {top: top + 'px'};
 67         this._setNewValue(top);
 68     }
 69 
 70     _setTrackPosition() {
 71         const bounds = this._element.bounds()
 72         const topPercent = this._value * 100 / 255;
 73         const top = topPercent * bounds.outerHeight / 100;
 74         this._pointer.styles = {top: top + 'px'};
 75     }
 76     
 77     get value() {
 78         return this._value;
 79     }
 80 
 81     set value(value) {
 82         this._value = value;
 83         this._setTrackPosition();
 84     }
 85 
 86 }
