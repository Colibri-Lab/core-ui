 61     _movingHandler(e) {
 62         const windowElement = e.currentTarget.closest('.app-component-window');
 63         const windowContainer = windowElement.querySelector('.app-component-window-container');
 64 
 65         const point = windowElement.tag('movingPoint');
 66 
 67         windowContainer.css('left', (e.pageX - point.left - parseInt(windowContainer.css('margin-left'))) + 'px');
 68         windowContainer.css('top', (e.pageY - point.top - parseInt(windowContainer.css('margin-top'))) + 'px');
 69     }
 70 
 71     _movingStartHandler(e) {
 72         
 73         if(e.target.is('button')) {
 74             return false;
 75         }
 76 
 77         const windowElement = e.currentTarget.closest('.app-component-window');
 78         const windowComponent = windowElement.tag('component');
 79         windowComponent.moving = true;
 80         windowElement.addEventListener('mousemove', windowComponent._movingHandler, true);
 81         windowElement.addEventListener('mouseup', windowComponent._movingStopHandler, true);
 82         windowElement.tag('movingPoint', {left: e.layerX, top: e.layerY});
 83     }
 84 
 85     _movingStopHandler(e) {
 86         const windowElement = e.currentTarget.closest('.app-component-window');
 87         const windowComponent = windowElement.tag('component');
 88         windowComponent.moving = false;
 89         windowElement.removeEventListener('mousemove', windowComponent._movingHandler, true);
 90         windowElement.removeEventListener('mouseup', windowComponent._movingStopHandler);
 91     }

137     MinimizeToggle() {
138         const content = this._minimizedGetContentMethod(this);
139         if(content) {
140             this._minimizedContent.html(content);
141         }
142 
143         if(this._minimizable && this._state === 'normal') {
144 
145             this.AddClass('-minimized');
146             super.width = this._minimizedSize[0];
147             super.height = this._minimizedSize[1];
148             super.right = this._minimizedPosition[0];
149             super.bottom = this._minimizedPosition[1];
150             this._state = 'minimized';
151 
152 
153         } else if(this._minimizable && this._state === 'minimized') {
154             this.RemoveClass('-minimized');   
155             super.width = null;
156             super.height = null;
157             super.right = null;
158             super.bottom = null;
159             this._state = 'normal';
160 
161         }
