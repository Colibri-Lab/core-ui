1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.PaneSwitcher = class extends Colibri.UI.Pane {
  7     
  8     constructor(name, container) {
  9         /* создаем компонент и передаем шаблон */
 10         super(name, container);
 11         this.AddClass('colibri-ui-paneswitcher');
 12 
 13         this.AddHandler('ChildsProcessed', (event, args) => this.ForEach((name, component) => component.Disconnect()));
 14 
 15     }
 16 
 17     /**
 18      * Show/Hide element
 19      * @type {Number|String}
 20      */
 21     get value() {
 22         return this._value;
 23     }
 24     /**
 25      * Show/Hide element
 26      * @type {Number|String}
 27      */
 28     set value(value) {
 29         this._value = value;
 30         this._showValue();
 31     }
 32     _showValue() {
 33         
 34         this.ForEach((name, component) => component.Disconnect())
 35         this.Children(this._value).ConnectTo(this);
 36 
 37     }
 38 
 39 }
