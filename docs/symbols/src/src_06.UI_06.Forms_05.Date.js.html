1 Colibri.UI.Forms.Date = class extends Colibri.UI.Forms.Field {
  2 
  3     RenderFieldContainer() {
  4 
  5         this.AddClass('app-component-date-field');
  6 
  7 
  8         const contentContainer = this.contentContainer;
  9 
 10         this._input = new Colibri.UI.DateSelector(this._name + '-input', contentContainer);
 11         this._input.shown = true;
 12         this._input.AddHandler('Changed', (event, args) => this.Dispatch('Changed', Object.assign(args || {}, {component: this})));
 13         this._input.AddHandler('KeyUp', (event, args) => this.Dispatch('KeyUp', args));
 14         this._input.AddHandler('Clicked', (event, args) => {
 15             this.Focus();
 16             this.Dispatch('Clicked', args);
 17             args.domEvent.stopPropagation();
 18             return false;
 19         });
 20         
 21         this._icon = new Colibri.UI.Icon(this.name + '-clear', contentContainer);
 22         this._icon.shown = true;
 23         this._icon.value = Colibri.UI.ClearIcon;
 24         this._icon.AddHandler('Clicked', (event, args) => {
 25             this.value = null;
 26         });
 27 
 28 
 29         this._input.AddHandler('PopupOpened', (event, args) => this.AddClass('-opened'));
 30         this._input.AddHandler('PopupClosed', (event, args) => this.RemoveClass('-opened'));
 31 
 32         if(this._fieldData?.params?.readonly === undefined) {
 33             this.readonly = false;    
 34         }
 35         else {
 36             this.readonly = this._fieldData?.params?.readonly;
 37         }
 38         if(this._fieldData?.params?.enabled === undefined) {
 39             this.enabled = true;
 40         }
 41         else {
 42             this.enabled = this._fieldData.params.enabled;
 43         }
 44 
 45     }
 46 
 47     Focus() {
 48         this._input.Focus();
 49     }
 50 
 51     get readonly() {
 52         return this._input.readonly;
 53     }
 54 
 55     set readonly(value) {

 68 
 69     get value() {

 72     }
 73 
 74     set value(value) {
 75         if(typeof value == 'string') {
 76             value = new Date(value);
 77         }

 79     }

 81
