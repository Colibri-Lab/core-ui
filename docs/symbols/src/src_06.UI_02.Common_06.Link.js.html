1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Link = class extends Colibri.UI.Component {
  7     constructor(name, container, value) {
  8         super(name, container, Element.create('a'));
  9         this.AddClass('app-component-link');
 10         this.value = value;
 11     }
 12 
 13     get href() {
 14         return this._element.attr('href');
 15     }
 16     set href(value) {
 17         this._element.attr('href', value);
 18     }
 19 
 20     get navigate() {
 21         return this._navigate;
 22     }
 23     set navigate(value) {
 24         this._navigate = value;
 25         this._element.attr('href', '#' + Date.Mc());
 26         this.AddHandler('Clicked', (event, args) => {
 27             App.Router.Navigate(this._navigate?.url ?? '/', this._navigate?.options ?? {});
 28             args.domEvent.stopPropagation();
 29             args.domEvent.preventDefault();
 30             return false;
 31         });
 32     }
 33 
 34     get value() {
 35         return this._element.html();
 36     }
 37     set value(value) {
 38         this._element.html(value);
 39     }
 40 
 41     get target() {
 42         return this._element.attr('target');
 43     }
 44     set target(value) {
 45         this._element.attr('target', value);
 46     }
 47 
 48     get download() {
 49         return this._element.attr('download');
 50     }
 51     set download(value) {
 52         this._element.attr('download', value);
 53     }
 54        
 55     /**
 56      * Enable/Disable
 57      * @type {boolean}
 58      */
 59     get enabled() {
 60         return this._element.attr('disabled') !== 'disabled';
 61     }
 62     /**
 63      * Enable/Disable
 64      * @type {boolean}
 65      */
 66     set enabled(value) {
 67         this._element.attr('disabled', value === true || value === 'true' ? null : 'disabled');
 68     }
 69 
 70 }
