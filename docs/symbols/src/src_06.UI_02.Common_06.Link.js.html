1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Link = class extends Colibri.UI.Component {
  7     /**
  8      * @constructor
  9      * @param {string} name name of component
 10      * @param {Element|Colibri.UI.Component} container container element and component
 11      */
 12     constructor(name, container, value) {
 13         super(name, container, Element.create('a'));
 14         this.AddClass('app-component-link');
 15         this.value = value;
 16     }
 17 
 18     /**
 19      * Href string
 20      * @type {string}
 21      */
 22     get href() {
 23         return this._element.attr('href');
 24     }
 25     /**
 26      * Href string
 27      * @type {string}
 28      */
 29     set href(value) {
 30         this._element.attr('href', value);
 31     }
 32 
 33     /**
 34      * Navigate to
 35      * @type {{url,options}}
 36      */
 37     get navigate() {
 38         return this._navigate;
 39     }
 40     /**
 41      * Navigate to
 42      * @type {{url,options}}
 43      */
 44     set navigate(value) {
 45         this._navigate = value;
 46         this._element.attr('href', '#' + Date.Mc());
 47         this.AddHandler('Clicked', (event, args) => {
 48             App.Router.Navigate(this._navigate?.url ?? '/', this._navigate?.options ?? {});
 49             args.domEvent.stopPropagation();
 50             args.domEvent.preventDefault();
 51             return false;
 52         });
 53     }
 54 
 55     /**
 56      * Value string
 57      * @type {string}
 58      */
 59     get value() {
 60         return this._element.html();
 61     }
 62     /**
 63      * Value string
 64      * @type {string}
 65      */
 66     set value(value) {
 67         this._element.html(value);
 68     }
 69 
 70     /**
 71      * Target value
 72      * @type {_blank,_self,_top}
 73      */
 74     get target() {
 75         return this._element.attr('target');
 76     }
 77     /**
 78      * Target value
 79      * @type {_blank,_self,_top}
 80      */
 81     set target(value) {
 82         this._element.attr('target', value);
 83     }
 84 
 85     /**
 86      * Download attribute
 87      * @type {string}
 88      */
 89     get download() {
 90         return this._element.attr('download');
 91     }
 92     /**
 93      * Download attribute
 94      * @type {string}
 95      */
 96     set download(value) {
 97         this._element.attr('download', value);
 98     }
 99        
100     /**
101      * Enable/Disable
102      * @type {boolean}
103      */
104     get enabled() {
105         return this._element.attr('disabled') !== 'disabled';
106     }
107     /**
108      * Enable/Disable
109      * @type {boolean}
110      */
111     set enabled(value) {
112         this._element.attr('disabled', value === true || value === 'true' ? null : 'disabled');
113     }
114 
115 }
