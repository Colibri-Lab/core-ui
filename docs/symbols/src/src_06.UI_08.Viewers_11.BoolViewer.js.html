1 Colibri.UI.BoolViewer = class extends Colibri.UI.Viewer {
  2 
  3     constructor(name, container, element = null, root = null) {
  4         super(name, container, element || Element.create('span'), root);
  5         this.AddClass('app-bool-viewer-component');
  6     }
  7 
  8     set value(value) {
  9         this._value = value;
 10         this._showValue();
 11     }
 12 
 13     get value() {
 14         return this._value ?? null;
 15     }
 16 
 17     get field() {
 18         return this._field;
 19     }
 20 
 21     set field(field) {
 22         this._field = field;
 23         this._showValue();
 24     }
 25 
 26     _showValue() {
 27 
 28         if(!this._field || this._value === undefined) {
 29             this._element.html('');
 30             return;
 31         }
 32 
 33         
 34         let value = this._value[this._field.selector?.value || 'value'] ?? this._value;
 35         value = this._convertValue(value);
 36 
 37         const values = Object.values(this._field.values);
 38         const found = values.filter(o => o.value == value);
 39 
 40         if(this._field?.params?.colors) {
 41             this.RemoveClass(this._field.params.colors.map(v => v!=''));
 42             if(value) {
 43                 this.AddClass(this._field.params.colors[0]);
 44             } else {
 45                 this.AddClass(this._field.params.colors[1]);
 46             }
 47         }
 48 
 49         this._element.html(found.length == 1 ? found[0].title : value);
 50     }
 51 
 52 
 53 }
