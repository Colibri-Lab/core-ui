  7     /**
  8      * Validate UK post code
  9      * @param {string} string uk post code
 10      * @returns {boolean}
 11      */
 12     static IsUKPostCode(string) {
 13         var postcodeRegEx = /[A-Z]{1,2}[0-9A-Z]{1,2} ?[0-9][A-Z]{2}/i; 
 14         return postcodeRegEx.test(string); 
 15     }
 16 
 17     /**
 18      * Validate Russian tax code
 19      * @param {string} value russian tax code
 20      * @returns {boolean}
 21      */
 22     static IsValidRuINN(value){
 23         //первая цифра ИНН может быть нулём, поэтому он не может быть числом
 24         if(typeof value !== 'string'){
 25             console.error(`ИНН должен быть строкой. Получено: ${typeof value}`);
 26             return false;
 27         }
 28         
 29         //ИНН юрлиц - 10 символов, ИП - 12 символов
 30         if(value.length !== 10 && value.length !== 12){
 31             console.error(`Некорректная длинна ИНН: ${value.length}`);
 32             return false;
 33         }
 34         
 35         //преобразуем строку в массив цифр
 36         const arNumbers = value.split('');
 37         if(arNumbers.length === 0){
 38             console.error('Не удалось разобрать строку в массив символов');
 39             return false;
 40         }
 41         
 42         //проверим что у нас в массиве только цифры
 43         for(let symbol of arNumbers){
 44             if(isNaN(Number(symbol))){
 45                 console.error(`Некорректный символ "${symbol}" в ИНН`);
 46                 return false;
 47             }
 48         }
 49         
 50         //формула для юрлиц и ИП отличается
 51         if(arNumbers.length === 10){
 52             //переменная для итоговой суммы
 53             let checkSum;
 54             
 55             //каждую цифру мы умножаем на свой коэффициент
 56             //а потом получаем остаток от деления на 11 и на 10
 57             checkSum = ((
 58                 2 * arNumbers[0] +
 59                 4 * arNumbers[1] +
 60                 10 * arNumbers[2] +
 61                 3 * arNumbers[3] +
 62                 5 * arNumbers[4] +
 63                 9 * arNumbers[5] +
 64                 4 * arNumbers[6] +
 65                 6 * arNumbers[7] +
 66                 8 * arNumbers[8]
 67             ) % 11 ) % 10;
 68             
 69             //проверяем что десятый символ ИНН совпадает с контрольной суммой
 70             if(checkSum === Number(arNumbers[9])){
 71                 return true;
 72             }else{
 73                 console.error(
 74                     `Контрольная сумма не совпала с десятым символом ${checkSum} != ${arNumbers[9]}`
 75                 );
 76                 return false;
 77             }
 78             
 79         //код для ИП
 80         }else if(arNumbers.length === 12){
 81             //в этом случае будет две контрольные суммы
 82             let checkSumOne, checkSumTwo;
 83             
 84             checkSumOne = ((
 85                 7 * arNumbers[0] +
 86                 2 * arNumbers[1] +
 87                 4 * arNumbers[2] +
 88                 10 * arNumbers[3] +
 89                 3 * arNumbers[4] +
 90                 5 * arNumbers[5] +
 91                 9 * arNumbers[6] +
 92                 4 * arNumbers[7] +
 93                 6 * arNumbers[8] +
 94                 8 * arNumbers[9]
 95             ) % 11 ) % 10;
 96             
 97             checkSumTwo = ((
 98                 3 * arNumbers[0] +
 99                 7 * arNumbers[1] +
100                 2 * arNumbers[2] +
101                 4 * arNumbers[3] +
102                 10 * arNumbers[4] +
103                 3 * arNumbers[5] +
104                 5 * arNumbers[6] +
105                 9 * arNumbers[7] +
106                 4 * arNumbers[8] +
107                 6 * arNumbers[9] +
108                 8 * arNumbers[10]
109             ) % 11 ) % 10;
110             
111             //в этом случае мы проверяем 11 и 12 символы
112             if(checkSumOne === Number(arNumbers[10]) &&
113                 checkSumTwo === Number(arNumbers[11]))
114             {
115                 return true;
116                 
117             }else if(checkSumOne !== Number(arNumbers[10])){
118                 console.error(
119                     `Первая контрольная сумма не совпала с одиннадцатым символом ${checkSumOne} != ${arNumbers[10]}`
120                 );
121                 return false;
122                 
123             }else if(checkSumTwo !== Number(arNumbers[11])){
124                 console.error(
125                     `Вторая контрольная сумма не совпала с двенадцатым символом ${checkSumTwo} != ${arNumbers[11]}`
126                 );
127                 return false;
128             }
129         }
130     }
