1 /**
  2  * @class
  3  * @extends Colibri.UI.FlexBox
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.YearQuarterSelector = class extends Colibri.UI.FlexBox {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      * @param {Array} values values to show
 13      */    
 14     constructor(name, container, values = null) {
 15         super(name, container);
 16         this.AddClass('app-component-year-quarter-selector');
 17 
 18         this._values = values;
 19 
 20         this._yearSelector = new Colibri.UI.Selector(this.name + '-year', this);
 21         this._quarterSelector = new Colibri.UI.Selector(this.name + '-quarter', this);
 22         this._yearSelector.searchable = false;
 23         this._yearSelector.readonly = false;
 24         this._quarterSelector.searchable = false;
 25         this._quarterSelector.readonly = false;
 26 
 27         this._quartersTexts = [
 28             '#{ui-yearquarter-selector-1}',
 29             '#{ui-yearquarter-selector-2}',
 30             '#{ui-yearquarter-selector-3}',
 31             '#{ui-yearquarter-selector-4}',
 32         ];
 33         
 34         this._yearSelector.shown = true;
 35         this._quarterSelector.shown = true;
 36 
 37         this._yearSelector.AddHandler('Changed', (event, args) => this.__yearSelectorChanged(event, args));
 38         this._quarterSelector.AddHandler('Changed', (event, args) => this.__quarterSelectorChanged(event, args));
 39 
 40     }
 41     
 42     /** @protected */
 43     _registerEvents() {
 44         super._registerEvents();
 45         this.RegisterEvent('Changed', false, 'Когда значение изменилось');
 46     }
 47 
 48     /**
 49      * @private
 50      * @param {Colibri.Events.Event} event event object
 51      * @param {*} args event arguments
 52      */ 
 53     __yearSelectorChanged(event, args) {
 54         const selectedYear = this._yearSelector.value.value;
 55         
 56         const quarters = [];
 57         let selectedQuarter = null;
 58         for(const period of this._values) {
 59             if(period.year === selectedYear) {
 60                 selectedQuarter = period.quarters[0];
 61                 for(const quarter of period.quarters) {
 62                     quarters.push({value: quarter, title: this._quartersTexts[quarter - 1]});
 63                 }
 64             }
 65         }
 66 
 67         this._quarterSelector.values = quarters;
 68         this._quarterSelector.value = selectedQuarter;
 69         this._value = {year: selectedYear, quarter: selectedQuarter};
 70         this.Dispatch('Changed', Object.assign(args, {value: this._value}));
 71     }
 72 
 73     /**
 74      * @private
 75      * @param {Colibri.Events.Event} event event object
 76      * @param {*} args event arguments
 77      */ 
 78     __quarterSelectorChanged(event, args) {
 79         this._value = {year: this._yearSelector.value.value, quarter: this._quarterSelector.value.value};
 80         this.Dispatch('Changed', Object.assign(args, {value: this._value}));
 81     }
 82 
 83     /**
 84      * Quarters placeholder text
 85      * @type {String}
 86      */
 87     get quarterPlaceholder() {
 88         return this._quarterSelector.placeholder;
 89     }
 90     /**
 91      * Quarters placeholder text
 92      * @type {String}
 93      */
 94     set quarterPlaceholder(value) {
 95         this._quarterSelector.placeholder = value;
 96     }
 97 
 98     /**
 99      * Years placeholder text
100      * @type {String}
101      */
102     get yearPlaceholder() {
103         return this._yearSelector.placeholder;
104     }
105     /**
106      * Years placeholder text
107      * @type {String}
108      */
109     set yearPlaceholder(value) {
110         this._yearSelector.placeholder = value;
111     }
112 
113     /**
114      * Array of periods {Number year, Array quarters}
115      * @type {Array}
116      */
117     get values() {
118         return this._values;
119     }
120     /**
121      * Array of periods {Number year, Array quarters}
122      * @type {Array}
123      */
124     set values(value) {
125         this._values = value;
126         this._showValues();
127     }
128     /** @private */
129     _showValues() {
130         
131         if(!this._values) {
132             return;
133         }
134 
135         const selectedYear = this._value?.year ?? this._values[0].year;
136         const selectedQuarter = this._value?.quarter ?? this._values[0].quarters[0];
137         
138         const quarters = [];
139         const years = [];
140         for(const period of this._values) {
141             years.push({value: period.year, title: period.year});
142             if(period.year === selectedYear) {
143                 for(const quarter of period.quarters) {
144                     quarters.push({value: quarter, title: this._quartersTexts[quarter - 1]});
145                 }
146             }
147         }
148 
149         years.sort((a, b) => a.value > b.value ? -1 : 1);
150 
151         this._yearSelector.values = years;
152         this._quarterSelector.values = quarters;
153         this._yearSelector.value = selectedYear;
154         this._quarterSelector.value = selectedQuarter;
155 
156     }
157 
158     /**
159      * Array [year, quarter] or string year-quarter
160      * @type {Array|String}
161      */
162     get value() {
163         return this._value;
164     }
165     /**
166      * Array [year, quarter] or string year-quarter
167      * @type {Array|String}
168      */
169     set value(value) {
170         this._value = value;
171         this._showValue();
172     }
173     /** @private */
174     _showValue() {
175         
176         if(typeof value === 'string') {
177             this._value = this._value.split('-');
178         } 
179 
180         this._showValues();
181 
182     }
183 
184 }
