1 /**
  2  * @class
  3  * @extends Colibri.UI.FlexBox
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.YearMonthSelector = class extends Colibri.UI.FlexBox {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      * @param {Array} values values to show
 13      */    
 14     constructor(name, container, values = null) {
 15         super(name, container);
 16         this.AddClass('app-component-year-month-selector');
 17 
 18         this._values = values;
 19 
 20         this._selector = new Colibri.UI.Selector(this.name + '-selector', this);
 21         this._selector.searchable = false;
 22         this._selector.readonly = false;
 23         this._selector.shown = true;
 24 
 25         this._selector.AddHandler('Changed', (event, args) => this.__selectorChanged(event, args));
 26 
 27     }
 28     
 29     _registerEvents() {
 30         super._registerEvents();
 31         this.RegisterEvent('Changed', false, 'Когда значение изменилось');
 32     }
 33 
 34     __selectorChanged(event, args) {
 35         this.Dispatch('Changed', Object.assign(args, {value: this._selector.value.value}));
 36     }
 37 
 38     /**
 39      * Placeholder text
 40      * @type {String}
 41      */
 42     get placeholder() {
 43         return this._yearSelector.placeholder;
 44     }
 45     /**
 46      * Placeholder text
 47      * @type {String}
 48      */
 49     set placeholder(value) {
 50         this._yearSelector.placeholder = value;
 51     }
 52 
 53     /**
 54      * Array of periods {Number year, Array quarters}
 55      * @type {Array}
 56      */
 57     get values() {
 58         return this._values;
 59     }
 60     /**
 61      * Array of periods {Number year, Array quarters}
 62      * @type {Array}
 63      */
 64     set values(value) {
 65         this._values = value;
 66         this._showValues();
 67     }
 68 
 69     _showValues() {
 70         
 71         if(!this._values) {
 72             return;
 73         }
 74 
 75         const ret = [];
 76         const values = this._values.sort((a, b) => a > b ? 1 : -1);
 77         for(const date of values) {
 78             ret.push(date.intlFormat(false, true));
 79         }
 80 
 81         this._selector.values = ret;
 82 
 83     }
 84 
 85     /**
 86      * Array [year, quarter] or string year-quarter
 87      * @type {Array|String}
 88      */
 89     get value() {
 90         return this._value;
 91     }
 92     /**
 93      * Array [year, quarter] or string year-quarter
 94      * @type {Array|String}
 95      */
 96     set value(value) {
 97         this._value = value;
 98         this._showValue();
 99     }
100     _showValue() {
101         
102         if(typeof value === 'string') {
103             this._value = this._value.split('-');
104         } 
105 
106         this._showValues();
107 
108     }
109 
110 }
