1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Editor
  5  */
  6 Colibri.UI.TextAreaEditor = class extends Colibri.UI.Editor {
  7     /**
  8      * @constructor
  9      * @param {string} name name of component
 10      * @param {Element|Colibri.UI.Component} container container element and component
 11      */ 
 12     constructor(name, container) {
 13         super(name, container, Element.create('textarea'));
 14         this.AddClass('app-textarea-editor-component');
 15         
 16         this._element.addEventListener('focus', (e) => this.Focus());
 17         this._element.addEventListener('blur', (e) => this.Blur());
 18         this._element.addEventListener('change', (e) => this.__elementChanged(e));
 19         this._element.addEventListener('keydown', (e) => this.__elementChanged(e));
 20 
 21 
 22     }
 23 
 24     /** @private */
 25     __elementChanged(e) {
 26         if(this.value) {
 27             this._setFilled();
 28         } else {
 29             this._unsetFilled();
 30         }
 31 
 32     }
 33 
 34     /** @private */
 35     _updateFieldData() {
 36         this.placeholder = this._fieldData?.placeholder;
 37     }
 38 
 39     /**
 40      * Validate
 41      */
 42     Validate() {
 43         
 44     }
 45 
 46     /**
 47      * Focus on editor
 48      */
 49     Focus() {
 50         this._element.focus();
 51         // this._element.select();
 52         this.parent.parent.AddClass('-focused');
 53     } 
 54 
 55     /**
 56      * Remove focus from editor
 57      */
 58     Blur() {
 59         this.parent.parent.RemoveClass('-focused');
 60     }
 61 
 62     /**
 63      * Readonly
 64      * @type {boolean}
 65      */
 66     get readonly() {
 67         return this._fieldData.readonly;
 68     }  
 69  
 70     /**
 71      * Readonly
 72      * @type {boolean}
 73      */
 74     set readonly(value) {
 75         this._fieldData.readonly = value === true || value === 'true';
 76         if(value === true || value === 'true') {
 77             this._element.attr('readonly', 'readonly');
 78         }
 79         else {
 80             this._element.attr('readonly', null);
 81         }
 82     }
 83 
 84     /**
 85      * Placeholder
 86      * @type {string}
 87      */
 88     get placeholder() {
 89         return this._element.attr('placeholder');
 90     }
 91 
 92     /**
 93      * Placeholder
 94      * @type {string}
 95      */
 96     set placeholder(value) {
 97         this._element.attr('placeholder', value ? value[Lang.Current] ?? value : '');
 98     }
 99 
100     /**
101      * Value
102      * @type {string}
103      */
104     get value() {
105         return this._element.value;
106     }
107 
108     /**
109      * Value
110      * @type {string}
111      */
112     set value(value) {
113         this._element.value = value;
114         this.Validate();
115         if(value) {
116             this._setFilled();
117         } else {
118             this._unsetFilled();
119         }
120 
121     }
122 
123     /**
124      * Enable/Disable
125      * @type {boolean}
126      */
127     get enabled() {
128         return this._element.attr('disabled') != 'disabled';
129     }
130 
131     /**
132      * Enable/Disable
133      * @type {boolean}
134      */
135     set enabled(value) {
136         if(value) {
137             this.RemoveClass('ui-disabled');
138             this._element.attr('disabled', null);
139         }
140         else {
141             this.AddClass('ui-disabled');
142             this._element.attr('disabled', 'disabled');
143         }
144     }
145 
146 }
