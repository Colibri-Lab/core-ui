 78     /** @protected */
 79     _registerEvents() {
 80         this.RegisterEvent('OrientationChanged', false, 'Когда ориентация была изменена');
 81         this.RegisterEvent('ThemeChanged', false, 'Когда тема изменена');
 82         this.RegisterEvent('NotificationTapped', false, 'When push notification is tapped');
 83     }

 85     /**
 86      * Detects the platform and initializes theme detection and push notifications.
 87      * @private
 88      */
 89     _detect() {
 90         if(!window.hasOwnProperty("cordova")){
 91             this._platform = 'web';
 92             this._theme = window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches ? Colibri.Devices.Device.Dark : Colibri.Devices.Device.Light;
 93         }
 94         else {
 95             this._platform = cordova.platformId;
 96             if(this.isIOs) {
 97                 this._themeDetectionPlugin = this.Plugin('ThemeDetection');
 98                 this._themeDetectionPlugin.isDarkModeEnabled((result) => {
 99                     this._theme = result.value ? Colibri.Devices.Device.Dark : Colibri.Devices.Device.Light;
100                 }, () => {
101                     this._theme = Colibri.Devices.Device.Light;
102                 })
103             } else {
104                 this._themeDetectionPlugin = this.Plugin('AutoTheme');
105                 if(this._themeDetectionPlugin) {
106                     this._themeDetectionPlugin.getTheme((isdark) => {
107                         this._theme = isdark ? Colibri.Devices.Device.Dark : Colibri.Devices.Device.Light;
108                     });
109                 }
110             }
111             this._pushNotifications = this.Plugin('pushNotification');
112             if(this._pushNotifications) {
113                 this._pushNotifications.registration((token) => {
114                     console.log(token);
115                 }, e => console.log(e));
116                 this._pushNotifications.tapped((payload) => {
117                     this.Dispatch('NotificationTapped', {payload: payload})
118                 }, e => console.log(e));
119             }    
120 
121             this._localNotifications = new Colibri.Devices.LocalNotifications(this);
122         }
123 
124     }

126     /**
127      * Binds events related to the device.
128      * @private
129      */
130     _bindDeviceEvents() {
131         window.addEventListener("orientationchange", () => {
132             const old = this._currentOrientation;
133             this._currentOrientation = screen.orientation.type;
134             this.Dispatch('OrientationChanged', {current: this._currentOrientation, old: old});
135         });
136         window.matchMedia("(prefers-color-scheme: dark)").addEventListener("change", (e) => {
137             const old = this._theme;
138             this._theme = e.matches ? Colibri.Devices.Device.Dark : Colibri.Devices.Device.Light;
139             this.Dispatch('ThemeChanged', {current: this._theme, old: old});
140         });
141         if(this._themeDetectionPlugin) {
142             window.onThemeChange = (isdark) => {
143                 const old = this._theme;
144                 this._theme = isdark ? Colibri.Devices.Device.Dark : Colibri.Devices.Device.Light;
145                 this.Dispatch('ThemeChanged', {current: this._theme, old: old});
146             };
147         }
148         
149         
150     }

200     /**
201      * Sets the background mode of the device.
202      * @param {boolean} value - The value to set for background mode.
203      * @throws {string} Throws an error if the 'cordova-plugin-background-mode' plugin is not enabled.
204      */
205     set backgroundMode(value) {
206         if(!cordova?.plugins?.backgroundMode) {
207             throw 'Please enable \'cordova-plugin-background-mode\' plugin';
208         }
209 
210         this._backgroundMode = value;
211         cordova.plugins.backgroundMode.setEnabled(value);
212         if(value) {
213             cordova.plugins.backgroundMode.setDefaults({ silent: false });
214             cordova.plugins.backgroundMode.on('activate', function () {
215                 cordova.plugins.backgroundMode.disableWebViewOptimizations();
216             });
217         }
218     
219     }

228         }

253     /**
254      * Retrieves the safe area of the device.
255      * @returns {Promise<{top: number, bottom: number}>} A promise that resolves with the safe area object containing top and bottom values.
256      */
257     SafeArea() {
258         return new Promise((resolve, reject) => {
259             if(this.isIOs) {
260                 if(!window?.plugins?.safearea) {
261                     reject('Can not fund \'cordova-plugin-safearea\' plugin');
262                 }
263                 window.plugins.safearea.get((result) => {
264                     resolve(result);
265                 }, (e) => {
266                     reject(e);
267                 });
268             } else {
269                 resolve({top: 0, bottom: 0});
270             }
271         });
272     }

274     /**
275      * Retrieves a plugin based on the provided query.
276      * @param {string} query - The query string to retrieve the plugin.
277      * @returns {*} The plugin object if found, otherwise undefined.
278      */
279     Plugin(query) {
280         let plugin = eval('cordova.' + query);
281         if(!plugin) {
282             plugin = eval('navigator.' + query);
283         }
284         if(!plugin) {
285             plugin = eval('cordova.plugins.' + query);
286         }
287         if(!plugin) {
288             plugin = eval('window.' + query);
289         }
290         if(!plugin) {
291             plugin = eval(query);
292         }
293         return plugin;
294     }
