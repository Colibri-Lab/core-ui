 34     __addButtonContainerForRowSelection() {
 35 
 36         this._checkboxContainer = new Colibri.UI.Component('button-container-for-row-selection', this._title, Element.create('td'));
 37         this._checkboxContainer.AddClass('app-ui-row-cell');
 38         this._checkboxContainer.shown = this.grid.showCheckboxes;
 39         if (this.grid.showCheckboxes) {
 40             this._checkboxContainer.AddClass('input-checkbox-shown');
 41         }
 42 
 43         this._checkbox = new Colibri.UI.Checkbox('row-checkbox', this._checkboxContainer);
 44         this._checkbox.hasThirdState = true;
 45         this._checkbox.shown = true;
 46 
 47         this._checkbox.AddHandler('Changed', (event, args) => {
 48             this.ForEach((name, row) => {
 49                 if(row.shown) {
 50                     row.checked = args.value;
 51                 }
 52                 else {
 53                     row.checked = false;
 54                 }
 55             });    
 56             this.grid.header.checkbox.thirdState = this.grid.rowsCount > this.grid.checked.length;
 57             this.checkbox.thirdState = this.rowsCount > this.checked.length;
 58             this.grid.Dispatch('RowSelected');
 59         });
 60 
 61     }

 72         this.RegisterEvent('RowDisposed', false, 'Поднимается, когда удалили строку');
 73         this.RegisterEvent('GridCellsChanged', false, 'Поднимается, когда все строки сообщили об изменении ячеек (sticky/dispose)');

 77     Add(name, value, index = null) {
 78         this.shown = true;
 79         let newRow = new Colibri.UI.Grid.Row(name, this);
 80         newRow.value = value;
 81         newRow.hasContextMenu = this.grid.hasContextMenu;
 82 
 83         // добавили, смотрим есть ли сортировка
 84         if(this.grid?.tag?.params?.sort) {
 85             const foundIndex = this.grid.tag.params.sort(newRow, this);
 86             this.Children(name, newRow, foundIndex);
 87         }
 88         else if(index) {
 89             this.Children(name, newRow, index);
 90         }
 91 
 92         this.Dispatch('RowAdded', {row: newRow});
 93 
 94         newRow.AddHandler('RowUpdated', (event, args) => {
 95             if(this.grid?.tag?.params?.sort) {
 96                 const foundIndex = this.grid.tag.params.sort(newRow, this);
 97                 this.Children(newRow.name, newRow, foundIndex);
 98             }
 99             this.Dispatch('RowUpdated', {row: args.row});
100         });
101 
102         newRow.AddHandler('RowStickyChange', (event, args) => {
103             this.Dispatch('RowStickyChange', args);
104         });
105 
106         newRow.AddHandler('CellClicked', (event, args) => {
107             this.Dispatch('CellClicked', args);
108         });
109 
110         newRow.AddHandler('CellDoubleClicked', (event, args) => {
111             this.Dispatch('CellDoubleClicked', args);
112         });
113 
114         newRow.AddHandler('RowSelected', (event, args) => {
115             if (this.grid.selectionMode === Colibri.UI.Grid.FullRow) {
116                 if (!this.grid.multiple) { 
117                     this.grid.UnselectAllRows();
118                 }
119                 //args.row.selected = !args.row.selected;
120                 this.Dispatch('RowSelected', args);
121             }
122         });
123 
124         newRow.AddHandler('ComponentDisposed', (event, args) => {
125             this.Dispatch('RowDisposed', {row: newRow});
126         });
127 
128         newRow.AddHandler('CellHorizontalStickyChanged', (event, args) => {
129             this._tempCountRowsReportedCellsChange = this._tempCountRowsReportedCellsChange + 1;
130             if (this.rowsCount === this._tempCountRowsReportedCellsChange) {
131                 this._tempCountRowsReportedCellsChange = 0;
132                 this.Dispatch('GridCellsChanged', {rows: this});
133             }
134         });
135 
136         newRow.AddHandler('CellDisposed', (event, args) => {
137             this._tempCountRowsReportedCellsChange = this._tempCountRowsReportedCellsChange + 1;
138             if (this.rowsCount === this._tempCountRowsReportedCellsChange) {
139                 this._tempCountRowsReportedCellsChange = 0;
140                 this.Dispatch('GridCellsChanged', {rows: this});
141             }
142         });
143 
144         newRow.AddHandler('StickyChanged', (event, args) => {
145             this.Dispatch('StickyChanged', args);
146         });
147 
148         this._titleCellCountSpan.value = ' (' + this.rowsCount + ')';
149 
150         return newRow;
151     }

372     SortById(callback) {
373         
374         let list = [];
375         this.ForEveryRow((name, row) => {
376             list.push(row.value.id);
377         });
378 
379         list = list.sort((a, b) => callback(a, b));
380         let index = 0;
381         for(const id of list) {
382             const row = this.Children('data' + id);
383             this.Children(row.name, row, index);
384             index++;
385         }
386         
387 
388     }

390     Update(value) {
391         
392         const existing = [];
393         for(const item of value) {
394             const exists = this.Children('data' + item.id);
395             if(exists) {
396                 exists.value = item;
397             } else {
398                 this.Add('data' + item.id, item);
399             }
400             existing.push(item.id);
401         }
402 
403         this.ForEach((name, row) => {
404             if(existing.indexOf(row.value.id) === -1) {
405                 row.Dispose();
406             }
407         });
408 
409     }
