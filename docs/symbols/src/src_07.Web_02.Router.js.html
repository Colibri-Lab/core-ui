1 /**
  2  * Represents a router for handling web routes.
  3  * @class
  4  * @extends Colibri.Events.Dispatcher
  5  * @memberof Colibri.Web
  6  */
  7 Colibri.Web.Router = class extends Colibri.Events.Dispatcher {
  8 
  9     /**
 10      * Creates an instance of Router.
 11      * @param {string} type - The type of router ('hash' or 'history').
 12      */
 13     constructor(type) {
 14         super();
 15 
 16         this._type = type;
 17         this._history = [];
 18         this._url = null;
 19         this._path = null;
 20         this._options = {};
 21         this._routeHandlers = [];
 22 
 23         this._registerEvents();
 24         this._preventNextEvent = false;
 25 
 26         this._handleHashChange = (e) => {
 27             if(this._preventNextEvent) {
 28                 this._preventNextEvent = false;
 29                 return;
 30             }
 31             if(this._url !== App.Request.uri || this._options !== App.Request.query) {
 32                 this._url = App.Request.uri;
 33                 this._path = App.Request.uri.split('/').filter(v => v != '');
 34                 this._options = App.Request.query;
 35                 this._history.push({url: this._url, options: this._options});
 36                 this._processRoutePatterns();
 37                 this.Dispatch('RouteChanged', {url: this._url, options: this._options});
 38             }
 39         };
 40         this._handlePopState = (e) => {
 41             if(this._preventNextEvent) {
 42                 this._preventNextEvent = false;
 43                 return;
 44             }
 45             if(this._url !== App.Request.uri || this._options !== App.Request.query) {
 46                 this._url = App.Request.uri;
 47                 this._path = App.Request.uri.split('/').filter(v => v != '');
 48                 this._options = App.Request.query;
 49                 this._history.push({url: this._url, options: this._options});
 50                 this._processRoutePatterns();
 51                 this.Dispatch('RouteChanged', {url: this._url, options: this._options});
 52             }
 53         };
 54 
 55         if(type == Colibri.Web.Router.RouteOnHash) {
 56             this._initRouterOnHash();
 57         }
 58         else if(type == Colibri.Web.Router.RouteOnHistory) {
 59             this._initRouterOnHistory();
 60         }
 61 
 62         
 63     }

 65     /**
 66      * Creates a URI string with the current URL and query parameters.
 67      * @returns {string} - The URI string.
 68      */
 69     CreateUri() {
 70         return this._url + '?' + Object.toQueryString(this._options, ['&', '=']);
 71     }

 73     /**
 74      * Handles the DOM ready event. Initializes the router with the current URL and query parameters.
 75      */
 76     HandleDomReady() {
 77         this._url = App.Request.uri;
 78         this._path = App.Request.uri.split('/').filter(v => v != '');
 79         this._options = App.Request.query;
 80         this._history.push({url: this._url, options: this._options});
 81         this._processRoutePatterns();
 82         this.Dispatch('RouteChanged', {url: this._url, options: this._options});
 83     }

 85     /**
 86      * Registers events for the router.
 87      * @private
 88      */
 89     _registerEvents() {
 90         this.RegisterEvent('RouteChanged', false, 'При изменении раута');
 91     }

 94      * Processes route patterns.
 95      */
 96     ProcessPatterns() {
 97         this._processRoutePatterns();

100     /**
101      * Initializes the router based on hash changes.
102      * @private
103      */
104     _initRouterOnHash() {
105         window.removeEventListener('popstate', this._handlePopState);
106         window.addEventListener('hashchange', this._handleHashChange);
107     }
108 
109     /**
110      * Initializes the router based on history changes.
111      * @private
112      */
113     _initRouterOnHistory() {
114         window.removeEventListener('hashchange', this._handleHashChange);
115         window.addEventListener('popstate', this._handlePopState);
116     }
117 
118     /**
119      * Processes route patterns.
120      * @private
121      */
122     _processRoutePatterns() {
123         const routePatters = Object.keys(this._routeHandlers);
124         for(let i=0; i < routePatters.length; i++) {
125             const pattern = routePatters[i];
126             const rex = new RegExp(pattern);
127             const u = this._url + (Object.countKeys(this._options) > 0 ? '?' + String.fromObject(this._options, ['&', '=']) : '');
128             if(rex.test(u)) {
129                 const handlers = this._routeHandlers[pattern];
130                 for(let j=0; j < handlers.length; j++) {
131                     const handler = handlers[j];
132                     if(handler(this._url, this._options, this._path) === false) {
133                         return false;
134                     }
135                 }
136             }
137         }
138         return true;
139     }

141     /**
142      * Adds a route pattern and its handler.
143      * @param {string} routePattern - The route pattern to match.
144      * @param {Function} handler - The handler function for the route.
145      * @param {boolean} [prepend=false] - Whether to add the handler at the beginning.
146      */
147     AddRoutePattern(routePattern, handler, prepend = false) {
148 
149         if(routePattern instanceof Array) {
150             routePattern.forEach((en) => {
151                 this.AddRoutePattern(en, handler, prepend);
152             });
153         }
154         else {
155             if(!this._routeHandlers[routePattern]) {
156                 this._routeHandlers[routePattern] = [];
157             }
158     
159             for (let i = 0; i < this._routeHandlers[routePattern].length; i++) {
160                 const h = this._routeHandlers[routePattern][i];
161                 if (h === handler) {
162                     this._routeHandlers[routePattern].splice(i, 1);
163                 }
164             }
165     
166             if (prepend) {
167                 this._routeHandlers[routePattern].splice(0, 0, handler);
168             } else {
169                 this._routeHandlers[routePattern].push(handler);
170             }    
171         }
172 
173 
174     }

176     /**
177      * Constructs a URL with query parameters.
178      * @param {string} url The base URL.
179      * @param {object} options The query parameters.
180      * @returns {string} The constructed URL.
181      */
182     Url(url, options) {
183         return url + (Object.countKeys(options) > 0 ? '?' + String.fromObject(options, ['&', '=']) : '');
184     }

186     /**
187      * Checks if the URL has changed.
188      * @param {string} url - The URL to check.
189      * @param {Object} options - The options for the URL.
190      * @returns {boolean} - Whether the URL has changed.
191      * @private
192      */
193     _isChanged(url, options) {
194         const u = url + (Object.countKeys(options) > 0 ? '?' + String.fromObject(options, ['&', '=']) : '');
195         let isChanged = false;
196         if(this._type == Colibri.Web.Router.RouteOnHash) {
197             isChanged = location.hash != '#' + u;
198         }
199         else if(this._type == Colibri.Web.Router.RouteOnHistory) {
200             isChanged = location.pathname + location.search != u;
201         }
202         return isChanged;
203     }

205     /**
206      * Sets the URL and options in the address bar.
207      * @param {string} url - The URL to set.
208      * @param {Object} options - The options for the URL.
209      * @param {boolean} replaceOnHistory - Whether to replace the current entry in the history.
210      * @param {boolean} preventNextEvent - Whether to prevent the next event.
211      * @private
212      */
213     _setToAddressBar(url, options, replaceOnHistory = false, preventNextEvent = false) {
214 
215         this._preventNextEvent = preventNextEvent;
216 
217         const u = url + (Object.countKeys(options) > 0 ? '?' + String.fromObject(options, ['&', '=']) : '') + this.GetSafeParamsAsString();
218         if(this._type == Colibri.Web.Router.RouteOnHash) {
219             location.hash = '#' + u;
220         } else if(this._type == Colibri.Web.Router.RouteOnHistory) {
221             const state = {url: this._url, options: this._options};
222             if(replaceOnHistory) {
223                 history.replaceState(state, '', u);
224             } else {
225                 history.pushState(state, '', u);
226             }
227             dispatchEvent(new PopStateEvent('popstate', { state: state }));
228         }
229     }

231     /**
232      * Navigates to a URL with options.
233      * @param {string} url - The URL to navigate to.
234      * @param {Object} [options={}] - The options for the URL.
235      * @param {boolean} [replaceOnHistory=false] - Whether to replace the current entry in the history.
236      * @param {boolean} [setOnHistory=false] - Whether to add the navigation to the history.
237      * @param {string} [target='_self'] - The target window or tab.
238      * @returns {string} - The URL that was navigated to.
239      */
240     Navigate(url, options = {}, replaceOnHistory = false, setOnHistory = false, target = '_self') {
241 
242         const isChanged = this._isChanged(url, options);
243 
244         const u = url + (Object.countKeys(options) > 0 ? '?' + String.fromObject(options, ['&', '=']) : '');
245         if(target === '_blank') {
246             window.open(this._type == Colibri.Web.Router.RouteOnHash ? '#' + u : u);
247             return u;
248         }
249 
250         if(setOnHistory) {
251             this._url = url;
252             this._path = (this._url ?? '/').split('/').filter(v => v != '');
253             this._options = options;
254             this._history.push({url: this._url, options: this._options});
255             
256             this._setToAddressBar(url, options, replaceOnHistory, true);
257 
258             return;
259         }
260 
261 
262         if(isChanged) {
263             this._setToAddressBar(url, options, replaceOnHistory);
264         } else {
265             
266             // this._url = url;
267             // this._path = (this._url ?? '/').split('/').filter(v => v != '');
268             // this._options = options;
269             // this._history.push({url: this._url, options: this._options});
270 
271             // this._processRoutePatterns();
272             // this.Dispatch('RouteChanged', {url: this._url, options: this._options});
273 
274         }
275 
276         return u;
277 
278     }

364     /**
365      * Gets the string representation of safe query parameters.
366      * @returns {string} - The string representation of safe query parameters.
367      * @private
368      */
369     GetSafeParamsAsString() {
370         let ret = [];
371         for(const param of this.safeParams) {
372             if(this.options[param]) {
373                 ret.push(param + '=' + this.options[param]);
374             }
375         }
376         return ret.join('&');
377     }
