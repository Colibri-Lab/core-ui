1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.Radio = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      * @param {string} fieldName name of field to generate
 13      */
 14     constructor(name, container, fieldName) {
 15         super(name, container, Element.create('span'));
 16         this.AddClass('app-component-radio');
 17 
 18         this._fieldName = fieldName;
 19         this._renderInput();
 20 
 21         this._enabled = true;
 22         this._readonly = false;
 23 
 24         this._handleEvents();
 25     }
 26 
 27     /** @protected */
 28     _registerEvents() {
 29         super._registerEvents();
 30         this.RegisterEvent('Changed', false, 'Поднимается, когда изменил состояние');
 31     }
 32 
 33     /** @protected */
 34     _handleEvents() {
 35         this.AddHandler('Clicked', (event, args) => {
 36             if (!this._readonly && this._enabled) {
 37                 this._setChecked(!this._input.checked);
 38                 this.Dispatch('Changed', {value: this._input.checked});
 39             }
 40 
 41             args.domEvent.stopPropagation();
 42             args.domEvent.preventDefault();
 43             return false;
 44         });
 45         this.AddHandler('KeyUp', (event, args) => {
 46             if(args.domEvent.code === 'Space') {
 47                 this.Dispatch('Clicked', args);
 48             }
 49             args.domEvent.stopPropagation();
 50             args.domEvent.preventDefault();
 51             return false;
 52         });
 53         this._checkIcon.AddHandler('ReceiveFocus', (event, args) => this.Dispatch('ReceiveFocus', args));
 54         this._checkIcon.AddHandler('LoosedFocus', (event, args) => this.Dispatch('LoosedFocus', args));
 55 
 56     }
 57 
 58     /**
 59      * Render input
 60      * @private
 61      */
 62     _renderInput() {
 63         this._checkIcon = new Colibri.UI.Icon(this.name + '-icon', this);
 64         this._checkIcon.shown = true;
 65         this._setIcon();
 66 
 67         this._input = Element.create('input', {id: this.name + '-input-' + Date.Now().getTime(), name: this._fieldName, type: 'radio'});
 68         this._input.tag('component', this);
 69         this._element.append(this._input);
 70     }
 71 
 72     /**
 73      * Set the icon
 74      * @private
 75      */
 76     _setIcon() {
 77         this._checkIcon.value = Colibri.UI.AltRadioMarkIcon;
 78     }
 79 
 80     /**
 81      * Focus on input
 82      */
 83     Focus() {
 84         this._input.focus();
 85     }
 86 
 87     /**
 88      * Checkbox is checked
 89      * @type {boolean}
 90      */
 91     get checked() {
 92         return this._input.checked;
 93     }
 94     /**
 95      * Checkbox is checked
 96      * @type {boolean}
 97      */
 98     set checked(value) {
 99         this._setChecked(value);
100     }
101     /** @private */
102     _setChecked(value, process = true) {
103         this._input.checked = value;
104         if (value) {
105             this.AddClass('-checked');
106         } else {
107             this.RemoveClass('-checked');
108         }
109 
110         if(process) {
111             this._unsetAllByGroup();
112         }
113 
114     }
115 
116     /** @private */
117     _unsetAllByGroup() {
118         document.querySelectorAll('input[type=radio][name="' + this._fieldName + '"]').forEach(el => {
119             if(el != this._input) {
120                 const component = el.tag('component');
121                 component._setChecked(false, false);
122             }
123         });
124     }
125 
126     /**
127      * Enable/Disable component
128      * @type {boolean}
129      */
130     set enabled(value) {
131         this._enabled = (value === true || value === 'true')
132         this._input.disabled = !this._enabled;
133         super.enabled = value;
134     }
135     /**
136      * Enable/Disable component
137      * @type {boolean}
138      */
139     get enabled() {
140         return this._enabled;
141     }
142 
143     /**
144      * Is checkbox readonly
145      * @type {boolean}
146      */
147     get readonly() {
148         return this._readonly;
149     }
150     /**
151      * Is checkbox readonly
152      * @type {boolean}
153      */
154     set readonly(value) {
155         this._readonly = (value === true || value === 'true');
156         super.readonly = this._readonly;
157     }
158 
159     /**
160      * Placeholder
161      * @type {string}
162      */
163     get placeholder() {
164         return this._placeholder?.value;
165     }
166     /**
167      * Placeholder
168      * @type {string}
169      */
170     set placeholder(value) {
171         this._setPlaceholder(value ? value[Lang.Current] ?? value : '');
172     }
173     /** @private */
174     _setPlaceholder(value) {
175         if(!value) {
176             this._placeholder.Dispose();
177         }
178         else {
179             this._placeholder = new Colibri.UI.TextSpan(this.name + '_placeholder', this);
180             this._placeholder.shown = true;
181             this._placeholder.value = value;
182         }
183     }
184 
185     /**
186      * Tab index
187      * @type {number}
188      */
189     get tabIndex() {
190         if (this._placeholder) {
191             return this._placeholder && this._placeholder.tabIndex;
192         }
193         else if(this._checkIcon) {
194             return this._checkIcon && this._checkIcon.tabIndex;
195         }
196         
197     }
198     /**
199      * Tab index
200      * @type {number}
201      */
202     set tabIndex(value) {
203         if (this._placeholder) {
204             this._placeholder.tabIndex = value === true ? Colibri.UI.tabIndex++ : value;
205         }
206         else if(this._checkIcon) {
207             this._checkIcon.tabIndex = value === true ? Colibri.UI.tabIndex++ : value;
208         }
209     }
210 
211 }
