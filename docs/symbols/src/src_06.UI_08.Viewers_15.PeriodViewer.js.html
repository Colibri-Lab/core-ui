1 /**

  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Viewer
  5  */
  6 Colibri.UI.PeriodViewer = class extends Colibri.UI.Viewer {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container element and component
 12      * @param {Element|string} element element to generate childs
 13      * @param {Colibri.UI.Component|null} root root component 
 14      */ 
 15     constructor(name, container, element = null, root = null) {
 16         super(name, container, element || Element.create('span'), root);
 17         this.AddClass('app-period-viewer-component');
 18 
 19         this._value = null;
 20 
 21     }
 22 
 23     get value() {
 24         return this.ContainsClass('app-is-read-component');
 25     }
 26 
 27     set value(value) {
 28         let _value = value;
 29         if(typeof _value === 'string') {
 30             _value = value.toDate();
 31             if(_value.toString() === 'Invalid Date') {
 32                 super.value = value;
 33                 return;
 34             }
 35         }
 36 
 37         this._value = _value;
 38 
 39         let date1 = _value[0];
 40         let date2 = _value[1];
 41 
 42         let days = 0;
 43         if (date1 && date2 && date1 !== 'Invalid Date' && date2 !== 'Invalid Date') {
 44             days = parseInt((date2.toDate().getTime() - date1.toDate().getTime()) / 1000 / 86400) + 1;
 45             super.value = date1.toDate().toShortRUString() + ' – ' + date2.toDate().toShortRUString() + ', ' + days.formatSequence(['день', 'дня', 'дней'], true);
 46         } else {
 47             super.value = date1.toDate().toShortRUString() + ' – -, ' + days.formatSequence(['день', 'дня', 'дней'], true);
 48         }
 49     }
 50 
 51 }
 52 Colibri.UI.Viewer.Register('Colibri.UI.PeriodViewer', '#{ui-viewers-period}');
