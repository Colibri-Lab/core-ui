1 Colibri.UI.FormValidator = class extends Colibri.Events.Dispatcher {
  2 
  3     _form = null;
  4     _validators = null;
  5 
  6     constructor(form) {
  7         super();
  8 
  9         this.RegisterEvent('Validated', false, 'Когда валидатор завершил валидацию');
 10 
 11         this._validators = [];
 12         this._form = form;
 13         this._form.AddHandler('FieldsRendered', (event, args) => {
 14             this._createValidators();
 15         });
 16         this._createValidators();
 17     }
 18 
 19     _createValidators() {
 20         this._validators = [];
 21         this._form.ForEach((name, component) => {
 22             const fieldValidator = new Colibri.UI.FieldValidator(component, this._form);
 23             fieldValidator.AddHandler('Validated', (event, args) => {
 24                 this.Dispatch('Validated', args);
 25             });
 26             this._validators.push(fieldValidator);
 27         });
 28     }
 29 
 30     /**
 31      *
 32      * @param {boolean} messages отображать ли сообщения
 33      * @param {boolean} breakFirst прерывается на первом не валидном поле
 34      * @param {string} className
 35      * @returns {boolean}
 36      * @constructor
 37      */
 38     Validate(messages = true, breakFirst = true, className = 'app-validate-error') {
 39 
 40         let needValidate = true;
 41         if (this._form.tag.params && this._form.tag.params.validate) {
 42             const method = eval(this._form.tag.params.validate);
 43             needValidate = method(this._form, this);
 44         }
 45 
 46         if (!needValidate) {
 47             return true;
 48         }
 49 
 50         if(breakFirst) {
 51             this._validators.forEach(validator => validator.Clear());
 52         }
 53 
 54         this._validated = true;
 55         for (let validator of this._validators) {
 56             const _validated = validator.Validate(messages, className);
 57             if(!_validated) {
 58                 this._validated = false;
 59                 if(breakFirst) {
 60                     break;
 61                 }
 62             }
 63         }
 64 
 65         return this._validated;
 66     }
 67 
 68     Invalidate(field, message, className = 'app-validate-error') {
 69         const fieldObject = this._form.FindField(field);
 70         if(!fieldObject) {
 71             return;
 72         }
 73 
 74         const validator = fieldObject.container.tag('validator');
 75         validator.Invalidate(message, className);
 76     }
 77 
 78     get form() {
 79         return this._form;
 80     }
