1 Colibri.UI.Input = class extends Colibri.UI.Component {

  3     constructor(name, container, element) {
  4         super(name, container, Element.create('div', {class: 'app-ui-component'}));
  5 
  6         this.GenerateChildren(element, this._element);
  7 
  8         this._hasClearIcon = true;
  9 
 10         this.AddClass('app-input-component');
 11 
 12         this._fillTimeoutValue = 500;
 13 
 14         new Colibri.UI.Icon('icon', this);
 15         new Colibri.UI.Pane('loadingicon', this);
 16 
 17         this._input = Element.create('input', { type: 'text', placeholder: this.placeholder || '' });
 18         this._element.append(this._input);
 19 
 20         new Colibri.UI.Pane('clear', this);
 21 
 22         this.Children('icon').shown = true;
 23         this.Children('clear').html = Colibri.UI.ClearIcon;
 24         this.Children('loadingicon').html = Colibri.UI.LoadingIcon;
 25         this.Children('icon').value = Colibri.UI.SearchIcon;
 26 
 27         this._input.addEventListener('keyup', (e) => {
 28 
 29             if(['Tab', 'Escape', 'Enter', 'ArrowDown', 'ArrowUp', 'Space'].indexOf(e.code) !== -1) {
 30                 return true;
 31             }
 32 
 33             if(!!this.maxlength && this.type != 'text') {
 34                 if( (this.value + '').length > this.maxlength ) {
 35                     this.value = (this.value + '').substring(0, this.maxlength);
 36                     e.preventDefault();
 37                     e.stopPropagation();
 38                     return false;
 39                 }
 40             }
 41 
 42             if(this.readonly) {
 43                 this.Children('clear').shown = false;
 44             } else {
 45                 this.Children('clear').shown = this._hasClearIcon && this._input.value.length > 0;
 46             } 
 47 
 48             this.Dispatch('KeyUp', { value: this.value, domEvent: e });
 49 
 50             if(this._fillTimeout) {
 51                 clearTimeout(this._fillTimeout);
 52                 this._fillTimeout = null;
 53             }
 54 
 55             this._fillTimeout = setTimeout(() => {
 56                 this.Dispatch('Filled', { value: this.value });
 57             }, this._fillTimeoutValue);
 58 
 59         });
 60 
 61         this._input.addEventListener('change', (e) => {
 62             this.Dispatch('Changed', { value: this.value, domEvent: e });
 63         });
 64 
 65         this._input.addEventListener('keydown', (e) => {
 66             this.Dispatch('KeyDown', { value: this.value, domEvent: e });
 67         });
 68 
 69         this._input.addEventListener('focus', (e) => this.Dispatch('ReceiveFocus', { domEvent: e }));
 70         this._input.addEventListener('blur', (e) => this.Dispatch('LoosedFocus', { domEvent: e }));
 71 
 72         this._input.addEventListener('mousedown', (e) => {
 73             e.target.focus();
 74             //e.target.select();
 75             return false;
 76         });
 77 
 78         this.Children('clear').AddHandler('Clicked', (event, args) => {
 79             if(!this.enabled) {
 80                 return;
 81             }
 82             this._input.value = '';
 83             this._input.focus();
 84             // this._input.select();
 85             this.Children('clear').shown = false;
 86             this.loading = false;
 87             this.Dispatch('Cleared', args);
 88         });
 89 
 90         this._input.addEventListener('blur', (e) => {
 91             this.Dispatch('LoosedFocus', { value: this.value, domEvent: e });
 92         });
 93 
 94         this.AddHandler('LoosedFocus', (event, args) => {
 95             if(this._popup) {
 96                 this._popup.Hide();
 97                 this._popup.Dispose();
 98                 this._popup = null;
 99             }
100         });
101         this.AddHandler('ReceiveFocus', (event, args) => {
102             this._showSuggestions();
103         });
104 
105     }

107     _registerEvents() {
108         super._registerEvents();
109         this.RegisterEvent('KeyUp', false, 'Поднимается, когда клавиша поднята');
110         this.RegisterEvent('KeyDown', false, 'Поднимается, когда клавиша нажата');
111         this.RegisterEvent('Changed', false, 'Поднимается, когда содержимое поля обновлено');
112         this.RegisterEvent('Filled', false, 'Поднимается, когда содержимое поля обновлено и не изменилось в течении fillTimeout мс');
113         this.RegisterEvent('Cleared', false, 'Поднимается, когда содержимое поля очищено с помощью крестика');
114     }

347     _createPopup(values) {
348         const popup = new Colibri.UI.PopupList('select-popup', document.body, this._multiple, this.__render, this._titleField, this._valueField, this._groupField);
349         popup.parent = this;
350         popup.multiple = this._multiple;
351         const el = this.container.closest('[namespace]');
352         el && popup.container.attr('namespace', el.attr('namespace'));
353         //заполнение списка перед хэндлерами, чтобы не сработал SelectionChanged
354         popup.FillItems(values, this._value);
355         popup.AddHandler('MouseDown', (event, args) => {
356             this._input.value = popup?.selected?.value;
357         });
358         return popup;
359     }
