  9     constructor(name, container) {
 10         super(name, container, Element.create('td'));
 11 
 12         this.AddClass('app-ui-row-cell');
 13 
 14         this._valueContainer = new Colibri.UI.TextSpan('span', this);
 15         this._valueContainer.AddClass('app-ui-row-cell-value-container');
 16         this._valueContainer.shown = true;
 17 
 18         this._stickyHorizontally = false;
 19         this._stickyVertically = false;
 20         this._selected = false;
 21         this._activated = false;
 22         this._parentColumn = null;
 23         this._value = null;
 24 
 25         this._parentColumn = this.grid.header.FindColumn(this.columnName);
 26         this._editor = this._parentColumn?.editor;
 27         this._viewer = this._parentColumn?.viewer;
 28         this._editorObject = null;
 29         this._viewerObject = null;
 30         this._createViewer();
 31         this._createEditor();
 32         this.valign = this._parentColumn?.valign;
 33         this.halign = this._parentColumn?.halign;
 34 
 35         this._handleEvents();
 36         if(this._parentColumn?.editorAllways ?? false) {
 37             this.EditValue(false);
 38         }
 39     }

 41     _registerEvents() {
 42         super._registerEvents();
 43         this.RegisterEvent('CellClicked', false, 'Поднимается, когда щелкнули по ячейке');
 44         this.RegisterEvent('CellDoubleClicked', false, 'Поднимается, когда дважды щелкнули по ячейке');
 45         this.RegisterEvent('CellDisposed', false, 'Поднимается, когда удалили ячейку');
 46         this.RegisterEvent('CellHorizontalStickyChanged', false, 'Поднимается, когда ячейка меняет липкость по горизонтали');
 47         this.RegisterEvent('CellVerticalStickyChanged', false, 'Поднимается, когда ячейка меняет липкость по вертикали');
 48         this.RegisterEvent('ViewerClicked', false, 'Кликнули на элемент отображения ячейки');
 49         this.RegisterEvent('EditorChanged', false, 'Изменилось значение в редакторе');
 50     }

 52     _handleEvents() {
 53 
 54         this.AddHandler('DoubleClicked', (event, args) => {
 55             this.Dispatch('CellDoubleClicked', {cell: this, row: this.parentRow});
 56         });
 57 
 58         this.AddHandler('ComponentDisposed', (event, args) => {
 59             this.Dispatch('CellDisposed', {cell: this});
 60         });
 61 
 62 
 63         this.parentRow.AddHandler('RowStickyChange', (event, args) => {
 64             this.stickyVertically = args.row.sticky;
 65         });
 66 
 67         this.parentRow.AddHandler('RowPositionChange', (event, args) => {
 68             if (this.stickyVertically) {
 69                 this._element.css('top', args.row._positionTop + 'px');
 70             }
 71         });
 72 
 73         this.AddHandler('ViewerClicked', (event, args) => {
 74             this.grid.Dispatch('CellViewerClicked', {cell: this, field: this.columnName, data: this.parentRow.value});
 75             this.Dispatch('CellClicked', {cell: this});
 76             this.EditValue();
 77         });
 78 
 79         this.AddHandler('EditorChanged', (event, args) => {
 80             if(this.value == args.value) {
 81                 return true;
 82             }
 83 
 84             const oldValue = this.value;
 85             this.value = args.value;
 86             this.grid.Dispatch('CellValueChanged', {cell: this, row: this.parentRow, value: event.sender.value, oldValue: oldValue});
 87             this.grid.Dispatch('CellEditorChanged', {cell: this, field: this.columnName, data: this.parentRow.value});
 88         });
 89 
 90     }

217     set parentColumn(value) {
218         this._parentColumn = value;
219         if(this.parentColumn !== null) {
220 
221             this.stickyHorizontally = this.parentColumn.sticky;
222             this.parentColumn.AddHandler('ColumnStickyChange', (event, args) => {
223                 this.stickyHorizontally = args.column.sticky;
224             });
225 
226             this.parentColumn.AddHandler('ColumnDisposed', (event, args) => {
227                 const parent = this.parent;
228                 this.Dispose();
229                 if(parent?.hasContextMenu) {
230                     parent?.lastCell && parent.lastCell.CreateContextMenuButton();   
231                 }
232             });
233 
234             this.parentColumn.AddHandler('ColumnPositionChange', (event, args) => {
235                 if (this.stickyHorizontally) {
236                     this.left = args.column._positionLeft;
237                     if(args.column.ContainsClass('-last-sticky')) {
238                         this.AddClass('-last-sticky');   
239                     }
240                 }
241             });
242         }

378     _generateViewFromParams(value) {
379         const column = this.grid.header.FindColumn(this.columnName);
380         if(column.tag.params && column.tag.params.render) {
381             const ret = column.tag.params.render.apply(this, [this, this.parentRow]);
382             if(ret) {
383                 return ret;
384             }
385         }
386         else {
387             return this._value;
388         }
389     }

391     _createViewer() {
392         if(this._viewer && !this._viewerObject) {
393             const viewer = eval(this._viewer);
394             if(!viewer) {
395                 throw 'Can not find viewer component: ' + this._viewer;
396             }
397             this._viewerObject = new viewer(this.name + '_viewer', this);
398             this._viewerObject.AddHandler('Clicked', (event, args) => this.Dispatch('ViewerClicked', {value: event.sender.value}));
399             this._viewerObject.shown = true;
400             this._viewerObject.field = this.parentColumn.tag;
401             this._viewerObject.download = this.parentColumn.download;
402             this._viewerObject.tag = {column: this.parentColumn, row: this.parentRow, cell: this};
403             Object.forEach(this.parentColumn.tag.viewerAttrs, (name, value) => {
404                 this._viewerObject[name] = value;
405             });
406 
407             this._valueContainer.shown = false;
408         }
409         else if(!this._viewer && this._viewerObject) {
410             this._viewerObject.Dispose();
411             this._valueContainer.shown = true;
412         }
413     }
414 
415     _createEditor() {
416         if(this._editor && !this._editorObject) {
417             const editor = eval(this._editor);
418             this._editorObject = new editor(this.name + '_editor', this);
419             this._editorObject.field = this.parentColumn.tag;
420             this._editorObject.download = this.parentColumn.download;  
421             this._editorObject.tag = {column: this.parentColumn, row: this.parentRow, cell: this};
422             this._editorObject.AddHandler('KeyDown', (event, args) => {
423                 if(args.domEvent.keyCode == 13 && !this._editorObject.invalid) {
424                     this.Dispatch('EditorChanged', {value: event.sender.value});
425                     this.EndEdit();
426                 }
427                 else if(args.domEvent.keyCode == 27) {
428                     this._editorObject.Dispatch('LoosedFocus');
429                 }
430                 args.domEvent.stopPropagation();
431             });
432             this._editorObject.AddHandler('Changed', (event, args) => {
433                 if(!this._editorObject.invalid) {
434                     this.Dispatch('EditorChanged', {value: event.sender.value});
435                 }
436             });
437             this._editorObject.AddHandler('LoosedFocus', (event, args) => {
438                 this.EndEdit();
439             });
440         }
441         else if(!this._editor && this._editorObject) {
442             this._editorObject.Dispose();
443         }
444     }
445 
446     EditValue(setFocus = true) {
447         if(this._editorObject) {
448             Colibri.Common.Delay(50).then(() => {
449                 this.AddClass('-editing');
450                 this.parentRow.AddClass('-editing');
451                 this._valueContainer.Hide();
452                 this._viewerObject && this._viewerObject.Hide();
453                 this._editorObject.field = this._parentColumn.tag;
454                 this._editorObject.Show();
455                 if(setFocus) {
456                     this._editorObject.Focus();
457                 }
458                 this._editorObject.editedObject = this.parentRow.value; 
459                 this._editorObject.value = this.value;   
460             });
461             return true;
462         }
463         return false;
464     }
465 
466     EndEdit(hide = true) {
467         if(this._parentColumn.editorAllways) {
468             return;
469         }
470         this.RemoveClass('-editing');
471         this.parentRow.RemoveClass('-editing');
472         if(this._viewerObject) {
473             this._viewerObject.Show();
474         }
475         else {
476             this._valueContainer.Show();
477         }
478         hide && this._editorObject && this._editorObject.Hide();
479     }

481     CreateContextMenuButton() {
482         let contextMenuParent = this.Children(this._name + '-contextmenu-icon-parent');
483         if(!this.parent.hasContextMenu || contextMenuParent) {
484             return;
485         }
486 
487         contextMenuParent = new Colibri.UI.Component(this._name + '-contextmenu-icon-parent', this);
488         contextMenuParent.AddClass('app-contextmenu-icon-component');
489         contextMenuParent.shown = true;
490         
491         const contextMenuIcon = new Colibri.UI.Icon(this._name + '-contextmenu-icon', contextMenuParent);
492         contextMenuIcon.shown = true;
493         contextMenuIcon.value = Colibri.UI.ContextMenuIcon;
494         contextMenuIcon.AddHandler(['Clicked', 'DoubleClicked'], (event, args) => {
495             this.parent.Dispatch('ContextMenuIconClicked', args);
496             args.domEvent.stopPropagation();
497             args.domEvent.preventDefault();
498             args.domEvent.returnValue = true;
499             return false;
500         });
501     }
502 
503     RemoveContextMenuButton() {
504         if(this.parent.hasContextMenu && this.Children(this._name + '-contextmenu-icon-parent')) {
505             this.Children(this._name + '-contextmenu-icon-parent').Dispose();
506         }
507     }
