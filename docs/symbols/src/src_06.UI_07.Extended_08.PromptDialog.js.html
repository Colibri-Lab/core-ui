1 /**
  2  * @class
  3  * @extends Colibri.UI.Window
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.PromptDialog = class extends Colibri.UI.Window {
  7 
  8     constructor(name, container, width) {
  9         super(name, container, Colibri.UI.Templates['Colibri.UI.PromptDialog'], '', width);
 10         this.AddClass('app-prompt-dialog-component');
 11 
 12         this._callback = null;
 13 
 14         this.AddHandler('KeyDown', (event, args) => {
 15             if(args.domEvent.code == 'Enter' || args.domEvent.code == 'NumpadEnter') {
 16                 this._save.Dispatch('Clicked');
 17                 args.domEvent.stopPropagation();
 18                 args.domEvent.preventDefault();
 19                 return false;
 20             }
 21             else if(args.domEvent.code == 'Escape') {
 22                 this._cancel.Dispatch('Clicked');
 23                 args.domEvent.stopPropagation();
 24                 args.domEvent.preventDefault();
 25                 return false;
 26             }
 27         })
 28 
 29 
 30     }
 31 
 32     /**
 33      * Показывает диалог
 34      * @param {Function(dialogResult)} callback результат диалога, true - да, false - нет 
 35      */
 36     Show(title, fields, button) {
 37     
 38     
 39         return new Promise((resolve, reject) => {
 40             this.title = title;
 41 
 42             this._form = this.Children('form');
 43             this._save = this.Children('btn-save');
 44             this._cancel = this.Children('btn-cancel');
 45             this._save.enabled = false;
 46 
 47             if(!this._validator) {
 48                 this._validator = new Colibri.UI.FormValidator(this._form);
 49             }
 50             this._form.fields = fields;
 51             this._form.value = {};
 52 
 53 
 54             this._save.value = button || '#{ui-prompt-ok}';
 55             super.Show();
 56 
 57             this._form.Children('firstChild').Focus();
 58 
 59             this._validator.ClearHandlers();
 60             this._validator.AddHandler('Validated', (event, args) => this._save.enabled = this._validator.Validate());
 61 
 62             this._save.ClearHandlers();
 63             this._save.AddHandler('Clicked', (event, args) => {
 64                 resolve(this._form.value);
 65                 this.Hide();
 66             });
 67     
 68             this._cancel.ClearHandlers();
 69             this._cancel.AddHandler('Clicked', (event, args) => {
 70                 reject();
 71                 this.Hide();
 72             });
 73 
 74         });
 75 
 76     }
 77 
 78 }
