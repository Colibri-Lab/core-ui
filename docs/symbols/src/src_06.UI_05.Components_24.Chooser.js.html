 74     ShowChooser() {
 75         if(this._chooser) {
 76             const component = this._chooser;
 77             if(!this._chooserObject) {
 78                 this._chooserObject = new component(this.name + '-chooser', document.body, this._selector?.params || {}, this._values, this._selector.title, this._selector.value);
 79                 this._chooserObject.AddHandler('Choosed', (event, args) => {
 80                     this.value = args.value;
 81                     this.valueObject = args.valueObject;
 82                     this.Dispatch('Changed', {});
 83                 });
 84             }
 85             this._chooserObject.Show();
 86         } 
 87         else {
 88             this.Dispatch('ChooserClicked', {currentValue: this._value});
 89         }
 90         return false; 
 91     }

 93     /**
 94      * Регистрация обработчиков событий
 95      */
 96     _handleEvents() {
 97 
 98         this._input.AddHandler('KeyUp', (event, args) => this.Dispatch('KeyUp', args));
 99 
100         this._input.AddHandler('Filled', (event, args) => this.__Filled(event, args));
101         this._input.AddHandler('Cleared', (event, args) => this.__Cleared(event, args));
102 
103         this._arrow.addEventListener('click', (e) => {
104             if(!this.readonly) {
105                 this.__Filled(null, {search: false});
106             }
107             return false; 
108         });
109 
110         // перехватить keydown и обработать Escape
111         this._input.AddHandler('KeyDown', (event, args) => {
112 
113             if(['Enter', 'NumpadEnter'].indexOf(args.domEvent.code) !== -1) {
114                 if(!this.readonly) {
115                     this.__Filled(null, {search: false});
116                 }
117                 args.domEvent.stopPropagation();
118                 args.domEvent.preventDefault();
119                 return false;
120             }
121 
122 
123             return this.Dispatch('KeyDown', args);        
124         });
125 
126     }

134     __Filled(event, args) {
135         this._itemSelected = false;
136         this.__BeforeFilled().then((ret) => {
137             if(ret === false || this._itemSelected === true) {
138                 return;
139             }
140             
141             this.ShowChooser();
142             this._renderValue(false);
143         });
144 
145         args?.domEvent?.preventDefault();
146         args?.domEvent?.stopPropagation();
147 
148         return false;

154         this._renderValue();

157         args.domEvent?.stopPropagation();

214      * Установить выбранное значение
215      * @param {Object|Array} value
216      */
217     _setValue(value) {
218 
219         if( value && this.multiple && !Array.isArray(value) ) {
220             throw 'Value must be array of an object/string, becose it is multiple';
221         }
222         this._value = value;

225     /**
226      * Отобразить значения
227      */
228     _renderValue() {
229 
230         const set = () => {
231             // this._value = [string] | [{title, value}] | null
232             if(!this._value) {
233                 this._input.value = '';
234                 return;
235             }
236             
237             if (!this.multiple) {
238                 if(!Object.isObject(this._value)) {
239                     this._value = Array.findObject(this.values, this._valueField, this._value);
240                 }
241                 const v = Object.isObject(this._value) ? (this._value[this._titleField] ?? this._value[this._valueField] ?? '') : this._value
242                 this._input.value = Lang ? Lang.Translate(v) : v;
243             } else {
244                 const values = this._value.map((v) => {
245                     if(!Object.isObject(v)) {
246                         v = Array.findObject(this.values, this._valueField, v);
247                     }    
248                     v = Object.isObject(v) ? (v[this._titleField] ?? v[this._valueField] ?? '') : v;
249                     return Lang ? Lang.Translate(v) : v;
250                 });
251                 this._input.value = values.join(', ');
252             }
253             this._input.toolTip = this._input.isValueExceeded ? this._input.value : '';
254         }
255         
256         if(!this.values) {
257             Colibri.Common.Wait(Array.isArray(this.values)).then(() => {
258                 set();
259             })
260         } else {
261             set();
262         }
263 
264         
265     }
