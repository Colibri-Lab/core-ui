1 /**
  2  * @class
  3  * @extends Colibri.UI.Component
  4  * @memberof Colibri.UI
  5  */
  6 Colibri.UI.LoadingBallun = class extends Colibri.UI.Component {
  7 
  8     /**
  9      * @constructor
 10      * @param {string} name name of component
 11      * @param {Element|Colibri.UI.Component} container container of component
 12      */
 13     constructor(name, container) {
 14         /* создаем компонент и передаем шаблон */
 15         super(name, container, Element.create('div'));
 16         this.AddClass('colibri-ui-loadingballun');
 17 
 18         this._iconObject = new Colibri.UI.Icon(this.name + '-icon', this);
 19         this._textObject = new Colibri.UI.TextSpan(this.name + '-text', this);
 20 
 21         this._iconObject.shown = true;
 22         this._textObject.shown = false;
 23 
 24         this._iconObject.value = Colibri.UI.LoadingIcon;
 25 
 26     }
 27 
 28     set shown(value) {
 29         super.shown = value;
 30         if(value) {
 31             this.BringToFront();
 32         }
 33         else {
 34             this.SendToBack();
 35         }
 36     }
 37 
 38     /**
 39      * Текст
 40      * @type {string}
 41      */
 42     get text() {
 43         return this._text;
 44     }
 45     /**
 46      * Текст
 47      * @type {string}
 48      */
 49     set text(value) {
 50         this._text = value;
 51         this._showText();
 52     }
 53     _showText() {
 54         this._textObject.shown = !!this._text;
 55         this._textObject.value = this._text;
 56     }
 57 
 58     /**
 59      * Иконка
 60      * @type {string}
 61      */
 62     get icon() {
 63         return this._icon;
 64     }
 65     /**
 66      * Иконка
 67      * @type {string}
 68      */
 69     set icon(value) {
 70         this._icon = value;
 71         this._showIcon();
 72     }
 73     _showIcon() {
 74         this._iconObject.value = this._icon;       
 75     }
 76 
 77     Start(loadingText, completeText, errorText, completeIcon, errorIcon) {
 78         this._completeIcon = completeIcon;
 79         this._errorIcon = errorIcon;
 80         this._errorText = errorText;
 81         this._completeText = completeText;
 82         this.AddClass('-loading');
 83         this.RemoveClass('-complete');
 84         this.RemoveClass('-error');
 85         this.shown = true;
 86         this.icon = Colibri.UI.LoadingIcon;
 87         this.text = loadingText;
 88     }
 89 
 90     Error() {
 91         this.RemoveClass('-loading');
 92         this.AddClass('-error');
 93         this.text = this._errorText;
 94         this.icon = this._errorIcon;            
 95         Colibri.Common.Delay(3000).then(() => {
 96             this.shown = false;
 97         });
 98     }
 99 
100     Complete() {
101         this.RemoveClass('-loading');
102         this.AddClass('-complete');
103         this.text = this._completeText;
104         this.icon = this._completeIcon;
105         Colibri.Common.Delay(5000).then(() => {
106             this.shown = false;
107         });
108     }
109 
110 }
