1 /**
  2  * @class
  3  * @memberof Colibri.UI
  4  * @extends Colibri.UI.Editor
  5  */
  6 Colibri.UI.NumberEditor = class extends Colibri.UI.Editor {
  7     constructor(name, container) {
  8         super(name, container, Element.create('input', {type: 'number'}));
  9         this.AddClass('app-number-editor-component');
 10 
 11         this._element.addEventListener('focus', (e) => this.Focus());
 12         this._element.addEventListener('blur', (e) => this.Blur());
 13         this._element.addEventListener('change', (e) => this.__elementChanged(e));
 14         this._element.addEventListener('keydown', (e) => this.__elementChanged(e));
 15 
 16 
 17     }
 18 
 19     __elementChanged(e) {
 20         if(this.value) {
 21             this._setFilled();
 22         } else {
 23             this._unsetFilled();
 24         }
 25 
 26     }
 27 
 28     Validate() {
 29         
 30     }
 31 
 32 
 33     Focus() {
 34         this._element.focus();
 35         // this._element.select();
 36         this.parent.parent.AddClass('-focused');
 37     } 
 38 
 39     Blur() {
 40         this.parent.parent.RemoveClass('-focused');
 41     }
 42 
 43     _updateFieldData() {
 44         if(this._fieldData?.placeholder) {
 45             this._element.attr('placeholder', this._fieldData?.placeholder);
 46         }
 47         if(this._fieldData?.params?.readonly) {
 48             this.readonly = this._fieldData?.params?.readonly;
 49         }
 50         if(this._fieldData?.params?.align) {
 51             this._element.css('text-align', this._fieldData?.params?.align);
 52         }
 53     }
 54 
 55     get readonly() {
 56         return this._fieldData.readonly;
 57     }  
 58  
 59     set readonly(value) {
 60         this._fieldData.readonly = value === true || value === 'true';
 61         if(value === true || value === 'true') {
 62             this._element.attr('readonly', 'readonly');
 63         }
 64         else {
 65             this._element.attr('readonly', null);
 66         }
 67     }
 68 
 69     get placeholder() {
 70         return this._element.attr('placeholder');
 71     }
 72 
 73     set placeholder(value) {
 74         this._element.attr('placeholder', value ? value[Lang.Current] ?? value : '');
 75     }
 76 
 77     get value() {
 78         let v = this._element.value; 
 79         if(this._fieldData?.params?.isShare) {
 80             v = v / 100;
 81         }
 82         return v;
 83     }
 84 
 85     set value(value) {
 86 
 87         if(this._fieldData?.params?.isShare) {
 88             value = value * 100;
 89         }
 90 
 91         this._element.value = value ?? '';
 92         this.Validate();
 93         if(value) {
 94             this._setFilled();
 95         } else {
 96             this._unsetFilled();
 97         }
 98     }
 99 
100     get enabled() {
101         return this._element.attr('disabled') != 'disabled';
102     }
103 
104     set enabled(value) {
105         if(value) {
106             this.RemoveClass('ui-disabled');
107             this._element.attr('disabled', null);
108         }
109         else {
110             this.AddClass('ui-disabled');
111             this._element.attr('disabled', 'disabled');
112         }
113     }
114 
115 }
