  6 Colibri.UI.MonthYearViewer = class extends Colibri.UI.Viewer {
  7 
  8     constructor(name, container, element = null, root = null) {
  9         super(name, container, element || Element.create('span'), root);
 10         this.AddClass('app-monthyear-viewer-component');

 12 
 13     set value(value) {
 14         value = this._convertValue(value);
 15         if(typeof value == 'string') {
 16             value = (value + '-01').toDate();
 17         }
 18         const formatter = new Intl.DateTimeFormat('ru-RU', {month: 'short', year: 'numeric'});
 19         super.value = formatter.format(value)
 20     }
 21 
 22 
 23 }
