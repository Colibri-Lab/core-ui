1 /**
  2  * @class
  3  * @extends Colibri.UI.Forms.Field
  4  * @memberof Colibri.UI.Forms
  5  */
  6 Colibri.UI.Forms.Checkbox = class extends Colibri.UI.Forms.Field {
  7 
  8     RenderFieldContainer() {
  9 
 10         this.AddClass('app-component-checkbox-field');
 11 
 12         const contentContainer = this.contentContainer;
 13 
 14         this._input = new Colibri.UI.Checkbox(this._name + '-input', contentContainer);
 15         this._input.shown = true;
 16 
 17         this._label = new Colibri.UI.TextSpan(this._name + '-label', contentContainer);
 18         this._label.shown = true;
 19         this._label.AddClass('app-component-checkbox-label');
 20         this._label.value = this._fieldData.placeholder ? this._fieldData.placeholder[Lang.Current] ?? this._fieldData.placeholder : '';
 21 
 22         if(this._fieldData?.params?.readonly === undefined) {
 23             this.readonly = false;    
 24         }
 25         else {
 26             this.readonly = this._fieldData?.params?.readonly;
 27         }
 28         if(this._fieldData?.params?.enabled === undefined) {
 29             this.enabled = true;
 30         }
 31         else {
 32             this.enabled = this._fieldData.params.enabled;
 33         }
 34         this.value = this._fieldData?.default ?? false;
 35 
 36         this._handleEvents();
 37     }
 38 
 39     _handleEvents() {
 40         this._input.AddHandler('Changed', (event, args) => this.Dispatch('Changed', Object.assign(args || {}, {component: this})));
 41         this._input.AddHandler('Clicked', (event, args) => {
 42             this.Dispatch('Clicked', args)
 43             args.domEvent.stopPropagation();
 44             args.domEvent.preventDefault();
 45             return false;
 46         });
 47         this._input.AddHandler('ReceiveFocus', (event, args) => this.Dispatch('ReceiveFocus', args));
 48         this._input.AddHandler('LoosedFocus', (event, args) => this.Dispatch('LoosedFocus', args));
 49 
 50         this._label.AddHandler('Clicked', (event, args) => {
 51             this._input.Dispatch('Clicked', args);
 52             args.domEvent.stopPropagation();
 53             return false;
 54         });
 55     }
 56 
 57     Focus() {
 58         this._input.Focus();
 59     }
 60 
 61     get readonly() {
 62         return this._input.readonly;
 63     }
 64 
 65     set readonly(value) {
 66         value = this._convertProperty('Boolean', value);
 67         this._input.readonly = value;
 68         if(value) {
 69             this.AddClass('app-component-readonly');
 70         } else {
 71             this.RemoveClass('app-component-readonly');
 72         }
 73     }
 74 
 75     get value() {
 76         return this._input.checked;
 77     }
 78 
 79     set value(value) {
 80         this._input.checked = value === 'true' || value === true || value === 1 || value === '1';
 81     }
 82 
 83     get enabled() {
 84         return this._input.enabled;
 85     }
 86 
 87     set enabled(value) {
 88         value = this._convertProperty('Boolean', value);
 89         this._input.enabled = value;
 90     }
 91 
 92     /**
 93      * Индекс табуляции
 94      * @todo проверить правильно ли получаю tabIndex и исправить
 95      * @type {number}
 96      */
 97     get tabIndex() {
 98         return this._input && this._input.tabIndex;
 99     }
100     set tabIndex(value) {
101         if (this._input) {
102             this._input.tabIndex = value;
103         }
104     }
105 }
