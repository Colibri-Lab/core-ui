1 /**
  2  * @class
  3  * @extends Colibri.UI.Forms.Field
  4  * @memberof Colibri.UI.Forms
  5  */
  6 Colibri.UI.Forms.Checkbox = class extends Colibri.UI.Forms.Field {
  7 
  8     /**
  9      * Render field component
 10      */
 11     RenderFieldContainer() {
 12 
 13         this.AddClass('app-component-checkbox-field');
 14 
 15         const contentContainer = this.contentContainer;
 16 
 17         this._input = new Colibri.UI.Checkbox(this._name + '-input', contentContainer);
 18         this._input.shown = true;
 19 
 20         this._label = new Colibri.UI.TextSpan(this._name + '-label', contentContainer);
 21         this._label.shown = true;
 22         this._label.AddClass('app-component-checkbox-label');
 23         this._label.value = this._fieldData.placeholder ? this._fieldData.placeholder[Lang.Current] ?? this._fieldData.placeholder : '';
 24 
 25         if(this._fieldData?.params?.readonly === undefined) {
 26             this.readonly = false;    
 27         }
 28         else {
 29             this.readonly = this._fieldData?.params?.readonly;
 30         }
 31         if(this._fieldData?.params?.enabled === undefined) {
 32             this.enabled = true;
 33         }
 34         else {
 35             this.enabled = this._fieldData.params.enabled;
 36         }
 37         this.value = this._fieldData?.default ?? false;
 38 
 39         this._handleEvents();
 40     }
 41 
 42     /** @protected */
 43     _handleEvents() {
 44         this._input.AddHandler('Changed', (event, args) => this.Dispatch('Changed', Object.assign(args || {}, {component: this})));
 45         this._input.AddHandler('Clicked', (event, args) => {
 46             this.Dispatch('Clicked', args)
 47             args.domEvent.stopPropagation();
 48             args.domEvent.preventDefault();
 49             return false;
 50         });
 51         this._input.AddHandler('ReceiveFocus', (event, args) => this.Dispatch('ReceiveFocus', args));
 52         this._input.AddHandler('LoosedFocus', (event, args) => this.Dispatch('LoosedFocus', args));
 53 
 54         this._label.AddHandler('Clicked', (event, args) => {
 55             this._input.Dispatch('Clicked', args);
 56             args.domEvent.stopPropagation();
 57             return false;
 58         });
 59     }
 60 
 61     /**
 62      * Focus on component
 63      */
 64     Focus() {
 65         this._input.Focus();
 66     }
 67 
 68     /**
 69      * Readonly
 70      * @type {boolean}
 71      */
 72     get readonly() {
 73         return this._input.readonly;
 74     }
 75 
 76     /**
 77      * Readonly
 78      * @type {boolean}
 79      */
 80     set readonly(value) {
 81         value = this._convertProperty('Boolean', value);
 82         this._input.readonly = value;
 83         if(value) {
 84             this.AddClass('app-component-readonly');
 85         } else {
 86             this.RemoveClass('app-component-readonly');
 87         }
 88     }
 89 
 90     /**
 91      * Value
 92      * @type {boolean}
 93      */
 94     get value() {
 95         return this._input.checked;
 96     }
 97 
 98     /**
 99      * Value
100      * @type {boolean}
101      */
102     set value(value) {
103         this._input.checked = value === 'true' || value === true || value === 1 || value === '1';
104     }
105 
106     /**
107      * Enable/Disable
108      * @type {boolean}
109      */ 
110     get enabled() {
111         return this._input.enabled;
112     }
113 
114     /**
115      * Enable/Disable
116      * @type {boolean}
117      */ 
118     set enabled(value) {
119         value = this._convertProperty('Boolean', value);
120         this._input.enabled = value;
121     }
122 
123     /**
124      * Tab index
125      * @type {number}
126      */
127     get tabIndex() {
128         return this._input && this._input.tabIndex;
129     }
130     /**
131      * Tab index
132      * @type {number}
133      */
134     set tabIndex(value) {
135         if (this._input) {
136             this._input.tabIndex = value;
137         }
138     }
139 }
