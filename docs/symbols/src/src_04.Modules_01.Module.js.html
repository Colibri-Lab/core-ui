 66     /**
 67      * Handles routing for the module.
 68      * @param {string} pattern - The routing pattern.
 69      * @param {string|Function} event - The event associated with the routing pattern.
 70      * @public 
 71      */
 72     HandleRoute(pattern, event) {
 73         pattern = ('/' + this._moduleEntry.fromCamelCase('-') + '/' + pattern).replaceAll('//', '/');
 74         let handler = event;
 75         if(typeof event === 'string') {
 76             handler = (url, options) => this.Dispatch(event, {url: url, options: options});
 77         }
 78         App.Router.AddRoutePattern(pattern, handler);
 79     }

 84      * @param {Object} options - Additional navigation options.
 85      * @public 
 86      */
 87     RouteTo(url, options = {}) {
 88         url = ('/' + this._moduleEntry.fromCamelCase('-') + '/' + url).replaceAll('//', '/');
 89         App.Router.Navigate(url, options);
 90     }

128     /**
129      * Makes an RPC (Remote Procedure Call) to the specified controller and method.
130      * @param {string} controller - The name of the controller to call.
131      * @param {string} method - The name of the method to call.
132      * @param {Object|null} params - Parameters to pass to the method.
133      * @param {Object} headers - Additional headers for the RPC request.
134      * @param {boolean} withCredentials - Whether to include credentials in the request.
135      * @param {string} requestKeyword - The request keyword.
136      * @returns {Promise} - A promise representing the result of the RPC call.
137      * @public 
138      */
139     Call(controller, method, params = null, headers= {}, withCredentials= true, requestKeyword = Date.Mc()) {
140         if(!this._useAuthorizationCookie) {
141             headers = Object.assign(headers, {
142                 Authorization: Colibri.Common.Cookie.Get(this._authorizationCookieName)
143             });
144             return super.Call(controller, method, params, headers, withCredentials, requestKeyword);
145         }
146         else { 
147             return super.Call(controller, method, params, headers, withCredentials, requestKeyword);
148         }
149     }

151     /**

155     _deferedCalls = [];

157     /**
158      * Call results by requestKeyword
159      * @protected
160      */
161     _deferedResults = {};

163     /**
164      * Start timer for defered calls
165      * @param {number} timeout timeout for defer call
166      * @param {string} deferedController controller for executing defered calls
167      * @param {string} deferedMethod defered calls method
168      */
169     _startDeferedTimer(timeout = 500, deferedController, deferedMethod) {
170 
171         Colibri.Common.StartTimer(this._moduleEntry.toLowerCase() + '-defered-timer', timeout, () => {
172 
173             if(this._deferedCalls.length > 0) {
174                 while(this._deferedCalls.length > 0) {
175                     const currentCalls = [].concat(this._deferedCalls.splice(0, 10));
176                     setTimeout(() => {
177                         this.Call(deferedController, deferedMethod, {calls: currentCalls}).then((response) => {
178                             const results = response.result;
179                             for(const result of results) {
180                                 this._deferedResults['_' + result.requestKeyword] = result;
181                             }
182                         });
183                     }, 5);
184                 }
185 
186                 
187             }
188 
189         });
190 
191     }
192 
193     /**
194      * Stop timer
195      */
196     _stopDeferedTimer() {
197         Colibri.Common.StopTimer(this.name + '-defered-timer');
198     }

200     /**
201      * Makes an RPC (Remote Procedure Call) to the specified controller and method.
202      * @param {string} controller - The name of the controller to call.
203      * @param {string} method - The name of the method to call.
204      * @param {Object|null} params - Parameters to pass to the method.
205      * @param {Object} headers - Additional headers for the RPC request.
206      * @param {boolean} withCredentials - Whether to include credentials in the request.
207      * @param {string} requestKeyword - The request keyword.
208      * @returns {Promise} - A promise representing the result of the RPC call.
209      * @public 
210      */
211     Defer(controller, method, params = null, headers= {}, withCredentials= true, requestKeyword = Date.Mc()) {
212 
213         this._deferedCalls.push({
214             controller, 
215             method, 
216             params, 
217             headers, 
218             withCredentials, 
219             requestKeyword
220         });
221 
222         return new Promise((resolve, reject) => {
223             Colibri.Common.StartTimer('request_' + requestKeyword, 100, () => {
224                 if(!!this._deferedResults['_' + requestKeyword]) {
225                     Colibri.Common.StopTimer('request_' + requestKeyword);
226                     const response = this._deferedResults['_' + requestKeyword];
227                     delete this._deferedResults['_' + requestKeyword];
228                     if(response.code != 200) {
229                         reject(response);
230                     } else {
231                         resolve(response);
232                     }    
233                 }
234             });
235         }); 
236         
237         
238     }
