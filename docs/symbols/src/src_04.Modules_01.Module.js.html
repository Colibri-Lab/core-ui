1 /**
  2  * Represents a module class in the Colibri framework.
  3  * This class extends Colibri.IO.RpcRequest and provides functionality for managing modules.
  4  * @class 
  5  * @extends Colibri.IO.RpcRequest
  6  * @memberof Colibri.Modules
  7  */
  8 Colibri.Modules.Module = class extends Colibri.IO.RpcRequest {
  9 
 10     /**
 11      * Constructs an instance of the Colibri.Modules.Module class.
 12      * @public 
 13      * @param {string} entryName - The name of the module.
 14      * @param {string} [type] - The type of data.
 15      * @param {string} remoteDomain - The remote domain for the module.
 16      * @param {string} urlResolver - The URL resolver for the module.
 17      */
 18     constructor(entryName, type, remoteDomain, urlResolver) {
 19         super(entryName, type, remoteDomain, urlResolver);
 20 
 21         this._routes = {};
 22 
 23         this.RegisterEvents();
 24         this.RegisterEventHandlers();
 25         
 26     }
 27 
 28     /**
 29      * Initializes the module.
 30      * This method is automatically called during module construction.
 31      * @public 
 32      */
 33     InitializeModule() {
 34 
 35         this._authorizationCookieName = 'ss-jwt';
 36         this._useAuthorizationCookie = true;
 37         if(!this.remoteDomain && App.RemoteDomain) {
 38             this.remoteDomain = App.RemoteDomain;
 39         }
 40 
 41     }
 42 
 43     /**
 44      * Registers events for the module.
 45      * This method is automatically called during module construction.
 46      * @public 
 47      */
 48     RegisterEvents() {
 49         // Тут регистрируем все события (свои события)
 50     }
 51 
 52     /**
 53      * Registers event handlers for the module.
 54      * This method is automatically called during module construction.
 55      * @public 
 56      */
 57     RegisterEventHandlers() {
 58         // Тут регистрируем обарботчики событий
 59     }
 60 
 61     /**
 62      * Handles routing for the module.
 63      * @param {string} pattern - The routing pattern.
 64      * @param {string|Function} event - The event associated with the routing pattern.
 65      * @public 
 66      */
 67     HandleRoute(pattern, event) {
 68         pattern = ('/' + this._moduleEntry.fromCamelCase('-') + '/' + pattern).replaceAll('//', '/');
 69         let handler = event;
 70         if(typeof event === 'string') {
 71             handler = (url, options) => this.Dispatch(event, {url: url, options: options});
 72         }
 73         App.Router.AddRoutePattern(pattern, handler);
 74     }
 75 
 76     /**
 77      * Navigates to a specified URL within the module.
 78      * @param {string} url - The URL to navigate to.
 79      * @param {Object} options - Additional navigation options.
 80      * @public 
 81      */
 82     RouteTo(url, options = {}) {
 83         url = ('/' + this._moduleEntry.fromCamelCase('-') + '/' + url).replaceAll('//', '/');
 84         App.Router.Navigate(url, options);
 85     }
 86 
 87     /**
 88      * Setter for enabling/disabling the use of authorization cookies.
 89      * @param {boolean} value - Whether to use authorization cookies.
 90      * @public 
 91      */
 92     set useAuthorizationCookie(value) {
 93         this._useAuthorizationCookie = value;
 94     }
 95 
 96     /**
 97      * Getter for checking if authorization cookies are enabled.
 98      * @returns {boolean} - Whether authorization cookies are used.
 99      * @public 
100      */
101     get useAuthorizationCookie() {
102         return this._useAuthorizationCookie;
103     }
104 
105     /**
106      * Getter for retrieving the name of the authorization cookie.
107      * @returns {string} - The name of the authorization cookie.
108      * @public 
109      */
110     get authorizationCookieName() {
111         return this._authorizationCookieName;
112     }
113 
114     /**
115      * Setter for defining the name of the authorization cookie.
116      * @param {string} value - The name of the authorization cookie.
117      * @public 
118      */
119     set authorizationCookieName(value) {
120         this._authorizationCookieName = value;
121     }
122 
123     /**
124      * Makes an RPC (Remote Procedure Call) to the specified controller and method.
125      * @param {string} controller - The name of the controller to call.
126      * @param {string} method - The name of the method to call.
127      * @param {Object|null} params - Parameters to pass to the method.
128      * @param {Object} headers - Additional headers for the RPC request.
129      * @param {boolean} withCredentials - Whether to include credentials in the request.
130      * @param {string} requestKeyword - The request keyword.
131      * @returns {Promise} - A promise representing the result of the RPC call.
132      * @public 
133      */
134     Call(controller, method, params = null, headers= {}, withCredentials= true, requestKeyword = Date.Mc()) {
135         if(!this._useAuthorizationCookie) {
136             headers = Object.assign(headers, {
137                 Authorization: Colibri.Common.Cookie.Get(this._authorizationCookieName)
138             });
139             return super.Call(controller, method, params, headers, withCredentials, requestKeyword);
140         }
141         else { 
142             return super.Call(controller, method, params, headers, withCredentials, requestKeyword);
143         }
144     }
