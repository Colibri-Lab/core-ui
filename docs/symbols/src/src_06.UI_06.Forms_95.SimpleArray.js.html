  6 Colibri.UI.Forms.SimpleArray = class extends Colibri.UI.Forms.Field {
  7 
  8     RenderFieldContainer() {
  9 
 10         this.AddClass('app-component-simplearray-field');
 11         this._enabled = true;
 12 
 13         const contentContainer = this.contentContainer;
 14 
 15         this._grid = new Colibri.UI.Grid('grid', contentContainer);
 16         this._grid.shown = true;
 17         this._grid.rows.title = '';
 18 
 19         if(this._fieldData?.params?.readonly === undefined) {
 20             this.readonly = false;    
 21         }
 22         else {
 23             this.readonly = this._fieldData?.params?.readonly;
 24         }
 25         if(this._fieldData?.params?.enabled === undefined) {
 26             this.enabled = true;
 27         }
 28         else {
 29             this.enabled = this._fieldData.params.enabled;
 30         }
 31 
 32         this._fieldData.params.simplearraywidth = (this._fieldData.params?.simplearraywidth ?? 1);
 33         this._fieldData.params.simplearrayheight = (this._fieldData.params?.simplearrayheight ?? 1);
 34 
 35         const column = this._grid.header.columns.Add('col0', '');
 36         column.width = 50;
 37         column.align = 'right';
 38         
 39         for(let i=0; i < this._fieldData.params?.simplearraywidth; i++) {
 40             const column = this._grid.header.columns.Add('col' + (i + 1), (i + 1));
 41             column.editor = Colibri.UI.TextEditor;
 42             column.editorAllways = true;
 43         }
 44 
 45         for(let j=0; j < this._fieldData.params?.simplearrayheight; j++) {
 46             let val = {col0: j + 1};
 47             for(let i=0; i < this._fieldData.params?.simplearraywidth; i++) {
 48                 val['col' + (i + 1)] = '';
 49             }   
 50             this._grid.rows.Add('item' + (j + 1), val);
 51         }
 52 
 53         this._grid.AddHandler('CellEditorChanged', (event, args) => this.Dispatch('Changed', {component: this}));
 54 
 55     } 
 56 
 57     _registerEvents() {
 58         super._registerEvents();
 59         // если нужно добавить что то
 60     }
 61 
 62     Focus() {
 63         this.contentContainer.Children('firstChild').Focus();
 64     }
 65 
 66     get readonly() {
 67         return this._grid.readonly;
 68     }
 69 
 70     set readonly(value) {
 71         value = this._convertProperty('Boolean', value);
 72         this._grid.readonly = value;
 73     }
 74 
 75     get enabled() {
 76         return this._enabled ?? true;
 77     }
 78 
 79     set enabled(value) {
 80         value = this._convertProperty('Boolean', value);
 81         if(this._enabled != value) {
 82             this._enabled = value;
 83             this.contentContainer.ForEach((name, component) => {
 84                 component.enabled = this._enabled; 
 85             });
 86             this._link && (this._link.enabled = this._enabled);
 87         }
 88     }
 89 
 90     get value() {
 91 
 92         let val = [];
 93         this._grid.ForEveryRow((name, row) => {
 94             let v = [];
 95             Object.forEach(row.value, (key, value) => {
 96                 if(key != 'col0') {
 97                     v.push(value);
 98                 }
 99             });
100             val.push(v);
101         });
102         return val;
103     }
104 
105     set value(value) {
106         if(!Array.isArray(value)) {
107             value = [];
108         }
109         for(let i=0; i < value.length; i++) {
110             let v = {col0: i + 1};
111             for(let j=0; j < value[i].length; j++) {
112                 v['col' + (j + 1)] = value[i][j] ?? '';
113             }    
114             this._grid.rows.Children('item' + (i + 1)).value = v;
115         }
116     }
117 
118 
119     set tabIndex(value) {
120         this._grid.tabIndex = value;
121     }
