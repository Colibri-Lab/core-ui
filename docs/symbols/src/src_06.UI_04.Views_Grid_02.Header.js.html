  2 /**
  3  * Grid header component
  4  * @class
  5  * @extends Colibri.UI.Component
  6  * @memberof Colibri.UI.Grid
  7  */
  8 Colibri.UI.Grid.Header = class extends Colibri.UI.Component {
  9 
 10     constructor(name, container) {
 11         super(name, container, Element.create('thead'));
 12 
 13         this._sticky = true;
 14         this._grid = this.parent.parent;
 15 
 16         this._addDefaultColumns();
 17 
 18     }
 19 
 20     _registerEvents() {
 21         super._registerEvents();
 22         this.RegisterEvent('ColumnAdded', false, 'Поднимается, когда добавляется колонка');
 23         this.RegisterEvent('ColumnStickyChange', false, 'Поднимается, когда колонка меняет липкость');
 24         this.RegisterEvent('ColumnClicked', false, 'Поднимается, когда щелкнули по колонке в заголовке');
 25         this.RegisterEvent('ColumnDisposed', false, 'Поднимается, когда удалили колонку');
 26     }
 27 
 28     _handleEvents(columns, checkbox) {
 29         columns.AddHandler('ColumnAdded', (event, args) => {
 30             this.Dispatch('ColumnAdded', Object.assign(args, {columns: columns}));
 31         });
 32 
 33         columns.AddHandler('ColumnStickyChange', (event, args) => {
 34             this.Dispatch('ColumnStickyChange', Object.assign(args, {columns: columns}));
 35         });
 36 
 37         columns.AddHandler('ColumnClicked', (event, args) => {
 38             this.Dispatch('ColumnClicked', Object.assign(args, {columns: columns}));
 39         });
 40 
 41         columns.AddHandler('ColumnDisposed', (event, args) => {
 42             this.Dispatch('ColumnDisposed', Object.assign(args, {columns: columns}));
 43         });
 44         checkbox.AddHandler('Changed', (event, args) => {
 45             Object.forEach(this.grid.groups, (name, group) => {
 46                 group.ForEach((name, row) => {
 47                     if(row.shown) {
 48                         row.checked = args.value;
 49                     }
 50                     else {
 51                         row.checked = false;
 52                     }
 53                 });    
 54                 group.checkbox.checked = args.value;
 55             });
 56             this.checkbox.thirdState = this.grid.rowsCount > this.grid.checked.length;
 57             Object.forEach(this.grid.groups, (name, group) => {
 58                 group.checkbox.thirdState = group.rowsCount > group.checked.length;
 59             });
 60             this.grid.Dispatch('RowSelected');
 61         });
 62     }
 63 
 64     AddColumns(name) {
 65         const columns = new Colibri.UI.Grid.Columns(name, this);
 66         columns.AddClass('app-ui-header-columns');
 67         columns.shown = true;
 68 
 69         let columnSelectionButtons = columns.Add('button-container-for-row-selection', '');
 70         columnSelectionButtons.shown = this.grid.showCheckboxes;
 71 
 72         const checkbox = new Colibri.UI.Checkbox('row-checkbox', columnSelectionButtons);
 73         checkbox.hasThirdState = true;
 74         checkbox.shown = true;
 75 
 76         checkbox.AddHandler('ContextMenuItemClicked', (event, args) => {
 77             this.grid.Dispatch('RowsCheckboxContextMenuItemClicked', Object.assign(args, {rows: this, checkbox: this._checkbox})); 
 78         });
 79         checkbox.AddHandler('MouseDown', (event, args) => {
 80             this.grid.Dispatch('RowsCheckboxClicked', {rows: this, checkbox: this._checkbox, domEvent: args.domEvent}); 
 81         });
 82 
 83 
 84         this._handleEvents(columns, checkbox);
 85         return [columns, checkbox];
 86     }
 87 
 88     _addDefaultColumns() {
 89         const newColumns = this.AddColumns('app-ui-header-columns');
 90         this._columns = newColumns[0];
 91         this._checkbox = newColumns[1];
 92     }
 93 
 94     Reset() {
 95         this.Clear();
 96         this._addDefaultColumns();
 97     }
 98 
 99 
100     /**
101      * Gets checkbox
102      * @type {Boolean}
103      * @readonly
104      */
105     get checkbox() {
106         return this._checkbox;
107     }
108 
109     get columns() {
110         return this._columns;
111     }
112 
113     get grid() {
114         return this._grid;
115     }
116 
117     get sticky() {
118         return this._sticky;
119     }
120 
121     FindColumn(colName) {
122         let found = null;
123         this.ForEach((name, columns) => {
124             if(columns.Children(colName)) {
125                 found = columns.Children(colName);
126                 return false;
127             }
128         });
129         return found;
130     }
131 
132     FindAllColumns() {
133         let ret = {};
134         this.ForEach((name, columns) => {
135             columns.ForEach((n, col) => {
136                 if(n != 'button-container-for-row-selection') {
137                     ret[n] = col;
138                 }
139             });
140         });
141 
142         // we need to reorder list
143         const hasIndexes = Array.sum(Object.values(ret).map(col => !!col.index ? 1 : 0)) > 0;
144         if(!hasIndexes) {
145             return ret;
146         } else {
147             let colArray = [];
148             for(const col of Object.values(ret)) {
149                 if(!!col.index) {
150                     colArray[parseInt(col.index)] = col;
151                 }
152             }
153             ret = {};
154             colArray.sort();
155             for(const col of colArray) {
156                 if(col) {
157                     ret[col.name] = col;
158                 }
159             }
160         }
161         return ret;
162     }
163 
164     /**
165      * @type {Number}
166      * @readonly
167      */
168     get columnsCount() {
169         let count = 0;
170         this._columns.ForEach((name, column) => {
171             if(name != 'button-container-for-row-selection') {
172                 count += parseInt(column.colspan ?? 1);
173             }
174         })
175         return count;
176     }
